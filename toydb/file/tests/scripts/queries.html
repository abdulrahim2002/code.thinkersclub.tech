     68 # Try a complex multi-way join with multiple joins of the same table. Uses GROUP

     86 # Explain that query.
     87 > EXPLAIN SELECT m.id, m.title, g.name AS genre, s.name AS studio, m.rating \
     88   FROM movies m JOIN genres g ON m.genre_id = g.id, \
     89     studios s JOIN movies good ON good.studio_id = s.id AND good.rating >= 8 \
     90   WHERE m.studio_id = s.id \
     91   GROUP BY m.id, m.title, g.name, s.name, m.rating, m.released \
     92   ORDER BY m.rating DESC, m.released ASC, m.id ASC
     93 ---
