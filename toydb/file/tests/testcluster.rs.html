     19 /// Runs a toyDB cluster using the built binary in a temporary directory. The
     20 /// cluster will be killed and removed when dropped.
     21 ///
     22 /// This runs the cluster as child processes using the built binary instead of
     23 /// spawning in-memory threads for a couple of reasons: it avoids having to
     24 /// gracefully shut down the server (which is complicated by e.g.
     25 /// TcpListener::accept() not being interruptable), and it tests the entire
     26 /// server (and eventually the toySQL client) end-to-end.
     27 pub struct TestCluster {
     28     servers: BTreeMap<NodeID, TestServer>,
     29     #[allow(dead_code)]
     30     dir: tempfile::TempDir, // deleted when dropped
     31 }
     32 
     33 type NodePorts = BTreeMap<NodeID, (u16, u16)>; // raft,sql on localhost
     34 
     35 impl TestCluster {
     36     /// Runs and returns a test cluster. It keeps running until dropped.
     37     pub fn run(nodes: u8) -> Result<Self, Box<dyn Error>> {
     38         // Create temporary directory.
     39         let dir = tempfile::TempDir::with_prefix("toydb")?;
     40 
     41         // Allocate port numbers for nodes.
     42         let ports: NodePorts = (1..=nodes)
     43             .map(|id| (id, (RAFT_BASE_PORT + id as u16, SQL_BASE_PORT + id as u16)))
     44             .collect();
     45 
     46         // Start nodes.
     47         let mut servers = BTreeMap::new();
     48         for id in 1..=nodes {
     49             let dir = dir.path().join(format!("toydb{id}"));
     50             servers.insert(id, TestServer::run(id, &dir, &ports)?);
     51         }
     52 
     53         // Wait for the nodes to be ready, by fetching the server status.
     54         let started = std::time::Instant::now();
     55         for server in servers.values_mut() {
     56             while let Err(error) = server.connect().and_then(|mut c| Ok(c.status()?)) {
     57                 server.assert_alive();
     58                 if started.elapsed() >= TIMEOUT {
     59                     return Err(error);
     60                 }
     61                 std::thread::sleep(Duration::from_millis(200));
     62             }
     63         }
     64 
     65         Ok(Self { servers, dir })
     66     }
     67 
     68     /// Connects to a random cluster node using a Rust client. Testing with
     69     /// toysql is too annoying, since we have to deal with rustyline, PTYs,
     70     /// echoing, multiline editing, etc.
     71     pub fn connect(&self) -> Result<Client, Box<dyn Error>> {
     72         let id = rand::thread_rng().gen_range(1..=self.servers.len()) as NodeID;
     73         self.servers.get(&id).unwrap().connect()
     74     }
     75 }

     77 /// A toyDB server.
     78 pub struct TestServer {
     79     id: NodeID,
     80     child: std::process::Child,
     81     sql_port: u16,
     82 }
     83 
     84 impl TestServer {
     85     /// Runs a toyDB server.
     86     fn run(id: NodeID, dir: &Path, ports: &NodePorts) -> Result<Self, Box<dyn Error>> {
     87         // Build and write the configuration file.
     88         let configfile = dir.join("toydb.yaml");
     89         std::fs::create_dir_all(dir)?;
     90         std::fs::write(&configfile, Self::build_config(id, dir, ports)?)?;
     91 
     92         // Build the binary.
     93         //
     94         // TODO: this may contribute to slow tests, consider building once.
     95         let build = escargot::CargoBuild::new().bin("toydb").run()?;
     96 
     97         // Spawn process. Discard output.
     98         let child = build
     99             .command()
    100             .args(["-c", &configfile.to_string_lossy()])
    101             .stdout(std::process::Stdio::null())
    102             .stderr(std::process::Stdio::null())
    103             .spawn()?;
    104 
    105         let (_, sql_port) = ports.get(&id).copied().expect("node not in ports");
    106         Ok(Self { id, child, sql_port })
    107     }
    108 
    109     /// Generates a config file for the given node.
    110     fn build_config(id: NodeID, dir: &Path, ports: &NodePorts) -> Result<String, Box<dyn Error>> {
    111         let (raft_port, sql_port) = ports.get(&id).expect("node not in ports");
    112         let mut cfg = String::new();
    113         writeln!(cfg, "id: {id}")?;
    114         writeln!(cfg, "data_dir: {}", dir.to_string_lossy())?;
    115         writeln!(cfg, "listen_raft: localhost:{raft_port}")?;
    116         writeln!(cfg, "listen_sql: localhost:{sql_port}")?;
    117         writeln!(cfg, "peers: {{")?;
    118         for (peer_id, (peer_raft_port, _)) in ports.iter().filter(|(peer, _)| **peer != id) {
    119             writeln!(cfg, "  '{peer_id}': localhost:{peer_raft_port},")?;
    120         }
    121         writeln!(cfg, "}}")?;
    122         Ok(cfg)
    123     }
    124 
    125     /// Asserts that the server is still running.
    126     fn assert_alive(&mut self) {
    127         if let Some(status) = self.child.try_wait().expect("failed to check exit status") {
    128             panic!("node {id} exited with status {status}", id = self.id)
    129         }
    130     }
    131 
    132     /// Connects to the server using a regular client.
    133     fn connect(&self) -> Result<Client, Box<dyn Error>> {
    134         Ok(Client::connect(("localhost", self.sql_port))?)
    135     }
    136 }

    138 impl Drop for TestServer {
    139     // Kills the child process when dropped.
    140     fn drop(&mut self) {
    141         self.child.kill().expect("failed to kill node");
    142         self.child.wait().expect("failed to wait for node to terminate");
    143     }
