      1 #!/usr/bin/env bash
      2 #
      3 # This script builds and runs a 5-node toyDB cluster listening on ports
      4 # 9601-9605. Config and data is stored under the toydb* directories.
      5 # To connect a toysql client to node 5 on port 9605, run:
      6 #
      7 # cargo run --release --bin toysql
      8 
      9 set -euo pipefail
     10 
     11 # Change into the script directory.
     12 cd "$(dirname $0)"
     13 
     14 # Build toyDB using release optimizations.
     15 cargo build --release --bin toydb
     16 
     17 # Start nodes 1-5 in the background, prefixing their output with the node ID.
     18 echo "Starting 5 nodes on ports 9601-9605 with data under cluster/*/data/."
     19 echo "To connect to node 5, run: cargo run --release --bin toysql"
     20 echo ""
     21 
     22 for ID in 1 2 3 4 5; do
     23     (cargo run -q --release -- -c toydb$ID/toydb.yaml 2>&1 | sed -e "s/\\(.*\\)/toydb$ID \\1/g") &
     24 done
     25 
     26 # Wait for the background processes to exit. Kill all toyDB processes when the
     27 # script exits (e.g. via Ctrl-C).
     28 trap 'kill $(jobs -p)' EXIT
     29 wait < <(jobs -p)
