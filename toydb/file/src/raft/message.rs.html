     23 /// A message sent between Raft nodes. Messages are sent asynchronously (i.e.
     24 /// they are not request/response) and may be dropped or reordered.
     25 ///
     26 /// In practice, they are sent across a TCP connection and crossbeam channels
     27 /// ensuring messages are not dropped or reordered as long as the connection
     28 /// remains intact. A message and its response are sent across separate TCP
     29 /// connections (outbound from their respective senders).
     30 #[derive(Clone, Debug, PartialEq, Serialize, Deserialize)]
     31 pub enum Message {
     32     /// Candidates campaign for leadership by soliciting votes from peers.
     33     /// Votes will only be granted if the candidate's log is at least as
     34     /// up-to-date as the voter.
     35     Campaign {
     36         /// The index of the candidate's last log entry.
     37         last_index: Index,
     38         /// The term of the candidate's last log entry.
     39         last_term: Term,
     40     },
     41 
     42     /// Followers may vote for a single candidate per term, but only if the
     43     /// candidate's log is at least as up-to-date as the follower. Candidates
     44     /// implicitly vote for themselves.
     45     CampaignResponse {
     46         /// If true, the follower granted the candidate a vote. A false response
     47         /// isn't necessary, but is emitted for clarity.
     48         vote: bool,
     49     },
     50 
     51     /// Leaders send periodic heartbeats. This serves several purposes:
     52     ///
     53     /// * Inform nodes about the leader, and prevent elections.
     54     /// * Detect lost appends and reads, as a retry mechanism.
     55     /// * Advance followers' commit indexes, so they can apply entries.
     56     ///
     57     /// The Raft paper does not have a distinct heartbeat message, and instead
     58     /// uses an empty AppendEntries RPC, but we choose to add one for better
     59     /// separation of concerns.
     60     Heartbeat {
     61         /// The index of the leader's last log entry. The term is the leader's
     62         /// current term, since it appends a noop entry on election win. The
     63         /// follower compares this to its own log to determine if it's
     64         /// up-to-date.
     65         last_index: Index,
     66         /// The index of the leader's last committed log entry. Followers use
     67         /// this to advance their commit index and apply entries. It's only safe
     68         /// to commit this if the local log matches last_index, such that the
     69         /// follower's log is identical to the leader at the commit index.
     70         commit_index: Index,
     71         /// The leader's latest read sequence number in this term.
     72         read_seq: ReadSequence,
     73     },
     74 
     75     /// Followers respond to leader heartbeats if they still consider it leader.
     76     HeartbeatResponse {
     77         /// If non-zero, the heartbeat's last_index which was matched in the
     78         /// follower's log. Otherwise, the follower is either divergent or
     79         /// lagging behind the leader.
     80         match_index: Index,
     81         /// The heartbeat's read sequence number.
     82         read_seq: ReadSequence,
     83     },
     84 
     85     /// Leaders replicate log entries to followers by appending to their logs
     86     /// after the given base entry.
     87     ///
     88     /// If the base entry matches the follower's log then their logs are
     89     /// identical up to it (see section 5.3 in the Raft paper), and the entries
     90     /// can be appended -- possibly replacing conflicting entries. Otherwise,
     91     /// the append is rejected and the leader must retry an earlier base index
     92     /// until a common base is found.
     93     ///
     94     /// Empty appends messages (no entries) are used to probe follower logs for
     95     /// a common match index in the case of divergent logs, restarted nodes, or
     96     /// dropped messages. This is typically done by sending probes with a
     97     /// decrementing base index until a match is found, at which point the
     98     /// subsequent entries can be sent.
     99     Append {
    100         /// The index of the log entry to append after.
    101         base_index: Index,
    102         /// The  term of the base entry.
    103         base_term: Term,
    104         /// Log entries to append. Must start at base_index + 1.
    105         entries: Vec<Entry>,
    106     },
    107 
    108     /// Followers accept or reject appends from the leader depending on whether
    109     /// the base entry matches their log.
    110     AppendResponse {
    111         /// If non-zero, the follower appended entries up to this index. The
    112         /// entire log up to this index is consistent with the leader. If no
    113         /// entries were sent (a probe), this will be the matching base index.
    114         match_index: Index,
    115         /// If non-zero, the follower rejected an append at this base index
    116         /// because the base index/term did not match its log. If the follower's
    117         /// log is shorter than the base index, the reject index will be lowered
    118         /// to the index after its last local index, to avoid probing each
    119         /// missing index.
    120         reject_index: Index,
    121     },
    122 
    123     /// Leaders need to confirm they are still the leader before serving reads,
    124     /// to guarantee linearizability in case a different leader has been
    125     /// estalished elsewhere. Read requests are served once the sequence number
    126     /// has been confirmed by a quorum.
    127     Read { seq: ReadSequence },
    128 
    129     /// Followers confirm leadership at the read sequence numbers.
    130     ReadResponse { seq: ReadSequence },
    131 
    132     /// A client request. This can be submitted to the leader, or to a follower
    133     /// which will forward it to its leader. If there is no leader, or the
    134     /// leader or term changes, the request is aborted with an Error::Abort
    135     /// ClientResponse and the client must retry.
    136     ClientRequest {
    137         /// The request ID. Must be globally unique for the request duration.
    138         id: RequestID,
    139         /// The request itself.
    140         request: Request,
    141     },
    142 
    143     /// A client response.
    144     ClientResponse {
    145         /// The ID of the original ClientRequest.
    146         id: RequestID,
    147         /// The response, or an error.
    148         response: Result<Response>,
    149     },
    150 }
    151 
    152 /// A client request ID. Must be globally unique for the duration of the
    153 /// request. For simplicity, a random UUIDv4 is used -- the node ID and process
    154 /// ID could be incorporated for further collision avoidance, but it does not
    155 /// matter at this scale.
    156 pub type RequestID = uuid::Uuid;
    157 
    158 /// A read sequence number, used to confirm leadership for linearizable reads.
    159 pub type ReadSequence = u64;

    161 /// A client request, typically passed through to the state machine.
    162 #[derive(Clone, Debug, PartialEq, Serialize, Deserialize)]
    163 pub enum Request {
    164     /// A state machine read command, executed via `State::read`. This is not
    165     /// replicated, and only evaluated on the leader.
    166     Read(Vec<u8>),
    167     /// A state machine write command, executed via `State::apply`. This is
    168     /// replicated across all nodes, and must produce a deterministic result.
    169     Write(Vec<u8>),
    170     /// Requests Raft cluster status from the leader.
    171     Status,
    172 }
    173 
    174 impl encoding::Value for Request {}
