     43 /// A Raft node with a dynamic role. This implements the Raft distributed
     44 /// consensus protocol, see the `raft` module documentation for more info.
     45 ///
     46 /// The node is driven synchronously by processing inbound messages via `step()`
     47 /// and by advancing time via `tick()`. These methods consume the node and
     48 /// return a new one with a possibly different role. Outbound messages are sent
     49 /// via the given `tx` channel, and must be delivered to peers or clients.
     50 ///
     51 /// This enum is the public interface to the node, with a closed set of roles.
     52 /// It wraps the `RawNode<Role>` types, which implement the actual node logic.
     53 /// The enum allows ergonomic use across role transitions since it can represent
     54 /// all roles, e.g.: `node = node.step()?`.
     55 pub enum Node {
     56     /// A candidate campaigns for leadership.
     57     Candidate(RawNode<Candidate>),
     58     /// A follower replicates entries from a leader.
     59     Follower(RawNode<Follower>),
     60     /// A leader processes client requests and replicates entries to followers.
     61     Leader(RawNode<Leader>),
     62 }

     64 /// Helper macro which calls a closure on the inner RawNode<Role>.
     65 macro_rules! with_rawnode {
     66     // Node is moved.
     67     ($node:expr, $closure:expr) => {{
     68         fn with<R: Role, T>(node: RawNode<R>, f: impl FnOnce(RawNode<R>) -> T) -> T {
     69             f(node)
     70         }
     71         match $node {
     72             Node::Candidate(node) => with(node, $closure),
     73             Node::Follower(node) => with(node, $closure),
     74             Node::Leader(node) => with(node, $closure),
     75         }
     76     }};
     77     // Node is borrowed (ref).
     78     (ref $node:expr, $closure:expr) => {{
     79         fn with<R: Role, T>(node: &RawNode<R>, f: impl FnOnce(&RawNode<R>) -> T) -> T {
     80             f(node)
     81         }
     82         match $node {
     83             Node::Candidate(ref node) => with(node, $closure),
     84             Node::Follower(ref node) => with(node, $closure),
     85             Node::Leader(ref node) => with(node, $closure),
     86         }
     87     }};
     88     // Node is mutably borrowed (ref mut).
     89     (ref mut $node:expr, $closure:expr) => {{
     90         fn with<R: Role, T>(node: &mut RawNode<R>, f: impl FnOnce(&mut RawNode<R>) -> T) -> T {
     91             f(node)
     92         }
     93         match $node {
     94             Node::Candidate(ref mut node) => with(node, $closure),
     95             Node::Follower(ref mut node) => with(node, $closure),
     96             Node::Leader(ref mut node) => with(node, $closure),
     97         }
     98     }};
     99 }

    103     /// hear from a leader or otherwise transitioning to candidate and
    104     /// campaigning for leadership. In the case of a single-node cluster (no
    105     /// peers), the node immediately transitions to leader when created.

    107         id: NodeID,
    108         peers: HashSet<NodeID>,
    109         log: Log,
    110         state: Box<dyn State>,
    111         tx: Sender<Envelope>,

    118         }
    119         Ok(node.into())
    120     }

    169 /// A Raft node with role R.
    170 ///
    171 /// This implements the typestate pattern, where individual node states (roles)
    172 /// are encoded as RawNode<Role>. See: http://cliffle.com/blog/rust-typestate/
    173 pub struct RawNode<R: Role> {
    174     /// The node ID. Must be unique in this cluster.
    175     id: NodeID,
    176     /// The IDs of the other nodes in the cluster. Does not change while
    177     /// running. Can change on restart, but all nodes must have the same node
    178     /// set to avoid multiple leaders (i.e. split brain).
    179     peers: HashSet<NodeID>,
    180     /// The Raft log, containing client commands to be executed.
    181     log: Log,
    182     /// The Raft state machine, on which client commands are executed.
    183     state: Box<dyn State>,
    184     /// Channel for sending outbound messages to other nodes.
    185     tx: Sender<Envelope>,
    186     /// Node options.
    187     opts: Options,
    188     /// Role-specific state.
    189     role: R,
    190 }

    221     /// Returns the quorum value of the given unsorted vector, in descending

    224         assert_eq!(values.len(), self.cluster_size(), "vector size must match cluster size");

    279     /// Creates a new node as a leaderless follower.
    280     fn new(
    281         id: NodeID,
    282         peers: HashSet<NodeID>,
    283         log: Log,
    284         state: Box<dyn State>,
    285         tx: Sender<Envelope>,
    286         opts: Options,
    287     ) -> Result<Self> {
    288         if peers.contains(&id) {
    289             return errinput!("node ID {id} can't be in peers");
    290         }
    291         let role = Follower::new(None, 0);
    292         let mut node = Self { id, peers, log, state, tx, opts, role };
    293         node.role.election_timeout = node.random_election_timeout();
    294 
    295         // Apply any pending entries following restart. Unlike the Raft log,
    296         // state machine writes are not flushed to durable storage, so a tail of
    297         // writes may be lost if the OS crashes or restarts.
    298         node.maybe_apply()?;
    299         Ok(node)
    300     }

    302     /// Transitions the follower into a candidate, by campaigning for
    303     /// leadership in a new term.
    304     fn into_candidate(mut self) -> Result<RawNode<Candidate>> {
    305         // Abort any forwarded requests. These must be retried with new leader.
    306         self.abort_forwarded()?;
    307 
    308         // Apply any pending log entries, so that we're caught up if we win.
    309         self.maybe_apply()?;
    310 
    311         // Become candidate and campaign.
    312         let election_timeout = self.random_election_timeout();
    313         let mut node = self.into_role(Candidate::new(election_timeout));
    314         node.campaign()?;
    315 
    316         let (term, vote) = node.log.get_term();
    317         assert!(node.role.votes.contains(&node.id), "candidate did not vote for self");
    318         assert_ne!(term, 0, "candidate can't have term 0");
    319         assert_eq!(vote, Some(node.id), "log vote does not match self");
    320 
    321         Ok(node)
    322     }

    324     /// Transitions the follower into either a leaderless follower in a new term
    325     /// (e.g. if someone holds a new election) or a follower of a current leader.
    326     fn into_follower(mut self, term: Term, leader: Option<NodeID>) -> Result<RawNode<Follower>> {
    327         assert_ne!(term, 0, "can't become follower in term 0");
    328 
    329         // Abort any forwarded requests. These must be retried with new leader.
    330         self.abort_forwarded()?;
    331 
    332         if let Some(leader) = leader {
    333             // We found a leader in the current term.
    334             assert!(self.peers.contains(&leader), "leader is not a peer");
    335             assert_eq!(self.role.leader, None, "already have leader in term");
    336             assert_eq!(term, self.term(), "can't follow leader in different term");
    337             info!("Following leader {leader} in term {term}");
    338             self.role = Follower::new(Some(leader), self.role.election_timeout);
    339         } else {
    340             // We found a new term, but we don't know who the leader is yet.
    341             // We'll find out if we step a message from it.
    342             assert_ne!(term, self.term(), "can't become leaderless follower in current term");
    343             info!("Discovered new term {term}");
    344             self.log.set_term(term, None)?;
    345             self.role = Follower::new(None, self.random_election_timeout());
    346         }
    347         Ok(self)
    348     }

    350     /// Processes an inbound message.
    351     fn step(mut self, msg: Envelope) -> Result<Node> {
    352         // Past term: drop the message.
    353         if msg.term < self.term() {
    354             debug!("Dropping message from past term: {msg:?}");
    355             return Ok(self.into());
    356         }
    357         // Future term: become leaderless follower and step the message.
    358         if msg.term > self.term() {
    359             return self.into_follower(msg.term, None)?.step(msg);
    360         }
    361 
    362         // Record when we last saw a message from the leader (if any).
    363         if Some(msg.from) == self.role.leader {
    364             self.role.leader_seen = 0
    365         }
    366 
    367         match msg.message {
    368             // The leader sends periodic heartbeats. If we don't have a leader
    369             // yet, follow it. If the commit_index advances, apply commands.
    370             Message::Heartbeat { last_index, commit_index, read_seq } => {
    371                 assert!(commit_index <= last_index, "commit_index after last_index");
    372 
    373                 // Make sure the heartbeat is from our leader, or follow it.
    374                 match self.role.leader {
    375                     Some(leader) => assert_eq!(msg.from, leader, "multiple leaders in term"),
    376                     None => self = self.into_follower(msg.term, Some(msg.from))?,
    377                 }
    378 
    379                 // Attempt to match the leader's log and respond to the
    380                 // heartbeat. last_index always has the leader's term.
    381                 let match_index = if self.log.has(last_index, msg.term)? { last_index } else { 0 };
    382                 self.send(msg.from, Message::HeartbeatResponse { match_index, read_seq })?;
    383 
    384                 // Advance the commit index and apply entries. We can only do
    385                 // this if we matched the leader's last_index, which implies
    386                 // that the logs are identical up to match_index. This also
    387                 // implies that the commit_index is present in our log.
    388                 if match_index != 0 && commit_index > self.log.get_commit_index().0 {
    389                     self.log.commit(commit_index)?;
    390                     self.maybe_apply()?;
    391                 }
    392             }
    393 
    394             // Append log entries from the leader to the local log.
    395             Message::Append { base_index, base_term, entries } => {
    396                 if let Some(first) = entries.first() {
    397                     assert_eq!(base_index, first.index - 1, "base index mismatch");
    398                 }
    399 
    400                 // Make sure the append is from our leader, or follow it.
    401                 match self.role.leader {
    402                     Some(leader) => assert_eq!(msg.from, leader, "multiple leaders in term"),
    403                     None => self = self.into_follower(msg.term, Some(msg.from))?,
    404                 }
    405 
    406                 // If the base entry matches our log, append the entries.
    407                 let (mut reject_index, mut match_index) = (0, 0);
    408                 if base_index == 0 || self.log.has(base_index, base_term)? {
    409                     match_index = entries.last().map(|e| e.index).unwrap_or(base_index);
    410                     self.log.splice(entries)?;
    411                 } else {
    412                     // Otherwise, reject the base index. If the local log is
    413                     // shorter than the base index, lower the reject index to
    414                     // skip all missing entries.
    415                     reject_index = std::cmp::min(base_index, self.log.get_last_index().0 + 1);
    416                 }
    417                 self.send(msg.from, Message::AppendResponse { reject_index, match_index })?;
    418             }
    419 
    420             // Confirm the leader's read sequence number.
    421             Message::Read { seq } => {
    422                 // Make sure the read is from our leader, or follow it.
    423                 match self.role.leader {
    424                     Some(leader) => assert_eq!(msg.from, leader, "multiple leaders in term"),
    425                     None => self = self.into_follower(msg.term, Some(msg.from))?,
    426                 }
    427 
    428                 // Confirm the read.
    429                 self.send(msg.from, Message::ReadResponse { seq })?;
    430             }
    431 
    432             // A candidate is requesting our vote. We'll only grant one.
    433             Message::Campaign { last_index, last_term } => {
    434                 // Don't vote if we already voted for someone else in this term.
    435                 // We can repeat our vote though.
    436                 if let (_, Some(vote)) = self.log.get_term() {
    437                     if msg.from != vote {
    438                         self.send(msg.from, Message::CampaignResponse { vote: false })?;
    439                         return Ok(self.into());
    440                     }
    441                 }
    442 
    443                 // Don't vote if our log is newer than the candidate's log.
    444                 // This ensures that an elected leader has all committed
    445                 // entries, see section 5.4.1 in the Raft paper.
    446                 let (log_index, log_term) = self.log.get_last_index();
    447                 if log_term > last_term || log_term == last_term && log_index > last_index {
    448                     self.send(msg.from, Message::CampaignResponse { vote: false })?;
    449                     return Ok(self.into());
    450                 }
    451 
    452                 // Grant the vote.
    453                 info!("Voting for {} in term {} election", msg.from, msg.term);
    454                 self.log.set_term(msg.term, Some(msg.from))?;
    455                 self.send(msg.from, Message::CampaignResponse { vote: true })?;
    456             }
    457 
    458             // Forward client requests to the leader, or abort them if there is
    459             // none. These will not be retried, the client should use timeouts.
    460             // Local client requests use our node ID as the sender.
    461             Message::ClientRequest { id, request: _ } => {
    462                 assert_eq!(msg.from, self.id, "client request from other node");
    463 
    464                 if let Some(leader) = self.role.leader {
    465                     debug!("Forwarding request to leader {leader}: {msg:?}");
    466                     self.role.forwarded.insert(id);
    467                     self.send(leader, msg.message)?
    468                 } else {
    469                     let response = Err(Error::Abort);
    470                     self.send(msg.from, Message::ClientResponse { id, response })?
    471                 }
    472             }
    473 
    474             // Client responses from the leader are passed on to the client.
    475             Message::ClientResponse { id, response } => {
    476                 assert_eq!(Some(msg.from), self.role.leader, "client response from non-leader");
    477 
    478                 if self.role.forwarded.remove(&id) {
    479                     self.send(self.id, Message::ClientResponse { id, response })?;
    480                 }
    481             }
    482 
    483             // We may receive a vote after we lost an election, ignore it.
    484             Message::CampaignResponse { .. } => {}
    485 
    486             // We're not leader this term, so we shouldn't see these.
    487             Message::HeartbeatResponse { .. }
    488             | Message::AppendResponse { .. }
    489             | Message::ReadResponse { .. } => {
    490                 panic!("unexpected message {msg:?}")
    491             }
    492         };
    493         Ok(self.into())
    494     }
    495 
    496     /// Processes a logical clock tick.
    497     fn tick(mut self) -> Result<Node> {
    498         self.role.leader_seen += 1;
    499         if self.role.leader_seen >= self.role.election_timeout {
    500             return Ok(self.into_candidate()?.into());
    501         }
    502         Ok(self.into())
    503     }
    504 
    505     /// Aborts all forwarded requests (e.g. on term/leader changes).
    506     fn abort_forwarded(&mut self) -> Result<()> {
    507         // Sort by ID for test determinism.
    508         for id in std::mem::take(&mut self.role.forwarded).into_iter().sorted() {
    509             debug!("Aborting forwarded request {id}");
    510             self.send(self.id, Message::ClientResponse { id, response: Err(Error::Abort) })?;
    511         }
    512         Ok(())
    513     }

    515     /// Applies any pending log entries.
    516     fn maybe_apply(&mut self) -> Result<()> {
    517         let mut iter = self.log.scan_apply(self.state.get_applied_index());
    518         while let Some(entry) = iter.next().transpose()? {
    519             debug!("Applying {entry:?}");
    520             // Throw away the result, since only the leader responds to clients.
    521             // This includes errors -- any non-deterministic errors (e.g. IO
    522             // errors) must panic instead to avoid replica divergence.
    523             _ = self.state.apply(entry);
    524         }
    525         Ok(())
    526     }

    549     /// Transitions the candidate to a follower. We either lost the election and
    550     /// follow the winner, or we discovered a new term and step into it as a
    551     /// leaderless follower.
    552     fn into_follower(mut self, term: Term, leader: Option<NodeID>) -> Result<RawNode<Follower>> {
    553         let election_timeout = self.random_election_timeout();
    554         if let Some(leader) = leader {
    555             // We lost the election, follow the winner.
    556             assert_eq!(term, self.term(), "can't follow leader in different term");
    557             info!("Lost election, following leader {leader} in term {term}");
    558             Ok(self.into_role(Follower::new(Some(leader), election_timeout)))
    559         } else {
    560             // We found a new term, but we don't necessarily know who the leader
    561             // is yet. We'll find out when we step a message from it.
    562             assert_ne!(term, self.term(), "can't become leaderless follower in current term");
    563             info!("Discovered new term {term}");
    564             self.log.set_term(term, None)?;
    565             Ok(self.into_role(Follower::new(None, election_timeout)))
    566         }
    567     }

    569     /// Transitions the candidate to a leader. We won the election.
    570     fn into_leader(self) -> Result<RawNode<Leader>> {
    571         let (term, vote) = self.log.get_term();
    572         assert_ne!(term, 0, "leaders can't have term 0");
    573         assert_eq!(vote, Some(self.id), "leader did not vote for self");
    574 
    575         info!("Won election for term {term}, becoming leader");
    576         let peers = self.peers.clone();
    577         let (last_index, _) = self.log.get_last_index();
    578         let mut node = self.into_role(Leader::new(peers, last_index));
    579 
    580         // Propose an empty command when assuming leadership, to disambiguate
    581         // previous entries in the log. See section 5.4.2 in the Raft paper.
    582         // We do this prior to the heartbeat, to avoid a wasted replication
    583         // roundtrip if the heartbeat response indicates the peer is behind.
    584         node.propose(None)?;
    585         node.maybe_commit_and_apply()?;
    586         node.heartbeat()?;
    587 
    588         Ok(node)
    589     }
    590 
    591     /// Processes an inbound message.
    592     fn step(mut self, msg: Envelope) -> Result<Node> {
    593         // Past term: drop the message.
    594         if msg.term < self.term() {
    595             debug!("Dropping message from past term: {msg:?}");
    596             return Ok(self.into());
    597         }
    598         // Future term: become leaderless follower and step the message.
    599         if msg.term > self.term() {
    600             return self.into_follower(msg.term, None)?.step(msg);
    601         }
    602 
    603         match msg.message {
    604             // If we received a vote, record it. If the vote gives us quorum,
    605             // assume leadership.
    606             Message::CampaignResponse { vote: true } => {
    607                 self.role.votes.insert(msg.from);
    608                 if self.role.votes.len() >= self.quorum_size() {
    609                     return Ok(self.into_leader()?.into());
    610                 }
    611             }
    612 
    613             // We didn't get the vote. :(
    614             Message::CampaignResponse { vote: false } => {}
    615 
    616             // Don't grant votes for other candidates.
    617             Message::Campaign { .. } => {
    618                 self.send(msg.from, Message::CampaignResponse { vote: false })?
    619             }
    620 
    621             // If we hear from a leader in this term, we lost the election.
    622             // Follow it and step the message.
    623             Message::Heartbeat { .. } | Message::Append { .. } | Message::Read { .. } => {
    624                 return self.into_follower(msg.term, Some(msg.from))?.step(msg);
    625             }
    626 
    627             // Abort client requests while campaigning. The client must retry.
    628             Message::ClientRequest { id, request: _ } => {
    629                 self.send(msg.from, Message::ClientResponse { id, response: Err(Error::Abort) })?;
    630             }
    631 
    632             // We're not a leader in this term, nor are we forwarding requests,
    633             // so we shouldn't see these.
    634             Message::HeartbeatResponse { .. }
    635             | Message::AppendResponse { .. }
    636             | Message::ReadResponse { .. }
    637             | Message::ClientResponse { .. } => panic!("unexpected message {msg:?}"),
    638         }
    639         Ok(self.into())
    640     }
    641 
    642     /// Processes a logical clock tick.
    643     fn tick(mut self) -> Result<Node> {
    644         self.role.election_duration += 1;
    645         if self.role.election_duration >= self.role.election_timeout {
    646             self.campaign()?;
    647         }
    648         Ok(self.into())
    649     }
    650 
    651     /// Hold a new election by increasing the term, voting for ourself, and
    652     /// soliciting votes from all peers.
    653     fn campaign(&mut self) -> Result<()> {
    654         let term = self.term() + 1;
    655         info!("Starting new election for term {term}");
    656         self.role = Candidate::new(self.random_election_timeout());
    657         self.role.votes.insert(self.id); // vote for ourself
    658         self.log.set_term(term, Some(self.id))?;
    659 
    660         let (last_index, last_term) = self.log.get_last_index();
    661         self.broadcast(Message::Campaign { last_index, last_term })
    662     }
    663 }

    665 // A leader serves client requests and replicates the log to followers.
    666 // If the leader loses leadership, all client requests are aborted.
    667 pub struct Leader {
    668     /// Follower replication progress.
    669     progress: HashMap<NodeID, Progress>,
    670     /// Tracks pending write requests by log index. Added when the write is
    671     /// proposed and appended to the leader's log, and removed when the command
    672     /// is applied to the state machine, returning the result to the client.
    673     writes: HashMap<Index, Write>,
    674     /// Tracks pending read requests. For linearizability, read requests are
    675     /// assigned a sequence number and only executed once a quorum of nodes have
    676     /// confirmed it. Otherwise, an old leader may serve stale reads if a new
    677     /// leader has been elected elsewhere.
    678     reads: VecDeque<Read>,
    679     /// The read sequence number used for the last read. Initialized to 0 in
    680     /// this term, and incremented for every read command.
    681     read_seq: ReadSequence,
    682     /// Number of ticks since last heartbeat.
    683     since_heartbeat: Ticks,
    684 }
    685 
    686 /// Follower replication progress (in this term).
    687 struct Progress {
    688     /// The highest index where the follower's log is known to match the leader.
    689     /// Initialized to 0, increases monotonically.
    690     match_index: Index,
    691     /// The next index to replicate to the follower. Initialized to
    692     /// last_index+1, decreased when probing log mismatches. Always in
    693     /// the range [match_index+1, last_index+1].
    694     ///
    695     /// Entries pending transmission are in the range [next_index, last_index].
    696     /// Unacknowledged entries are in the range [match_index+1, next_index).
    697     next_index: Index,
    698     /// The last read sequence number confirmed by the peer. To avoid stale
    699     /// reads on leader changes, a read is only served once its sequence number
    700     /// is confirmed by a quorum.
    701     read_seq: ReadSequence,
    702 }
    703 
    704 impl Progress {
    705     /// Attempts to advance a follower's match index, returning true if it did.
    706     /// If next_index is below it, it is advanced to the following index.
    707     fn advance(&mut self, match_index: Index) -> bool {
    708         if match_index <= self.match_index {
    709             return false;
    710         }
    711         self.match_index = match_index;
    712         self.next_index = std::cmp::max(self.next_index, match_index + 1);
    713         true
    714     }
    715 
    716     /// Attempts to advance a follower's read_seq, returning true if it did.
    717     fn advance_read(&mut self, read_seq: ReadSequence) -> bool {
    718         if read_seq <= self.read_seq {
    719             return false;
    720         }
    721         self.read_seq = read_seq;
    722         true
    723     }
    724 
    725     /// Attempts to regress a follower's next index to the given index, returning
    726     /// true if it did. Won't regress below match_index + 1.
    727     fn regress_next(&mut self, next_index: Index) -> bool {
    728         if next_index >= self.next_index || self.next_index <= self.match_index + 1 {
    729             return false;
    730         }
    731         self.next_index = std::cmp::max(next_index, self.match_index + 1);
    732         true
    733     }
    734 }

    759         let next_index = last_index + 1;

    776 impl RawNode<Leader> {
    777     /// Transitions the leader into a follower. This can only happen if we
    778     /// discover a new term, so we become a leaderless follower. Stepping the
    779     /// received message may then follow the new leader, if there is one.
    780     fn into_follower(mut self, term: Term) -> Result<RawNode<Follower>> {
    781         assert!(term > self.term(), "leader can only become follower in later term");
    782         info!("Discovered new term {term}");
    783 
    784         // Abort in-flight requests. The client must retry. Sort the requests
    785         // by ID for test determinism.
    786         for write in std::mem::take(&mut self.role.writes).into_values().sorted_by_key(|w| w.id) {
    787             let response = Err(Error::Abort);
    788             self.send(write.from, Message::ClientResponse { id: write.id, response })?;
    789         }
    790         for read in std::mem::take(&mut self.role.reads).into_iter().sorted_by_key(|r| r.id) {
    791             let response = Err(Error::Abort);
    792             self.send(read.from, Message::ClientResponse { id: read.id, response })?;
    793         }
    794 
    795         self.log.set_term(term, None)?;
    796         let election_timeout = self.random_election_timeout();
    797         Ok(self.into_role(Follower::new(None, election_timeout)))
    798     }

    800     /// Processes an inbound message.
    801     fn step(mut self, msg: Envelope) -> Result<Node> {
    802         // Past term: drop the message.
    803         if msg.term < self.term() {
    804             debug!("Dropping message from past term: {msg:?}");
    805             return Ok(self.into());
    806         }
    807         // Future term: become leaderless follower and step the message.
    808         if msg.term > self.term() {
    809             return self.into_follower(msg.term)?.step(msg);
    810         }
    811 
    812         match msg.message {
    813             // A follower received our heartbeat and confirms our leadership.
    814             // We may be able to execute new reads, and we may find that the
    815             // follower's log is lagging and requires us to catch it up.
    816             Message::HeartbeatResponse { match_index, read_seq } => {
    817                 let (last_index, _) = self.log.get_last_index();
    818                 assert!(match_index <= last_index, "future match index");
    819                 assert!(read_seq <= self.role.read_seq, "future read sequence number");
    820 
    821                 // If the read sequence number advances, try to execute reads.
    822                 if self.progress(msg.from).advance_read(read_seq) {
    823                     self.maybe_read()?;
    824                 }
    825 
    826                 // If the follower didn't match our last index, an append to it
    827                 // must have failed (or it's catching up). Probe it to discover
    828                 // a matching entry and start replicating. Move next_index back
    829                 // to last_index since the follower just told us it doesn't have
    830                 // it (or a previous last_index).
    831                 if match_index == 0 {
    832                     self.progress(msg.from).regress_next(last_index);
    833                     self.maybe_send_append(msg.from, true)?;
    834                 }
    835 
    836                 // If the follower's match index advances, an append response
    837                 // got lost. Try to commit and apply.
    838                 //
    839                 // We don't need to eagerly send any pending entries, since any
    840                 // proposals made after this heartbeat was sent should have been
    841                 // eagerly replicated in steady state. If not, the next
    842                 // heartbeat will trigger a probe above.
    843                 if self.progress(msg.from).advance(match_index) {
    844                     self.maybe_commit_and_apply()?;
    845                 }
    846             }
    847 
    848             // A follower appended our log entries (or a probe found a match).
    849             // Record its progress and attempt to commit and apply.
    850             Message::AppendResponse { match_index, reject_index: 0 } if match_index > 0 => {
    851                 let (last_index, _) = self.log.get_last_index();
    852                 assert!(match_index <= last_index, "future match index");
    853 
    854                 if self.progress(msg.from).advance(match_index) {
    855                     self.maybe_commit_and_apply()?;
    856                 }
    857 
    858                 // Eagerly send any further pending entries. This may be a
    859                 // successful probe response, or the peer may be lagging and
    860                 // we're catching it up one MAX_APPEND_ENTRIES batch at a time.
    861                 self.maybe_send_append(msg.from, false)?;
    862             }
    863 
    864             // A follower confirmed our read sequence number. If it advances,
    865             // try to execute reads.
    866             Message::ReadResponse { seq } => {
    867                 if self.progress(msg.from).advance_read(seq) {
    868                     self.maybe_read()?;
    869                 }
    870             }
    871 
    872             // A follower rejected an append because the base entry in
    873             // reject_index did not match its log. Probe the previous entry by
    874             // sending an empty append until we find a common base.
    875             //
    876             // This linear probing can be slow with long divergent logs, but we
    877             // keep it simple. See also section 5.3 in the Raft paper.
    878             Message::AppendResponse { reject_index, match_index: 0 } if reject_index > 0 => {
    879                 let (last_index, _) = self.log.get_last_index();
    880                 assert!(reject_index <= last_index, "future reject index");
    881 
    882                 // If the rejected base index is at or below the match index,
    883                 // the rejection is stale and can be ignored.
    884                 if reject_index <= self.progress(msg.from).match_index {
    885                     return Ok(self.into());
    886                 }
    887 
    888                 // Probe below the reject index, if we haven't already moved
    889                 // next_index below it. This avoids sending duplicate probes
    890                 // (heartbeats will trigger retries if they're lost).
    891                 if self.progress(msg.from).regress_next(reject_index) {
    892                     self.maybe_send_append(msg.from, true)?;
    893                 }
    894             }
    895 
    896             // AppendResponses must set either match_index or reject_index.
    897             Message::AppendResponse { .. } => panic!("invalid message {msg:?}"),
    898 
    899             // A client submitted a write request. Propose it, and wait until
    900             // it's replicated and applied to the state machine before returning
    901             // the response to the client.
    902             Message::ClientRequest { id, request: Request::Write(command) } => {
    903                 let index = self.propose(Some(command))?;
    904                 self.role.writes.insert(index, Write { from: msg.from, id });
    905                 if self.cluster_size() == 1 {
    906                     self.maybe_commit_and_apply()?;
    907                 }
    908             }
    909 
    910             // A client submitted a read request. To ensure linearizability, we
    911             // must confirm that we are still the leader by sending the read's
    912             // sequence number and wait for quorum confirmation.
    913             Message::ClientRequest { id, request: Request::Read(command) } => {
    914                 self.role.read_seq += 1;
    915                 let read = Read { seq: self.role.read_seq, from: msg.from, id, command };
    916                 self.role.reads.push_back(read);
    917                 self.broadcast(Message::Read { seq: self.role.read_seq })?;
    918                 if self.cluster_size() == 1 {
    919                     self.maybe_read()?;
    920                 }
    921             }
    922 
    923             // A client submitted a status command.
    924             Message::ClientRequest { id, request: Request::Status } => {
    925                 let response = self.status().map(Response::Status);
    926                 self.send(msg.from, Message::ClientResponse { id, response })?;
    927             }
    928 
    929             // Don't grant any votes (we've already voted for ourself).
    930             Message::Campaign { .. } => {
    931                 self.send(msg.from, Message::CampaignResponse { vote: false })?
    932             }
    933 
    934             // Votes can come in after we won the election, ignore them.
    935             Message::CampaignResponse { .. } => {}
    936 
    937             // There can't be another leader in this term.
    938             Message::Heartbeat { .. } | Message::Append { .. } | Message::Read { .. } => {
    939                 panic!("saw other leader {} in term {}", msg.from, msg.term);
    940             }
    941 
    942             // Leaders don't proxy client requests.
    943             Message::ClientResponse { .. } => panic!("unexpected message {msg:?}"),
    944         }
    945 
    946         Ok(self.into())
    947     }

    949     /// Processes a logical clock tick.
    950     fn tick(mut self) -> Result<Node> {
    951         self.role.since_heartbeat += 1;
    952         if self.role.since_heartbeat >= self.opts.heartbeat_interval {
    953             self.heartbeat()?;
    954         }
    955         Ok(self.into())
    956     }

    958     /// Broadcasts a heartbeat to all peers.
    959     fn heartbeat(&mut self) -> Result<()> {
    960         let (last_index, last_term) = self.log.get_last_index();
    961         let (commit_index, _) = self.log.get_commit_index();
    962         let read_seq = self.role.read_seq;
    963         assert_eq!(last_term, self.term(), "leader's last_term not in current term");
    964 
    965         self.role.since_heartbeat = 0;
    966         self.broadcast(Message::Heartbeat { last_index, commit_index, read_seq })
    967     }

    969     /// Proposes a command for consensus by appending it to our log and
    970     /// replicating it to peers. If successful, it will eventually be committed
    971     /// and applied to the state machine.
    972     fn propose(&mut self, command: Option<Vec<u8>>) -> Result<Index> {
    973         let index = self.log.append(command)?;
    974         for peer in self.peers.iter().copied().sorted() {
    975             // Eagerly send the entry to the peer if it's in steady state and
    976             // we've sent all previous entries. Otherwise, the peer is lagging
    977             // and we're probing past entries for a match.
    978             if index == self.progress(peer).next_index {
    979                 self.maybe_send_append(peer, false)?;
    980             }
    981         }
    982         Ok(index)
    983     }

    985     /// Commits new entries that have been replicated to a quorum and applies
    986     /// them to the state machine, returning results to clients.
    987     fn maybe_commit_and_apply(&mut self) -> Result<Index> {
    988         // Determine the new commit index by quorum.
    989         let (last_index, _) = self.log.get_last_index();
    990         let quorum_index = self.quorum_value(
    991             self.role.progress.values().map(|p| p.match_index).chain([last_index]).collect(),
    992         );
    993 
    994         // If the commit index doesn't advance, do nothing. We don't assert on
    995         // this, since the quorum value may regress e.g. following a restart or
    996         // leader change where followers are initialized with match index 0.
    997         let (old_index, old_term) = self.log.get_commit_index();
    998         if quorum_index <= old_index {
    999             return Ok(old_index);
   1000         }
   1001 
   1002         // We can only safely commit an entry from our own term (see section
   1003         // 5.4.2 in Raft paper).
   1004         match self.log.get(quorum_index)? {
   1005             Some(entry) if entry.term == self.term() => {}
   1006             Some(_) => return Ok(old_index),
   1007             None => panic!("commit index {quorum_index} missing"),
   1008         }
   1009 
   1010         // Commit entries.
   1011         self.log.commit(quorum_index)?;
   1012 
   1013         // Apply entries and respond to clients.
   1014         let term = self.term();
   1015         let mut iter = self.log.scan_apply(self.state.get_applied_index());
   1016         while let Some(entry) = iter.next().transpose()? {
   1017             debug!("Applying {entry:?}");
   1018             let write = self.role.writes.remove(&entry.index);
   1019             let result = self.state.apply(entry);
   1020 
   1021             if let Some(Write { id, from: to }) = write {
   1022                 let message = Message::ClientResponse { id, response: result.map(Response::Write) };
   1023                 Self::send_with(&self.tx, Envelope { from: self.id, term, to, message })?;
   1024             }
   1025         }
   1026         drop(iter);
   1027 
   1028         // If the commit term changed, there may be pending reads waiting for us
   1029         // to commit and apply an entry from our own term. Execute them.
   1030         if old_term != self.term() {
   1031             self.maybe_read()?;
   1032         }
   1033 
   1034         Ok(quorum_index)
   1035     }
   1036 
   1037     /// Executes any ready read requests (with confirmed sequence numbers).
   1038     fn maybe_read(&mut self) -> Result<()> {
   1039         if self.role.reads.is_empty() {
   1040             return Ok(());
   1041         }
   1042 
   1043         // It's only safe to read if we've committed and applied an entry from
   1044         // our own term (the leader appends an entry when elected). Otherwise we
   1045         // may be behind on application and serve stale reads.
   1046         let (commit_index, commit_term) = self.log.get_commit_index();
   1047         let applied_index = self.state.get_applied_index();
   1048         if commit_term < self.term() || applied_index < commit_index {
   1049             return Ok(());
   1050         }
   1051 
   1052         // Determine the maximum read sequence confirmed by quorum.
   1053         let quorum_read_seq = self.quorum_value(
   1054             self.role.progress.values().map(|p| p.read_seq).chain([self.role.read_seq]).collect(),
   1055         );
   1056 
   1057         // Execute ready reads. The VecDeque is ordered by read_seq, so we
   1058         // can keep pulling until we hit quorum_read_seq.
   1059         while let Some(read) = self.role.reads.front() {
   1060             if read.seq > quorum_read_seq {
   1061                 break;
   1062             }
   1063             let read = self.role.reads.pop_front().unwrap();
   1064             let response = self.state.read(read.command).map(Response::Read);
   1065             self.send(read.from, Message::ClientResponse { id: read.id, response })?;
   1066         }
   1067         Ok(())
   1068     }
   1069 
   1070     // Sends a batch of pending log entries to a follower in the
   1071     // [next_index,last_index] range, limited by max_append_entries.
   1072     //
   1073     // If probe is true, an empty append probe with base_index of next_index-1
   1074     // is sent to check if the base entry is present in the follower's log. If
   1075     // it is, the actual entries are sent next -- otherwise, next_index is
   1076     // decremented and another probe is sent until a match is found. See section
   1077     // 5.3 in the Raft paper.
   1078     //
   1079     // The probe is skipped if the follower is up-to-date (according to
   1080     // match_index and last_index). If the probe's base_index has already been
   1081     // confirmed via match_index, an actual append is sent instead.
   1082     fn maybe_send_append(&mut self, peer: NodeID, mut probe: bool) -> Result<()> {
   1083         let (last_index, _) = self.log.get_last_index();
   1084         let progress = self.role.progress.get_mut(&peer).expect("unknown node");
   1085         assert_ne!(progress.next_index, 0, "invalid next_index");
   1086         assert!(progress.next_index > progress.match_index, "invalid next_index <= match_index");
   1087         assert!(progress.match_index <= last_index, "invalid match_index > last_index");
   1088         assert!(progress.next_index <= last_index + 1, "invalid next_index > last_index + 1");
   1089 
   1090         // If the peer is caught up, there's no point sending an append.
   1091         if progress.match_index == last_index {
   1092             return Ok(());
   1093         }
   1094 
   1095         // If a probe was requested, but the base_index has already been
   1096         // confirmed via match_index, there is no point in probing. Just send
   1097         // the entries instead.
   1098         probe = probe && progress.next_index > progress.match_index + 1;
   1099 
   1100         // If there are no pending entries, and this is not a probe, there's
   1101         // nothing more to send until we get a response from the follower.
   1102         if progress.next_index > last_index && !probe {
   1103             return Ok(());
   1104         }
   1105 
   1106         // Fetch the base and entries.
   1107         let (base_index, base_term) = match progress.next_index {
   1108             0 => panic!("next_index=0 for node {peer}"),
   1109             1 => (0, 0),
   1110             next => self.log.get(next - 1)?.map(|e| (e.index, e.term)).expect("missing base entry"),
   1111         };
   1112         let entries = match probe {
   1113             false => self
   1114                 .log
   1115                 .scan(progress.next_index..)
   1116                 .take(self.opts.max_append_entries)
   1117                 .try_collect()?,
   1118             true => Vec::new(),
   1119         };
   1120 
   1121         // Optimistically assume the entries will be accepted by the follower,

   1284         fn run(&mut self, command: &goldenscript::Command) -> Result<String, Box<dyn Error>> {
   1285             let mut output = String::new();
   1286             match command.name.as_str() {
   1287                 // campaign [ID...]
   1288                 // Transition the given nodes to candidates and campaign.
   1289                 "campaign" => {
   1290                     let ids = self.parse_ids_or_all(&command.args)?;
   1291                     self.campaign(&ids, &mut output)?;
   1292                 }
   1293 
   1294                 // cluster nodes=N [leader=ID] [heartbeat_interval=N] [election_timeout=N] [max_append_entries=N]
   1295                 // Creates a new Raft cluster.
   1296                 "cluster" => {
   1297                     let mut opts = Options::default();
   1298                     let mut args = command.consume_args();
   1299                     let nodes = args.lookup_parse("nodes")?.unwrap_or(0);
   1300                     let leader = args.lookup_parse("leader")?;
   1301                     if let Some(heartbeat_interval) = args.lookup_parse("heartbeat_interval")? {
   1302                         opts.heartbeat_interval = heartbeat_interval;
   1303                     };
   1304                     if let Some(election_timeout) = args.lookup_parse("election_timeout")? {
   1305                         opts.election_timeout_range = election_timeout..election_timeout + 1;
   1306                     }
   1307                     if let Some(max_append_entries) = args.lookup_parse("max_append_entries")? {
   1308                         opts.max_append_entries = max_append_entries;
   1309                     }
   1310                     args.reject_rest()?;
   1311                     self.cluster(nodes, leader, opts, &mut output)?;
   1312                 }
   1313 
   1314                 // deliver [from=ID] [ID...]
   1315                 // Delivers (steps) pending messages to the given nodes. If from
   1316                 // is given, only messages from the given node is delivered, the
   1317                 // others are left pending.
   1318                 "deliver" => {
   1319                     let mut args = command.consume_args();
   1320                     let from = args.lookup_parse("from")?;
   1321                     let ids = self.parse_ids_or_all(&args.rest())?;
   1322                     self.deliver(&ids, from, &mut output)?;

   1324

   1326                 // Sends a client request to the given node to read the given
   1327                 // key from the state machine (key/value store).
   1328                 "get" => {
   1329                     let mut args = command.consume_args();

   1331                     let key = args.next_pos().ok_or("must specify key")?.value.clone();
   1332                     args.reject_rest()?;
   1333                     let request = Request::Read(KVCommand::Get { key }.encode());
   1334                     self.request(id, request, &mut output)?;
   1335                 }
   1336 
   1337                 // heal [ID...]
   1338                 // Heals all network partitions for the given nodes.
   1339                 "heal" => {
   1340                     let ids = self.parse_ids_or_all(&command.args)?;
   1341                     self.heal(&ids, &mut output)?;
   1342                 }
   1343 
   1344                 // heartbeat ID...
   1345                 // Sends a heartbeat from the given leader nodes.
   1346                 "heartbeat" => {

   1348                     self.heartbeat(&ids, &mut output)?;
   1349                 }
   1350 
   1351                 // log [ID...]
   1352                 // Outputs the current Raft log for the given nodes.
   1353                 "log" => {
   1354                     let ids = self.parse_ids_or_all(&command.args)?;
   1355                     self.log(&ids, &mut output)?;
   1356                 }
   1357 
   1358                 // partition ID...
   1359                 // Partitions the given nodes away from the rest of the cluster.
   1360                 // They can still communicate with each other, unless they were
   1361                 // previously partitioned.
   1362                 "partition" => {
   1363                     let ids = self.parse_ids_or_error(&command.args)?;
   1364                     self.partition(&ids, &mut output)?;
   1365                 }
   1366 
   1367                 // put ID KEY=VALUE
   1368                 // Sends a client request to the given node to write a key/value
   1369                 // pair to the state machine (key/value store).
   1370                 "put" => {
   1371                     let mut args = command.consume_args();
   1372                     let id = args.next_pos().ok_or("must specify node ID")?.parse()?;
   1373                     let kv = args.next_key().ok_or("must specify key/value pair")?.clone();
   1374                     let (key, value) = (kv.key.unwrap(), kv.value);
   1375                     args.reject_rest()?;
   1376                     let request = Request::Write(KVCommand::Put { key, value }.encode());
   1377                     self.request(id, request, &mut output)?;
   1378                 }
   1379 
   1380                 // restart [commit_index=INDEX] [applied_index=INDEX] [ID...]
   1381                 // Restarts the given nodes (or all nodes). They retain their
   1382                 // log and state, unless applied_index is given (which reverts
   1383                 // the state machine to the given index, or 0 if empty).
   1384                 // commit_index may be given to regress the commit index (it
   1385                 // is not flushed to durable storage).
   1386                 "restart" => {
   1387                     let mut args = command.consume_args();
   1388                     let applied_index = args.lookup_parse("applied_index")?;
   1389                     let commit_index = args.lookup_parse("commit_index")?;
   1390                     let ids = self.parse_ids_or_all(&args.rest())?;
   1391                     self.restart(&ids, commit_index, applied_index, &mut output)?;
   1392                 }
   1393 
   1394                 // stabilize [heartbeat=BOOL] [ID...]
   1395                 // Stabilizes the given nodes by repeatedly delivering messages
   1396                 // until no more messages are pending. If heartbeat is true, also
   1397                 // emits a heartbeat from the leader and restabilizes, e.g. to
   1398                 // propagate the commit index.
   1399                 "stabilize" => {
   1400                     let mut args = command.consume_args();
   1401                     let heartbeat = args.lookup_parse("heartbeat")?.unwrap_or(false);
   1402                     let ids = self.parse_ids_or_all(&args.rest())?;
   1403                     self.stabilize(&ids, heartbeat, &mut output)?;
   1404                 }
   1405 
   1406                 // state [ID...]
   1407                 // Prints the current state machine contents on the given nodes.
   1408                 "state" => {
   1409                     let ids = self.parse_ids_or_all(&command.args)?;
   1410                     self.state(&ids, &mut output)?;
   1411                 }
   1412 
   1413                 // status [request=BOOL] [ID...]
   1414                 // Prints the current node status of the given nodes. If request
   1415                 // is true, sends a status client request to a single node,
   1416                 // otherwise fetches status directly from each node.
   1417                 "status" => {
   1418                     let mut args = command.consume_args();
   1419                     let request = args.lookup_parse("request")?.unwrap_or(false);
   1420                     let ids = self.parse_ids_or_all(&args.rest())?;
   1421                     if request {
   1422                         let [id] = *ids.as_slice() else {
   1423                             return Err("request=true requires 1 node ID".into());
   1424                         };
   1425                         self.request(id, Request::Status, &mut output)?;
   1426                     } else {
   1427                         self.status(&ids, &mut output)?;
   1428                     }
   1429                 }
   1430 
   1431                 // step ID JSON
   1432                 // Steps a manually generated JSON message on the given node.
   1433                 "step" => {
   1434                     let mut args = command.consume_args();
   1435                     let id = args.next_pos().ok_or("node ID not given")?.parse()?;
   1436                     let raw = &args.next_pos().ok_or("message not given")?.value;
   1437                     let msg = serde_json::from_str(raw)?;
   1438                     args.reject_rest()?;
   1439                     self.transition(id, |n| n.step(msg), &mut output)?;
   1440                 }
   1441 
   1442                 // tick [ID...]
   1443                 // Ticks the given nodes.
   1444                 "tick" => {
   1445                     let ids = self.parse_ids_or_all(&command.args)?;
   1446                     for id in ids {
   1447                         self.transition(id, |n| n.tick(), &mut output)?;
   1448                     }
   1449                 }
   1450 
   1451                 name => return Err(format!("unknown command {name}").into()),
   1452             }
   1453             Ok(output)
   1454         }
   1455     }

   1510         /// Transitions nodes to candidates and campaign in a new term.
   1511         fn campaign(&mut self, ids: &[NodeID], output: &mut String) -> Result<(), Box<dyn Error>> {
   1512             let campaign = |node| match node {
   1513                 Node::Candidate(mut node) => {

   1517                 Node::Follower(node) => Ok(node.into_candidate()?.into()),
   1518                 Node::Leader(node) => {
   1519                     let term = node.term();
   1520                     Ok(node.into_follower(term + 1)?.into_candidate()?.into())
   1521                 }
   1522             };

   1524                 self.transition(id, campaign, output)?;
   1525             }

   1529         /// Creates a Raft cluster.
   1530         fn cluster(
   1531             &mut self,
   1532             nodes: u8,
   1533             leader: Option<NodeID>,
   1534             opts: Options,
   1535             output: &mut String,
   1536         ) -> Result<(), Box<dyn Error>> {
   1537             if !self.ids.is_empty() {
   1538                 return Err("cluster already exists".into());
   1539             }
   1540             if nodes == 0 {
   1541                 return Err("cluster can't have 0 nodes".into());
   1542             }
   1543 
   1544             self.ids = (1..=nodes).collect();
   1545 
   1546             for id in self.ids.clone() {
   1547                 let peers = self.ids.iter().copied().filter(|i| i != &id).collect();
   1548                 self.add_node(id, peers, opts.clone())?;
   1549             }
   1550 
   1551             // Promote leader if requested. Suppress output.
   1552             if let Some(id) = leader {
   1553                 let quiet = &mut String::new();
   1554                 let Some(Node::Follower(node)) = self.nodes.remove(&id) else {
   1555                     return Err(format!("invalid leader {id}").into());
   1556                 };
   1557                 self.nodes.insert(id, node.into_candidate()?.into_leader()?.into());
   1558                 self.receive(id, quiet)?;
   1559                 self.stabilize(&self.ids.clone(), true, quiet)?;
   1560             }
   1561 
   1562             // Drain any initial applied entries.
   1563             for applied_rx in self.applied_rx.values_mut() {
   1564                 while applied_rx.try_recv().is_ok() {}
   1565             }
   1566 
   1567             // Output final cluster status.
   1568             self.status(&self.ids, output)
   1569         }

   1571         /// Delivers pending messages to the given nodes. If from is given, only
   1572         /// delivers messages from that node. Returns the number of delivered
   1573         /// messages.
   1574         fn deliver(
   1575             &mut self,
   1576             ids: &[NodeID],
   1577             from: Option<NodeID>,
   1578             output: &mut String,
   1579         ) -> Result<usize, Box<dyn Error>> {
   1580             // Take a snapshot of the pending queues before delivering any
   1581             // messages. This avoids outbound messages in response to delivery
   1582             // being delivered to higher node IDs in the same loop, which can
   1583             // give unintuitive results.
   1584             let mut step = Vec::new();
   1585             for id in ids.iter().copied() {
   1586                 let Some(pending) = self.nodes_pending.remove(&id) else {
   1587                     return Err(format!("unknown node {id}").into());
   1588                 };
   1589                 let (deliver, requeue) =
   1590                     pending.into_iter().partition(|msg| from.is_none() || from == Some(msg.from));
   1591                 self.nodes_pending.insert(id, requeue);
   1592                 step.extend(deliver);
   1593             }
   1594 
   1595             let delivered = step.len();
   1596             for msg in step {
   1597                 self.transition(msg.to, |node| node.step(msg), output)?;
   1598             }
   1599             Ok(delivered)
   1600         }

   1606                 self.disconnected.insert(id, HashSet::new());
   1607                 for peers in self.disconnected.values_mut() {
   1608                     peers.remove(&id);
   1609                 }
   1610             }
   1611             output.push_str(&Self::format_disconnected(&self.disconnected));
   1612             Ok(())
   1613         }

   1646         /// Partitions the given nodes from all other nodes in the cluster

   1648         /// unless they were previously partitioned.
   1649         fn partition(&mut self, ids: &[NodeID], output: &mut String) -> Result<(), Box<dyn Error>> {
   1650             let ids = HashSet::<NodeID>::from_iter(ids.iter().copied());
   1651             for id in ids.iter().copied() {

   1653                     self.disconnected.entry(id).or_default().insert(peer);
   1654                     self.disconnected.entry(peer).or_default().insert(id);
   1655                 }
   1656             }
   1657             output.push_str(&Self::format_disconnected(&self.disconnected));
   1658             Ok(())

   1661         /// Receives outbound messages from a node, prints them, and queues them
   1662         /// for delivery. Returns the number of received messages.
   1663         fn receive(&mut self, id: NodeID, output: &mut String) -> Result<u32, Box<dyn Error>> {
   1664             let rx = self.nodes_rx.get_mut(&id).ok_or(format!("unknown node {id}"))?;
   1665             let mut count = 0;
   1666             for msg in rx.try_iter() {
   1667                 count += 1;
   1668                 let (from, term, to) = (msg.from, msg.term, msg.to); // simplify formatting
   1669                 let msgfmt = Self::format_message(&msg.message);
   1670 
   1671                 // If the peer is disconnected, drop the message and output it.
   1672                 if self.disconnected[&msg.from].contains(&msg.to) {
   1673                     writeln!(
   1674                         output,
   1675                         "n{from}@{term}  n{to} {}",
   1676                         Self::format_strikethrough(&msgfmt),
   1677                     )?;
   1678                     continue;
   1679                 }
   1680 
   1681                 // Intercept and output client responses.
   1682                 if msg.from == msg.to {
   1683                     let Message::ClientResponse { id, response } = &msg.message else {
   1684                         return Err(format!("invalid self-addressed message: {msg:?}").into());
   1685                     };
   1686                     writeln!(output, "n{from}@{term}  c{to} {msgfmt}")?;
   1687                     let request = &self.requests.remove(id).ok_or("unknown request id")?;
   1688                     writeln!(
   1689                         output,
   1690                         "c{to}@{term} {}  {}",
   1691                         Self::format_request(request),
   1692                         Self::format_response(response),
   1693                     )?;
   1694                     continue;
   1695                 }
   1696 
   1697                 // Output the message and queue it for delivery.
   1698                 writeln!(output, "n{from}@{term}  n{to} {msgfmt}")?;
   1699                 self.nodes_pending.get_mut(&msg.to).ok_or(format!("unknown node {to}"))?.push(msg);
   1700             }
   1701             Ok(count)
   1702         }

   1706             &mut self,
   1707             id: NodeID,
   1708             request: Request,
   1709             output: &mut String,
   1710         ) -> Result<(), Box<dyn Error>> {
   1711             let request_id = uuid::Uuid::from_u64_pair(0, self.next_request_id);
   1712             self.next_request_id += 1;

   1726         /// Restarts the given nodes. If commit_index or applied_index are
   1727         /// given, the log commit index or state machine will regress.
   1728         fn restart(
   1729             &mut self,
   1730             ids: &[NodeID],
   1731             commit_index: Option<Index>,
   1732             applied_index: Option<Index>,
   1733             output: &mut String,
   1734         ) -> Result<(), Box<dyn Error>> {
   1735             for id in ids.iter().copied() {
   1736                 let node = self.nodes.remove(&id).ok_or(format!("unknown node {id}"))?;
   1737                 let peers = node.peers();
   1738                 let opts = node.options();
   1739                 let (log, mut state) = node.dismantle();
   1740                 let mut log = Log::new(log.engine)?; // reset log
   1741 
   1742                 // If requested, regress the commit index.
   1743                 if let Some(commit_index) = commit_index {
   1744                     if commit_index > log.get_commit_index().0 {
   1745                         return Err(format!("commit_index={commit_index} beyond current").into());
   1746                     }
   1747                     let commit_term = match log.get(commit_index)? {
   1748                         Some(e) => e.term,
   1749                         None if commit_index == 0 => 0,
   1750                         None => return Err(format!("unknown commit_index={commit_index}").into()),
   1751                     };
   1752                     log.engine.set(
   1753                         &crate::raft::log::Key::CommitIndex.encode(),
   1754                         bincode::serialize(&(commit_index, commit_term)),
   1755                     )?;
   1756                     // Reset the log again.
   1757                     log = Log::new(log.engine)?;
   1758                 }
   1759 
   1760                 // If requested, wipe the state machine and reapply up to the
   1761                 // requested applied index.
   1762                 if let Some(applied_index) = applied_index {
   1763                     if applied_index > log.get_commit_index().0 {
   1764                         return Err(format!("applied_index={applied_index} beyond commit").into());
   1765                     }
   1766                     state = teststate::KV::new();
   1767                     let mut scan = log.scan(..=applied_index);
   1768                     while let Some(entry) = scan.next().transpose()? {
   1769                         _ = state.apply(entry); // apply errors are returned to client
   1770                     }
   1771                     assert_eq!(state.get_applied_index(), applied_index, "wrong applied index");
   1772                 }
   1773 
   1774                 // Add node, and run a noop transition to output applied entries.
   1775                 self.add_node_with(id, peers, log, state, opts)?;
   1776                 self.transition(id, Ok, output)?;
   1777             }
   1778             // Output restarted node status.
   1779             self.status(ids, output)
   1780         }

   1782         /// Stabilizes the given nodes by repeatedly delivering pending messages
   1783         /// until no new messages are generated. If heartbeat is true, leaders
   1784         /// then emit a heartbeat and restabilize again, e.g. to propagate the
   1785         /// commit index.
   1786         fn stabilize(
   1787             &mut self,
   1788             ids: &[NodeID],
   1789             heartbeat: bool,
   1790             output: &mut String,
   1791         ) -> Result<(), Box<dyn Error>> {
   1792             while self.deliver(ids, None, output)? > 0 {}
   1793             // If requested, heartbeat the current leader (with the highest
   1794             // term) and re-stabilize the nodes.
   1795             if heartbeat {
   1796                 let leader = self
   1797                     .nodes
   1798                     .values()
   1799                     .sorted_by_key(|n| n.term())
   1800                     .rev()
   1801                     .find(|n| matches!(n, Node::Leader(_)));
   1802                 if let Some(leader) = leader {
   1803                     self.heartbeat(&[leader.id()], output)?;
   1804                     self.stabilize(ids, false, output)?;
   1805                 }
   1806             }
   1807             Ok(())
   1808         }

   1830             for id in ids {

   1855         /// Applies a node transition (typically a step or tick), and outputs
   1856         /// relevant changes.
   1857         fn transition(
   1858             &mut self,
   1859             id: NodeID,
   1860             f: impl FnOnce(Node) -> crate::error::Result<Node>,
   1861             output: &mut String,
   1862         ) -> Result<(), Box<dyn Error>> {
   1863             let mut node = self.nodes.remove(&id).ok_or(format!("unknown node {id}"))?;
   1864 
   1865             // Fetch pre-transition info.
   1866             let old_noderole = Self::format_node_role(&node);
   1867             let (old_commit_index, _) = node.get_commit_index();
   1868             let mut old_entries = node.scan_log()?.into_iter();
   1869 
   1870             // Apply the transition.
   1871             node = f(node)?;
   1872 
   1873             // Fetch post-transition info.
   1874             let nodefmt = Self::format_node(&node);
   1875             let noderole = Self::format_node_role(&node);
   1876             let (commit_index, commit_term) = node.get_commit_index();
   1877 
   1878             let entries = node.scan_log()?.into_iter();
   1879             let appended: Vec<Entry> = entries
   1880                 .skip_while(|e| Some(e.term) == old_entries.next().map(|e| e.term))
   1881                 .collect();
   1882 
   1883             self.nodes.insert(id, node);
   1884 
   1885             // Output relevant changes.
   1886             if old_noderole != noderole {
   1887                 writeln!(output, "{old_noderole}  {noderole}")?
   1888             }
   1889             for entry in appended {
   1890                 writeln!(output, "{nodefmt} append {}", Self::format_entry(&entry))?
   1891             }
   1892             if old_commit_index != commit_index {
   1893                 writeln!(output, "{nodefmt} commit {commit_index}@{commit_term}")?;
   1894             }
   1895             for entry in self.applied_rx[&id].try_iter() {
   1896                 writeln!(output, "{nodefmt} apply {}", Self::format_entry(&entry))?
   1897             }
   1898 
   1899             // Receive any outbound messages.
   1900             self.receive(id, output)?;
   1901             Ok(())
   1902         }

   1949         /// Formats network partitions.
   1950         fn format_disconnected(disconnected: &HashMap<NodeID, HashSet<NodeID>>) -> String {
   1951             // Return early if the cluster is fully connected.
   1952             if disconnected.iter().all(|(_, peers)| peers.is_empty()) {
   1953                 return format!(
   1954                     "{} fully connected\n",
   1955                     disconnected.keys().sorted().map(|id| format!("n{id}")).join(" ")
   1956                 );
   1957             }
   1958 
   1959             let mut output = String::new();
   1960 
   1961             // Separate symmetric and asymmetric partitions.
   1962             let mut symmetric: HashMap<NodeID, HashSet<NodeID>> = HashMap::new();
   1963             let mut asymmetric: HashMap<NodeID, HashSet<NodeID>> = HashMap::new();
   1964             for (id, peers) in disconnected {
   1965                 for peer in peers {
   1966                     if disconnected[peer].contains(id) {
   1967                         symmetric.entry(*id).or_default().insert(*peer);
   1968                     } else {
   1969                         asymmetric.entry(*id).or_default().insert(*peer);
   1970                     }
   1971                 }
   1972             }
   1973 
   1974             // Anchor the symmetric partitions at the node with the largest number
   1975             // of disconnects, otherwise the smallest (first) ID.
   1976             for (id, peers) in &symmetric.clone() {
   1977                 for peer in peers {
   1978                     // Recompute the peer set sizes for each iteration, since we
   1979                     // modify the peer set below.
   1980                     let len = symmetric.get(id).map(|p| p.len()).unwrap_or(0);
   1981                     let peer_len = symmetric.get(peer).map(|p| p.len()).unwrap_or(0);
   1982                     // If this peer set is the smallest (or we're the higher ID),
   1983                     // remove the entry. We may no longer be in the map.
   1984                     if len < peer_len || len == peer_len && id > peer {
   1985                         if let Some(peers) = symmetric.get_mut(id) {
   1986                             peers.remove(peer);
   1987                             if peers.is_empty() {
   1988                                 symmetric.remove(id);
   1989                             }
   1990                         }
   1991                     }
   1992                 }
   1993             }
   1994 
   1995             // The values (HashSets) correspond to the RHS of a partition. Let's
   1996             // group the LHS of the partition as well, from smallest to largest,
   1997             // separately for symmetric and asymmetric partitions. The vector
   1998             // contains (LHS, RHS, symmetric) groupings for each partition.
   1999             let mut grouped: Vec<(HashSet<NodeID>, HashSet<NodeID>, bool)> = Vec::new();
   2000             for (id, peers, symm) in symmetric
   2001                 .into_iter()
   2002                 .map(|(i, p)| (i, p, true))
   2003                 .chain(asymmetric.into_iter().map(|(i, p)| (i, p, false)))
   2004                 .sorted_by_key(|(id, _, symm)| (*id, !symm))
   2005             {
   2006                 // Look for an existing LHS group with the same RHS, and insert
   2007                 // this node into it. Otherwise, create a new LHS group.
   2008                 match grouped.iter_mut().find(|(_, rhs, s)| peers == *rhs && symm == *s) {
   2009                     Some((lhs, _, _)) => _ = lhs.insert(id),
   2010                     None => grouped.push((HashSet::from([id]), peers, symm)),
   2011                 }
   2012             }
   2013 
   2014             // Display the groups.
   2015             for (lhs, rhs, symm) in grouped {
   2016                 let lhs = lhs.iter().sorted().map(|id| format!("n{id}")).join(" ");
   2017                 let sep = if symm { '' } else { '' };
   2018                 let rhs = rhs.iter().sorted().map(|id| format!("n{id}")).join(" ");
   2019                 writeln!(output, "{lhs} {sep} {rhs}").unwrap();
   2020             }
   2021 
   2022             output
   2023         }
