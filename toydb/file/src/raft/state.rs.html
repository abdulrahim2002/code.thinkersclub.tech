      4 /// A Raft-managed state machine. Raft itself does not care what the state
      5 /// machine is, nor what the commands and results do -- it will simply apply
      6 /// arbitrary binary commands sequentially from the Raft log, returning an
      7 /// arbitrary binary result to the client.
      8 ///
      9 /// Since commands are applied identically across all replicas, they must be
     10 /// deterministic and yield the same state and result across all replicas too.
     11 /// Otherwise, the replicas will diverge, and different replicas will produce
     12 /// different results.
     13 ///
     14 /// Write commands (`Request::Write`) are replicated and applied on all replicas
     15 /// via `State::apply`. The state machine must keep track of the last applied
     16 /// index and return it via `State::get_applied_index`. Read commands
     17 /// (`Request::Read`) are only executed on a single replica via `State::read`
     18 /// and must not make any state changes.
     19 pub trait State: Send {
     20     /// Returns the last applied index from the state machine.
     21     ///
     22     /// This must correspond to the current state of the state machine, since it
     23     /// determines which command to apply next. In particular, a node crash may
     24     /// result in partial command application or data loss, which must be
     25     /// handled appropriately.
     26     fn get_applied_index(&self) -> Index;
     27 
     28     /// Applies a log entry to the state machine, returning a client result.
     29     /// Errors are considered applied and propagated back to the client.
     30     ///
     31     /// This is executed on all replicas, so the result must be deterministic:
     32     /// it must yield the same state and result on all replicas, even if the
     33     /// command is reapplied following a node crash.
     34     ///
     35     /// Any non-deterministic apply error (e.g. an IO error) must panic and
     36     /// crash the node -- if it instead returns an error to the client, the
     37     /// command is considered applied and replica states will diverge. The state
     38     /// machine is responsible for panicing when appropriate.
     39     ///
     40     /// The entry may contain a noop command, which is committed by Raft during
     41     /// leader changes. This still needs to be applied to the state machine to
     42     /// properly update the applied index, and should return an empty result.
     43     fn apply(&mut self, entry: Entry) -> Result<Vec<u8>>;
     44 
     45     /// Executes a read command in the state machine, returning a client result.
     46     /// Errors are also propagated back to the client.
     47     ///
     48     /// This is only executed on a single replica/node, so it must not result in
     49     /// any state changes (i.e. it must not write).
     50     fn read(&self, command: Vec<u8>) -> Result<Vec<u8>>;
     51 }

     92     /// A simple string key/value store. Takes KVCommands.
     93     pub struct KV {
     94         applied_index: Index,
     95         data: BTreeMap<String, String>,
     96     }
     97 
     98     impl KV {
     99         pub fn new() -> Box<Self> {
    100             Box::new(Self { applied_index: 0, data: BTreeMap::new() })
    101         }
    102     }
    103 
    104     impl State for KV {
    105         fn get_applied_index(&self) -> Index {
    106             self.applied_index
    107         }
    108 
    109         fn apply(&mut self, entry: Entry) -> Result<Vec<u8>> {
    110             let command = entry.command.as_deref().map(KVCommand::decode).transpose()?;
    111             let response = match command {
    112                 Some(KVCommand::Put { key, value }) => {
    113                     self.data.insert(key, value);
    114                     KVResponse::Put(entry.index).encode()
    115                 }
    116                 Some(c @ (KVCommand::Get { .. } | KVCommand::Scan)) => {
    117                     panic!("{c} submitted as write command")
    118                 }
    119                 None => Vec::new(),
    120             };
    121             self.applied_index = entry.index;
    122             Ok(response)
    123         }
    124 
    125         fn read(&self, command: Vec<u8>) -> Result<Vec<u8>> {
    126             match KVCommand::decode(&command)? {
    127                 KVCommand::Get { key } => {
    128                     Ok(KVResponse::Get(self.data.get(&key).cloned()).encode())
    129                 }
    130                 KVCommand::Scan => Ok(KVResponse::Scan(self.data.clone()).encode()),
    131                 c @ KVCommand::Put { .. } => panic!("{c} submitted as read command"),
    132             }
    133         }
    134     }

    136     /// A KV command. Returns the corresponding KVResponse.
    137     #[derive(Serialize, Deserialize)]
    138     pub enum KVCommand {
    139         /// Fetches the value of the given key.
    140         Get { key: String },
    141         /// Stores the given key/value pair, returning the applied index.
    142         Put { key: String, value: String },
    143         /// Returns all key/value pairs.
    144         Scan,
    145     }

    147     impl encoding::Value for KVCommand {}

    149     impl std::fmt::Display for KVCommand {
    150         fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
    151             match self {
    152                 Self::Get { key } => write!(f, "get {key}"),
    153                 Self::Put { key, value } => write!(f, "put {key}={value}"),
    154                 Self::Scan => write!(f, "scan"),
    155             }
    156         }
    157     }

    159     /// A KVCommand response.
    160     #[derive(Serialize, Deserialize)]
    161     pub enum KVResponse {
    162         /// Get returns the key's value, or None if it does not exist.
    163         Get(Option<String>),
    164         /// Put returns the applied index of the command.
    165         Put(Index),
    166         /// Scan returns the key/value pairs.
    167         Scan(BTreeMap<String, String>),
    168     }
    169 
    170     impl encoding::Value for KVResponse {}
    171 
    172     impl std::fmt::Display for KVResponse {
    173         fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
    174             match self {
    175                 Self::Get(Some(value)) => write!(f, "{value}"),
    176                 Self::Get(None) => write!(f, "None"),
    177                 Self::Put(applied_index) => write!(f, "{applied_index}"),
    178                 Self::Scan(kvs) => {
    179                     write!(f, "{}", kvs.iter().map(|(k, v)| format!("{k}={v}")).join(","))
    180                 }
    181             }
    182         }
    183     }
