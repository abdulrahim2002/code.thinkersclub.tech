     11 /// A SQL engine. This provides low-level CRUD (create, read, update, delete)
     12 /// operations for table rows, a schema catalog for accessing and modifying
     13 /// table schemas, and interactive SQL sessions that execute client SQL
     14 /// statements. All engine access is transactional with snapshot isolation.
     15 pub trait Engine<'a>: Sized {
     16     /// The engine's transaction type. This provides both row-level CRUD
     17     /// operations as well as transactional access to the schema catalog. It
     18     /// can't outlive the engine.
     19     type Transaction: Transaction + Catalog + 'a;
     20 
     21     /// Begins a read-write transaction.
     22     fn begin(&'a self) -> Result<Self::Transaction>;
     23     /// Begins a read-only transaction.
     24     fn begin_read_only(&'a self) -> Result<Self::Transaction>;
     25     /// Begins a read-only transaction as of a historical version.
     26     fn begin_as_of(&'a self, version: mvcc::Version) -> Result<Self::Transaction>;
     27 
     28     /// Creates a session for executing SQL statements. Can't outlive engine.
     29     fn session(&'a self) -> Session<'a, Self> {
     30         Session::new(self)
     31     }
     32 }
     33 
     34 /// A SQL transaction. Executes transactional CRUD operations on table rows.
     35 /// Provides snapshot isolation (see `storage::mvcc` module for details).
     36 ///
     37 /// All methods operate on row batches rather than single rows to amortize the
     38 /// cost. With the Raft engine, each call results in a Raft roundtrip, and we'd
     39 /// rather not have to do that for every single row that's modified.
     40 pub trait Transaction {
     41     /// The transaction's internal MVCC state.
     42     fn state(&self) -> &mvcc::TransactionState;
     43     /// The transaction's MVCC version. Unique for read/write transactions.
     44     fn version(&self) -> mvcc::Version {
     45         self.state().version
     46     }
     47     /// Whether the transaction is read-only.
     48     fn read_only(&self) -> bool {
     49         self.state().read_only
     50     }
     51 
     52     /// Commits the transaction.
     53     fn commit(self) -> Result<()>;
     54     /// Rolls back the transaction.
     55     fn rollback(self) -> Result<()>;
     56 
     57     /// Deletes table rows by primary key, if they exist.
     58     fn delete(&self, table: &str, ids: &[Value]) -> Result<()>;
     59     /// Fetches table rows by primary key, if they exist.
     60     fn get(&self, table: &str, ids: &[Value]) -> Result<Vec<Row>>;
     61     /// Inserts new table rows.
     62     fn insert(&self, table: &str, rows: Vec<Row>) -> Result<()>;
     63     /// Looks up a set of primary keys by index values. BTreeSet for testing.
     64     fn lookup_index(&self, table: &str, column: &str, values: &[Value]) -> Result<BTreeSet<Value>>;
     65     /// Scans a table's rows, optionally applying the given filter.
     66     fn scan(&self, table: &str, filter: Option<Expression>) -> Result<Rows>;
     67     /// Updates table rows by primary key. Uses BTreeMap for testing.
     68     fn update(&self, table: &str, rows: BTreeMap<Value, Row>) -> Result<()>;
     69 }

     71 /// The catalog stores table schema information. It must be implemented for
     72 /// Engine::Transaction, and is thus fully transactional. For simplicity, it
     73 /// only supports creating and dropping tables. There are no ALTER TABLE schema
     74 /// changes, nor CREATE INDEX -- everything has to be specified when the table
     75 /// is initially created.
     76 ///
     77 /// This type is separate from Transaction, even though Engine::Transaction
     78 /// requires transactions to implement it. This allows better control of when
     79 /// catalog access should be used (i.e. during planning, not execution).
     80 pub trait Catalog {
     81     /// Creates a new table. Errors if it already exists.
     82     fn create_table(&self, table: Table) -> Result<()>;
     83     /// Drops a table. Errors if it does not exist, unless if_exists is true.
     84     /// Returns true if the table existed and was deleted.
     85     fn drop_table(&self, table: &str, if_exists: bool) -> Result<bool>;
     86     /// Fetches a table schema, or None if it doesn't exist.
     87     fn get_table(&self, table: &str) -> Result<Option<Table>>;
     88     /// Returns a list of all table schemas.
     89     fn list_tables(&self) -> Result<Vec<Table>>;
     90 
     91     /// Fetches a table schema, or errors if it does not exist.
     92     fn must_get_table(&self, table: &str) -> Result<Table> {
     93         self.get_table(table)?.ok_or_else(|| errinput!("table {table} does not exist"))
     94     }
     95 }
