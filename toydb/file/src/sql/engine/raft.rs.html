     14 /// A Raft-based SQL engine. This dispatches to the `Local` engine for local
     15 /// storage and processing on each node, but plumbs read/write commands through
     16 /// Raft for distributed consensus.
     17 ///
     18 /// The `Raft` engine itself is simply a Raft client which sends `raft::Request`
     19 /// requests to the local Raft node for processing. These requests are received
     20 /// and processed by the Raft engine's `State` state machine running below Raft
     21 /// on each node, which forwards the commands to a `Local` SQL engine which
     22 /// actually executes them using a `storage::Engine` for local storage.
     23 ///
     24 /// For more details on how SQL statements flow through the engine, see the
     25 /// `sql` module documentation.
     26 pub struct Raft {
     27     /// Sends requests to the local Raft node, along with a response channel.
     28     tx: Sender<(raft::Request, Sender<Result<raft::Response>>)>,
     29 }
     30 
     31 impl Raft {
     32     /// The unversioned key used to store the applied index. Just uses a string
     33     /// for simplicity.
     34     pub const APPLIED_INDEX_KEY: &'static [u8] = b"applied_index";
     35 
     36     /// Creates a new Raft-based SQL engine, given a Raft request channel to the
     37     /// local Raft node.
     38     pub fn new(tx: Sender<(raft::Request, Sender<Result<raft::Response>>)>) -> Self {
     39         Self { tx }
     40     }

     42     /// Creates the Raft-managed state machine for the Raft engine. Receives
     43     /// commands from the Raft engine and executes them on a `Local` engine.
     44     pub fn new_state<E: storage::Engine + 'static>(engine: E) -> Result<State<E>> {
     45         State::new(engine)
     46     }

     48     /// Executes a request against the Raft cluster, waiting for the response.
     49     fn execute(&self, request: raft::Request) -> Result<raft::Response> {
     50         let (response_tx, response_rx) = crossbeam::channel::bounded(1);
     51         self.tx.send((request, response_tx))?;
     52         response_rx.recv()?
     53     }
     54 
     55     /// Writes through Raft, deserializing the response into the return type.
     56     fn write<V: DeserializeOwned>(&self, write: Write) -> Result<V> {
     57         match self.execute(raft::Request::Write(write.encode()))? {
     58             raft::Response::Write(response) => bincode::deserialize(&response),
     59             response => errdata!("unexpected Raft write response {response:?}"),
     60         }
     61     }
     62 
     63     /// Reads from Raft, deserializing the response into the return type.
     64     fn read<V: DeserializeOwned>(&self, read: Read) -> Result<V> {
     65         match self.execute(raft::Request::Read(read.encode()))? {
     66             raft::Response::Read(response) => bincode::deserialize(&response),
     67             response => errdata!("unexpected Raft read response {response:?}"),
     68         }
     69     }
     70 
     71     /// Raft SQL engine status.
     72     pub fn status(&self) -> Result<Status> {
     73         let raft = match self.execute(raft::Request::Status)? {
     74             raft::Response::Status(status) => status,
     75             response => return errdata!("unexpected Raft status response {response:?}"),
     76         };
     77         let mvcc = self.read(Read::Status)?;
     78         Ok(Status { raft, mvcc })
     79     }
     80 }

     98 /// A Raft SQL engine transaction.
     99 ///
    100 /// This keeps track of the transaction state in memory, for the benefit of
    101 /// read-only transactions. An `mvcc::Transaction` normally encapsulates this
    102 /// and manages it in memory, but since `mvcc::Transaction` runs below Raft, it
    103 /// can't maintain this state between individual requests (which could execute
    104 /// on different leaders). Instead, we use `mvcc::Transaction::resume` to resume
    105 /// the transaction using the provided transaction state for each request.
    106 pub struct Transaction<'a> {
    107     /// The Raft SQL engine, used to communicate with Raft.
    108     engine: &'a Raft,
    109     /// The MVCC transaction state.
    110     state: mvcc::TransactionState,
    111 }
    112 
    113 impl<'a> Transaction<'a> {
    114     /// Starts a transaction in the given mode.
    115     fn begin(engine: &'a Raft, read_only: bool, as_of: Option<mvcc::Version>) -> Result<Self> {
    116         assert!(as_of.is_none() || read_only, "can't use as_of without read_only");
    117         // Read-only transactions don't need to persist anything, they just need
    118         // to grab the current transaction state, so submit them as reads to
    119         // avoid a replication roundtrip.
    120         let state = if read_only || as_of.is_some() {
    121             engine.read(Read::BeginReadOnly { as_of })?
    122         } else {
    123             engine.write(Write::Begin)?
    124         };
    125         Ok(Self { engine, state })
    126     }
    127 }
    128 
    129 impl<'a> super::Transaction for Transaction<'a> {
    130     fn state(&self) -> &mvcc::TransactionState {
    131         &self.state
    132     }
    133 
    134     fn commit(self) -> Result<()> {
    135         if self.state.read_only {
    136             return Ok(()); // noop
    137         }
    138         self.engine.write(Write::Commit(self.state.into()))
    139     }
    140 
    141     fn rollback(self) -> Result<()> {
    142         if self.state.read_only {
    143             return Ok(()); // noop
    144         }
    145         self.engine.write(Write::Rollback(self.state.into()))
    146     }
    147 
    148     fn delete(&self, table: &str, ids: &[Value]) -> Result<()> {
    149         self.engine.write(Write::Delete {
    150             txn: (&self.state).into(),
    151             table: table.into(),
    152             ids: ids.into(),
    153         })
    154     }
    155 
    156     fn get(&self, table: &str, ids: &[Value]) -> Result<Vec<Row>> {
    157         self.engine.read(Read::Get {
    158             txn: (&self.state).into(),
    159             table: table.into(),
    160             ids: ids.into(),
    161         })
    162     }
    163 
    164     fn insert(&self, table: &str, rows: Vec<Row>) -> Result<()> {
    165         self.engine.write(Write::Insert { txn: (&self.state).into(), table: table.into(), rows })
    166     }
    167 
    168     fn lookup_index(&self, table: &str, column: &str, values: &[Value]) -> Result<BTreeSet<Value>> {
    169         self.engine.read(Read::LookupIndex {
    170             txn: (&self.state).into(),
    171             table: table.into(),
    172             column: column.into(),
    173             values: values.into(),
    174         })
    175     }
    176 
    177     fn scan(&self, table: &str, filter: Option<Expression>) -> Result<Rows> {
    178         let scan: Vec<Row> = self.engine.read(Read::Scan {
    179             txn: (&self.state).into(),
    180             table: table.into(),
    181             filter,
    182         })?;
    183         Ok(Box::new(scan.into_iter().map(Ok)))
    184     }
    185 
    186     fn update(&self, table: &str, rows: BTreeMap<Value, Row>) -> Result<()> {
    187         self.engine.write(Write::Update { txn: (&self.state).into(), table: table.into(), rows })
    188     }
    189 }

    213 /// The state machine for the Raft SQL engine. Receives commands from the Raft
    214 /// SQL engine and dispatches to a `Local` SQL engine which does the actual
    215 /// work, using a `storage::Engine` for storage.
    216 ///
    217 /// For simplicity, we don't attempt to stream large requests or responses,
    218 /// instead simply delivering them as one large chunk. This means that e.g. a
    219 /// full table scan will pull the entire table into memory, serialize it, and
    220 /// send it across the network as one message. The simplest way to address this
    221 /// would likely be to send row batches, but this is fine for our purposes.
    222 pub struct State<E: storage::Engine + 'static> {
    223     /// The local SQL engine.
    224     local: super::Local<E>,
    225     /// The last applied index. This tells Raft which command to apply next.
    226     applied_index: raft::Index,
    227 }
    228 
    229 impl<E: storage::Engine> State<E> {
    230     /// Creates a new Raft state maching using the given storage engine for
    231     /// local storage.
    232     pub fn new(engine: E) -> Result<Self> {
    233         let local = super::Local::new(engine);
    234         let applied_index = local
    235             .get_unversioned(Raft::APPLIED_INDEX_KEY)?
    236             .map(|b| bincode::deserialize(&b))
    237             .transpose()?
    238             .unwrap_or(0);
    239         Ok(State { local, applied_index })
    240     }
    241 
    242     /// Executes a write command.
    243     fn write(&self, command: Write) -> Result<Vec<u8>> {
    244         Ok(match command {
    245             Write::Begin => self.local.begin()?.state().encode(),
    246             Write::Commit(txn) => {
    247                 bincode::serialize(&self.local.resume(txn.into_owned())?.commit()?)
    248             }
    249             Write::Rollback(txn) => {
    250                 bincode::serialize(&self.local.resume(txn.into_owned())?.rollback()?)
    251             }
    252 
    253             Write::Delete { txn, table, ids } => {
    254                 bincode::serialize(&self.local.resume(txn.into_owned())?.delete(&table, &ids)?)
    255             }
    256             Write::Insert { txn, table, rows } => {
    257                 bincode::serialize(&self.local.resume(txn.into_owned())?.insert(&table, rows)?)
    258             }
    259             Write::Update { txn, table, rows } => {
    260                 bincode::serialize(&self.local.resume(txn.into_owned())?.update(&table, rows)?)
    261             }
    262 
    263             Write::CreateTable { txn, schema } => {
    264                 bincode::serialize(&self.local.resume(txn.into_owned())?.create_table(schema)?)
    265             }
    266             Write::DropTable { txn, table, if_exists } => bincode::serialize(
    267                 &self.local.resume(txn.into_owned())?.drop_table(&table, if_exists)?,
    268             ),
    269         })
    270     }
    271 }

    273 impl<E: storage::Engine> raft::State for State<E> {
    274     fn get_applied_index(&self) -> raft::Index {
    275         self.applied_index
    276     }
    277 
    278     fn apply(&mut self, entry: raft::Entry) -> Result<Vec<u8>> {
    279         assert_eq!(entry.index, self.applied_index + 1, "entry index not after applied index");
    280 
    281         let result = match &entry.command {
    282             Some(command) => match self.write(Write::decode(command)?) {
    283                 // Panic on non-deterministic apply failures, to prevent replica
    284                 // divergence. See [`raft::State`] docs for details.
    285                 Err(e) if !e.is_deterministic() => panic!("non-deterministic apply failure: {e}"),
    286                 result => result,
    287             },
    288             // Raft submits noop commands on leader changes. Ignore them, but
    289             // record the applied index below.
    290             None => Ok(Vec::new()),
    291         };
    292 
    293         // Persist the applied index. We don't have to flush, because it's ok to
    294         // lose a tail of the state machine writes (e.g. if the machine
    295         // crashes). Raft will replay the log from the last known applied index.
    296         self.applied_index = entry.index;
    297         self.local.set_unversioned(Raft::APPLIED_INDEX_KEY, bincode::serialize(&entry.index))?;
    298         result
    299     }
    300 
    301     fn read(&self, command: Vec<u8>) -> Result<Vec<u8>> {
    302         Ok(match Read::decode(&command)? {
    303             Read::BeginReadOnly { as_of } => {
    304                 let txn = match as_of {
    305                     Some(version) => self.local.begin_as_of(version)?,
    306                     None => self.local.begin_read_only()?,
    307                 };
    308                 txn.state().encode()
    309             }
    310             Read::Status => self.local.mvcc.status()?.encode(),
    311 
    312             Read::Get { txn, table, ids } => {
    313                 self.local.resume(txn.into_owned())?.get(&table, &ids)?.encode()
    314             }
    315             Read::LookupIndex { txn, table, column, values } => self
    316                 .local
    317                 .resume(txn.into_owned())?
    318                 .lookup_index(&table, &column, &values)?
    319                 .encode(),
    320             Read::Scan { txn, table, filter } => {
    321                 // For simplicity, buffer the entire scan. See `State` comment.
    322                 self.local
    323                     .resume(txn.into_owned())?
    324                     .scan(&table, filter)?
    325                     .collect::<Result<Vec<Row>>>()?
    326                     .encode()
    327             }
    328 
    329             Read::GetTable { txn, table } => {
    330                 self.local.resume(txn.into_owned())?.get_table(&table)?.encode()
    331             }
    332             Read::ListTables { txn } => {
    333                 self.local.resume(txn.into_owned())?.list_tables()?.encode()
    334             }
    335         })
    336     }
    337 }
