     15 /// A SQL client session. Executes raw SQL statements against a SQL engine and
     16 /// handles transaction control.
     17 pub struct Session<'a, E: Engine<'a>> {
     18     /// The SQL engine.
     19     engine: &'a E,
     20     /// The current transaction, if any.
     21     txn: Option<E::Transaction>,
     22 }

     24 impl<'a, E: Engine<'a>> Session<'a, E> {
     25     /// Creates a new session using the given SQL engine.
     26     pub fn new(engine: &'a E) -> Self {
     27         Self { engine, txn: None }
     28     }
     29 
     30     /// Executes a client statement.
     31     pub fn execute(&mut self, statement: &str) -> Result<StatementResult> {
     32         // Parse and execute the statement. Transaction control is done here,
     33         // other statements are executed by the SQL engine.
     34         Ok(match Parser::new(statement).parse()? {
     35             ast::Statement::Begin { read_only, as_of } => {
     36                 if self.txn.is_some() {
     37                     return errinput!("already in a transaction");
     38                 }
     39                 let txn = match (read_only, as_of) {
     40                     (false, None) => self.engine.begin()?,
     41                     (true, None) => self.engine.begin_read_only()?,
     42                     (true, Some(as_of)) => self.engine.begin_as_of(as_of)?,
     43                     (false, Some(_)) => {
     44                         return errinput!("can't start read-write transaction in a given version")
     45                     }
     46                 };
     47                 let state = txn.state().clone();
     48                 self.txn = Some(txn);
     49                 StatementResult::Begin(state)
     50             }
     51             ast::Statement::Commit => {
     52                 let Some(txn) = self.txn.take() else {
     53                     return errinput!("not in a transaction");
     54                 };
     55                 let version = txn.version();
     56                 txn.commit()?;
     57                 StatementResult::Commit { version }
     58             }
     59             ast::Statement::Rollback => {
     60                 let Some(txn) = self.txn.take() else {
     61                     return errinput!("not in a transaction");
     62                 };
     63                 let version = txn.version();
     64                 txn.rollback()?;
     65                 StatementResult::Rollback { version }
     66             }
     67             ast::Statement::Explain(statement) => self.with_txn(true, |txn| {
     68                 Ok(StatementResult::Explain(Plan::build(*statement, txn)?.optimize()?))
     69             })?,
     70             statement => {
     71                 let read_only = matches!(statement, ast::Statement::Select { .. });
     72                 self.with_txn(read_only, |txn| {
     73                     Plan::build(statement, txn)?.optimize()?.execute(txn)?.try_into()
     74                 })?
     75             }
     76         })
     77     }

     79     /// Runs a closure in the session's explicit transaction, if there is one,
     80     /// otherwise a temporary implicit transaction. If read_only is true, uses a
     81     /// read-only implicit transaction. Does not retry errors.
     82     pub fn with_txn<F, T>(&mut self, read_only: bool, f: F) -> Result<T>
     83     where
     84         F: FnOnce(&mut E::Transaction) -> Result<T>,
     85     {
     86         // Use the current explicit transaction, if there is one.
     87         if let Some(ref mut txn) = self.txn {
     88             return f(txn);
     89         }
     90         // Otherwise, use an implicit transaction. Doing this session-side
     91         // results in additional Raft roundtrips to begin and complete the
     92         // transaction -- we could avoid this if the below-Raft engine supported
     93         // implicit transactions, but we keep it simple.
     94         let mut txn = match read_only {
     95             true => self.engine.begin_read_only()?,
     96             false => self.engine.begin()?,
     97         };
     98         let result = f(&mut txn);
     99         match result {
    100             Ok(_) => txn.commit()?,
    101             Err(_) => txn.rollback()?,
    102         }
    103         result
    104     }
    105 }

    114 /// If the session has an open transaction when dropped, roll it back.
    115 impl<'a, E: Engine<'a>> Drop for Session<'a, E> {
    116     fn drop(&mut self) {
    117         if let Some(txn) = self.txn.take() {
    118             if let Err(error) = txn.rollback() {
    119                 error!("implicit transaction rollback failed: {error}")
    120             }
    121         }
    122     }
    123 }

    125 /// A session statement result. Sent across the wire to SQL clients.
    126 #[derive(Debug, PartialEq, Serialize, Deserialize)]
    127 pub enum StatementResult {
    128     Begin(mvcc::TransactionState),
    129     Commit { version: mvcc::Version },
    130     Rollback { version: mvcc::Version },
    131     Explain(Plan),
    132     CreateTable { name: String },
    133     DropTable { name: String, existed: bool },
    134     Delete { count: u64 },
    135     Insert { count: u64 },
    136     Update { count: u64 },
    137     // For simplicity, we buffer and send the entire set of rows as a vector
    138     // instead of streaming them to the client. Streaming reads haven't been
    139     // implemented from Raft either, so they're buffered all the way through.
    140     Select { columns: Vec<Label>, rows: Vec<Row> },
    141 }

    143 /// Converts an execution result into a statement result.
    144 impl TryFrom<ExecutionResult> for StatementResult {
    145     type Error = Error;
    146 
    147     fn try_from(result: ExecutionResult) -> Result<Self> {
    148         Ok(match result {
    149             ExecutionResult::CreateTable { name } => Self::CreateTable { name },
    150             ExecutionResult::DropTable { name, existed } => Self::DropTable { name, existed },
    151             ExecutionResult::Delete { count } => Self::Delete { count },
    152             ExecutionResult::Insert { count } => Self::Insert { count },
    153             ExecutionResult::Update { count } => Self::Update { count },
    154             ExecutionResult::Select { rows, columns } => {
    155                 // We buffer the entire set of rows, for simplicity.
    156                 Self::Select { columns, rows: rows.try_collect()? }
    157             }
    158         })
    159     }
    160 }

    162 /// Converts a query result into a row iterator.
    163 impl TryFrom<StatementResult> for Rows {
    164     type Error = Error;
    165 
    166     fn try_from(result: StatementResult) -> Result<Self> {
    167         let StatementResult::Select { rows, .. } = result else {
    168             return errdata!("expected select result, found {result:?}");
    169         };
    170         Ok(Box::new(rows.into_iter().map(Ok)))
    171     }
    172 }
