      4 /// The lexer (lexical analyzer) preprocesses raw SQL strings into a sequence of
      5 /// lexical tokens (e.g. keyword, number, string, etc), which are passed on to
      6 /// the SQL parser. In doing so, it strips away basic syntactic noise such as
      7 /// whitespace, case, and quotes, and performs initial symbol validation.
      8 pub struct Lexer<'a> {
      9     chars: std::iter::Peekable<std::str::Chars<'a>>,
     10 }
     11 
     12 /// A lexical token.
     13 ///
     14 /// For simplicity, these carry owned String copies rather than &str references
     15 /// into the original input string. This causes frequent allocations, but that's
     16 /// fine for our purposes here.
     17 #[derive(Clone, Debug, PartialEq)]
     18 pub enum Token {
     19     /// A numeric string, with digits, decimal points, and/or exponents. Leading
     20     /// signs (e.g. -) are separate tokens.
     21     Number(String),
     22     /// A Unicode string, with quotes stripped and escape sequences resolved.
     23     String(String),
     24     /// An identifier, with any quotes stripped.
     25     Ident(String),
     26     /// A SQL keyword.
     27     Keyword(Keyword),
     28     Period,             // .
     29     Equal,              // =
     30     NotEqual,           // !=
     31     GreaterThan,        // >
     32     GreaterThanOrEqual, // >=
     33     LessThan,           // <
     34     LessThanOrEqual,    // <=
     35     LessOrGreaterThan,  // <>
     36     Plus,               // +
     37     Minus,              // -
     38     Asterisk,           // *
     39     Slash,              // /
     40     Caret,              // ^
     41     Percent,            // %
     42     Exclamation,        // !
     43     Question,           // ?
     44     Comma,              // ,
     45     Semicolon,          // ;
     46     OpenParen,          // (
     47     CloseParen,         // )
     48 }

     64             Self::LessOrGreaterThan => "<>",

    312 /// The lexer is used as a token iterator.
    313 impl<'a> Iterator for Lexer<'a> {
    314     type Item = Result<Token>;
    315 
    316     fn next(&mut self) -> Option<Result<Token>> {
    317         match self.scan() {
    318             Ok(Some(token)) => Some(Ok(token)),
    319             // If there's any remaining chars, the lexer didn't recognize them.
    320             // Otherwise, we're done lexing.
    321             Ok(None) => self.chars.peek().map(|c| errinput!("unexpected character {c}")),
    322             Err(err) => Some(Err(err)),
    323         }
    324     }
    325 }
    326 
    327 impl<'a> Lexer<'a> {
    328     /// Creates a new lexer for the given string.
    329     pub fn new(input: &'a str) -> Lexer<'a> {
    330         Lexer { chars: input.chars().peekable() }
    331     }
    332 
    333     /// Returns the next character if it satisfies the predicate.
    334     fn next_if(&mut self, predicate: impl Fn(char) -> bool) -> Option<char> {
    335         self.chars.peek().filter(|&&c| predicate(c))?;
    336         self.chars.next()
    337     }
    338 
    339     /// Applies a function to the next character, returning its result and
    340     /// consuming the next character if it's Some.
    341     fn next_if_map<T>(&mut self, map: impl Fn(char) -> Option<T>) -> Option<T> {
    342         let value = self.chars.peek().and_then(|&c| map(c))?;
    343         self.chars.next();
    344         Some(value)
    345     }
    346 
    347     /// Returns true if the next character is the given character, consuming it.
    348     fn next_is(&mut self, c: char) -> bool {
    349         self.next_if(|n| n == c).is_some()
    350     }
    351 
    352     /// Scans the next token, if any.
    353     fn scan(&mut self) -> Result<Option<Token>> {
    354         // Ignore whitespace.
    355         self.skip_whitespace();
    356         // The first character tells us the token type.
    357         match self.chars.peek() {
    358             Some('\'') => self.scan_string(),
    359             Some('"') => self.scan_ident_quoted(),
    360             Some(c) if c.is_ascii_digit() => Ok(self.scan_number()),
    361             Some(c) if c.is_alphabetic() => Ok(self.scan_ident_or_keyword()),
    362             Some(_) => Ok(self.scan_symbol()),
    363             None => Ok(None),
    364         }
    365     }
    366 
    367     /// Scans the next identifier or keyword, if any. It's converted to
    368     /// lowercase, by SQL convention.
    369     fn scan_ident_or_keyword(&mut self) -> Option<Token> {
    370         // The first character must be alphabetic. The rest can be numeric.
    371         let mut name = self.next_if(|c| c.is_alphabetic())?.to_lowercase().to_string();
    372         while let Some(c) = self.next_if(|c| c.is_alphanumeric() || c == '_') {
    373             name.extend(c.to_lowercase())
    374         }
    375         // Check if the identifier matches a keyword.
    376         match Keyword::try_from(name.as_str()).ok() {
    377             Some(keyword) => Some(Token::Keyword(keyword)),
    378             None => Some(Token::Ident(name)),
    379         }
    380     }
    381 
    382     /// Scans the next quoted identifier, if any. Case is preserved.
    383     fn scan_ident_quoted(&mut self) -> Result<Option<Token>> {
    384         if !self.next_is('"') {
    385             return Ok(None);
    386         }
    387         let mut ident = String::new();
    388         loop {
    389             match self.chars.next() {
    390                 // "" is the escape sequence for ".
    391                 Some('"') if self.next_is('"') => ident.push('"'),
    392                 Some('"') => break,
    393                 Some(c) => ident.push(c),
    394                 None => return errinput!("unexpected end of quoted identifier"),
    395             }
    396         }
    397         Ok(Some(Token::Ident(ident)))
    398     }
    399 
    400     /// Scans the next number, if any.
    401     fn scan_number(&mut self) -> Option<Token> {
    402         // Scan the integer part. There must be one digit.
    403         let mut number = self.next_if(|c| c.is_ascii_digit())?.to_string();
    404         while let Some(c) = self.next_if(|c| c.is_ascii_digit()) {
    405             number.push(c)
    406         }
    407         // Scan the fractional part, if any.
    408         if self.next_is('.') {
    409             number.push('.');
    410             while let Some(dec) = self.next_if(|c| c.is_ascii_digit()) {
    411                 number.push(dec)
    412             }
    413         }
    414         // Scan the exponent, if any.
    415         if let Some(exp) = self.next_if(|c| c == 'e' || c == 'E') {
    416             number.push(exp);
    417             if let Some(sign) = self.next_if(|c| c == '+' || c == '-') {
    418                 number.push(sign)
    419             }
    420             while let Some(c) = self.next_if(|c| c.is_ascii_digit()) {
    421                 number.push(c)
    422             }
    423         }
    424         Some(Token::Number(number))
    425     }
    426 
    427     /// Scans the next quoted string literal, if any.
    428     fn scan_string(&mut self) -> Result<Option<Token>> {
    429         if !self.next_is('\'') {
    430             return Ok(None);
    431         }
    432         let mut string = String::new();
    433         loop {
    434             match self.chars.next() {
    435                 // '' is the escape sequence for '.
    436                 Some('\'') if self.next_is('\'') => string.push('\''),
    437                 Some('\'') => break,
    438                 Some(c) => string.push(c),
    439                 None => return errinput!("unexpected end of string literal"),
    440             }
    441         }
    442         Ok(Some(Token::String(string)))
    443     }
    444 
    445     /// Scans the next symbol token, if any.
    446     fn scan_symbol(&mut self) -> Option<Token> {
    447         let mut token = self.next_if_map(|c| {
    448             Some(match c {
    449                 '.' => Token::Period,
    450                 '=' => Token::Equal,
    451                 '>' => Token::GreaterThan,
    452                 '<' => Token::LessThan,
    453                 '+' => Token::Plus,
    454                 '-' => Token::Minus,
    455                 '*' => Token::Asterisk,
    456                 '/' => Token::Slash,
    457                 '^' => Token::Caret,
    458                 '%' => Token::Percent,
    459                 '!' => Token::Exclamation,
    460                 '?' => Token::Question,
    461                 ',' => Token::Comma,
    462                 ';' => Token::Semicolon,
    463                 '(' => Token::OpenParen,
    464                 ')' => Token::CloseParen,
    465                 _ => return None,
    466             })
    467         })?;
    468         // Handle two-character tokens, e.g. !=.
    469         token = match token {
    470             Token::Exclamation if self.next_is('=') => Token::NotEqual,
    471             Token::GreaterThan if self.next_is('=') => Token::GreaterThanOrEqual,
    472             Token::LessThan if self.next_is('>') => Token::LessOrGreaterThan,
    473             Token::LessThan if self.next_is('=') => Token::LessThanOrEqual,
    474             token => token,
    475         };
    476         Some(token)
    477     }
    478 
    479     /// Skips any whitespace.
    480     fn skip_whitespace(&mut self) {
    481         while self.next_if(|c| c.is_whitespace()).is_some() {}
    482     }
    483 }
    484 
    485 /// Returns true if the entire given string is a single valid identifier.
    486 pub fn is_ident(ident: &str) -> bool {
    487     let mut lexer = Lexer::new(ident);
    488     let Some(Ok(Token::Ident(_))) = lexer.next() else { return false };
    489     lexer.next().is_none() // if further tokens, it's not a lone identifier
