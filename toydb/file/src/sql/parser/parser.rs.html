      8 /// The SQL parser takes tokens from the lexer and parses the SQL syntax into an
      9 /// Abstract Syntax Tree (AST). This nested structure represents the syntactic
     10 /// structure of a SQL query (e.g. the SELECT and FROM clauses, values,
     11 /// arithmetic expressions, etc.). However, it only ensures the syntax is
     12 /// well-formed, and does not know whether e.g. a given table or column exists
     13 /// or which kind of join to use -- that is the job of the planner.
     14 pub struct Parser<'a> {
     15     pub lexer: std::iter::Peekable<Lexer<'a>>,
     16 }
     17 
     18 impl<'a> Parser<'a> {
     19     /// Creates a new parser for the given raw SQL string.
     20     pub fn new(statement: &str) -> Parser {
     21         Parser { lexer: Lexer::new(statement).peekable() }
     22     }

     24     /// Parses the input string into an AST statement. The whole string must be
     25     /// parsed as a single statement, ending with an optional semicolon.
     26     pub fn parse(&mut self) -> Result<ast::Statement> {
     27         let statement = self.parse_statement()?;
     28         self.next_is(Token::Semicolon);
     29         if let Some(token) = self.lexer.next().transpose()? {
     30             return errinput!("unexpected token {token}");
     31         }
     32         Ok(statement)
     33     }

     35     /// Fetches the next lexer token, or errors if none is found.
     36     fn next(&mut self) -> Result<Token> {
     37         self.lexer.next().transpose()?.ok_or_else(|| errinput!("unexpected end of input"))
     38     }

     40     /// Returns the next identifier, or errors if not found.
     41     fn next_ident(&mut self) -> Result<String> {
     42         match self.next()? {
     43             Token::Ident(ident) => Ok(ident),
     44             token => errinput!("expected identifier, got {token}"),
     45         }
     46     }

     48     /// Returns the next lexer token if it satisfies the predicate.
     49     fn next_if(&mut self, predicate: impl Fn(&Token) -> bool) -> Option<Token> {
     50         self.peek().unwrap_or(None).filter(|t| predicate(t))?;
     51         self.next().ok()
     52     }

     54     /// Passes the next lexer token through the closure, consuming it if the
     55     /// closure returns Some.
     56     fn next_if_map<T>(&mut self, f: impl Fn(&Token) -> Option<T>) -> Option<T> {
     57         let out = self.peek().unwrap_or(None).map(f)?;
     58         if out.is_some() {
     59             self.next().ok();
     60         }
     61         out
     62     }

     64     /// Grabs the next keyword if there is one.
     65     fn next_if_keyword(&mut self) -> Option<Keyword> {
     66         self.next_if_map(|token| match token {
     67             Token::Keyword(keyword) => Some(*keyword),
     68             _ => None,
     69         })
     70     }

     72     /// Consumes the next lexer token if it is the given token, returning true.
     73     fn next_is(&mut self, token: Token) -> bool {
     74         self.next_if(|t| t == &token).is_some()
     75     }

     77     /// Consumes the next lexer token if it's the expected token, or errors.
     78     fn expect(&mut self, expect: Token) -> Result<()> {
     79         let token = self.next()?;
     80         if token != expect {
     81             return errinput!("expected token {expect}, found {token}");
     82         }
     83         Ok(())
     84     }

     86     /// Consumes the next lexer token if it is the given token. Mostly
     87     /// equivalent to next_is(), but expresses intent better.
     88     fn skip(&mut self, token: Token) {
     89         self.next_is(token);

     92     /// Peeks the next lexer token if any, but transposes it for convenience.
     93     fn peek(&mut self) -> Result<Option<&Token>> {
     94         self.lexer.peek().map(|r| r.as_ref().map_err(|err| err.clone())).transpose()
     95     }

     97     /// Parses a SQL statement.
     98     fn parse_statement(&mut self) -> Result<ast::Statement> {
     99         let Some(token) = self.peek()? else {
    100             return errinput!("unexpected end of input");
    101         };
    102         match token {
    103             Token::Keyword(Keyword::Begin) => self.parse_begin(),
    104             Token::Keyword(Keyword::Commit) => self.parse_commit(),
    105             Token::Keyword(Keyword::Rollback) => self.parse_rollback(),
    106             Token::Keyword(Keyword::Explain) => self.parse_explain(),
    107 
    108             Token::Keyword(Keyword::Create) => self.parse_create_table(),
    109             Token::Keyword(Keyword::Drop) => self.parse_drop_table(),
    110 
    111             Token::Keyword(Keyword::Delete) => self.parse_delete(),
    112             Token::Keyword(Keyword::Insert) => self.parse_insert(),
    113             Token::Keyword(Keyword::Select) => self.parse_select(),
    114             Token::Keyword(Keyword::Update) => self.parse_update(),
    115 
    116             token => errinput!("unexpected token {token}"),
    117         }
    118     }

    120     /// Parses a BEGIN statement.
    121     fn parse_begin(&mut self) -> Result<ast::Statement> {
    122         self.expect(Keyword::Begin.into())?;
    123         self.skip(Keyword::Transaction.into());
    124 
    125         let mut read_only = false;
    126         if self.next_is(Keyword::Read.into()) {
    127             match self.next()? {
    128                 Token::Keyword(Keyword::Only) => read_only = true,
    129                 Token::Keyword(Keyword::Write) => {}
    130                 token => return errinput!("unexpected token {token}"),
    131             }
    132         }
    133 
    134         let mut as_of = None;
    135         if self.next_is(Keyword::As.into()) {
    136             self.expect(Keyword::Of.into())?;
    137             self.expect(Keyword::System.into())?;
    138             self.expect(Keyword::Time.into())?;
    139             match self.next()? {
    140                 Token::Number(n) => as_of = Some(n.parse()?),
    141                 token => return errinput!("unexpected token {token}, wanted number"),
    142             }
    143         }
    144         Ok(ast::Statement::Begin { read_only, as_of })
    145     }

    168     /// Parses a CREATE TABLE statement.
    169     fn parse_create_table(&mut self) -> Result<ast::Statement> {
    170         self.expect(Keyword::Create.into())?;
    171         self.expect(Keyword::Table.into())?;
    172         let name = self.next_ident()?;
    173         self.expect(Token::OpenParen)?;
    174         let mut columns = Vec::new();
    175         loop {
    176             columns.push(self.parse_create_table_column()?);
    177             if !self.next_is(Token::Comma) {
    178                 break;
    179             }
    180         }
    181         self.expect(Token::CloseParen)?;
    182         Ok(ast::Statement::CreateTable { name, columns })
    183     }

    185     /// Parses a CREATE TABLE column definition.
    186     fn parse_create_table_column(&mut self) -> Result<ast::Column> {
    187         let name = self.next_ident()?;
    188         let datatype = match self.next()? {
    189             Token::Keyword(Keyword::Bool | Keyword::Boolean) => DataType::Boolean,
    190             Token::Keyword(Keyword::Float | Keyword::Double) => DataType::Float,
    191             Token::Keyword(Keyword::Int | Keyword::Integer) => DataType::Integer,
    192             Token::Keyword(Keyword::String | Keyword::Text | Keyword::Varchar) => DataType::String,
    193             token => return errinput!("unexpected token {token}"),
    194         };
    195         let mut column = ast::Column {
    196             name,
    197             datatype,
    198             primary_key: false,
    199             nullable: None,
    200             default: None,
    201             unique: false,
    202             index: false,
    203             references: None,
    204         };
    205         while let Some(keyword) = self.next_if_keyword() {
    206             match keyword {
    207                 Keyword::Primary => {
    208                     self.expect(Keyword::Key.into())?;
    209                     column.primary_key = true;
    210                 }
    211                 Keyword::Null => {
    212                     if column.nullable.is_some() {
    213                         return errinput!("nullability already set for column {}", column.name);
    214                     }
    215                     column.nullable = Some(true)
    216                 }
    217                 Keyword::Not => {
    218                     self.expect(Keyword::Null.into())?;
    219                     if column.nullable.is_some() {
    220                         return errinput!("nullability already set for column {}", column.name);
    221                     }
    222                     column.nullable = Some(false)
    223                 }
    224                 Keyword::Default => column.default = Some(self.parse_expression()?),
    225                 Keyword::Unique => column.unique = true,
    226                 Keyword::Index => column.index = true,
    227                 Keyword::References => column.references = Some(self.next_ident()?),
    228                 keyword => return errinput!("unexpected keyword {keyword}"),
    229             }
    230         }
    231         Ok(column)
    232     }

    234     /// Parses a DROP TABLE statement.
    235     fn parse_drop_table(&mut self) -> Result<ast::Statement> {
    236         self.expect(Token::Keyword(Keyword::Drop))?;
    237         self.expect(Token::Keyword(Keyword::Table))?;
    238         let mut if_exists = false;
    239         if self.next_is(Keyword::If.into()) {
    240             self.expect(Token::Keyword(Keyword::Exists))?;
    241             if_exists = true;
    242         }
    243         let name = self.next_ident()?;
    244         Ok(ast::Statement::DropTable { name, if_exists })
    245     }

    255     /// Parses an INSERT statement.
    256     fn parse_insert(&mut self) -> Result<ast::Statement> {
    257         self.expect(Keyword::Insert.into())?;
    258         self.expect(Keyword::Into.into())?;
    259         let table = self.next_ident()?;
    260 
    261         let mut columns = None;
    262         if self.next_is(Token::OpenParen) {
    263             let columns = columns.insert(Vec::new());
    264             loop {
    265                 columns.push(self.next_ident()?);
    266                 if !self.next_is(Token::Comma) {
    267                     break;
    268                 }
    269             }
    270             self.expect(Token::CloseParen)?;
    271         }
    272 
    273         self.expect(Keyword::Values.into())?;
    274 
    275         let mut values = Vec::new();
    276         loop {
    277             let mut row = Vec::new();
    278             self.expect(Token::OpenParen)?;
    279             loop {
    280                 row.push(self.parse_expression()?);
    281                 if !self.next_is(Token::Comma) {
    282                     break;
    283                 }
    284             }
    285             self.expect(Token::CloseParen)?;
    286             values.push(row);
    287             if !self.next_is(Token::Comma) {
    288                 break;
    289             }
    290         }
    291 
    292         Ok(ast::Statement::Insert { table, columns, values })
    293     }

    295     /// Parses an UPDATE statement.
    296     fn parse_update(&mut self) -> Result<ast::Statement> {
    297         self.expect(Keyword::Update.into())?;
    298         let table = self.next_ident()?;
    299         self.expect(Keyword::Set.into())?;
    300         let mut set = std::collections::BTreeMap::new();
    301         loop {
    302             let column = self.next_ident()?;
    303             self.expect(Token::Equal)?;
    304             let expr = (!self.next_is(Keyword::Default.into()))
    305                 .then(|| self.parse_expression())
    306                 .transpose()?;
    307             if set.contains_key(&column) {
    308                 return errinput!("column {column} set multiple times");
    309             }
    310             set.insert(column, expr);
    311             if !self.next_is(Token::Comma) {
    312                 break;
    313             }
    314         }
    315         Ok(ast::Statement::Update { table, set, r#where: self.parse_where_clause()? })
    316     }

    338     /// Parses a SELECT clause, if present.
    339     fn parse_select_clause(&mut self) -> Result<Vec<(ast::Expression, Option<String>)>> {
    340         if !self.next_is(Keyword::Select.into()) {
    341             return Ok(Vec::new());
    342         }
    343         let mut select = Vec::new();
    344         loop {
    345             let expr = self.parse_expression()?;
    346             let mut label = None;
    347             if self.next_is(Keyword::As.into()) || matches!(self.peek()?, Some(Token::Ident(_))) {
    348                 if expr == ast::Expression::All {
    349                     return errinput!("can't alias *");
    350                 }
    351                 label = Some(self.next_ident()?);
    352             }
    353             select.push((expr, label));
    354             if !self.next_is(Token::Comma) {
    355                 break;
    356             }
    357         }
    358         Ok(select)
    359     }

    361     /// Parses a FROM clause, if present.
    362     fn parse_from_clause(&mut self) -> Result<Vec<ast::From>> {
    363         if !self.next_is(Keyword::From.into()) {
    364             return Ok(Vec::new());
    365         }
    366         let mut from = Vec::new();
    367         loop {
    368             let mut item = self.parse_from_table()?;
    369             while let Some(r#type) = self.parse_from_join()? {
    370                 let left = Box::new(item);
    371                 let right = Box::new(self.parse_from_table()?);
    372                 let mut predicate = None;
    373                 if r#type != ast::JoinType::Cross {
    374                     self.expect(Keyword::On.into())?;
    375                     predicate = Some(self.parse_expression()?)
    376                 }
    377                 item = ast::From::Join { left, right, r#type, predicate };
    378             }
    379             from.push(item);
    380             if !self.next_is(Token::Comma) {
    381                 break;
    382             }
    383         }
    384         Ok(from)
    385     }

    397     // Parses a FROM JOIN type, if present.
    398     fn parse_from_join(&mut self) -> Result<Option<ast::JoinType>> {
    399         if self.next_is(Keyword::Join.into()) {
    400             return Ok(Some(ast::JoinType::Inner));
    401         }
    402         if self.next_is(Keyword::Cross.into()) {
    403             self.expect(Keyword::Join.into())?;
    404             return Ok(Some(ast::JoinType::Cross));
    405         }
    406         if self.next_is(Keyword::Inner.into()) {
    407             self.expect(Keyword::Join.into())?;
    408             return Ok(Some(ast::JoinType::Inner));
    409         }
    410         if self.next_is(Keyword::Left.into()) {
    411             self.skip(Keyword::Outer.into());
    412             self.expect(Keyword::Join.into())?;
    413             return Ok(Some(ast::JoinType::Left));
    414         }
    415         if self.next_is(Keyword::Right.into()) {
    416             self.skip(Keyword::Outer.into());
    417             self.expect(Keyword::Join.into())?;
    418             return Ok(Some(ast::JoinType::Right));
    419         }
    420         Ok(None)
    421     }

    455     /// Parses an ORDER BY clause, if present.
    456     fn parse_order_by_clause(&mut self) -> Result<Vec<(ast::Expression, ast::Direction)>> {
    457         if !self.next_is(Keyword::Order.into()) {
    458             return Ok(Vec::new());
    459         }
    460         let mut order_by = Vec::new();
    461         self.expect(Keyword::By.into())?;
    462         loop {
    463             let expr = self.parse_expression()?;
    464             let order = self
    465                 .next_if_map(|token| match token {
    466                     Token::Keyword(Keyword::Asc) => Some(ast::Direction::Ascending),
    467                     Token::Keyword(Keyword::Desc) => Some(ast::Direction::Descending),
    468                     _ => None,
    469                 })
    470                 .unwrap_or(ast::Direction::Ascending);
    471             order_by.push((expr, order));
    472             if !self.next_is(Token::Comma) {
    473                 break;
    474             }
    475         }
    476         Ok(order_by)
    477     }

    485     }

    487     /// Parses an expression at the given minimum precedence.
    488     fn parse_expression_at(&mut self, min_precedence: Precedence) -> Result<ast::Expression> {
    489         // If there is a prefix operator, parse it and its right-hand operand.
    490         // Otherwise, parse the left-hand atom.
    491         let mut lhs = if let Some(prefix) = self.parse_prefix_operator(min_precedence) {
    492             let at_precedence = prefix.precedence() + prefix.associativity();
    493             prefix.build(self.parse_expression_at(at_precedence)?)
    494         } else {
    495             self.parse_expression_atom()?
    496         };
    497         // Apply any postfix operators for the left-hand atom.
    498         while let Some(postfix) = self.parse_postfix_operator(min_precedence)? {
    499             lhs = postfix.build(lhs)
    500         }
    501         // Apply any binary infix operators, parsing the right-hand operand.
    502         while let Some(infix) = self.parse_infix_operator(min_precedence) {
    503             let at_precedence = infix.precedence() + infix.associativity();
    504             let rhs = self.parse_expression_at(at_precedence)?;
    505             lhs = infix.build(lhs, rhs);
    506         }
    507         // Apply any postfix operators after the binary operator. Consider e.g.
    508         // 1 + NULL IS NULL.
    509         while let Some(postfix) = self.parse_postfix_operator(min_precedence)? {
    510             lhs = postfix.build(lhs)
    511         }
    512         Ok(lhs)
    513     }

    515     /// Parses an expression atom. This is either:
    516     ///
    517     /// * A literal value.
    518     /// * A column name.
    519     /// * A function call.
    520     /// * A parenthesized expression.
    521     fn parse_expression_atom(&mut self) -> Result<ast::Expression> {
    522         Ok(match self.next()? {
    523             // All columns.
    524             Token::Asterisk => ast::Expression::All,
    525 
    526             // Literal value.
    527             Token::Number(n) if n.chars().all(|c| c.is_ascii_digit()) => {
    528                 ast::Literal::Integer(n.parse()?).into()
    529             }
    530             Token::Number(n) => ast::Literal::Float(n.parse()?).into(),
    531             Token::String(s) => ast::Literal::String(s).into(),
    532             Token::Keyword(Keyword::True) => ast::Literal::Boolean(true).into(),
    533             Token::Keyword(Keyword::False) => ast::Literal::Boolean(false).into(),
    534             Token::Keyword(Keyword::Infinity) => ast::Literal::Float(std::f64::INFINITY).into(),
    535             Token::Keyword(Keyword::NaN) => ast::Literal::Float(std::f64::NAN).into(),
    536             Token::Keyword(Keyword::Null) => ast::Literal::Null.into(),
    537 
    538             // Function call.
    539             Token::Ident(name) if self.next_is(Token::OpenParen) => {
    540                 let mut args = Vec::new();
    541                 while !self.next_is(Token::CloseParen) {
    542                     if !args.is_empty() {
    543                         self.expect(Token::Comma)?;
    544                     }
    545                     args.push(self.parse_expression()?);
    546                 }
    547                 ast::Expression::Function(name, args)
    548             }
    549 
    550             // Column name, either qualified as table.column or unqualified.
    551             Token::Ident(table) if self.next_is(Token::Period) => {
    552                 ast::Expression::Column(Some(table), self.next_ident()?)
    553             }
    554             Token::Ident(column) => ast::Expression::Column(None, column),
    555 
    556             // Parenthesized expression.
    557             Token::OpenParen => {
    558                 let expr = self.parse_expression()?;
    559                 self.expect(Token::CloseParen)?;
    560                 expr
    561             }
    562 
    563             token => return errinput!("expected expression atom, found {token}"),
    564         })
    565     }

    567     /// Parses a prefix operator, if there is one and its precedence is at least
    568     /// min_precedence.
    569     fn parse_prefix_operator(&mut self, min_precedence: Precedence) -> Option<PrefixOperator> {
    570         self.next_if_map(|token| {
    571             let operator = match token {
    572                 Token::Keyword(Keyword::Not) => PrefixOperator::Not,
    573                 Token::Minus => PrefixOperator::Minus,
    574                 Token::Plus => PrefixOperator::Plus,
    575                 _ => return None,
    576             };
    577             Some(operator).filter(|op| op.precedence() >= min_precedence)
    578         })
    579     }

    581     /// Parses an infix operator, if there is one and its precedence is at least
    582     /// min_precedence.
    583     fn parse_infix_operator(&mut self, min_precedence: Precedence) -> Option<InfixOperator> {
    584         self.next_if_map(|token| {
    585             let operator = match token {
    586                 Token::Asterisk => InfixOperator::Multiply,
    587                 Token::Caret => InfixOperator::Exponentiate,
    588                 Token::Equal => InfixOperator::Equal,
    589                 Token::GreaterThan => InfixOperator::GreaterThan,
    590                 Token::GreaterThanOrEqual => InfixOperator::GreaterThanOrEqual,
    591                 Token::Keyword(Keyword::And) => InfixOperator::And,
    592                 Token::Keyword(Keyword::Like) => InfixOperator::Like,
    593                 Token::Keyword(Keyword::Or) => InfixOperator::Or,
    594                 Token::LessOrGreaterThan => InfixOperator::NotEqual,
    595                 Token::LessThan => InfixOperator::LessThan,
    596                 Token::LessThanOrEqual => InfixOperator::LessThanOrEqual,
    597                 Token::Minus => InfixOperator::Subtract,
    598                 Token::NotEqual => InfixOperator::NotEqual,
    599                 Token::Percent => InfixOperator::Remainder,
    600                 Token::Plus => InfixOperator::Add,
    601                 Token::Slash => InfixOperator::Divide,
    602                 _ => return None,
    603             };
    604             Some(operator).filter(|op| op.precedence() >= min_precedence)
    605         })
    606     }
    607 
    608     /// Parses a postfix operator, if there is one and its precedence is at
    609     /// least min_precedence.
    610     fn parse_postfix_operator(
    611         &mut self,
    612         min_precedence: Precedence,
    613     ) -> Result<Option<PostfixOperator>> {
    614         // Handle IS (NOT) NULL/NAN separately, since it's multiple tokens.
    615         if let Some(Token::Keyword(Keyword::Is)) = self.peek()? {
    616             // We can't consume tokens unless the precedence is satisfied, so we
    617             // assume IS NULL (they all have the same precedence).
    618             if PostfixOperator::Is(ast::Literal::Null).precedence() < min_precedence {
    619                 return Ok(None);
    620             }
    621             self.expect(Keyword::Is.into())?;
    622             let not = self.next_is(Keyword::Not.into());
    623             let value = match self.next()? {
    624                 Token::Keyword(Keyword::NaN) => ast::Literal::Float(f64::NAN),
    625                 Token::Keyword(Keyword::Null) => ast::Literal::Null,
    626                 token => return errinput!("unexpected token {token}"),
    627             };
    628             let operator = match not {
    629                 false => PostfixOperator::Is(value),
    630                 true => PostfixOperator::IsNot(value),
    631             };
    632             return Ok(Some(operator));
    633         }
    634 
    635         Ok(self.next_if_map(|token| {
    636             let operator = match token {
    637                 Token::Exclamation => PostfixOperator::Factorial,
    638                 _ => return None,
    639             };
    640             Some(operator).filter(|op| op.precedence() >= min_precedence)
    641         }))
    642     }
    643 }

    645 /// Operator precedence.
    646 type Precedence = u8;
    647 
    648 const LEFT_ASSOCIATIVE: Precedence = 1;
    649 const RIGHT_ASSOCIATIVE: Precedence = 0;
    650 
    651 /// Prefix operators.
    652 enum PrefixOperator {
    653     Minus, // -a
    654     Not,   // NOT a
    655     Plus,  // +a
    656 }
    657 
    658 impl PrefixOperator {
    659     /// The operator precedence.
    660     fn precedence(&self) -> Precedence {
    661         match self {
    662             Self::Not => 3,
    663             Self::Minus | Self::Plus => 10,
    664         }
    665     }
    666 
    667     // The operator associativity. Prefix operators are right-associative by
    668     // definition.
    669     fn associativity(&self) -> Precedence {
    670         RIGHT_ASSOCIATIVE
    671     }
    672 
    673     /// Builds an AST expression for the operator.
    674     fn build(self, rhs: ast::Expression) -> ast::Expression {
    675         let rhs = Box::new(rhs);
    676         match self {
    677             Self::Plus => ast::Operator::Identity(rhs).into(),
    678             Self::Minus => ast::Operator::Negate(rhs).into(),
    679             Self::Not => ast::Operator::Not(rhs).into(),
    680         }
    681     }
    682 }
    683 
    684 /// Infix operators.
    685 enum InfixOperator {
    686     Add,                // a + b
    687     And,                // a AND b
    688     Divide,             // a / b
    689     Equal,              // a = b
    690     Exponentiate,       // a ^ b
    691     GreaterThan,        // a > b
    692     GreaterThanOrEqual, // a >= b
    693     LessThan,           // a < b
    694     LessThanOrEqual,    // a <= b
    695     Like,               // a LIKE b
    696     Multiply,           // a * b
    697     NotEqual,           // a != b
    698     Or,                 // a OR b
    699     Remainder,          // a % b
    700     Subtract,           // a - b
    701 }
    702 
    703 impl InfixOperator {
    704     /// The operator precedence.
    705     ///
    706     /// Mostly follows Postgres, except IS and LIKE having same precedence as =.
    707     /// This is similar to SQLite and MySQL.
    708     fn precedence(&self) -> Precedence {
    709         match self {
    710             Self::Or => 1,
    711             Self::And => 2,
    712             // Self::Not => 3
    713             Self::Equal | Self::NotEqual | Self::Like => 4, // and Self::Is
    714             Self::GreaterThan
    715             | Self::GreaterThanOrEqual
    716             | Self::LessThan
    717             | Self::LessThanOrEqual => 5,
    718             Self::Add | Self::Subtract => 6,
    719             Self::Multiply | Self::Divide | Self::Remainder => 7,
    720             Self::Exponentiate => 8,
    721         }
    722     }
    723 
    724     /// The operator associativity.
    725     fn associativity(&self) -> Precedence {
    726         match self {
    727             Self::Exponentiate => RIGHT_ASSOCIATIVE,
    728             _ => LEFT_ASSOCIATIVE,
    729         }
    730     }
    731 
    732     /// Builds an AST expression for the infix operator.
    733     fn build(self, lhs: ast::Expression, rhs: ast::Expression) -> ast::Expression {
    734         let (lhs, rhs) = (Box::new(lhs), Box::new(rhs));
    735         match self {
    736             Self::Add => ast::Operator::Add(lhs, rhs).into(),
    737             Self::And => ast::Operator::And(lhs, rhs).into(),
    738             Self::Divide => ast::Operator::Divide(lhs, rhs).into(),
    739             Self::Equal => ast::Operator::Equal(lhs, rhs).into(),
    740             Self::Exponentiate => ast::Operator::Exponentiate(lhs, rhs).into(),
    741             Self::GreaterThan => ast::Operator::GreaterThan(lhs, rhs).into(),
    742             Self::GreaterThanOrEqual => ast::Operator::GreaterThanOrEqual(lhs, rhs).into(),
    743             Self::LessThan => ast::Operator::LessThan(lhs, rhs).into(),
    744             Self::LessThanOrEqual => ast::Operator::LessThanOrEqual(lhs, rhs).into(),
    745             Self::Like => ast::Operator::Like(lhs, rhs).into(),
    746             Self::Multiply => ast::Operator::Multiply(lhs, rhs).into(),
    747             Self::NotEqual => ast::Operator::NotEqual(lhs, rhs).into(),
    748             Self::Or => ast::Operator::Or(lhs, rhs).into(),
    749             Self::Remainder => ast::Operator::Remainder(lhs, rhs).into(),
    750             Self::Subtract => ast::Operator::Subtract(lhs, rhs).into(),
    751         }
    752     }
    753 }
    754 
    755 /// Postfix operators.
    756 enum PostfixOperator {
    757     Factorial,           // a!
    758     Is(ast::Literal),    // a IS NULL | NAN
    759     IsNot(ast::Literal), // a IS NOT NULL | NAN
    760 }
    761 
    762 impl PostfixOperator {
    763     // The operator precedence.
    764     fn precedence(&self) -> Precedence {
    765         match self {
    766             Self::Is(_) | Self::IsNot(_) => 4,
    767             Self::Factorial => 9,
    768         }
    769     }
    770 
    771     /// Builds an AST expression for the operator.
    772     fn build(self, lhs: ast::Expression) -> ast::Expression {
    773         let lhs = Box::new(lhs);
    774         match self {
    775             Self::Factorial => ast::Operator::Factorial(lhs).into(),
    776             Self::Is(v) => ast::Operator::Is(lhs, v).into(),
    777             Self::IsNot(v) => ast::Operator::Not(ast::Operator::Is(lhs, v).into()).into(),
    778         }
    779     }
