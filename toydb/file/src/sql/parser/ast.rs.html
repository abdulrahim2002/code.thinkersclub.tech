      5 /// The statement is the root node of the Abstract Syntax Tree, and describes
      6 /// the syntactic structure of a SQL query. It is built from a raw SQL string by
      7 /// the parser, and passed on to the planner which validates it and builds an
      8 /// execution plan from it.
      9 #[derive(Debug)]
     10 pub enum Statement {
     11     /// Begin a new transaction.
     12     Begin { read_only: bool, as_of: Option<u64> },
     13     /// Commit a transaction.
     14     Commit,
     15     /// Roll back a transaction.
     16     Rollback,
     17     /// Explain a statement.
     18     Explain(Box<Statement>),
     19     /// Create a new table.
     20     CreateTable { name: String, columns: Vec<Column> },
     21     /// Drop a table.
     22     DropTable { name: String, if_exists: bool },
     23     /// Delete matching rows.
     24     Delete { table: String, r#where: Option<Expression> },
     25     /// Insert new rows into a table.
     26     Insert {
     27         table: String,
     28         columns: Option<Vec<String>>, // columns given in values, using default for rest
     29         values: Vec<Vec<Expression>>, // rows to insert
     30     },
     31     /// Update rows in a table.
     32     Update {
     33         table: String,
     34         set: BTreeMap<String, Option<Expression>>, // column â†’ value, None for default value
     35         r#where: Option<Expression>,
     36     },
     37     /// Select matching rows.
     38     Select {
     39         select: Vec<(Expression, Option<String>)>, // optional column aliases
     40         from: Vec<From>,
     41         r#where: Option<Expression>,
     42         group_by: Vec<Expression>,
     43         having: Option<Expression>,
     44         order_by: Vec<(Expression, Direction)>,
     45         offset: Option<Expression>,
     46         limit: Option<Expression>,
     47     },

    124 /// To allow using Expressions and Literals in e.g. hashmaps, implement simple
    125 /// equality and hash for all types, including Null and f64::NAN. This is not
    126 /// used for expression evaluation (handled by sql::types::Expression), where
    127 /// these values should not be considered equal to themselves, only in lookups.
    128 impl std::cmp::PartialEq for Literal {
    129     fn eq(&self, other: &Self) -> bool {
    130         match (self, other) {
    131             (Self::Boolean(l), Self::Boolean(r)) => l == r,
    132             (Self::Integer(l), Self::Integer(r)) => l == r,
    133             // Implies NaN == NaN but -NaN != NaN. Similarly with +/-0.0.
    134             (Self::Float(l), Self::Float(r)) => l.to_bits() == r.to_bits(),
    135             (Self::String(l), Self::String(r)) => l == r,
    136             (l, r) => core::mem::discriminant(l) == core::mem::discriminant(r),
    137         }
    138     }
    139 }
    140 
    141 impl std::cmp::Eq for Literal {}
    142 
    143 impl std::hash::Hash for Literal {
    144     fn hash<H: std::hash::Hasher>(&self, state: &mut H) {
    145         core::mem::discriminant(self).hash(state);
    146         match self {
    147             Self::Null => {}
    148             Self::Boolean(v) => v.hash(state),
    149             Self::Integer(v) => v.hash(state),
    150             Self::Float(v) => v.to_bits().hash(state),
    151             Self::String(v) => v.hash(state),
    152         }
    153     }
    154 }

    156 /// Expression operators.
    157 ///
    158 /// Since this is a recursive data structure, we have to box each child
    159 /// expression, which incurs a heap allocation. There are clever ways to get
    160 /// around this, but we keep it simple.
    161 #[derive(Clone, Debug, Eq, Hash, PartialEq)]
    162 pub enum Operator {
    163     And(Box<Expression>, Box<Expression>), // a AND b
    164     Not(Box<Expression>),                  // NOT a
    165     Or(Box<Expression>, Box<Expression>),  // a OR b
    166 
    167     Equal(Box<Expression>, Box<Expression>),       // a = b
    168     GreaterThan(Box<Expression>, Box<Expression>), // a > b
    169     GreaterThanOrEqual(Box<Expression>, Box<Expression>), // a != b
    170     Is(Box<Expression>, Literal),                  // IS NULL or IS NAN
    171     LessThan(Box<Expression>, Box<Expression>),    // a < b
    172     LessThanOrEqual(Box<Expression>, Box<Expression>), // a <= b
    173     NotEqual(Box<Expression>, Box<Expression>),    // a != b
    174 
    175     Add(Box<Expression>, Box<Expression>),          // a + b
    176     Divide(Box<Expression>, Box<Expression>),       // a / b
    177     Exponentiate(Box<Expression>, Box<Expression>), // a ^ b
    178     Factorial(Box<Expression>),                     // a!
    179     Identity(Box<Expression>),                      // +a
    180     Multiply(Box<Expression>, Box<Expression>),     // a * b
    181     Negate(Box<Expression>),                        // -a
    182     Remainder(Box<Expression>, Box<Expression>),    // a % b
    183     Subtract(Box<Expression>, Box<Expression>),     // a - b
    184 
    185     Like(Box<Expression>, Box<Expression>), // a LIKE b
    186 }
    187 
    188 impl Expression {
    189     /// Walks the expression tree depth-first, calling a closure for every node.
    190     /// Halts and returns false if the closure returns false.
    191     pub fn walk(&self, visitor: &mut impl FnMut(&Expression) -> bool) -> bool {
    192         use Operator::*;
    193         if !visitor(self) {
    194             return false;
    195         }
    196         match self {
    197             Self::Operator(Add(lhs, rhs))
    198             | Self::Operator(And(lhs, rhs))
    199             | Self::Operator(Divide(lhs, rhs))
    200             | Self::Operator(Equal(lhs, rhs))
    201             | Self::Operator(Exponentiate(lhs, rhs))
    202             | Self::Operator(GreaterThan(lhs, rhs))
    203             | Self::Operator(GreaterThanOrEqual(lhs, rhs))
    204             | Self::Operator(LessThan(lhs, rhs))
    205             | Self::Operator(LessThanOrEqual(lhs, rhs))
    206             | Self::Operator(Like(lhs, rhs))
    207             | Self::Operator(Multiply(lhs, rhs))
    208             | Self::Operator(NotEqual(lhs, rhs))
    209             | Self::Operator(Or(lhs, rhs))
    210             | Self::Operator(Remainder(lhs, rhs))
    211             | Self::Operator(Subtract(lhs, rhs)) => lhs.walk(visitor) && rhs.walk(visitor),
    212 
    213             Self::Operator(Factorial(expr))
    214             | Self::Operator(Identity(expr))
    215             | Self::Operator(Is(expr, _))
    216             | Self::Operator(Negate(expr))
    217             | Self::Operator(Not(expr)) => expr.walk(visitor),
    218 
    219             Self::Function(_, exprs) => exprs.iter().any(|expr| expr.walk(visitor)),
    220 
    221             Self::All | Self::Column(_, _) | Self::Literal(_) => true,
    222         }
    223     }
    224 
    225     /// Walks the expression tree depth-first while calling a closure until it
    226     /// returns true. This is the inverse of walk().
    227     pub fn contains(&self, visitor: &impl Fn(&Expression) -> bool) -> bool {
    228         !self.walk(&mut |expr| !visitor(expr))
    229     }
    230 
    231     /// Find and collects expressions for which the given closure returns true,
    232     /// adding them to c. Does not recurse into matching expressions.
    233     pub fn collect(&self, visitor: &impl Fn(&Expression) -> bool, c: &mut Vec<Expression>) {
    234         if visitor(self) {
    235             c.push(self.clone());
    236             return;
    237         }
    238         use Operator::*;
    239         match self {
    240             Self::Operator(Add(lhs, rhs))
    241             | Self::Operator(And(lhs, rhs))
    242             | Self::Operator(Divide(lhs, rhs))
    243             | Self::Operator(Equal(lhs, rhs))
    244             | Self::Operator(Exponentiate(lhs, rhs))
    245             | Self::Operator(GreaterThan(lhs, rhs))
    246             | Self::Operator(GreaterThanOrEqual(lhs, rhs))
    247             | Self::Operator(LessThan(lhs, rhs))
    248             | Self::Operator(LessThanOrEqual(lhs, rhs))
    249             | Self::Operator(Like(lhs, rhs))
    250             | Self::Operator(Multiply(lhs, rhs))
    251             | Self::Operator(NotEqual(lhs, rhs))
    252             | Self::Operator(Or(lhs, rhs))
    253             | Self::Operator(Remainder(lhs, rhs))
    254             | Self::Operator(Subtract(lhs, rhs)) => {
    255                 lhs.collect(visitor, c);
    256                 rhs.collect(visitor, c);
    257             }
    258 
    259             Self::Operator(Factorial(expr))
    260             | Self::Operator(Identity(expr))
    261             | Self::Operator(Is(expr, _))
    262             | Self::Operator(Negate(expr))
    263             | Self::Operator(Not(expr)) => expr.collect(visitor, c),
    264 
    265             Self::Function(_, exprs) => exprs.iter().for_each(|expr| expr.collect(visitor, c)),
    266 
    267             Self::All | Self::Column(_, _) | Self::Literal(_) => {}
    268         }
    269     }
    270 }
