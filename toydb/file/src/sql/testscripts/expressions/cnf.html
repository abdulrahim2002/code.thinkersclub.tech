      1 # Tests conversion of logical expressions into canonical normal form.
      2 
      3 # Noop for non-boolean expressions.
      4 [cnf]> 1 + 2
      5 ---
      6 3 ← 1 + 2
      7 
      8 # Applies De Morgan's laws.
      9 [cnf]> NOT (TRUE AND FALSE)
     10 ---
     11 TRUE ← NOT TRUE OR NOT FALSE
     12 
     13 [cnf]> NOT (TRUE OR FALSE)
     14 ---
     15 FALSE ← NOT TRUE AND NOT FALSE
     16 
     17 # NOTs are pushed into the expression.
     18 [cnf]> NOT (TRUE AND TRUE AND TRUE OR TRUE)
     19 ---
     20 FALSE ← (NOT TRUE OR NOT TRUE OR NOT TRUE) AND NOT TRUE
     21 
     22 # ORs are converted to ANDs by the distributive law.
     23 [cnf]> (TRUE AND FALSE) OR (FALSE AND TRUE)
     24 ---
     25 FALSE ← (TRUE OR FALSE) AND (TRUE OR TRUE) AND (FALSE OR FALSE) AND (FALSE OR TRUE)
     26 
     27 # This is also true when combined with De Morgan's laws.
     28 [cnf]> NOT ((TRUE OR FALSE) AND (TRUE OR FALSE))
     29 ---
     30 FALSE ← (NOT TRUE OR NOT TRUE) AND (NOT TRUE OR NOT FALSE) AND (NOT FALSE OR NOT TRUE) AND (NOT FALSE OR NOT FALSE)
