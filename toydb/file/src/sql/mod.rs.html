      1 //! Implements a SQL execution engine. A SQL statement flows through the engine
      2 //! as follows:
      3 //!
      4 //! 1. The `toySQL` client connects to the server, which creates a new
      5 //!    `sql::engine::Session` in `Server::sql_session`.
      6 //!
      7 //! 2. `toySQL` submits a SQL `SELECT` string, which the server executes via
      8 //!     `Session::execute`.
      9 //!
     10 //! 3. `Session::execute` calls `Parser::parse` to parse the SQL `SELECT` string
     11 //!     into an `ast::Statement::Select` AST (Abstract Syntax Tree). The parser
     12 //!     uses the `Lexer` for initial tokenization.
     13 //!     
     14 //! 4. `Session::execute` obtains a new read-only `sql::engine::Transaction` via
     15 //!    `Session::with_txn`. We'll gloss over the details here.
     16 //!
     17 //! 5. `Session::execute` calls `Plan::build` to construct an execution plan
     18 //!    from the AST via the `Planner`, using the `Transaction`'s
     19 //!    `sql::engine::Catalog` trait to look up table schema information.
     20 //!
     21 //! 6. `Session::execute` calls `Plan::optimize` to optimize the execution plan
     22 //!    via the optimizers in `sql::planner::optimizer`. This e.g. performs
     23 //!    filter pushdown to filter rows during storage scans, uses secondary
     24 //!    indexes where appropriate, and chooses more efficient join types.
     25 //!
     26 //! 7. `Session::execute` calls `Plan::execute` to actually execute the plan,
     27 //!    using the `Transaction` to access the `sql::engine::Engine`.  It uses the
     28 //!    executors in `sql::execution` to recursively execute the
     29 //!    `sql::planner::Node` nodes, which stream and process `sql::types::Row`
     30 //!    vectors via `sql::types::Rows` iterators.
     31 //!
     32 //! 8. At the tip of the execution plan there's typically a `Node::Scan` which
     33 //!    performs full table scans from storage. It is executed by
     34 //!    `sql::execution::source::scan`, which calls `Transaction::scan`.
     35 //!
     36 //! 9. The upper `sql::engine::Raft` engine submits a `Read::Scan` request to
     37 //!    Raft via `Raft::read` and `Raft::execute`. This is submitted through the
     38 //!    crossbeam channel `Raft::tx`, which is routed to the local Raft node in
     39 //!    `Server::raft_route` via `raft::Node::step`.
     40 //!
     41 //! 10. We'll skip Raft details, but see the `raft` module documentation. The
     42 //!     `Read::Scan` request eventually makes its way to the SQL state machine
     43 //!     `sql::engine::raft::State` that's managed by Raft. Since this is a read
     44 //!     request, it is executed only on the leader node, calling `State::read`.
     45 //!
     46 //! 11. `State` wraps the `sql::engine::Local` SQL execution engine that runs
     47 //!     on each node, using local storage. `State::read` calls
     48 //!     `Transaction::scan` using a `Local::Transaction`.
     49 //!
     50 //! 12. The `Local` engine uses a `storage::BitCask` engine for local storage,
     51 //!     with `storage::mvcc` providing transactions. See their documentation
     52 //!     for details.
     53 //!
     54 //! 13. `Transaction::scan` uses `sql::engine::KeyPrefix::Table` to obtain the
     55 //!     key prefix for the scanned table, encoded via `encoding::keycode`. It
     56 //!     scans rows under this prefix by calling `MVCC::scan_prefix`, which in
     57 //!     turn dispatches to `BitCask::scan_prefix`. It returns a row iterator.
     58 //!
     59 //! 14. A row iterator is propagated back up through the stack:
     60 //!     `BitCask` → `MVCC` → `Local` → `State` → `Raft` → `scan` → `Plan::execute`
     61 //!
     62 //! 15. `Plan::execute` collects the results in a `ExecutionResult::Select`,
     63 //!     and returns it to `Session::execute`. It in turns returns it to
     64 //!     `Server::sql_session`, which encodes it and sends it across the wire
     65 //!     to `toySQL`, which displays them to the user.
     66 //!
     67 //! TODO: expand this into a "Life of a SQL statement" document.
     68

    103     /// Runs SQL goldenscripts.
    104     fn test_goldenscript(path: &std::path::Path) {
    105         // The runner's Session can't borrow from an Engine in the same struct,
    106         // so pass an engine reference. Use both BitCask and Memory engines and
    107         // mirror operations across them. Emit engine operations to op_rx.
    108         let (op_tx, op_rx) = crossbeam::channel::unbounded();
    109         let tempdir = tempfile::TempDir::with_prefix("toydb").expect("tempdir failed");
    110         let bitcask =
    111             storage::BitCask::new(tempdir.path().join("bitcask")).expect("bitcask failed");
    112         let memory = storage::Memory::new();
    113         let engine =
    114             Local::new(testengine::Emit::new(testengine::Mirror::new(bitcask, memory), op_tx));
    115         let mut runner = SQLRunner::new(&engine, op_rx);
    116 
    117         goldenscript::run(&mut runner, path).expect("goldenscript failed")
    118     }

    125     /// The SQL test runner.
    126     struct SQLRunner<'a> {
    127         engine: &'a TestEngine,
    128         sessions: HashMap<String, Session<'a, TestEngine>>,
    129         op_rx: Receiver<testengine::Operation>,
    130     }

    132     type TestEngine =
    133         Local<testengine::Emit<testengine::Mirror<storage::BitCask, storage::Memory>>>;

    135     impl<'a> SQLRunner<'a> {
    136         fn new(engine: &'a TestEngine, op_rx: Receiver<testengine::Operation>) -> Self {
    137             Self { engine, sessions: HashMap::new(), op_rx }
    138         }
    139     }

    141     impl<'a> goldenscript::Runner for SQLRunner<'a> {
    142         fn run(&mut self, command: &goldenscript::Command) -> Result<String, Box<dyn Error>> {
    143             let mut output = String::new();
    144 
    145             // Obtain a session based on the command prefix ("" if none).
    146             let prefix = command.prefix.clone().unwrap_or_default();
    147             let session = self.sessions.entry(prefix).or_insert_with(|| self.engine.session());
    148 
    149             // Handle runner commands.
    150             match command.name.as_str() {
    151                 // dump
    152                 "dump" => {
    153                     command.consume_args().reject_rest()?;
    154                     let mut engine = self.engine.mvcc.engine.lock().expect("mutex failed");
    155                     let mut iter = engine.scan(..);
    156                     while let Some((key, value)) = iter.next().transpose()? {
    157                         let fmtkv = format::MVCC::<format::SQL>::key_value(&key, &value);
    158                         let rawkv = format::Raw::key_value(&key, &value);
    159                         writeln!(output, "{fmtkv} [{rawkv}]",)?;
    160                     }
    161                     return Ok(output);
    162                 }
    163 
    164                 // schema [TABLE...]
    165                 "schema" => {
    166                     let mut args = command.consume_args();
    167                     let tables = args.rest_pos().iter().map(|arg| arg.value.clone()).collect_vec();
    168                     args.reject_rest()?;
    169 
    170                     let schemas = if tables.is_empty() {
    171                         session.with_txn(true, |txn| txn.list_tables())?
    172                     } else {
    173                         tables
    174                             .into_iter()
    175                             .map(|t| session.with_txn(true, |txn| txn.must_get_table(&t)))
    176                             .try_collect()?
    177                     };
    178                     return Ok(schemas.into_iter().join("\n"));
    179                 }
    180 
    181                 // Otherwise, fall through to SQL execution.
    182                 _ => {}
    183             }
    184 
    185             // The entire command is the SQL statement. There are no args.
    186             if !command.args.is_empty() {
    187                 return Err("SQL statements should be given as a command with no args".into());
    188             }
    189             let input = &command.name;
    190             let mut tags = command.tags.clone();
    191 
    192             // Output the plan if requested.
    193             if tags.remove("plan") {
    194                 let ast = Parser::new(input).parse()?;
    195                 let plan =
    196                     session.with_txn(true, |txn| Planner::new(txn).build(ast)?.optimize())?;
    197                 writeln!(output, "{plan}")?;
    198             }
    199 
    200             // Output plan optimizations if requested.
    201             if tags.remove("opt") {
    202                 if tags.contains("plan") {
    203                     return Err("using both plan and opt is redundant".into());
    204                 }
    205                 let ast = Parser::new(input).parse()?;
    206                 let plan = session.with_txn(true, |txn| Planner::new(txn).build(ast))?;
    207                 let Plan::Select(mut root) = plan else {
    208                     return Err("can only use opt with SELECT plans".into());
    209                 };
    210                 writeln!(output, "{}", format!("Initial:\n{root}").replace('\n', "\n   "))?;
    211                 for (name, optimizer) in OPTIMIZERS {
    212                     let prev = root.clone();
    213                     root = optimizer(root)?;
    214                     if root != prev {
    215                         writeln!(output, "{}", format!("{name}:\n{root}").replace('\n', "\n   "))?;
    216                     }
    217                 }
    218             }
    219 
    220             // Execute the statement.
    221             let result = session.execute(input)?;
    222 
    223             // Output engine ops if requested.
    224             if tags.remove("ops") {
    225                 while let Ok(op) = self.op_rx.try_recv() {
    226                     match op {
    227                         testengine::Operation::Delete { key } => {
    228                             let fmtkey = format::MVCC::<format::SQL>::key(&key);
    229                             let rawkey = format::Raw::key(&key);
    230                             writeln!(output, "delete {fmtkey} [{rawkey}]")?;
    231                         }
    232                         testengine::Operation::Flush => writeln!(output, "flush")?,
    233                         testengine::Operation::Set { key, value } => {
    234                             let fmtkv = format::MVCC::<format::SQL>::key_value(&key, &value);
    235                             let rawkv = format::Raw::key_value(&key, &value);
    236                             writeln!(output, "set {fmtkv} [{rawkv}]")?;
    237                         }
    238                     }
    239                 }
    240             }
    241 
    242             // Output the result if requested. SELECT results are always output.
    243             match result {
    244                 StatementResult::Select { columns, rows } => {
    245                     if tags.remove("header") {
    246                         writeln!(output, "{}", columns.into_iter().join(", "))?;
    247                     }
    248                     for row in rows {
    249                         writeln!(output, "{}", row.into_iter().join(", "))?;
    250                     }
    251                 }
    252                 result if tags.remove("result") => writeln!(output, "{result:?}")?,
    253                 _ => {}
    254             }
    255 
    256             // Reject unknown tags.
    257             if let Some(tag) = tags.iter().next() {
    258                 return Err(format!("unknown tag {tag}").into());
    259             }
    260 
    261             Ok(output)
    262         }
    263 
    264         /// Drain unprocessed operations after each command.
    265         fn end_command(&mut self, _: &goldenscript::Command) -> Result<String, Box<dyn Error>> {
    266             while self.op_rx.try_recv().is_ok() {}
    267             Ok(String::new())
    268         }
    269     }

    271     /// A test runner for expressions. Evaluates expressions to values, and
    272     /// optionally emits the expression tree.
    273     struct ExpressionRunner;
    274 
    275     type Catalog<'a> = <Local<storage::Memory> as Engine<'a>>::Transaction;
    276 
    277     impl goldenscript::Runner for ExpressionRunner {
    278         fn run(&mut self, command: &goldenscript::Command) -> Result<String, Box<dyn Error>> {
    279             let mut output = String::new();
    280 
    281             // The entire command is the expression to evaluate. There are no args.
    282             if !command.args.is_empty() {
    283                 return Err("expressions should be given as a command with no args".into());
    284             }
    285             let input = &command.name;
    286             let mut tags = command.tags.clone();
    287 
    288             // Parse and build the expression.
    289             let mut parser = Parser::new(input);
    290             let ast = parser.parse_expression()?;
    291             if let Some(next) = parser.lexer.next().transpose()? {
    292                 return Err(format!("unconsumed token {next}").into());
    293             }
    294             let expr = Planner::<Catalog>::build_expression(ast, &Scope::new())?;
    295 
    296             // Evaluate the expression.
    297             let value = expr.evaluate(None)?;
    298             write!(output, "{value}")?;
    299 
    300             // If requested, convert the expression to conjunctive normal form
    301             // and dump it. Assert that it produces the same result.
    302             if tags.remove("cnf") {
    303                 let cnf = expr.clone().into_cnf();
    304                 assert_eq!(value, cnf.evaluate(None)?, "CNF result differs");
    305                 write!(output, " ← {}", cnf.format_constant())?;
    306             }
    307 
    308             // If requested, debug-dump the parsed expression.
    309             if tags.remove("expr") {
    310                 write!(output, " ← {:?}", expr)?;
    311             }
    312             writeln!(output)?;
    313 
    314             // Reject unknown tags.
    315             if let Some(tag) = tags.iter().next() {
    316                 return Err(format!("unknown tag {tag}").into());
    317             }
    318 
    319             Ok(output)
    320         }
    321     }
