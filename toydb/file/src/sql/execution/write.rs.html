      9 /// Deletes rows, taking primary keys from the source (i.e. DELETE) using the
     10 /// primary_key column index. Returns the number of rows deleted.
     11 pub fn delete(
     12     txn: &impl Transaction,
     13     table: String,
     14     primary_key: usize,
     15     source: Rows,
     16 ) -> Result<u64> {
     17     let ids: Vec<Value> =
     18         source.map_ok(|row| row.into_iter().nth(primary_key).expect("short row")).try_collect()?;
     19     let count = ids.len() as u64;
     20     txn.delete(&table, &ids)?;
     21     Ok(count)
     22 }

     24 /// Inserts rows into a table (i.e. INSERT) from the given source.
     25 ///
     26 /// If given, column_map contains the mapping of table -> source columns for all
     27 /// columns in source. Otherwise, every column in source is the corresponding
     28 /// column in table, but the source may not have all columns in table (there may
     29 /// be a missing tail).
     30 pub fn insert(
     31     txn: &impl Transaction,
     32     table: Table,
     33     column_map: Option<HashMap<usize, usize>>,
     34     mut source: Rows,
     35 ) -> Result<u64> {
     36     let mut rows = Vec::new();
     37     while let Some(values) = source.next().transpose()? {
     38         // Fast path: the row is already complete, with no column mapping.
     39         if values.len() == table.columns.len() && column_map.is_none() {
     40             rows.push(values);
     41             continue;
     42         }
     43         if values.len() > table.columns.len() {
     44             return errinput!("too many values for table {}", table.name);
     45         }
     46         if let Some(column_map) = &column_map {
     47             if column_map.len() != values.len() {
     48                 return errinput!("column and value counts do not match");
     49             }
     50         }
     51 
     52         // Map source columns to table columns, and fill in default values.
     53         let mut row = Vec::with_capacity(table.columns.len());
     54         for (i, column) in table.columns.iter().enumerate() {
     55             if column_map.is_none() && i < values.len() {
     56                 // Pass through the source column to the table column.
     57                 row.push(values[i].clone())
     58             } else if let Some(vi) = column_map.as_ref().and_then(|c| c.get(&i)).copied() {
     59                 // Map the source column to the table column.
     60                 row.push(values[vi].clone())
     61             } else if let Some(default) = &column.default {
     62                 // Column not given in source, use the default.
     63                 row.push(default.clone())
     64             } else {
     65                 return errinput!("no value given for column {} with no default", column.name);
     66             }
     67         }
     68         rows.push(row);
     69     }
     70     let count = rows.len() as u64;
     71     txn.insert(&table.name, rows)?;
     72     Ok(count)
     73 }

     75 /// Updates rows passed in from the source (i.e. UPDATE). Returns the number of
     76 /// rows updated.
     77 pub fn update(
     78     txn: &impl Transaction,
     79     table: String,
     80     primary_key: usize,
     81     mut source: Rows,
     82     expressions: Vec<(usize, Expression)>,
     83 ) -> Result<u64> {
     84     let mut updates = BTreeMap::new();
     85     while let Some(row) = source.next().transpose()? {
     86         let mut update = row.clone();
     87         for (column, expr) in &expressions {
     88             update[*column] = expr.evaluate(Some(&row))?;
     89         }
     90         let id = row.into_iter().nth(primary_key).expect("short row");
     91         updates.insert(id, update);
     92     }
     93     let count = updates.len() as u64;
     94     txn.update(&table, updates)?;
     95     Ok(count)
     96 }
