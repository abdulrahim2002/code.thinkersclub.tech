      8 /// Aggregates row values from the source according to the aggregates, using the
      9 /// group_by expressions as buckets. Emits rows with group_by buckets then
     10 /// aggregates in the given order.
     11 pub fn aggregate(
     12     mut source: Rows,
     13     group_by: Vec<Expression>,
     14     aggregates: Vec<Aggregate>,
     15 ) -> Result<Rows> {
     16     let mut aggregator = Aggregator::new(group_by, aggregates);
     17     while let Some(row) = source.next().transpose()? {
     18         aggregator.add(row)?;
     19     }
     20     aggregator.into_rows()
     21 }

     23 /// Computes bucketed aggregates for rows.
     24 struct Aggregator {
     25     /// Bucketed accumulators (by group_by values).
     26     buckets: BTreeMap<Vec<Value>, Vec<Accumulator>>,
     27     /// The set of empty accumulators. Used to create new buckets.
     28     empty: Vec<Accumulator>,
     29     /// Group by expressions. Indexes map to bucket values.
     30     group_by: Vec<Expression>,
     31     /// Expressions to accumulate. Indexes map to accumulators.
     32     expressions: Vec<Expression>,
     33 }
     34 
     35 impl Aggregator {
     36     /// Creates a new aggregator for the given GROUP BY buckets and aggregates.
     37     fn new(group_by: Vec<Expression>, aggregates: Vec<Aggregate>) -> Self {
     38         use Aggregate::*;
     39         let accumulators = aggregates.iter().map(Accumulator::new).collect();
     40         let expressions = aggregates
     41             .into_iter()
     42             .map(|aggregate| match aggregate {
     43                 Average(expr) | Count(expr) | Max(expr) | Min(expr) | Sum(expr) => expr,
     44             })
     45             .collect();
     46         Self { buckets: BTreeMap::new(), empty: accumulators, group_by, expressions }
     47     }
     48 
     49     /// Adds a row to the aggregator.
     50     fn add(&mut self, row: Row) -> Result<()> {
     51         // Compute the bucket value.
     52         let bucket: Vec<Value> =
     53             self.group_by.iter().map(|expr| expr.evaluate(Some(&row))).try_collect()?;
     54 
     55         // Compute and accumulate the input values.
     56         let accumulators = self.buckets.entry(bucket).or_insert_with(|| self.empty.clone());
     57         for (accumulator, expr) in accumulators.iter_mut().zip(&self.expressions) {
     58             accumulator.add(expr.evaluate(Some(&row))?)?;
     59         }
     60         Ok(())
     61     }
     62 
     63     /// Returns a row iterator over the aggregate result.
     64     fn into_rows(self) -> Result<Rows> {
     65         // If there were no rows and no group_by expressions, return a row of
     66         // empty accumulators, e.g. SELECT COUNT(*) FROM t WHERE FALSE
     67         if self.buckets.is_empty() && self.group_by.is_empty() {
     68             let result = self.empty.into_iter().map(|acc| acc.value()).collect();
     69             return Ok(Box::new(std::iter::once(result)));
     70         }
     71 
     72         // Emit the group_by and aggregate values for each bucket. We use an
     73         // intermediate vec since btree_map::IntoIter doesn't implement Clone
     74         // (required by Rows).
     75         let buckets = self.buckets.into_iter().collect_vec();
     76         Ok(Box::new(buckets.into_iter().map(|(bucket, accumulators)| {
     77             bucket
     78                 .into_iter()
     79                 .map(Ok)
     80                 .chain(accumulators.into_iter().map(|acc| acc.value()))
     81                 .collect()
     82         })))
     83     }
     84 }

     86 /// Accumulates aggregate values. Uses an enum rather than a trait since we need
     87 /// to keep these in a vector (could use boxed trait objects too).
     88 #[derive(Clone)]
     89 enum Accumulator {
     90     Average { count: i64, sum: Value },
     91     Count(i64),
     92     Max(Option<Value>),
     93     Min(Option<Value>),
     94     Sum(Option<Value>),
     95 }
     96 
     97 impl Accumulator {
     98     /// Creates a new accumulator from an aggregate kind.
     99     fn new(aggregate: &Aggregate) -> Self {
    100         match aggregate {
    101             Aggregate::Average(_) => Self::Average { count: 0, sum: Value::Integer(0) },
    102             Aggregate::Count(_) => Self::Count(0),
    103             Aggregate::Max(_) => Self::Max(None),
    104             Aggregate::Min(_) => Self::Min(None),
    105             Aggregate::Sum(_) => Self::Sum(None),
    106         }
    107     }
    108 
    109     /// Adds a value to the accumulator.
    110     fn add(&mut self, value: Value) -> Result<()> {
    111         // Aggregates ignore NULL values.
    112         if value == Value::Null {
    113             return Ok(());
    114         }
    115         match self {
    116             Self::Average { sum, count } => (*sum, *count) = (sum.checked_add(&value)?, *count + 1),
    117             Self::Count(count) => *count += 1,
    118             Self::Max(max @ None) => *max = Some(value),
    119             Self::Max(Some(max)) if value > *max => *max = value,
    120             Self::Max(Some(_)) => {}
    121             Self::Min(min @ None) => *min = Some(value),
    122             Self::Min(Some(min)) if value < *min => *min = value,
    123             Self::Min(Some(_)) => {}
    124             Self::Sum(sum @ None) => *sum = Some(Value::Integer(0).checked_add(&value)?),
    125             Self::Sum(Some(sum)) => *sum = sum.checked_add(&value)?,
    126         }
    127         Ok(())
    128     }
    129 
    130     /// Returns the aggregate value.
    131     fn value(self) -> Result<Value> {
    132         Ok(match self {
    133             Self::Average { count: 0, sum: _ } => Value::Null,
    134             Self::Average { count, sum } => sum.checked_div(&Value::Integer(count))?,
    135             Self::Count(count) => count.into(),
    136             Self::Max(Some(value)) | Self::Min(Some(value)) | Self::Sum(Some(value)) => value,
    137             Self::Max(None) | Self::Min(None) | Self::Sum(None) => Value::Null,
    138         })
    139     }
    140 }
