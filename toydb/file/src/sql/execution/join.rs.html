      9 /// A nested loop join. Iterates over the right source for every row in the left
     10 /// source, optionally filtering on the join predicate. If outer is true, and
     11 /// there are no matches in the right source for a row in the left source, a
     12 /// joined row with NULL values for the right source is returned (typically used
     13 /// for a LEFT JOIN).
     14 pub fn nested_loop(
     15     left: Rows,
     16     right: Rows,
     17     right_size: usize,
     18     predicate: Option<Expression>,
     19     outer: bool,
     20 ) -> Result<Rows> {
     21     Ok(Box::new(NestedLoopIterator::new(left, right, right_size, predicate, outer)?))
     22 }

     24 /// NestedLoopIterator implements nested loop joins.
     25 ///
     26 /// This could be trivially implemented with carthesian_product(), but we need
     27 /// to handle the left outer join case where there is no match in the right
     28 /// source.
     29 #[derive(Clone)]
     30 struct NestedLoopIterator {
     31     /// The left source.
     32     left: Peekable<Rows>,
     33     /// The right source.
     34     right: Rows,
     35     /// The initial right iterator state. Cloned to reset right.
     36     right_init: Rows,
     37     /// The column width of the right source.
     38     right_size: usize,
     39     /// True if a right match has been seen for the current left row.
     40     right_match: bool,
     41     /// The join predicate.
     42     predicate: Option<Expression>,
     43     /// If true, emit a row when there is no match in the right source.
     44     outer: bool,
     45 }
     46 
     47 impl NestedLoopIterator {
     48     fn new(
     49         left: Rows,
     50         right: Rows,
     51         right_size: usize,
     52         predicate: Option<Expression>,
     53         outer: bool,
     54     ) -> Result<Self> {
     55         let left = left.peekable();
     56         let right_init = right.clone();
     57         Ok(Self { left, right, right_init, right_size, right_match: false, predicate, outer })
     58     }
     59 
     60     // Returns the next joined row, if any.
     61     fn try_next(&mut self) -> Result<Option<Row>> {
     62         // While there is a valid left row, look for a right-hand match to return.
     63         while let Some(Ok(left_row)) = self.left.peek() {
     64             // If there is a match in the remaining right rows, return it.
     65             while let Some(right_row) = self.right.next().transpose()? {
     66                 // We could avoid cloning here unless we're actually emitting
     67                 // the row, but we keep it simple.
     68                 let row = left_row.iter().cloned().chain(right_row).collect();
     69                 let is_match = match &self.predicate {
     70                     Some(predicate) => match predicate.evaluate(Some(&row))? {
     71                         Value::Boolean(true) => true,
     72                         Value::Boolean(false) | Value::Null => false,
     73                         v => return errinput!("join predicate returned {v}, expected boolean"),
     74                     },
     75                     None => true,
     76                 };
     77                 if is_match {
     78                     self.right_match = true;
     79                     return Ok(Some(row));
     80                 }
     81             }
     82 
     83             // We reached the end of the right source, reset it.
     84             self.right = self.right_init.clone();
     85 
     86             // If there was no match for this row, and this is an outer join,
     87             // emit a row with right NULLs.
     88             if !self.right_match && self.outer {
     89                 let row = left_row
     90                     .iter()
     91                     .cloned()
     92                     .chain(std::iter::repeat(Value::Null).take(self.right_size))
     93                     .collect();
     94                 self.left.next();
     95                 return Ok(Some(row));
     96             }
     97 
     98             // Move onto the next left row.
     99             self.left.next();
    100             self.right_match = false;
    101         }
    102 
    103         // Otherwise, there's either a None or Err in left. Return it.
    104         self.left.next().transpose()
    105     }
    106 }

    108 impl Iterator for NestedLoopIterator {
    109     type Item = Result<Row>;
    110 
    111     fn next(&mut self) -> Option<Self::Item> {
    112         self.try_next().transpose()
    113     }
    114 }

    116 /// Executes a hash join. This builds a hash table of rows from the right source
    117 /// keyed on the join value, then iterates over the left source and looks up
    118 /// matching rows in the hash table. If outer is true, and there is no match
    119 /// in the right source for a row in the left source, a row with NULL values
    120 /// for the right source is emitted instead.
    121 pub fn hash(
    122     left: Rows,
    123     left_column: usize,
    124     right: Rows,
    125     right_column: usize,
    126     right_size: usize,
    127     outer: bool,
    128 ) -> Result<Rows> {
    129     // Build the hash table from the right source.
    130     let mut rows = right;
    131     let mut right: HashMap<Value, Vec<Row>> = HashMap::new();
    132     while let Some(row) = rows.next().transpose()? {
    133         let value = row[right_column].clone();
    134         if value.is_undefined() {
    135             continue; // NULL and NAN equality is always false
    136         }
    137         right.entry(value).or_default().push(row);
    138     }
    139 
    140     // Set up an iterator for an empty right row in the outer case.
    141     let empty = std::iter::repeat(Value::Null).take(right_size);
    142 
    143     // Set up the join iterator.
    144     let join = left.flat_map(move |result| -> Rows {
    145         // Pass through errors.
    146         let Ok(row) = result else {
    147             return Box::new(std::iter::once(result));
    148         };
    149         // Join the left row with any matching right rows.
    150         match right.get(&row[left_column]) {
    151             Some(matches) => Box::new(
    152                 std::iter::once(row)
    153                     .cartesian_product(matches.clone())
    154                     .map(|(l, r)| l.into_iter().chain(r).collect())
    155                     .map(Ok),
    156             ),
    157             None if outer => {
    158                 Box::new(std::iter::once(Ok(row.into_iter().chain(empty.clone()).collect())))
    159             }
    160             None => Box::new(std::iter::empty()),
    161         }
    162     });
    163     Ok(Box::new(join))
    164 }
