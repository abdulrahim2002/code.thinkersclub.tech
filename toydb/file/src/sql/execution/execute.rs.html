      7 /// Executes a plan, returning an execution result.
      8 ///
      9 /// Takes the transaction and catalog separately, even though Transaction must
     10 /// implement Catalog, to ensure the catalog is primarily used during planning.
     11 pub fn execute_plan(
     12     plan: Plan,
     13     txn: &impl Transaction,
     14     catalog: &impl Catalog,
     15 ) -> Result<ExecutionResult> {
     16     Ok(match plan {
     17         Plan::CreateTable { schema } => {
     18             let name = schema.name.clone();
     19             catalog.create_table(schema)?;
     20             ExecutionResult::CreateTable { name }
     21         }
     22 
     23         Plan::DropTable { table, if_exists } => {
     24             let existed = catalog.drop_table(&table, if_exists)?;
     25             ExecutionResult::DropTable { name: table, existed }
     26         }
     27 
     28         Plan::Delete { table, primary_key, source } => {
     29             let source = execute(source, txn)?;
     30             let count = write::delete(txn, table, primary_key, source)?;
     31             ExecutionResult::Delete { count }
     32         }
     33 
     34         Plan::Insert { table, column_map, source } => {
     35             let source = execute(source, txn)?;
     36             let count = write::insert(txn, table, column_map, source)?;
     37             ExecutionResult::Insert { count }
     38         }
     39 
     40         Plan::Select(root) => {
     41             let columns = (0..root.columns()).map(|i| root.column_label(i)).collect();
     42             let rows = execute(root, txn)?;
     43             ExecutionResult::Select { rows, columns }
     44         }
     45 
     46         Plan::Update { table, primary_key, source, expressions } => {
     47             let source = execute(source, txn)?;
     48             let count = write::update(txn, table.name, primary_key, source, expressions)?;
     49             ExecutionResult::Update { count }
     50         }
     51     })
     52 }

     54 /// Recursively executes a query plan node, returning a row iterator.
     55 ///
     56 /// Rows stream through the plan node tree from the branches to the root. Nodes
     57 /// recursively pull input rows upwards from their child node(s), process them,
     58 /// and hand the resulting rows off to their parent node.
     59 ///
     60 /// Below is an example of an (unoptimized) query plan:
     61 ///
     62 /// SELECT title, released, genres.name AS genre
     63 /// FROM movies INNER JOIN genres ON movies.genre_id = genres.id
     64 /// WHERE released >= 2000 ORDER BY released
     65 ///
     66 /// Order: movies.released desc
     67 /// └─ Projection: movies.title, movies.released, genres.name as genre
     68 ///    └─ Filter: movies.released >= 2000
     69 ///       └─ NestedLoopJoin: inner on movies.genre_id = genres.id
     70 ///          ├─ Scan: movies
     71 ///          └─ Scan: genres
     72 ///
     73 /// Rows flow from the tree leaves to the root. The Scan nodes read and emit
     74 /// table rows from storage. They are passed to the NestedLoopJoin node which
     75 /// joins the rows from the two tables, then the Filter node discards old
     76 /// movies, the Projection node picks out the requested columns, and the Order
     77 /// node sorts them before emitting the rows to the client.
     78 pub fn execute(node: Node, txn: &impl Transaction) -> Result<Rows> {
     79     Ok(match node {
     80         Node::Aggregate { source, group_by, aggregates } => {
     81             let source = execute(*source, txn)?;
     82             aggregate::aggregate(source, group_by, aggregates)?
     83         }
     84 
     85         Node::Filter { source, predicate } => {
     86             let source = execute(*source, txn)?;
     87             transform::filter(source, predicate)
     88         }
     89 
     90         Node::HashJoin { left, left_column, right, right_column, outer } => {
     91             let right_size = right.columns();
     92             let left = execute(*left, txn)?;
     93             let right = execute(*right, txn)?;
     94             join::hash(left, left_column, right, right_column, right_size, outer)?
     95         }
     96 
     97         Node::IndexLookup { table, column, values, alias: _ } => {
     98             let column = table.columns.into_iter().nth(column).expect("invalid column").name;
     99             let table = table.name;
    100             source::lookup_index(txn, table, column, values)?
    101         }
    102 
    103         Node::KeyLookup { table, keys, alias: _ } => source::lookup_key(txn, table.name, keys)?,
    104 
    105         Node::Limit { source, limit } => {
    106             let source = execute(*source, txn)?;
    107             transform::limit(source, limit)
    108         }
    109 
    110         Node::NestedLoopJoin { left, right, predicate, outer } => {
    111             let right_size = right.columns();
    112             let left = execute(*left, txn)?;
    113             let right = execute(*right, txn)?;
    114             join::nested_loop(left, right, right_size, predicate, outer)?
    115         }
    116 
    117         Node::Nothing { .. } => source::nothing(),
    118 
    119         Node::Offset { source, offset } => {
    120             let source = execute(*source, txn)?;
    121             transform::offset(source, offset)
    122         }
    123 
    124         Node::Order { source, key: orders } => {
    125             let source = execute(*source, txn)?;
    126             transform::order(source, orders)?
    127         }
    128 
    129         Node::Projection { source, expressions, aliases: _ } => {
    130             let source = execute(*source, txn)?;
    131             transform::project(source, expressions)
    132         }
    133 
    134         Node::Remap { source, targets } => {
    135             let source = execute(*source, txn)?;
    136             transform::remap(source, targets)
    137         }
    138 
    139         Node::Scan { table, filter, alias: _ } => source::scan(txn, table, filter)?,
    140 
    141         Node::Values { rows } => source::values(rows),
    142     })
    143 }
