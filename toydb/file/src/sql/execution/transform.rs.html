      8 /// Filters the input rows (i.e. WHERE).
      9 pub fn filter(source: Rows, predicate: Expression) -> Rows {
     10     Box::new(source.filter_map(move |r| {
     11         r.and_then(|row| match predicate.evaluate(Some(&row))? {
     12             Value::Boolean(true) => Ok(Some(row)),
     13             Value::Boolean(false) | Value::Null => Ok(None),
     14             value => errinput!("filter returned {value}, expected boolean",),
     15         })
     16         .transpose()
     17     }))
     18 }

     30 /// Sorts the rows (i.e. ORDER BY).
     31 pub fn order(source: Rows, order: Vec<(Expression, Direction)>) -> Result<Rows> {
     32     // We can't use sort_by_cached_key(), since expression evaluation is
     33     // fallible, and since we may have to vary the sort direction of each
     34     // expression. Precompute the sort values instead, and map them based on
     35     // the row index.
     36     let mut irows: Vec<_> = source.enumerate().map(|(i, r)| r.map(|row| (i, row))).try_collect()?;
     37     let mut sort_values = Vec::with_capacity(irows.len());
     38     for (_, row) in &irows {
     39         let values: Vec<_> = order.iter().map(|(e, _)| e.evaluate(Some(row))).try_collect()?;
     40         sort_values.push(values)
     41     }
     42 
     43     irows.sort_by(|&(a, _), &(b, _)| {
     44         let dirs = order.iter().map(|(_, dir)| dir);
     45         for (a, b, dir) in izip!(&sort_values[a], &sort_values[b], dirs) {
     46             match a.cmp(b) {
     47                 std::cmp::Ordering::Equal => {}
     48                 order if *dir == Direction::Descending => return order.reverse(),
     49                 order => return order,
     50             }
     51         }
     52         std::cmp::Ordering::Equal
     53     });
     54 
     55     Ok(Box::new(irows.into_iter().map(|(_, row)| Ok(row))))
     56 }

     58 /// Projects the rows using the given expressions (i.e. SELECT).
     59 pub fn project(source: Rows, expressions: Vec<Expression>) -> Rows {
     60     Box::new(source.map(move |result| {
     61         result.and_then(|row| expressions.iter().map(|e| e.evaluate(Some(&row))).collect())
     62     }))
     63 }

     65 /// Remaps source columns to target column indexes, or drops them if None.
     66 pub fn remap(source: Rows, targets: Vec<Option<usize>>) -> Rows {
     67     let size = targets.iter().filter_map(|v| *v).map(|i| i + 1).max().unwrap_or(0);
     68     Box::new(source.map_ok(move |row| {
     69         let mut out = vec![Value::Null; size];
     70         for (value, target) in row.into_iter().zip(&targets) {
     71             if let Some(index) = target {
     72                 out[*index] = value;
     73             }
     74         }
     75         out
     76     }))
