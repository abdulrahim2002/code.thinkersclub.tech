     25     /// Builds a plan for an AST statement.
     26     pub fn build(&mut self, statement: ast::Statement) -> Result<Plan> {
     27         use ast::Statement::*;
     28         match statement {
     29             CreateTable { name, columns } => self.build_create_table(name, columns),
     30             DropTable { name, if_exists } => Ok(Plan::DropTable { table: name, if_exists }),
     31             Delete { table, r#where } => self.build_delete(table, r#where),
     32             Insert { table, columns, values } => self.build_insert(table, columns, values),
     33             Update { table, set, r#where } => self.build_update(table, set, r#where),
     34             Select { select, from, r#where, group_by, having, order_by, offset, limit } => {
     35                 self.build_select(select, from, r#where, group_by, having, order_by, offset, limit)
     36             }
     37 
     38             // Transaction and explain statements are handled by Session.
     39             Begin { .. } | Commit | Rollback | Explain(_) => {
     40                 panic!("unexpected statement {statement:?}")
     41             }
     42         }
     43     }

     45     /// Builds a CREATE TABLE plan.
     46     fn build_create_table(&self, name: String, columns: Vec<ast::Column>) -> Result<Plan> {
     47         // Most schema validation happens during execution via Table.validate().
     48         // However, the AST specifies the primary key as a column field, while
     49         // the schema stores it as a column index, so we have to map that here.
     50         let Some(primary_key) = columns.iter().position(|c| c.primary_key) else {
     51             return errinput!("no primary key for table {name}");
     52         };
     53         if columns.iter().filter(|c| c.primary_key).count() > 1 {
     54             return errinput!("multiple primary keys for table {name}");
     55         }
     56         let columns = columns
     57             .into_iter()
     58             .map(|c| {
     59                 let nullable = c.nullable.unwrap_or(!c.primary_key);
     60                 Ok(Column {
     61                     name: c.name,
     62                     datatype: c.datatype,
     63                     nullable,
     64                     default: match c.default {
     65                         Some(expr) => Some(Self::evaluate_constant(expr)?),
     66                         None if nullable => Some(Value::Null),
     67                         None => None,
     68                     },
     69                     unique: c.unique || c.primary_key,
     70                     index: (c.index || c.unique || c.references.is_some()) && !c.primary_key,
     71                     references: c.references,
     72                 })
     73             })
     74             .collect::<Result<_>>()?;
     75         Ok(Plan::CreateTable { schema: Table { name, primary_key, columns } })
     76     }

     78     /// Builds a DELETE plan.
     79     fn build_delete(&self, table: String, r#where: Option<ast::Expression>) -> Result<Plan> {
     80         let table = self.catalog.must_get_table(&table)?;
     81         let scope = Scope::from_table(&table)?;
     82         let filter = r#where.map(|expr| Self::build_expression(expr, &scope)).transpose()?;
     83         Ok(Plan::Delete {
     84             table: table.name.clone(),
     85             primary_key: table.primary_key,
     86             source: Node::Scan { table, alias: None, filter },
     87         })
     88     }

     90     /// Builds an INSERT plan.
     91     fn build_insert(
     92         &self,
     93         table: String,
     94         columns: Option<Vec<String>>,
     95         values: Vec<Vec<ast::Expression>>,
     96     ) -> Result<Plan> {
     97         let table = self.catalog.must_get_table(&table)?;
     98         let mut column_map = None;
     99         if let Some(columns) = columns {
    100             let column_map = column_map.insert(HashMap::new());
    101             for (vidx, name) in columns.into_iter().enumerate() {
    102                 let Some(cidx) = table.columns.iter().position(|c| c.name == name) else {
    103                     return errinput!("unknown column {name} in table {}", table.name);
    104                 };
    105                 if column_map.insert(cidx, vidx).is_some() {
    106                     return errinput!("column {name} given multiple times");
    107                 }
    108             }
    109         }
    110         let scope = Scope::new();
    111         let rows = values
    112             .into_iter()
    113             .map(|exprs| {
    114                 exprs.into_iter().map(|expr| Self::build_expression(expr, &scope)).collect()
    115             })
    116             .try_collect()?;
    117         Ok(Plan::Insert { table, column_map, source: Node::Values { rows } })
    118     }

    120     /// Builds an UPDATE plan.
    121     fn build_update(
    122         &self,
    123         table: String,
    124         set: BTreeMap<String, Option<ast::Expression>>,
    125         r#where: Option<ast::Expression>,
    126     ) -> Result<Plan> {
    127         let table = self.catalog.must_get_table(&table)?;
    128         let scope = Scope::from_table(&table)?;
    129         let filter = r#where.map(|expr| Self::build_expression(expr, &scope)).transpose()?;
    130         let mut expressions = Vec::with_capacity(set.len());
    131         for (column, expr) in set {
    132             let index = scope.lookup_column(None, &column)?;
    133             let expr = match expr {
    134                 Some(expr) => Self::build_expression(expr, &scope)?,
    135                 None => match &table.columns[index].default {
    136                     Some(default) => Expression::Constant(default.clone()),
    137                     None => return errinput!("column {column} has no default value"),
    138                 },
    139             };
    140             expressions.push((index, expr));
    141         }
    142         Ok(Plan::Update {
    143             table: table.clone(),
    144             primary_key: table.primary_key,
    145             source: Node::Scan { table, alias: None, filter },
    146             expressions,
    147         })
    148     }

    150     /// Builds a SELECT plan.
    151     #[allow(clippy::too_many_arguments)]
    152     fn build_select(
    153         &self,
    154         mut select: Vec<(ast::Expression, Option<String>)>,
    155         from: Vec<ast::From>,
    156         r#where: Option<ast::Expression>,
    157         group_by: Vec<ast::Expression>,
    158         having: Option<ast::Expression>,
    159         order_by: Vec<(ast::Expression, ast::Direction)>,
    160         offset: Option<ast::Expression>,
    161         limit: Option<ast::Expression>,
    162     ) -> Result<Plan> {
    163         let mut scope = Scope::new();
    164 
    165         // Build FROM clause.
    166         let mut node = if !from.is_empty() {
    167             self.build_from_clause(from, &mut scope)?
    168         } else {
    169             // For a constant SELECT, emit a single empty row to project with.
    170             // This allows using aggregate functions and WHERE as normal.
    171             Node::Values { rows: vec![vec![]] }
    172         };
    173 
    174         // Expand out SELECT * to all FROM columns if there are multiple SELECT
    175         // expressions or a GROUP BY clause (to ensure all columns are in GROUP
    176         // BY). For simplicity, expressions only supports scalar values, so we
    177         // special-case the * tuple here.
    178         if select.contains(&(ast::Expression::All, None)) {
    179             if node.columns() == 0 {
    180                 return errinput!("SELECT * requires a FROM clause");
    181             }
    182             if select.len() > 1 || !group_by.is_empty() {
    183                 select = select
    184                     .into_iter()
    185                     .flat_map(|(expr, alias)| match expr {
    186                         ast::Expression::All => itertools::Either::Left(
    187                             (0..node.columns()).map(|i| (node.column_label(i).into(), None)),
    188                         ),
    189                         expr => itertools::Either::Right(std::iter::once((expr, alias))),
    190                     })
    191                     .collect();
    192             }
    193         }
    194 
    195         // Build WHERE clause.
    196         if let Some(r#where) = r#where {
    197             let predicate = Self::build_expression(r#where, &scope)?;
    198             node = Node::Filter { source: Box::new(node), predicate };
    199         }
    200 
    201         // Build aggregate functions and GROUP BY clause.
    202         let aggregates = Self::collect_aggregates(&select, &having, &order_by);
    203         if !group_by.is_empty() || !aggregates.is_empty() {
    204             node = self.build_aggregate(node, group_by, aggregates, &mut scope)?;
    205         }
    206 
    207         // Build SELECT clause. We can omit this for a trivial SELECT *.
    208         if select.as_slice() != [(ast::Expression::All, None)] {
    209             // Prepare the post-projection scope.
    210             let mut child_scope = scope.project(&select);
    211 
    212             // Build the SELECT column expressions and aliases.
    213             let mut expressions = Vec::with_capacity(select.len());
    214             let mut aliases = Vec::with_capacity(select.len());
    215             for (expr, alias) in select {
    216                 expressions.push(Self::build_expression(expr, &scope)?);
    217                 aliases.push(Label::from(alias));
    218             }
    219 
    220             // Add hidden columns for HAVING and ORDER BY columns not in SELECT.
    221             let hidden = self.build_select_hidden(&having, &order_by, &scope, &mut child_scope);
    222             aliases.extend(std::iter::repeat(Label::None).take(hidden.len()));
    223             expressions.extend(hidden);
    224 
    225             scope = child_scope;
    226             node = Node::Projection { source: Box::new(node), expressions, aliases };
    227         }
    228 
    229         // Build HAVING clause.
    230         if let Some(having) = having {
    231             if scope.aggregates.is_empty() {
    232                 return errinput!("HAVING requires GROUP BY or aggregate function");
    233             }
    234             let predicate = Self::build_expression(having, &scope)?;
    235             node = Node::Filter { source: Box::new(node), predicate };
    236         }
    237 
    238         // Build ORDER BY clause.
    239         if !order_by.is_empty() {
    240             let key = order_by
    241                 .into_iter()
    242                 .map(|(expr, dir)| Ok((Self::build_expression(expr, &scope)?, dir.into())))
    243                 .collect::<Result<_>>()?;
    244             node = Node::Order { source: Box::new(node), key };
    245         }
    246 
    247         // Build OFFSET clause.
    248         if let Some(offset) = offset {
    249             let offset = match Self::evaluate_constant(offset)? {
    250                 Value::Integer(offset) if offset >= 0 => offset as usize,
    251                 offset => return errinput!("invalid offset {offset}"),
    252             };
    253             node = Node::Offset { source: Box::new(node), offset }
    254         }
    255 
    256         // Build LIMIT clause.
    257         if let Some(limit) = limit {
    258             let limit = match Self::evaluate_constant(limit)? {
    259                 Value::Integer(limit) if limit >= 0 => limit as usize,
    260                 limit => return errinput!("invalid limit {limit}"),
    261             };
    262             node = Node::Limit { source: Box::new(node), limit }
    263         }
    264 
    265         // Remove any hidden columns before emitting the result.
    266         if let Some(targets) = scope.remap_hidden() {
    267             node = Node::Remap { source: Box::new(node), targets }
    268         }
    269 
    270         Ok(Plan::Select(node))
    271     }

    273     /// Builds a FROM clause consisting of one or more items. Each item is
    274     /// either a table or a join of two or more tables. All items are implicitly
    275     /// joined, e.g. "SELECT * FROM a, b" is an implicit full join of a and b.
    276     fn build_from_clause(&self, from: Vec<ast::From>, scope: &mut Scope) -> Result<Node> {
    277         // Build the first FROM item. A FROM clause must have at least one.
    278         let mut items = from.into_iter();
    279         let mut node = match items.next() {
    280             Some(from) => self.build_from(from, scope)?,
    281             None => return errinput!("no from items given"),
    282         };
    283 
    284         // Build and implicitly join additional items.
    285         for from in items {
    286             let right = self.build_from(from, scope)?;
    287             node = Node::NestedLoopJoin {
    288                 left: Box::new(node),
    289                 right: Box::new(right),
    290                 predicate: None,
    291                 outer: false,
    292             };
    293         }
    294         Ok(node)
    295     }

    297     /// Builds FROM items, which can either be a single table or a chained join
    298     /// of multiple tables, e.g. "SELECT * FROM a LEFT JOIN b ON b.a_id = a.id".
    299     fn build_from(&self, from: ast::From, parent_scope: &mut Scope) -> Result<Node> {
    300         // Each from item is built in its own scope, such that a join node only
    301         // sees the columns of its children. It's then merged into the parent.
    302         let mut scope = Scope::new();
    303 
    304         let node = match from {
    305             // A full table scan.
    306             ast::From::Table { name, alias } => {
    307                 let table = self.catalog.must_get_table(&name)?;
    308                 scope.add_table(&table, alias.as_deref())?;
    309                 Node::Scan { table, alias, filter: None }
    310             }
    311 
    312             // A two-way join. The left or right nodes may be chained joins.
    313             ast::From::Join { mut left, mut right, r#type, predicate } => {
    314                 // Right joins are built as a left join then column swap.
    315                 if r#type == ast::JoinType::Right {
    316                     (left, right) = (right, left)
    317                 }
    318 
    319                 // Build the left and right nodes.
    320                 let left = Box::new(self.build_from(*left, &mut scope)?);
    321                 let right = Box::new(self.build_from(*right, &mut scope)?);
    322                 let (left_size, right_size) = (left.columns(), right.columns());
    323 
    324                 // Build the join node.
    325                 let predicate = predicate.map(|e| Self::build_expression(e, &scope)).transpose()?;
    326                 let outer = r#type.is_outer();
    327                 let mut node = Node::NestedLoopJoin { left, right, predicate, outer };
    328 
    329                 // For right joins, swap the columns.
    330                 if r#type == ast::JoinType::Right {
    331                     let size = left_size + right_size;
    332                     let targets = (0..size).map(|i| Some((i + right_size) % size)).collect_vec();
    333                     scope = scope.remap(&targets);
    334                     node = Node::Remap { source: Box::new(node), targets }
    335                 }
    336                 node
    337             }
    338         };
    339 
    340         parent_scope.merge(scope)?;
    341         Ok(node)
    342     }

    344     /// Builds an aggregate node, which computes aggregates for a set of GROUP
    345     /// BY buckets. The aggregate functions have been collected from the SELECT,
    346     /// HAVING, and ORDER BY clauses.
    347     ///
    348     /// The ast::Expression for each aggregate function and GROUP BY expression
    349     /// is tracked in the Scope and mapped to the column index. Later nodes
    350     /// (i.e. SELECT, HAVING, and ORDER BY) can look up the column index of
    351     /// aggregate expressions while building expressions. Consider e.g.:
    352     ///
    353     /// SELECT SUM(a) / COUNT(*) FROM t GROUP BY b % 10 HAVING b % 10 >= 5 ORDER BY MAX(c)
    354     ///
    355     /// This will build an Aggregate node for SUM(a), COUNT(*), MAX(c) bucketed
    356     /// by b % 10. The SELECT can look up up SUM(a) and COUNT(*) to compute the
    357     /// division, and HAVING can look up b % 10 to compute the predicate.
    358     fn build_aggregate(
    359         &self,
    360         source: Node,
    361         mut group_by: Vec<ast::Expression>,
    362         mut aggregates: Vec<ast::Expression>,
    363         scope: &mut Scope,
    364     ) -> Result<Node> {
    365         // Construct a child scope with the group_by and aggregate AST
    366         // expressions, for lookups. Discard duplicate expressions.
    367         let mut child_scope = scope.spawn();
    368         group_by.retain(|expr| child_scope.add_aggregate(expr, scope).is_some());
    369         aggregates.retain(|expr| child_scope.add_aggregate(expr, scope).is_some());
    370 
    371         // Build the node from the remaining unique expressions.
    372         let group_by =
    373             group_by.into_iter().map(|expr| Self::build_expression(expr, scope)).try_collect()?;
    374         let aggregates = aggregates
    375             .into_iter()
    376             .map(|expr| Self::build_aggregate_function(expr, scope))
    377             .try_collect()?;
    378 
    379         *scope = child_scope;
    380         Ok(Node::Aggregate { source: Box::new(source), group_by, aggregates })
    381     }

    383     /// Builds an aggregate function from an AST expression.
    384     fn build_aggregate_function(expr: ast::Expression, scope: &Scope) -> Result<Aggregate> {
    385         let ast::Expression::Function(name, mut args) = expr else {
    386             panic!("aggregate expression must be function");
    387         };
    388         if args.len() != 1 {
    389             return errinput!("{name} takes 1 argument");
    390         }
    391         if args[0].contains(&|expr| Self::is_aggregate_function(expr)) {
    392             return errinput!("aggregate functions can't be nested");
    393         }
    394         // Special-case COUNT(*) since expressions don't support tuples.
    395         let expr = match (name.as_str(), args.remove(0)) {
    396             ("count", ast::Expression::All) => Expression::Constant(Value::Boolean(true)),
    397             (_, arg) => Self::build_expression(arg, scope)?,
    398         };
    399         Ok(match name.as_str() {
    400             "avg" => Aggregate::Average(expr),
    401             "count" => Aggregate::Count(expr),
    402             "min" => Aggregate::Min(expr),
    403             "max" => Aggregate::Max(expr),
    404             "sum" => Aggregate::Sum(expr),
    405             name => return errinput!("unknown aggregate function {name}"),
    406         })
    407     }

    409     /// Checks whether a given AST expression is an aggregate function.
    410     fn is_aggregate_function(expr: &ast::Expression) -> bool {
    411         if let ast::Expression::Function(name, _) = expr {
    412             return ["avg", "count", "max", "min", "sum"].contains(&name.as_str());
    413         }
    414         false
    415     }

    417     /// Collects aggregate functions from SELECT, HAVING, and ORDER BY clauses.
    418     fn collect_aggregates(
    419         select: &[(ast::Expression, Option<String>)],
    420         having: &Option<ast::Expression>,
    421         order_by: &[(ast::Expression, ast::Direction)],
    422     ) -> Vec<ast::Expression> {
    423         let select = select.iter().map(|(expr, _)| expr);
    424         let having = having.iter();
    425         let order_by = order_by.iter().map(|(expr, _)| expr);
    426         let mut aggregates = Vec::new();
    427         for expr in select.chain(having).chain(order_by) {
    428             expr.collect(&|expr| Self::is_aggregate_function(expr), &mut aggregates)
    429         }
    430         aggregates
    431     }

    433     /// Builds hidden columns for a projection to pass through columns that are
    434     /// used by downstream nodes. Consider e.g.:
    435     ///
    436     /// SELECT id FROM table ORDER BY value
    437     ///
    438     /// The ORDER BY node is evaluated after the SELECT projection (it may need
    439     /// to order on projected columns), but "value" isn't projected and thus
    440     /// isn't available to the ORDER BY node. We add a hidden "value" column to
    441     /// the projection to satisfy the ORDER BY.
    442     ///
    443     /// Hidden columns are tracked in the scope and stripped before the result
    444     /// is returned to the client.
    445     fn build_select_hidden(
    446         &self,
    447         having: &Option<ast::Expression>,
    448         order_by: &[(ast::Expression, ast::Direction)],
    449         scope: &Scope,
    450         child_scope: &mut Scope,
    451     ) -> Vec<Expression> {
    452         let mut hidden = Vec::new();
    453         for expr in having.iter().chain(order_by.iter().map(|(expr, _)| expr)) {
    454             expr.walk(&mut |expr| {
    455                 // If this is an aggregate or GROUP BY expression that isn't
    456                 // already available in the child scope, add a hidden column.
    457                 if let Some(index) = scope.lookup_aggregate(expr) {
    458                     if child_scope.lookup_aggregate(expr).is_none() {
    459                         child_scope.add_passthrough(scope, index, true);
    460                         hidden.push(Expression::Column(index));
    461                         return true;
    462                     }
    463                 }
    464 
    465                 // Look for column references that don't exist post-projection,
    466                 // but that do exist in the parent, and add hidden columns.
    467                 let ast::Expression::Column(table, column) = expr else {
    468                     return true;
    469                 };
    470                 if child_scope.lookup_column(table.as_deref(), column).is_ok() {
    471                     return true;
    472                 }
    473                 let Ok(index) = scope.lookup_column(table.as_deref(), column) else {
    474                     // If the parent lookup fails too (i.e. unknown column),
    475                     // ignore the error. It will be surfaced during building.
    476                     return true;
    477                 };
    478                 child_scope.add_passthrough(scope, index, true);
    479                 hidden.push(Expression::Column(index));
    480                 true
    481             });
    482         }
    483         hidden
    484     }

    486     /// Builds an expression from an AST expression, looking up columns and
    487     /// aggregate expressions in the scope.
    488     pub fn build_expression(expr: ast::Expression, scope: &Scope) -> Result<Expression> {
    489         use Expression::*;
    490 
    491         // Look up aggregate functions or GROUP BY expressions. These were added
    492         // to the scope when building the Aggregate node, if any.
    493         if let Some(index) = scope.lookup_aggregate(&expr) {
    494             return Ok(Column(index));
    495         }
    496 
    497         // Helper for building a boxed expression.
    498         let build = |expr: Box<ast::Expression>| -> Result<Box<Expression>> {
    499             Ok(Box::new(Self::build_expression(*expr, scope)?))
    500         };
    501 
    502         Ok(match expr {
    503             // For simplicity, expression evaluation only supports scalar
    504             // values, not compound types like tuples. Support for * is
    505             // therefore special-cased in SELECT and COUNT(*).
    506             ast::Expression::All => return errinput!("unsupported use of *"),
    507             ast::Expression::Literal(l) => Constant(match l {
    508                 ast::Literal::Null => Value::Null,
    509                 ast::Literal::Boolean(b) => Value::Boolean(b),
    510                 ast::Literal::Integer(i) => Value::Integer(i),
    511                 ast::Literal::Float(f) => Value::Float(f),
    512                 ast::Literal::String(s) => Value::String(s),
    513             }),
    514             ast::Expression::Column(table, name) => {
    515                 Column(scope.lookup_column(table.as_deref(), &name)?)
    516             }
    517             ast::Expression::Function(name, mut args) => match (name.as_str(), args.len()) {
    518                 // NB: aggregate functions are processed above.
    519                 ("sqrt", 1) => SquareRoot(build(Box::new(args.remove(0)))?),
    520                 (name, n) => return errinput!("unknown function {name} with {n} arguments"),
    521             },
    522             ast::Expression::Operator(op) => match op {
    523                 ast::Operator::And(lhs, rhs) => And(build(lhs)?, build(rhs)?),
    524                 ast::Operator::Not(expr) => Not(build(expr)?),
    525                 ast::Operator::Or(lhs, rhs) => Or(build(lhs)?, build(rhs)?),
    526 
    527                 ast::Operator::Equal(lhs, rhs) => Equal(build(lhs)?, build(rhs)?),
    528                 ast::Operator::GreaterThan(lhs, rhs) => GreaterThan(build(lhs)?, build(rhs)?),
    529                 ast::Operator::GreaterThanOrEqual(lhs, rhs) => Or(
    530                     GreaterThan(build(lhs.clone())?, build(rhs.clone())?).into(),
    531                     Equal(build(lhs)?, build(rhs)?).into(),
    532                 ),
    533                 ast::Operator::Is(expr, literal) => {
    534                     let expr = build(expr)?;
    535                     let value = match literal {
    536                         ast::Literal::Null => Value::Null,
    537                         ast::Literal::Float(f) if f.is_nan() => Value::Float(f),
    538                         value => panic!("invalid IS value {value:?}"), // enforced by parser
    539                     };
    540                     Is(expr, value)
    541                 }
    542                 ast::Operator::LessThan(lhs, rhs) => LessThan(build(lhs)?, build(rhs)?),
    543                 ast::Operator::LessThanOrEqual(lhs, rhs) => Or(
    544                     LessThan(build(lhs.clone())?, build(rhs.clone())?).into(),
    545                     Equal(build(lhs)?, build(rhs)?).into(),
    546                 ),
    547                 ast::Operator::Like(lhs, rhs) => Like(build(lhs)?, build(rhs)?),
    548                 ast::Operator::NotEqual(lhs, rhs) => Not(Equal(build(lhs)?, build(rhs)?).into()),
    549 
    550                 ast::Operator::Add(lhs, rhs) => Add(build(lhs)?, build(rhs)?),
    551                 ast::Operator::Divide(lhs, rhs) => Divide(build(lhs)?, build(rhs)?),
    552                 ast::Operator::Exponentiate(lhs, rhs) => Exponentiate(build(lhs)?, build(rhs)?),
    553                 ast::Operator::Factorial(expr) => Factorial(build(expr)?),
    554                 ast::Operator::Identity(expr) => Identity(build(expr)?),
    555                 ast::Operator::Remainder(lhs, rhs) => Remainder(build(lhs)?, build(rhs)?),
    556                 ast::Operator::Multiply(lhs, rhs) => Multiply(build(lhs)?, build(rhs)?),
    557                 ast::Operator::Negate(expr) => Negate(build(expr)?),
    558                 ast::Operator::Subtract(lhs, rhs) => Subtract(build(lhs)?, build(rhs)?),
    559             },
    560         })
    561     }
    562 
    563     /// Builds and evaluates a constant AST expression. Errors on column refs.
    564     fn evaluate_constant(expr: ast::Expression) -> Result<Value> {
    565         Self::build_expression(expr, &Scope::new())?.evaluate(None)
    566     }
    567 }

    569 /// A scope maps column/table names to input column indexes, for lookups during
    570 /// expression construction. It also tracks aggregate and GROUP BY expressions,
    571 /// as well as hidden columns.
    572 ///
    573 /// Expression evaluation generally happens in the context of an input row. This
    574 /// row may come directly from a single table, or it may be the result of a long
    575 /// chain of joins and projections. The scope keeps track of which columns are
    576 /// currently visible and what names they have. During expression planning, the
    577 /// scope is used to resolve column names to column indexes, which are placed in
    578 /// the plan and used during execution.
    579 pub struct Scope {
    580     /// The currently visible columns. If empty, only constant expressions can
    581     /// be used (no column references).
    582     columns: Vec<Label>,
    583     /// Index of currently visible tables, by query name (e.g. may be aliased).
    584     tables: HashSet<String>,
    585     /// Index of fully qualified table.column names to column indexes. Qualified
    586     /// names are always unique within a scope.
    587     qualified: HashMap<(String, String), usize>,
    588     /// Index of unqualified column names to column indexes. If a name points
    589     /// to multiple columns, lookups will fail with an ambiguous name error.
    590     unqualified: HashMap<String, Vec<usize>>,
    591     /// Index of aggregate and GROUP BY expressions to column indexes. This is
    592     /// used to track output columns of Aggregate nodes and look them up from
    593     /// expressions in downstream SELECT, HAVING, and ORDER BY clauses. If the
    594     /// node contains an (inner) Aggregate node, this is never empty.
    595     aggregates: HashMap<ast::Expression, usize>,
    596     /// Hidden columns. These are used to pass e.g. ORDER BY and HAVING
    597     /// expressions through SELECT projection nodes if the expressions aren't
    598     /// already projected. They should be removed before emitting results.
    599     hidden: HashSet<usize>,
    600 }
    601 
    602 impl Scope {
    603     /// Creates a new, empty scope.
    604     pub fn new() -> Self {
    605         Self {
    606             columns: Vec::new(),
    607             tables: HashSet::new(),
    608             qualified: HashMap::new(),
    609             unqualified: HashMap::new(),
    610             aggregates: HashMap::new(),
    611             hidden: HashSet::new(),
    612         }
    613     }
    614 
    615     /// Creates a scope from a table, using the table's original name.
    616     fn from_table(table: &Table) -> Result<Self> {
    617         let mut scope = Self::new();
    618         scope.add_table(table, None)?;
    619         Ok(scope)
    620     }
    621 
    622     /// Creates a new child scope that inherits from the parent scope.
    623     pub fn spawn(&self) -> Self {
    624         let mut child = Scope::new();
    625         child.tables = self.tables.clone(); // retain table names
    626         child
    627     }
    628 
    629     /// Adds a table to the scope. The label is either the table's original name
    630     /// or an alias, and must be unique. All table columns are added, in order.
    631     fn add_table(&mut self, table: &Table, alias: Option<&str>) -> Result<()> {
    632         let name = alias.unwrap_or(&table.name);
    633         if self.tables.contains(name) {
    634             return errinput!("duplicate table name {name}");
    635         }
    636         for column in &table.columns {
    637             self.add_column(Label::Qualified(name.to_string(), column.name.clone()));
    638         }
    639         self.tables.insert(name.to_string());
    640         Ok(())
    641     }
    642 
    643     /// Appends a column with the given label to the scope. Returns the column
    644     /// index.
    645     fn add_column(&mut self, label: Label) -> usize {
    646         let index = self.columns.len();
    647         if let Label::Qualified(table, column) = &label {
    648             self.qualified.insert((table.clone(), column.clone()), index);
    649         }
    650         if let Label::Qualified(_, name) | Label::Unqualified(name) = &label {
    651             self.unqualified.entry(name.clone()).or_default().push(index)
    652         }
    653         self.columns.push(label);
    654         index
    655     }
    656 
    657     /// Looks up a column index by name, if possible.
    658     fn lookup_column(&self, table: Option<&str>, name: &str) -> Result<usize> {
    659         let fmtname = || table.map(|table| format!("{table}.{name}")).unwrap_or(name.to_string());
    660         if self.columns.is_empty() {
    661             return errinput!("expression must be constant, found column {}", fmtname());
    662         }
    663         if let Some(table) = table {
    664             if !self.tables.contains(table) {
    665                 return errinput!("unknown table {table}");
    666             }
    667             if let Some(index) = self.qualified.get(&(table.to_string(), name.to_string())) {
    668                 return Ok(*index);
    669             }
    670         } else if let Some(indexes) = self.unqualified.get(name) {
    671             if indexes.len() > 1 {
    672                 return errinput!("ambiguous column {name}");
    673             }
    674             return Ok(indexes[0]);
    675         }
    676         if !self.aggregates.is_empty() {
    677             return errinput!(
    678                 "column {} must be used in an aggregate or GROUP BY expression",
    679                 fmtname()
    680             );
    681         }
    682         errinput!("unknown column {}", fmtname())
    683     }
    684 
    685     /// Adds an aggregate expression to the scope, returning the new column
    686     /// index or None if the expression already exists. This is either an
    687     /// aggregate function or a GROUP BY expression, used to look up the
    688     /// aggregate output column from e.g. SELECT, HAVING, and ORDER BY.
    689     fn add_aggregate(&mut self, expr: &ast::Expression, parent: &Scope) -> Option<usize> {
    690         if self.aggregates.contains_key(expr) {
    691             return None;
    692         }
    693         // If this is a simple column reference (i.e. GROUP BY foo), pass
    694         // through the column label from the parent scope for lookups.
    695         let mut label = Label::None;
    696         if let ast::Expression::Column(table, column) = expr {
    697             // Ignore errors, they will be emitted when building the expression.
    698             if let Ok(index) = parent.lookup_column(table.as_deref(), column.as_str()) {
    699                 label = parent.columns[index].clone();
    700             }
    701         }
    702         let index = self.add_column(label);
    703         self.aggregates.insert(expr.clone(), index);
    704         Some(index)
    705     }
    706 
    707     /// Looks up an aggregate column index by aggregate function or GROUP BY
    708     /// expression.
    709     fn lookup_aggregate(&self, expr: &ast::Expression) -> Option<usize> {
    710         self.aggregates.get(expr).copied()
    711     }
    712 
    713     /// Adds a column that passes through a column from the parent scope,
    714     /// retaining its properties. If hide is true, the column is hidden.
    715     fn add_passthrough(&mut self, parent: &Scope, parent_index: usize, hide: bool) -> usize {
    716         let index = self.add_column(parent.columns[parent_index].clone());
    717         for (expr, i) in &parent.aggregates {
    718             if *i == parent_index {
    719                 self.aggregates.entry(expr.clone()).or_insert(index);
    720             }
    721         }
    722         if hide || parent.hidden.contains(&parent_index) {
    723             self.hidden.insert(index);
    724         }
    725         index
    726     }
    727 
    728     /// Merges two scopes, by appending the given scope to self.
    729     fn merge(&mut self, scope: Scope) -> Result<()> {
    730         for table in scope.tables {
    731             if self.tables.contains(&table) {
    732                 return errinput!("duplicate table name {table}");
    733             }
    734             self.tables.insert(table);
    735         }
    736         let offset = self.columns.len();
    737         for label in scope.columns {
    738             self.add_column(label);
    739         }
    740         for (expr, index) in scope.aggregates {
    741             self.aggregates.entry(expr).or_insert(index + offset);
    742         }
    743         self.hidden.extend(scope.hidden.into_iter().map(|index| index + offset));
    744         Ok(())
    745     }
    746 
    747     /// Projects the scope via the given expressions and aliases, creating a new
    748     /// child scope with one column per expression. These may be a simple column
    749     /// reference (e.g. "SELECT a, b FROM table"), which passes through the
    750     /// corresponding column from the original scope and retains its qualified
    751     /// and unqualified names. Otherwise, for non-trivial column references, a
    752     /// new column is created for the expression. Explicit aliases may be given.
    753     fn project(&self, expressions: &[(ast::Expression, Option<String>)]) -> Self {
    754         let mut child = self.spawn();
    755         for (expr, alias) in expressions {
    756             // Use the alias if given, or look up any column references.
    757             let mut label = Label::None;
    758             if let Some(alias) = alias {
    759                 label = Label::Unqualified(alias.clone());
    760             } else if let ast::Expression::Column(table, column) = expr {
    761                 // Ignore errors, they will be surfaced in build_expression().
    762                 if let Ok(index) = self.lookup_column(table.as_deref(), column.as_str()) {
    763                     label = self.columns[index].clone();
    764                 }
    765             }
    766             let index = child.add_column(label);
    767             // If this is an aggregate query, then all projected expressions
    768             // must also be aggregates by definition (an aggregate node can only
    769             // emit aggregate functions or GROUP BY expressions).
    770             if !self.aggregates.is_empty() {
    771                 child.aggregates.entry(expr.clone()).or_insert(index);
    772             }
    773         }
    774         child
    775     }
    776 
    777     /// Remaps the scope using the given targets.
    778     fn remap(&self, targets: &[Option<usize>]) -> Self {
    779         let mut child = self.spawn();
    780         for index in remap_sources(targets).into_iter().flatten() {
    781             child.add_passthrough(self, index, false);
    782         }
    783         child
    784     }
    785 
    786     /// Removes hidden columns from the scope, returning their indexes or None
    787     /// if no columns are hidden.
    788     fn remove_hidden(&mut self) -> Option<HashSet<usize>> {
    789         if self.hidden.is_empty() {
    790             return None;
    791         }
    792         let hidden = std::mem::take(&mut self.hidden);
    793         let mut index = 0;
    794         self.columns.retain(|_| {
    795             let retain = !hidden.contains(&index);
    796             index += 1;
    797             retain
    798         });
    799         self.qualified.retain(|_, index| !hidden.contains(index));
    800         self.unqualified.iter_mut().for_each(|(_, vec)| vec.retain(|i| !hidden.contains(i)));
    801         self.unqualified.retain(|_, vec| !vec.is_empty());
    802         self.aggregates.retain(|_, index| !hidden.contains(index));
    803         Some(hidden)
    804     }
    805 
    806     /// Removes hidden columns from the scope and returns the remaining column
    807     /// indexes as a Remap targets vector, or None if no columns are hidden. A
    808     /// Remap targets vector maps parent column indexes to child column indexes,
    809     /// or None if a column should be dropped.
    810     fn remap_hidden(&mut self) -> Option<Vec<Option<usize>>> {
    811         let size = self.columns.len();
    812         let hidden = self.remove_hidden()?;
    813         let mut targets = vec![None; size];
    814         let mut index = 0;
    815         for (old_index, target) in targets.iter_mut().enumerate() {
    816             if !hidden.contains(&old_index) {
    817                 *target = Some(index);
    818                 index += 1;
    819             }
    820         }
    821         Some(targets)
    822     }
