     13 /// A statement execution plan. The root nodes can perform data modifications or
     14 /// schema changes, in addition to SELECT queries. Beyond the root, the plan is
     15 /// made up of a tree of inner plan nodes that stream and process rows via
     16 /// iterators. Below is an example of an (unoptimized) plan for the given query:
     17 ///
     18 /// SELECT title, released, genres.name AS genre
     19 /// FROM movies INNER JOIN genres ON movies.genre_id = genres.id
     20 /// WHERE released >= 2000 ORDER BY released
     21 ///
     22 /// Order: movies.released desc
     23 /// └─ Projection: movies.title, movies.released, genres.name as genre
     24 ///    └─ Filter: movies.released >= 2000
     25 ///       └─ NestedLoopJoin: inner on movies.genre_id = genres.id
     26 ///          ├─ Scan: movies
     27 ///          └─ Scan: genres
     28 ///
     29 /// Rows flow from the tree leaves to the root. The Scan nodes read and emit
     30 /// table rows from storage. They are passed to the NestedLoopJoin node which
     31 /// joins the rows from the two tables, then the Filter node discards old
     32 /// movies, the Projection node picks out the requested columns, and the Order
     33 /// node sorts them before emitting the rows to the client.
     34 #[derive(Clone, Debug, PartialEq, Serialize, Deserialize)]
     35 pub enum Plan {
     36     /// A CREATE TABLE plan. Creates a new table with the given schema. Errors
     37     /// if the table already exists or the schema is invalid.
     38     CreateTable { schema: Table },
     39     /// A DROP TABLE plan. Drops the given table. Errors if the table does not
     40     /// exist, unless if_exists is true.
     41     DropTable { table: String, if_exists: bool },
     42     /// A DELETE plan. Deletes rows in table that match the rows from source.
     43     /// primary_key specifies the primary key column index in the source rows.
     44     Delete { table: String, primary_key: usize, source: Node },
     45     /// An INSERT plan. Inserts rows from source (typically a Values node) into
     46     /// table. If column_map is given, it maps table → source column indexes and
     47     /// must have one entry for every column in source. Table columns not
     48     /// present in source will get the column's default value if set, or error.
     49     Insert { table: Table, column_map: Option<HashMap<usize, usize>>, source: Node },
     50     /// An UPDATE plan. Updates rows in table that match the rows from source,
     51     /// where primary_key specifies the primary key column index in the source
     52     /// rows. The given column/expression pairs specify the row updates to make,
     53     /// evaluated using the existing source row, which must be a complete row
     54     /// from the update table.
     55     Update { table: Table, primary_key: usize, source: Node, expressions: Vec<(usize, Expression)> },
     56     /// A SELECT plan. Recursively executes the query plan tree and returns the
     57     /// resulting rows.
     58     Select(Node),
     59 }

     72     /// Optimizes the plan, consuming it. See OPTIMIZERS for the list of
     73     /// optimizers.
     74     pub fn optimize(self) -> Result<Self> {
     75         let optimize = |node| OPTIMIZERS.iter().try_fold(node, |node, (_, opt)| opt(node));
     76         Ok(match self {
     77             Self::CreateTable { .. } | Self::DropTable { .. } => self,
     78             Self::Delete { table, primary_key, source } => {
     79                 Self::Delete { table, primary_key, source: optimize(source)? }
     80             }
     81             Self::Insert { table, column_map, source } => {
     82                 Self::Insert { table, column_map, source: optimize(source)? }
     83             }
     84             Self::Update { table, primary_key, source, expressions } => {
     85                 Self::Update { table, primary_key, source: optimize(source)?, expressions }
     86             }
     87             Self::Select(root) => Self::Select(optimize(root)?),
     88         })
     89     }

     92 /// A query plan node. Returns a row iterator, and can be nested.
     93 #[derive(Clone, Debug, PartialEq, Serialize, Deserialize)]
     94 pub enum Node {
     95     /// Computes the given aggregate values for the given group_by buckets
     96     /// across all rows in the source node. The group_by columns are emitted
     97     /// first, followed by the aggregate columns, in the given order.
     98     Aggregate { source: Box<Node>, group_by: Vec<Expression>, aggregates: Vec<Aggregate> },
     99     /// Filters source rows, by discarding rows for which the predicate
    100     /// evaluates to false.
    101     Filter { source: Box<Node>, predicate: Expression },
    102     /// Joins the left and right sources on the given columns by building an
    103     /// in-memory hashmap of the right source and looking up matches for each
    104     /// row in the left source. When outer is true (e.g. LEFT JOIN), a left row
    105     /// without a right match is emitted anyway, with NULLs for the right row.
    106     HashJoin {
    107         left: Box<Node>,
    108         left_column: usize,
    109         right: Box<Node>,
    110         right_column: usize,
    111         outer: bool,
    112     },
    113     /// Looks up the given values in a secondary index and emits matching rows.
    114     /// NULL and NaN values are considered equal, to allow IS NULL and IS NAN
    115     /// index lookups, as is -0.0 and 0.0.
    116     IndexLookup { table: Table, column: usize, values: Vec<Value>, alias: Option<String> },
    117     /// Looks up the given primary keys and emits their rows.
    118     KeyLookup { table: Table, keys: Vec<Value>, alias: Option<String> },
    119     /// Only emits the first limit rows from the source, discards the rest.
    120     Limit { source: Box<Node>, limit: usize },
    121     /// Joins the left and right sources on the given predicate by buffering the
    122     /// right source and iterating over it for every row in the left source.
    123     /// When outer is true (e.g. LEFT JOIN), a left row without a right match is
    124     /// emitted anyway, with NULLs for the right row.
    125     NestedLoopJoin { left: Box<Node>, right: Box<Node>, predicate: Option<Expression>, outer: bool },
    126     /// Nothing does not emit anything, and is used to short-circuit nodes that
    127     /// can't emit anything during optimization. It retains the column names of
    128     /// any replaced nodes for results headers and plan formatting.
    129     Nothing { columns: Vec<Label> },
    130     /// Discards the first offset rows from source, emits the rest.
    131     Offset { source: Box<Node>, offset: usize },
    132     /// Sorts the source rows by the given sort key. Buffers the entire row set
    133     /// in memory.
    134     Order { source: Box<Node>, key: Vec<(Expression, Direction)> },
    135     /// Projects the input rows by evaluating the given expressions. Aliases are
    136     /// only used when displaying the plan.
    137     Projection { source: Box<Node>, expressions: Vec<Expression>, aliases: Vec<Label> },
    138     /// Remaps source columns to the given target column index, or None to drop
    139     /// the column. Unspecified target columns yield Value::Null. The source →
    140     /// target mapping ensures a source column can only be mapped to a single
    141     /// target column, allowing the value to be moved rather than cloned.
    142     Remap { source: Box<Node>, targets: Vec<Option<usize>> },
    143     /// A full table scan, with an optional pushed-down filter. The schema is
    144     /// used during plan optimization. The alias is only used for formatting.
    145     Scan { table: Table, filter: Option<Expression>, alias: Option<String> },
    146     /// A constant set of values.
    147     Values { rows: Vec<Vec<Expression>> },
    148 }

    150 impl Node {
    151     /// Returns the number of columns emitted by the node.
    152     pub fn columns(&self) -> usize {
    153         match self {
    154             // Source nodes emit all table columns.
    155             Self::IndexLookup { table, .. }
    156             | Self::KeyLookup { table, .. }
    157             | Self::Scan { table, .. } => table.columns.len(),
    158 
    159             // Some nodes modify the column set.
    160             Self::Aggregate { aggregates, group_by, .. } => aggregates.len() + group_by.len(),
    161             Self::Projection { expressions, .. } => expressions.len(),
    162             Self::Remap { targets, .. } => {
    163                 targets.iter().filter_map(|v| *v).map(|i| i + 1).max().unwrap_or(0)
    164             }
    165 
    166             // Join nodes emit the combined columns.
    167             Self::HashJoin { left, right, .. } | Self::NestedLoopJoin { left, right, .. } => {
    168                 left.columns() + right.columns()
    169             }
    170 
    171             // Simple nodes just pass through the source columns.
    172             Self::Filter { source, .. }
    173             | Self::Limit { source, .. }
    174             | Self::Offset { source, .. }
    175             | Self::Order { source, .. } => source.columns(),
    176 
    177             // And some are trivial.
    178             Self::Nothing { columns } => columns.len(),
    179             Self::Values { rows } => rows.first().map(|row| row.len()).unwrap_or(0),
    180         }
    181     }

    183     /// Returns a label for a column, if any, by tracing the column through the
    184     /// plan tree. Only used for query result headers and plan display purposes,
    185     /// not to look up expression columns (see Scope).
    186     pub fn column_label(&self, index: usize) -> Label {
    187         match self {
    188             // Source nodes use the table/column name.
    189             Self::IndexLookup { table, alias, .. }
    190             | Self::KeyLookup { table, alias, .. }
    191             | Self::Scan { table, alias, .. } => Label::Qualified(
    192                 alias.as_ref().unwrap_or(&table.name).clone(),
    193                 table.columns[index].name.clone(),
    194             ),
    195 
    196             // Some nodes rearrange columns. Route them to the correct
    197             // upstream column where appropriate.
    198             Self::Aggregate { source, group_by, .. } => match group_by.get(index) {
    199                 Some(Expression::Column(index)) => source.column_label(*index),
    200                 Some(_) | None => Label::None,
    201             },
    202             Self::Projection { source, expressions, aliases } => match aliases.get(index) {
    203                 Some(Label::None) | None => match expressions.get(index) {
    204                     // Unaliased column references route to the source.
    205                     Some(Expression::Column(index)) => source.column_label(*index),
    206                     // Unaliased expressions don't have a name.
    207                     Some(_) | None => Label::None,
    208                 },
    209                 // Aliased columns use the alias.
    210                 Some(alias) => alias.clone(),
    211             },
    212             Self::Remap { source, targets } => targets
    213                 .iter()
    214                 .position(|t| t == &Some(index))
    215                 .map(|i| source.column_label(i))
    216                 .unwrap_or(Label::None),
    217 
    218             // Joins dispatch to the appropriate source.
    219             Self::HashJoin { left, right, .. } | Self::NestedLoopJoin { left, right, .. } => {
    220                 if index < left.columns() {
    221                     left.column_label(index)
    222                 } else {
    223                     right.column_label(index - left.columns())
    224                 }
    225             }
    226 
    227             // Simple nodes just dispatch to the source.
    228             Self::Filter { source, .. }
    229             | Self::Limit { source, .. }
    230             | Self::Offset { source, .. }
    231             | Self::Order { source, .. } => source.column_label(index),
    232 
    233             // Nothing nodes contain the original columns of replaced nodes.
    234             Self::Nothing { columns } => columns.get(index).cloned().unwrap_or(Label::None),
    235 
    236             // And some don't have any names at all.
    237             Self::Values { .. } => Label::None,
    238         }
    239     }

    241     /// Recursively transforms query nodes depth-first by applying the given
    242     /// closures before and after descending.
    243     pub fn transform(
    244         mut self,
    245         before: &impl Fn(Self) -> Result<Self>,
    246         after: &impl Fn(Self) -> Result<Self>,
    247     ) -> Result<Self> {
    248         // Helper for transforming boxed nodes.
    249         let xform = |mut node: Box<Node>| -> Result<Box<Node>> {
    250             *node = node.transform(before, after)?;
    251             Ok(node)
    252         };
    253 
    254         self = before(self)?;
    255         self = match self {
    256             Self::Aggregate { source, group_by, aggregates } => {
    257                 Self::Aggregate { source: xform(source)?, group_by, aggregates }
    258             }
    259             Self::Filter { source, predicate } => {
    260                 Self::Filter { source: xform(source)?, predicate }
    261             }
    262             Self::HashJoin { left, left_column, right, right_column, outer } => Self::HashJoin {
    263                 left: xform(left)?,
    264                 left_column,
    265                 right: xform(right)?,
    266                 right_column,
    267                 outer,
    268             },
    269             Self::Limit { source, limit } => Self::Limit { source: xform(source)?, limit },
    270             Self::NestedLoopJoin { left, right, predicate, outer } => {
    271                 Self::NestedLoopJoin { left: xform(left)?, right: xform(right)?, predicate, outer }
    272             }
    273             Self::Offset { source, offset } => Self::Offset { source: xform(source)?, offset },
    274             Self::Order { source, key } => Self::Order { source: xform(source)?, key },
    275             Self::Projection { source, expressions, aliases } => {
    276                 Self::Projection { source: xform(source)?, expressions, aliases }
    277             }
    278             Self::Remap { source, targets } => Self::Remap { source: xform(source)?, targets },
    279 
    280             Self::IndexLookup { .. }
    281             | Self::KeyLookup { .. }
    282             | Self::Nothing { .. }
    283             | Self::Scan { .. }
    284             | Self::Values { .. } => self,
    285         };
    286         self = after(self)?;
    287         Ok(self)
    288     }

    290     /// Recursively transforms all node expressions by calling the given
    291     /// closures on them before and after descending.
    292     pub fn transform_expressions(
    293         self,
    294         before: &impl Fn(Expression) -> Result<Expression>,
    295         after: &impl Fn(Expression) -> Result<Expression>,
    296     ) -> Result<Self> {
    297         Ok(match self {
    298             Self::Filter { source, mut predicate } => {
    299                 predicate = predicate.transform(before, after)?;
    300                 Self::Filter { source, predicate }
    301             }
    302             Self::NestedLoopJoin { left, right, predicate: Some(predicate), outer } => {
    303                 let predicate = Some(predicate.transform(before, after)?);
    304                 Self::NestedLoopJoin { left, right, predicate, outer }
    305             }
    306             Self::Order { source, mut key } => {
    307                 key = key
    308                     .into_iter()
    309                     .map(|(expr, dir)| Ok((expr.transform(before, after)?, dir)))
    310                     .collect::<Result<_>>()?;
    311                 Self::Order { source, key }
    312             }
    313             Self::Projection { source, mut expressions, aliases } => {
    314                 expressions = expressions
    315                     .into_iter()
    316                     .map(|expr| expr.transform(before, after))
    317                     .try_collect()?;
    318                 Self::Projection { source, expressions, aliases }
    319             }
    320             Self::Scan { table, alias, filter: Some(filter) } => {
    321                 let filter = Some(filter.transform(before, after)?);
    322                 Self::Scan { table, alias, filter }
    323             }
    324             Self::Values { mut rows } => {
    325                 rows = rows
    326                     .into_iter()
    327                     .map(|row| row.into_iter().map(|expr| expr.transform(before, after)).collect())
    328                     .try_collect()?;
    329                 Self::Values { rows }
    330             }
    331 
    332             Self::Aggregate { .. }
    333             | Self::HashJoin { .. }
    334             | Self::IndexLookup { .. }
    335             | Self::KeyLookup { .. }
    336             | Self::Limit { .. }
    337             | Self::NestedLoopJoin { predicate: None, .. }
    338             | Self::Nothing { .. }
    339             | Self::Offset { .. }
    340             | Self::Remap { .. }
    341             | Self::Scan { filter: None, .. } => self,
    342         })
    343     }
    344 }

    426 impl Node {
    427     /// Recursively formats the node. Prefix is used for tree branch lines. root
    428     /// is true if this is the root (first) node, and last_child is true if this
    429     /// is the last child node of the parent.
    430     pub fn format(
    431         &self,
    432         f: &mut std::fmt::Formatter<'_>,
    433         prefix: &str,
    434         root: bool,
    435         last_child: bool,
    436     ) -> std::fmt::Result {
    437         // If this is not the root node, emit a newline after the previous node.
    438         // This avoids a spurious newline at the end of the plan.
    439         if !root {
    440             writeln!(f)?;
    441         }
    442 
    443         // Prefix the node with a tree branch line. Modify the prefix for any
    444         // child nodes we'll recurse into.
    445         let prefix = if !last_child {
    446             write!(f, "{prefix}├─ ")?;
    447             format!("{prefix}│  ")
    448         } else if !root {
    449             write!(f, "{prefix}└─ ")?;
    450             format!("{prefix}   ")
    451         } else {
    452             write!(f, "{prefix}")?;
    453             prefix.to_string()
    454         };
    455 
    456         // Format the node.
    457         match self {
    458             Self::Aggregate { source, aggregates, group_by } => {
    459                 let aggregates = group_by
    460                     .iter()
    461                     .map(|group_by| group_by.format(source))
    462                     .chain(aggregates.iter().map(|agg| agg.format(source)))
    463                     .join(", ");
    464                 write!(f, "Aggregate: {aggregates}")?;
    465                 source.format(f, &prefix, false, true)?;
    466             }
    467             Self::Filter { source, predicate } => {
    468                 write!(f, "Filter: {}", predicate.format(source))?;
    469                 source.format(f, &prefix, false, true)?;
    470             }
    471             Self::HashJoin { left, left_column, right, right_column, outer } => {
    472                 let kind = if *outer { "outer" } else { "inner" };
    473                 let left_column = match left.column_label(*left_column) {
    474                     Label::None => format!("left #{left_column}"),
    475                     label => format!("{label}"),
    476                 };
    477                 let right_column = match right.column_label(*right_column) {
    478                     Label::None => format!("right #{right_column}"),
    479                     label => format!("{label}"),
    480                 };
    481                 write!(f, "HashJoin: {kind} on {left_column} = {right_column}")?;
    482                 left.format(f, &prefix, false, false)?;
    483                 right.format(f, &prefix, false, true)?;
    484             }
    485             Self::IndexLookup { table, column, alias, values } => {
    486                 let column = &table.columns[*column].name;
    487                 write!(f, "IndexLookup: {}.{column}", table.name)?;
    488                 if let Some(alias) = alias {
    489                     write!(f, " as {alias}.{column}")?;
    490                 }
    491                 if !values.is_empty() && values.len() < 10 {
    492                     write!(f, " ({})", values.iter().join(", "))?;
    493                 } else {
    494                     write!(f, " ({} values)", values.len())?;
    495                 }
    496             }
    497             Self::KeyLookup { table, alias, keys } => {
    498                 write!(f, "KeyLookup: {}", table.name)?;
    499                 if let Some(alias) = alias {
    500                     write!(f, " as {alias}")?;
    501                 }
    502                 if !keys.is_empty() && keys.len() < 10 {
    503                     write!(f, " ({})", keys.iter().join(", "))?;
    504                 } else {
    505                     write!(f, " ({} keys)", keys.len())?;
    506                 }
    507             }
    508             Self::Limit { source, limit } => {
    509                 write!(f, "Limit: {limit}")?;
    510                 source.format(f, &prefix, false, true)?;
    511             }
    512             Self::NestedLoopJoin { left, right, predicate, outer, .. } => {
    513                 let kind = if *outer { "outer" } else { "inner" };
    514                 write!(f, "NestedLoopJoin: {kind}")?;
    515                 if let Some(predicate) = predicate {
    516                     write!(f, " on {}", predicate.format(self))?;
    517                 }
    518                 left.format(f, &prefix, false, false)?;
    519                 right.format(f, &prefix, false, true)?;
    520             }
    521             Self::Nothing { .. } => write!(f, "Nothing")?,
    522             Self::Offset { source, offset } => {
    523                 write!(f, "Offset: {offset}")?;
    524                 source.format(f, &prefix, false, true)?;
    525             }
    526             Self::Order { source, key: orders } => {
    527                 let orders = orders
    528                     .iter()
    529                     .map(|(expr, dir)| format!("{} {dir}", expr.format(source)))
    530                     .join(", ");
    531                 write!(f, "Order: {orders}")?;
    532                 source.format(f, &prefix, false, true)?;
    533             }
    534             Self::Projection { source, expressions, aliases } => {
    535                 let expressions = expressions
    536                     .iter()
    537                     .enumerate()
    538                     .map(|(i, expr)| match aliases.get(i) {
    539                         Some(Label::None) | None => expr.format(source),
    540                         Some(alias) => format!("{} as {alias}", expr.format(source)),
    541                     })
    542                     .join(", ");
    543                 write!(f, "Projection: {expressions}")?;
    544                 source.format(f, &prefix, false, true)?;
    545             }
    546             Self::Remap { source, targets } => {
    547                 let remap = remap_sources(targets)
    548                     .into_iter()
    549                     .map(|from| match from {
    550                         Some(from) => match source.column_label(from) {
    551                             Label::None => format!("#{from}"),
    552                             label => label.to_string(),
    553                         },
    554                         None => "Null".to_string(),
    555                     })
    556                     .join(", ");
    557                 write!(f, "Remap: {remap}")?;
    558                 let dropped = targets
    559                     .iter()
    560                     .enumerate()
    561                     .filter_map(|(i, v)| {
    562                         v.is_none().then_some(match source.column_label(i) {
    563                             Label::None => format!("#{i}"),
    564                             label => format!("{label}"),
    565                         })
    566                     })
    567                     .join(", ");
    568                 if !dropped.is_empty() {
    569                     write!(f, " (dropped: {dropped})")?;
    570                 }
    571                 source.format(f, &prefix, false, true)?;
    572             }
    573             Self::Scan { table, alias, filter } => {
    574                 write!(f, "Scan: {}", table.name)?;
    575                 if let Some(alias) = alias {
    576                     write!(f, " as {alias}")?;
    577                 }
    578                 if let Some(filter) = filter {
    579                     write!(f, " ({})", filter.format(self))?;
    580                 }
    581             }
    582             Self::Values { rows, .. } => {
    583                 write!(f, "Values: ")?;
    584                 match rows.len() {
    585                     1 if rows[0].is_empty() => write!(f, "blank row")?,
    586                     1 => write!(f, "{}", rows[0].iter().map(|e| e.format(self)).join(", "))?,
    587                     n => write!(f, "{n} rows")?,
    588                 }
    589             }
    590         };
    591         Ok(())
    592     }
    593 }

    595 /// Inverts a Remap targets vector to a vector of source indexes, with None
    596 /// for columns that weren't targeted.
    597 pub fn remap_sources(targets: &[Option<usize>]) -> Vec<Option<usize>> {
    598     let size = targets.iter().filter_map(|v| *v).map(|i| i + 1).max().unwrap_or(0);
    599     let mut sources = vec![None; size];
    600     for (from, to) in targets.iter().enumerate() {
    601         if let Some(to) = to {
    602             sources[*to] = Some(from);
    603         }
    604     }
    605     sources
