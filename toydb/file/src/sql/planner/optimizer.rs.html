     20 /// Folds constant (sub)expressions by pre-evaluating them, instead of
     21 /// re-evaluating then for every row during execution.
     22 pub fn fold_constants(node: Node) -> Result<Node> {
     23     use Expression::*;
     24     use Value::*;
     25 
     26     // Transform expressions. Called after descending, to perform logical
     27     // short-circuiting on child expressions that have already been folded, and
     28     // to reduce the quadratic cost when an expression contains a column.
     29     let xform = |mut expr: Expression| {
     30         // If the expression is constant, evaluate it.
     31         //
     32         // This is a very simple approach, which doesn't handle more complex
     33         // cases such as 1 + a - 2 (which would require rearranging the
     34         // expression as 1 - 2 + a to evaluate the 1 - 2 branch).
     35         //
     36         // TODO: consider doing something better.
     37         if !expr.contains(&|expr| matches!(expr, Column(_))) {
     38             return expr.evaluate(None).map(Constant);
     39         }
     40 
     41         // If the expression is a logical operator, and one of the sides is
     42         // known, we may be able to short-circuit it.
     43         expr = match expr {
     44             And(lhs, rhs) => match (*lhs, *rhs) {
     45                 // If either side of an AND is false, the AND is false.
     46                 (Constant(Boolean(false)), _) | (_, Constant(Boolean(false))) => {
     47                     Constant(Boolean(false))
     48                 }
     49                 // If either side of an AND is true, the AND is redundant.
     50                 (Constant(Boolean(true)), expr) | (expr, Constant(Boolean(true))) => expr,
     51                 (lhs, rhs) => And(lhs.into(), rhs.into()),
     52             },
     53             Or(lhs, rhs) => match (*lhs, *rhs) {
     54                 // If either side of an OR is true, the OR is true.
     55                 (Constant(Boolean(true)), _) | (_, Constant(Boolean(true))) => {
     56                     Constant(Boolean(true))
     57                 }
     58                 // If either side of an OR is false, the OR is redundant.
     59                 (Constant(Boolean(false)), expr) | (expr, Constant(Boolean(false))) => expr,
     60                 (lhs, rhs) => Or(lhs.into(), rhs.into()),
     61             },
     62             expr => expr,
     63         };
     64         Ok(expr)
     65     };
     66 
     67     node.transform(&|node| node.transform_expressions(&Ok, &xform), &Ok)
     68 }

     70 /// Pushes filter predicates down into child nodes where possible. In
     71 /// particular, this can allow filtering during storage scans (below Raft),
     72 /// instead of reading and transmitting all rows then filtering.
     73 pub fn push_filters(node: Node) -> Result<Node> {
     74     /// Pushes an expression into a node if possible. Otherwise, returns the the
     75     /// unpushed expression.
     76     fn push_into(expr: Expression, target: &mut Node) -> Option<Expression> {
     77         match target {
     78             Node::Filter { predicate, .. } => {
     79                 // Temporarily swap the predicate to take ownership.
     80                 let rhs = std::mem::replace(predicate, Expression::Constant(Value::Null));
     81                 *predicate = Expression::And(expr.into(), rhs.into());
     82             }
     83             Node::NestedLoopJoin { predicate, .. } => {
     84                 *predicate = match predicate.take() {
     85                     Some(predicate) => Some(Expression::And(expr.into(), predicate.into())),
     86                     None => Some(expr),
     87                 };
     88             }
     89             Node::Scan { filter, .. } => {
     90                 *filter = match filter.take() {
     91                     Some(filter) => Some(Expression::And(expr.into(), filter.into())),
     92                     None => Some(expr),
     93                 };
     94             }
     95             // We don't handle HashJoin here, since we assume the join_type()
     96             // optimizer runs after this.
     97             Node::HashJoin { .. } => panic!("filter pushdown must run before join optimizer"),
     98             // Unable to push down, just return the original expression.
     99             _ => return Some(expr),
    100         }
    101         None
    102     }
    103 
    104     /// Pushes down a filter node if possible.
    105     fn push_filter(node: Node) -> Node {
    106         let Node::Filter { mut source, predicate } = node else { return node };
    107         // Attempt to push the filter into the source.
    108         if let Some(predicate) = push_into(predicate, &mut source) {
    109             // Push failed, return the original filter node.
    110             return Node::Filter { source, predicate };
    111         }
    112         // Push succeded, return the source that was pushed into. When we
    113         // replace this filter node with the source node, Node.transform() will
    114         // skip the source node since it now takes the place of the original
    115         // filter node. Transform the source manually.
    116         xform(*source)
    117     }
    118 
    119     // Pushes down parts of a join predicate into the left or right sources
    120     // where possible.
    121     fn push_join(node: Node) -> Node {
    122         let Node::NestedLoopJoin { mut left, mut right, predicate: Some(predicate), outer } = node
    123         else {
    124             return node;
    125         };
    126         // Convert the predicate into conjunctive normal form (an AND vector).
    127         let cnf = predicate.into_cnf_vec();
    128 
    129         // Push down expressions that don't reference both sources. Constant
    130         // expressions can be pushed down into both.
    131         let (mut push_left, mut push_right, mut predicate) = (Vec::new(), Vec::new(), Vec::new());
    132         for expr in cnf {
    133             let (mut ref_left, mut ref_right) = (false, false);
    134             expr.walk(&mut |expr| {
    135                 if let Expression::Column(index) = expr {
    136                     ref_left = ref_left || *index < left.columns();
    137                     ref_right = ref_right || *index >= left.columns();
    138                 }
    139                 !(ref_left && ref_right) // exit once both are referenced
    140             });
    141             match (ref_left, ref_right) {
    142                 (true, true) => predicate.push(expr),
    143                 (true, false) => push_left.push(expr),
    144                 (false, true) => push_right.push(expr),
    145                 (false, false) => {
    146                     push_left.push(expr.clone());
    147                     push_right.push(expr);
    148                 }
    149             }
    150         }
    151 
    152         // In the remaining cross-source expressions, look for equijoins where
    153         // one side also has constant value lookups. In this case we can copy
    154         // the constant lookups to the other side, to allow index lookups. This
    155         // commonly happens when joining a foreign key (which is indexed) on a
    156         // primary key, and we want to make use of the foreign key index, e.g.:
    157         //
    158         // SELECT m.name, g.name FROM movies m JOIN genres g ON m.genre_id = g.id AND g.id = 7;
    159         let left_lookups: HashMap<usize, usize> = push_left // column → push_left index
    160             .iter()
    161             .enumerate()
    162             .filter_map(|(i, expr)| expr.is_column_lookup().map(|column| (column, i)))
    163             .collect();
    164         let right_lookups: HashMap<usize, usize> = push_right // column → push_right index
    165             .iter()
    166             .enumerate()
    167             .filter_map(|(i, expr)| expr.is_column_lookup().map(|column| (column, i)))
    168             .collect();
    169 
    170         for expr in &predicate {
    171             // Find equijoins.
    172             let Expression::Equal(lhs, rhs) = expr else { continue };
    173             let Expression::Column(mut l) = **lhs else { continue };
    174             let Expression::Column(mut r) = **rhs else { continue };
    175 
    176             // The lhs may be a reference to the right source; swap them.
    177             if l > r {
    178                 (l, r) = (r, l)
    179             }
    180 
    181             // Check if either side is a column lookup, and copy it over.
    182             if let Some(expr) = left_lookups.get(&l).map(|i| push_left[*i].clone()) {
    183                 push_right.push(expr.replace_column(l, r));
    184             }
    185             if let Some(expr) = right_lookups.get(&r).map(|i| push_right[*i].clone()) {
    186                 push_left.push(expr.replace_column(r, l));
    187             }
    188         }
    189 
    190         // Push predicates down into the sources if possible.
    191         if let Some(expr) = Expression::and_vec(push_left) {
    192             if let Some(expr) = push_into(expr, &mut left) {
    193                 // Pushdown failed, put it back into the join predicate.
    194                 predicate.push(expr)
    195             }
    196         }
    197 
    198         if let Some(mut expr) = Expression::and_vec(push_right) {
    199             // Right columns have indexes in the joined row; shift them left.
    200             expr = expr.shift_column(-(left.columns() as isize));
    201             if let Some(mut expr) = push_into(expr, &mut right) {
    202                 // Pushdown failed, undo the column index shift.
    203                 expr = expr.shift_column(left.columns() as isize);
    204                 predicate.push(expr)
    205             }
    206         }
    207 
    208         // Leave any remaining predicates in the join node.
    209         let predicate = Expression::and_vec(predicate);
    210         Node::NestedLoopJoin { left, right, predicate, outer }
    211     }
    212 
    213     /// Applies pushdown transformations to a node.
    214     fn xform(node: Node) -> Node {
    215         push_join(push_filter(node))
    216     }
    217 
    218     // Push down before descending, so we can keep recursively pushing down.
    219     node.transform(&|node| Ok(xform(node)), &Ok)
    220 }

    222 /// Uses an index or primary key lookup for a filter when possible.
    223 pub fn index_lookup(node: Node) -> Result<Node> {
    224     let transform = |mut node| {
    225         // Only handle scan filters. filter_pushdown() must have pushed filters
    226         // into scan nodes first.
    227         let Node::Scan { table, alias, filter: Some(filter) } = node else { return node };
    228 
    229         // Convert the filter into conjunctive normal form (a list of ANDs).
    230         let mut cnf = filter.clone().into_cnf_vec();
    231 
    232         // Find the first expression that's either a primary key or secondary
    233         // index lookup. We could be more clever here, but this is fine.
    234         let Some((i, column)) = cnf.iter().enumerate().find_map(|(i, expr)| {
    235             expr.is_column_lookup()
    236                 .filter(|c| *c == table.primary_key || table.columns[*c].index)
    237                 .map(|column| (i, column))
    238         }) else {
    239             return Node::Scan { table, alias, filter: Some(filter) };
    240         };
    241 
    242         // Extract the lookup values and expression from the cnf vector.
    243         let values = cnf.remove(i).into_column_values(column);
    244 
    245         // Build the primary key or secondary index lookup node.
    246         if column == table.primary_key {
    247             node = Node::KeyLookup { table, keys: values, alias };
    248         } else {
    249             node = Node::IndexLookup { table, column, values, alias };
    250         }
    251 
    252         // If there's any remaining CNF expressions, add a filter node for them.
    253         if let Some(predicate) = Expression::and_vec(cnf) {
    254             node = Node::Filter { source: Box::new(node), predicate };
    255         }
    256 
    257         node
    258     };
    259     node.transform(&Ok, &|n| Ok(transform(n)))
    260 }

    262 /// Uses a hash join instead of a nested loop join for single-column equijoins.
    263 pub fn join_type(node: Node) -> Result<Node> {
    264     let xform = |node| match node {
    265         Node::NestedLoopJoin {
    266             left,
    267             right,
    268             predicate: Some(Expression::Equal(lhs, rhs)),
    269             outer,
    270         } => match (*lhs, *rhs) {
    271             (Expression::Column(mut left_column), Expression::Column(mut right_column)) => {
    272                 // The LHS column may be a column in the right table; swap them.
    273                 if right_column < left_column {
    274                     (left_column, right_column) = (right_column, left_column);
    275                 }
    276                 // The NestedLoopJoin predicate uses column indexes in the
    277                 // joined row, while the HashJoin uses column indexes in each
    278                 // individual table. Adjust the RHS column reference.
    279                 right_column -= left.columns();
    280                 Node::HashJoin { left, left_column, right, right_column, outer }
    281             }
    282             (lhs, rhs) => {
    283                 let predicate = Some(Expression::Equal(lhs.into(), rhs.into()));
    284                 Node::NestedLoopJoin { left, right, predicate, outer }
    285             }
    286         },
    287         node => node,
    288     };
    289     node.transform(&|node| Ok(xform(node)), &Ok)
    290 }

    292 /// Short-circuits useless nodes and expressions, by removing them and/or
    293 /// replacing them with Nothing nodes that yield no rows.
    294 pub fn short_circuit(node: Node) -> Result<Node> {
    295     use Expression::*;
    296     use Value::*;
    297 
    298     /// Creates a Nothing node with the columns of the original node.
    299     fn nothing(node: &Node) -> Node {
    300         let columns = (0..node.columns()).map(|i| node.column_label(i)).collect();
    301         Node::Nothing { columns }
    302     }
    303 
    304     let xform = |node| match node {
    305         // Filter nodes that always yield true are unnecessary: remove them.
    306         Node::Filter { source, predicate: Constant(Boolean(true)) } => *source,
    307 
    308         // Predicates that always yield true are unnecessary: remove them.
    309         Node::Scan { table, filter: Some(Constant(Boolean(true))), alias } => {
    310             Node::Scan { table, filter: None, alias }
    311         }
    312         Node::NestedLoopJoin { left, right, predicate: Some(Constant(Boolean(true))), outer } => {
    313             Node::NestedLoopJoin { left, right, predicate: None, outer }
    314         }
    315 
    316         // Short-circuit nodes that can't produce anything by replacing them
    317         // with a Nothing node, retaining the columns.
    318         ref node @ Node::Filter { predicate: Constant(Boolean(false) | Null), .. } => nothing(node),
    319         ref node @ Node::IndexLookup { ref values, .. } if values.is_empty() => nothing(node),
    320         ref node @ Node::KeyLookup { ref keys, .. } if keys.is_empty() => nothing(node),
    321         ref node @ Node::Limit { limit: 0, .. } => nothing(node),
    322         ref node @ Node::NestedLoopJoin {
    323             predicate: Some(Constant(Boolean(false) | Null)), ..
    324         } => nothing(node),
    325         ref node @ Node::Scan { filter: Some(Constant(Boolean(false) | Null)), .. } => {
    326             nothing(node)
    327         }
    328         ref node @ Node::Values { ref rows } if rows.is_empty() => nothing(node),
    329 
    330         // Short-circuit nodes that pull from a Nothing node.
    331         //
    332         // NB: does not short-circuit aggregation, since an aggregation over 0
    333         // rows should produce a result.
    334         ref node @ (Node::Filter { ref source, .. }
    335         | Node::HashJoin { left: ref source, .. }
    336         | Node::HashJoin { right: ref source, .. }
    337         | Node::NestedLoopJoin { left: ref source, .. }
    338         | Node::NestedLoopJoin { right: ref source, .. }
    339         | Node::Offset { ref source, .. }
    340         | Node::Order { ref source, .. }
    341         | Node::Projection { ref source, .. })
    342             if matches!(**source, Node::Nothing { .. }) =>
    343         {
    344             nothing(node)
    345         }
    346 
    347         // Remove noop projections that simply pass through the source columns.
    348         Node::Projection { source, expressions, aliases }
    349             if source.columns() == expressions.len()
    350                 && aliases.iter().all(|alias| *alias == Label::None)
    351                 && expressions
    352                     .iter()
    353                     .enumerate()
    354                     .all(|(i, expr)| matches!(expr, Expression::Column(c) if i == *c)) =>
    355         {
    356             *source
    357         }
    358 
    359         node => node,
    360     };
    361 
    362     // Transform after descending, to pull Nothing nodes upwards.
    363     node.transform(&Ok, &|node| Ok(xform(node)))
    364 }
