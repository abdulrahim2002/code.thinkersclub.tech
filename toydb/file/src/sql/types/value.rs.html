     10 /// A primitive SQL value.
     11 ///
     12 /// For simplicity, only a handful of representative scalar types are supported,
     13 /// no compound types or more compact variants.
     14 ///
     15 /// In SQL, neither Null nor floating point NaN are considered equal to
     16 /// themselves (they are unknown values). However, in code, we consider them
     17 /// equal and comparable. This is necessary to allow sorting and processing of
     18 /// these values (e.g. in index lookups, aggregation buckets, etc.). SQL
     19 /// expression evaluation have special handling of these values to produce the
     20 /// desired NULL != NULL and NAN != NAN semantics in SQL queries.
     21 ///
     22 /// Float -0.0 is considered equal to 0.0. It is normalized to 0.0 when stored.
     23 /// Similarly, -NaN is normalized to NaN.
     24 #[derive(Clone, Debug, Serialize, Deserialize)]
     25 pub enum Value {
     26     /// An unknown value of unknown type.
     27     Null,
     28     /// A boolean.
     29     Boolean(bool),
     30     /// A 64-bit signed integer.
     31     Integer(i64),
     32     /// A 64-bit floating point number.
     33     Float(f64),
     34     /// A UTF-8 encoded string.
     35     String(String),
     36 }

     40 // In code, consider Null and NaN equal, so that we can detect and process these
     41 // values (e.g. in index lookups, aggregation groups, etc). SQL expressions
     42 // handle them specially to provide their undefined value semantics.
     43 impl std::cmp::PartialEq for Value {
     44     fn eq(&self, other: &Self) -> bool {
     45         match (self, other) {
     46             (Self::Boolean(l), Self::Boolean(r)) => l == r,
     47             (Self::Integer(l), Self::Integer(r)) => l == r,
     48             (Self::Float(l), Self::Float(r)) => l == r || l.is_nan() && r.is_nan(),
     49             (Self::String(l), Self::String(r)) => l == r,
     50             (l, r) => core::mem::discriminant(l) == core::mem::discriminant(r),
     51         }
     52     }
     53 }

     57 impl std::hash::Hash for Value {
     58     fn hash<H: std::hash::Hasher>(&self, state: &mut H) {
     59         core::mem::discriminant(self).hash(state);
     60         // Normalize to treat +/-0.0 and +/-NAN as equal when hashing.
     61         match self.normalize_ref().as_ref() {
     62             Self::Null => {}
     63             Self::Boolean(v) => v.hash(state),
     64             Self::Integer(v) => v.hash(state),
     65             Self::Float(v) => v.to_bits().hash(state),
     66             Self::String(v) => v.hash(state),
     67         }
     68     }
     69 }

     71 // For ordering purposes, we consider NULL and NaN equal. We establish a total
     72 // order across all types, even though mixed types will rarely/never come up.
     73 impl Ord for Value {
     74     fn cmp(&self, other: &Self) -> std::cmp::Ordering {
     75         use std::cmp::Ordering::*;
     76         use Value::*;
     77         match (self, other) {
     78             (Null, Null) => Equal,
     79             (Boolean(a), Boolean(b)) => a.cmp(b),
     80             (Integer(a), Integer(b)) => a.cmp(b),
     81             (Integer(a), Float(b)) => (*a as f64).total_cmp(b),
     82             (Float(a), Integer(b)) => a.total_cmp(&(*b as f64)),
     83             (Float(a), Float(b)) => a.total_cmp(b),
     84             (String(a), String(b)) => a.cmp(b),
     85 
     86             (Null, _) => Less,
     87             (_, Null) => Greater,
     88             (Boolean(_), _) => Less,
     89             (_, Boolean(_)) => Greater,
     90             (Float(_), _) => Less,
     91             (_, Float(_)) => Greater,
     92             (Integer(_), _) => Less,
     93             (_, Integer(_)) => Greater,
     94             // String is ordered last.
     95         }
     96     }
     97 }

    106     /// Adds two values. Errors when invalid.
    107     pub fn checked_add(&self, other: &Self) -> Result<Self> {
    108         use Value::*;
    109         Ok(match (self, other) {
    110             (Integer(lhs), Integer(rhs)) => match lhs.checked_add(*rhs) {
    111                 Some(i) => Integer(i),
    112                 None => return errinput!("integer overflow"),
    113             },
    114             (Integer(lhs), Float(rhs)) => Float(*lhs as f64 + rhs),
    115             (Float(lhs), Integer(rhs)) => Float(lhs + *rhs as f64),
    116             (Float(lhs), Float(rhs)) => Float(lhs + rhs),
    117             (Null, Integer(_) | Float(_) | Null) => Null,
    118             (Integer(_) | Float(_), Null) => Null,
    119             (lhs, rhs) => return errinput!("can't add {lhs} and {rhs}"),
    120         })
    121     }

    155     /// Exponentiates two values. Errors when invalid.
    156     pub fn checked_pow(&self, other: &Self) -> Result<Self> {
    157         use Value::*;
    158         Ok(match (self, other) {
    159             (Integer(lhs), Integer(rhs)) if *rhs >= 0 => {
    160                 let rhs = (*rhs).try_into().or_else(|_| errinput!("integer overflow"))?;
    161                 match lhs.checked_pow(rhs) {
    162                     Some(i) => Integer(i),
    163                     None => return errinput!("integer overflow"),
    164                 }
    165             }
    166             (Integer(lhs), Integer(rhs)) => Float((*lhs as f64).powf(*rhs as f64)),
    167             (Integer(lhs), Float(rhs)) => Float((*lhs as f64).powf(*rhs)),
    168             (Float(lhs), Integer(rhs)) => Float((lhs).powi(*rhs as i32)),
    169             (Float(lhs), Float(rhs)) => Float((lhs).powf(*rhs)),
    170             (Integer(_) | Float(_), Null) => Null,
    171             (Null, Integer(_) | Float(_) | Null) => Null,
    172             (lhs, rhs) => return errinput!("can't exponentiate {lhs} and {rhs}"),
    173         })
    174     }

    176     /// Finds the remainder of two values. Errors when invalid.
    177     ///
    178     /// NB: uses the remainder, not modulo, like Postgres. For negative values,
    179     /// the result has the sign of the dividend, rather than always returning a
    180     /// positive value (modulo).
    181     pub fn checked_rem(&self, other: &Self) -> Result<Self> {
    182         use Value::*;
    183         Ok(match (self, other) {
    184             (Integer(_), Integer(0)) => return errinput!("can't divide by zero"),
    185             (Integer(lhs), Integer(rhs)) => Integer(lhs % rhs),
    186             (Integer(lhs), Float(rhs)) => Float(*lhs as f64 % rhs),
    187             (Float(lhs), Integer(rhs)) => Float(lhs % *rhs as f64),
    188             (Float(lhs), Float(rhs)) => Float(lhs % rhs),
    189             (Integer(_) | Float(_) | Null, Null) => Null,
    190             (Null, Integer(_) | Float(_)) => Null,
    191             (lhs, rhs) => return errinput!("can't take remainder of {lhs} and {rhs}"),
    192         })
    193     }

    195     /// Subtracts two values. Errors when invalid.
    196     pub fn checked_sub(&self, other: &Self) -> Result<Self> {
    197         use Value::*;

    205             (Float(lhs), Float(rhs)) => Float(lhs - rhs),

    228     /// Normalizes a value in place. Currently normalizes -0.0 and -NAN to 0.0
    229     /// and NAN respectively, which is the canonical value used e.g. in primary
    230     /// key and index lookups.
    231     pub fn normalize(&mut self) {
    232         if let Cow::Owned(normalized) = self.normalize_ref() {
    233             *self = normalized;
    234         }
    235     }
    236 
    237     /// Normalizes a borrowed value. Currently normalizes -0.0 and -NAN to 0.0
    238     /// and NAN respectively, which is the canonical value used e.g. in primary
    239     /// key and index lookups. Returns a Cow::Owned when changed, to avoid
    240     /// allocating in the common case where the value doesn't change.
    241     pub fn normalize_ref(&self) -> Cow<'_, Self> {
    242         if let Self::Float(f) = self {
    243             if (f.is_nan() || *f == -0.0) && f.is_sign_negative() {
    244                 return Cow::Owned(Self::Float(-f));
    245             }
    246         }
    247         Cow::Borrowed(self)
    248     }
    249 
    250     // Returns true if the value is already normalized.
    251     pub fn is_normalized(&self) -> bool {
    252         matches!(self.normalize_ref(), Cow::Borrowed(_))
    253     }
    254 }

    379 /// A column label, used in query results and plans.

    381 pub enum Label {
    382     /// No label.
    383     None,
    384     /// An unqualified column name.
    385     Unqualified(String),
    386     /// A fully qualified table/column name.
    387     Qualified(String, String),
    388 }

    390 impl std::fmt::Display for Label {
    391     fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
    392         match self {
    393             Self::None => write!(f, ""),
    394             Self::Unqualified(name) => write!(f, "{name}"),
    395             Self::Qualified(table, column) => write!(f, "{table}.{column}"),
    396         }
    397     }
