      8 /// An expression, made up of nested operations and values. Values are either
      9 /// constants or dynamic column references. Evaluates to a final value during
     10 /// query execution, using row values for column references.
     11 ///
     12 /// Since this is a recursive data structure, we have to box each child
     13 /// expression, which incurs a heap allocation per expression node. There are
     14 /// clever ways to avoid this, but we keep it simple.
     15 #[derive(Clone, Debug, PartialEq, Serialize, Deserialize)]
     16 pub enum Expression {
     17     /// A constant value.
     18     Constant(Value),
     19     /// A column reference. Used as row index when evaluating expressions.
     20     Column(usize),
     21 
     22     /// Logical AND of two booleans: a AND b.
     23     And(Box<Expression>, Box<Expression>),
     24     /// Logical OR of two booleans: a OR b.
     25     Or(Box<Expression>, Box<Expression>),
     26     /// Logical NOT of a boolean: NOT a.
     27     Not(Box<Expression>),
     28 
     29     /// Equality comparison of two values: a = b.
     30     Equal(Box<Expression>, Box<Expression>),
     31     /// Greater than comparison of two values: a > b.
     32     GreaterThan(Box<Expression>, Box<Expression>),
     33     /// Less than comparison of two values: a < b.
     34     LessThan(Box<Expression>, Box<Expression>),
     35     /// Checks for the given value: IS NULL or IS NAN.
     36     Is(Box<Expression>, Value),
     37 
     38     /// Adds two numbers: a + b.
     39     Add(Box<Expression>, Box<Expression>),
     40     /// Divides two numbers: a / b.
     41     Divide(Box<Expression>, Box<Expression>),
     42     /// Exponentiates two numbers, i.e. a ^ b.
     43     Exponentiate(Box<Expression>, Box<Expression>),
     44     /// Takes the factorial of a number: 4! = 4*3*2*1.
     45     Factorial(Box<Expression>),
     46     /// The identify function, which simply returns the same number: +a.
     47     Identity(Box<Expression>),
     48     /// Multiplies two numbers: a * b.
     49     Multiply(Box<Expression>, Box<Expression>),
     50     /// Negates the given number: -a.
     51     Negate(Box<Expression>),
     52     /// The remainder after dividing two numbers: a % b.
     53     Remainder(Box<Expression>, Box<Expression>),
     54     /// Takes the square root of a number: √a.
     55     SquareRoot(Box<Expression>),
     56     /// Subtracts two numbers: a - b.
     57     Subtract(Box<Expression>, Box<Expression>),
     58 
     59     // Checks if a string matches a pattern: a LIKE b.
     60     Like(Box<Expression>, Box<Expression>),
     61 }

     64     /// Formats the expression, using the given plan node to look up labels for
     65     /// numeric column references.
     66     pub fn format(&self, node: &Node) -> String {
     67         use Expression::*;
     68 
     69         // Precedence levels, for grouping. Matches the parser precedence.
     70         fn precedence(expr: &Expression) -> u8 {
     71             match expr {
     72                 Column(_) | Constant(_) | SquareRoot(_) => 11,
     73                 Identity(_) | Negate(_) => 10,
     74                 Factorial(_) => 9,
     75                 Exponentiate(_, _) => 8,
     76                 Multiply(_, _) | Divide(_, _) | Remainder(_, _) => 7,
     77                 Add(_, _) | Subtract(_, _) => 6,
     78                 GreaterThan(_, _) | LessThan(_, _) => 5,
     79                 Equal(_, _) | Like(_, _) | Is(_, _) => 4,
     80                 Not(_) => 3,
     81                 And(_, _) => 2,
     82                 Or(_, _) => 1,
     83             }
     84         }
     85 
     86         // Helper to format a boxed expression, grouping it with () if needed.
     87         let format = |expr: &Expression| {
     88             let mut string = expr.format(node);
     89             if precedence(expr) < precedence(self) {
     90                 string = format!("({string})");
     91             }
     92             string
     93         };
     94 
     95         match self {
     96             Constant(value) => format!("{value}"),
     97             Column(index) => match node.column_label(*index) {
     98                 Label::None => format!("#{index}"),
     99                 label => format!("{label}"),
    100             },
    101 
    102             And(lhs, rhs) => format!("{} AND {}", format(lhs), format(rhs)),
    103             Or(lhs, rhs) => format!("{} OR {}", format(lhs), format(rhs)),
    104             Not(expr) => format!("NOT {}", format(expr)),
    105 
    106             Equal(lhs, rhs) => format!("{} = {}", format(lhs), format(rhs)),
    107             GreaterThan(lhs, rhs) => format!("{} > {}", format(lhs), format(rhs)),
    108             LessThan(lhs, rhs) => format!("{} < {}", format(lhs), format(rhs)),
    109             Is(expr, Value::Null) => format!("{} IS NULL", format(expr)),
    110             Is(expr, Value::Float(f)) if f.is_nan() => format!("{} IS NAN", format(expr)),
    111             Is(_, v) => panic!("unexpected IS value {v}"),
    112 
    113             Add(lhs, rhs) => format!("{} + {}", format(lhs), format(rhs)),
    114             Divide(lhs, rhs) => format!("{} / {}", format(lhs), format(rhs)),
    115             Exponentiate(lhs, rhs) => format!("{} ^ {}", format(lhs), format(rhs)),
    116             Factorial(expr) => format!("{}!", format(expr)),
    117             Identity(expr) => format(expr),
    118             Multiply(lhs, rhs) => format!("{} * {}", format(lhs), format(rhs)),
    119             Negate(expr) => format!("-{}", format(expr)),
    120             Remainder(lhs, rhs) => format!("{} % {}", format(lhs), format(rhs)),
    121             SquareRoot(expr) => format!("sqrt({})", format(expr)),
    122             Subtract(lhs, rhs) => format!("{} - {}", format(lhs), format(rhs)),
    123 
    124             Like(lhs, rhs) => format!("{} LIKE {}", format(lhs), format(rhs)),
    125         }
    126     }

    133     /// Evaluates an expression, returning a value. Column references look up
    134     /// values in the given row. If None, any Column references will panic.
    135     pub fn evaluate(&self, row: Option<&Row>) -> Result<Value> {
    136         use Value::*;
    137         Ok(match self {
    138             // Constant values return themselves.
    139             Self::Constant(value) => value.clone(),
    140 
    141             // Column references look up a row value. The planner ensures that
    142             // only constant expressions are evaluated without a row.
    143             Self::Column(index) => match row {
    144                 Some(row) => row.get(*index).expect("short row").clone(),
    145                 None => panic!("can't reference column {index} with constant evaluation"),
    146             },
    147 
    148             // Logical AND. Inputs must be boolean or NULL. NULLs generally
    149             // yield NULL, except the special case NULL AND false == false.
    150             Self::And(lhs, rhs) => match (lhs.evaluate(row)?, rhs.evaluate(row)?) {
    151                 (Boolean(lhs), Boolean(rhs)) => Boolean(lhs && rhs),
    152                 (Boolean(b), Null) | (Null, Boolean(b)) if !b => Boolean(false),
    153                 (Boolean(_), Null) | (Null, Boolean(_)) | (Null, Null) => Null,
    154                 (lhs, rhs) => return errinput!("can't AND {lhs} and {rhs}"),
    155             },
    156 
    157             // Logical OR. Inputs must be boolean or NULL. NULLs generally
    158             // yield NULL, except the special case NULL OR true == true.
    159             Self::Or(lhs, rhs) => match (lhs.evaluate(row)?, rhs.evaluate(row)?) {
    160                 (Boolean(lhs), Boolean(rhs)) => Boolean(lhs || rhs),
    161                 (Boolean(b), Null) | (Null, Boolean(b)) if b => Boolean(true),
    162                 (Boolean(_), Null) | (Null, Boolean(_)) | (Null, Null) => Null,
    163                 (lhs, rhs) => return errinput!("can't OR {lhs} and {rhs}"),
    164             },
    165 
    166             // Logical NOT. Input must be boolean or NULL.
    167             Self::Not(expr) => match expr.evaluate(row)? {
    168                 Boolean(b) => Boolean(!b),
    169                 Null => Null,
    170                 value => return errinput!("can't NOT {value}"),
    171             },
    172 
    173             // Comparisons. Must be of same type, except floats and integers
    174             // which are interchangeable. NULLs yield NULL, NaNs yield NaN.
    175             //
    176             // Does not dispatch to Value.cmp() because sorting and comparisons
    177             // are different for f64 NaN and -0.0 values.
    178             #[allow(clippy::float_cmp)]
    179             Self::Equal(lhs, rhs) => match (lhs.evaluate(row)?, rhs.evaluate(row)?) {
    180                 (Boolean(lhs), Boolean(rhs)) => Boolean(lhs == rhs),
    181                 (Integer(lhs), Integer(rhs)) => Boolean(lhs == rhs),
    182                 (Integer(lhs), Float(rhs)) => Boolean(lhs as f64 == rhs),
    183                 (Float(lhs), Integer(rhs)) => Boolean(lhs == rhs as f64),
    184                 (Float(lhs), Float(rhs)) => Boolean(lhs == rhs),
    185                 (String(lhs), String(rhs)) => Boolean(lhs == rhs),
    186                 (Null, _) | (_, Null) => Null,
    187                 (lhs, rhs) => return errinput!("can't compare {lhs} and {rhs}"),
    188             },
    189 
    190             Self::GreaterThan(lhs, rhs) => match (lhs.evaluate(row)?, rhs.evaluate(row)?) {
    191                 #[allow(clippy::bool_comparison)]
    192                 (Boolean(lhs), Boolean(rhs)) => Boolean(lhs > rhs),
    193                 (Integer(lhs), Integer(rhs)) => Boolean(lhs > rhs),
    194                 (Integer(lhs), Float(rhs)) => Boolean(lhs as f64 > rhs),
    195                 (Float(lhs), Integer(rhs)) => Boolean(lhs > rhs as f64),
    196                 (Float(lhs), Float(rhs)) => Boolean(lhs > rhs),
    197                 (String(lhs), String(rhs)) => Boolean(lhs > rhs),
    198                 (Null, _) | (_, Null) => Null,
    199                 (lhs, rhs) => return errinput!("can't compare {lhs} and {rhs}"),
    200             },
    201 
    202             Self::LessThan(lhs, rhs) => match (lhs.evaluate(row)?, rhs.evaluate(row)?) {
    203                 #[allow(clippy::bool_comparison)]
    204                 (Boolean(lhs), Boolean(rhs)) => Boolean(lhs < rhs),
    205                 (Integer(lhs), Integer(rhs)) => Boolean(lhs < rhs),
    206                 (Integer(lhs), Float(rhs)) => Boolean((lhs as f64) < rhs),
    207                 (Float(lhs), Integer(rhs)) => Boolean(lhs < rhs as f64),
    208                 (Float(lhs), Float(rhs)) => Boolean(lhs < rhs),
    209                 (String(lhs), String(rhs)) => Boolean(lhs < rhs),
    210                 (Null, _) | (_, Null) => Null,
    211                 (lhs, rhs) => return errinput!("can't compare {lhs} and {rhs}"),
    212             },
    213 
    214             Self::Is(expr, Null) => Boolean(expr.evaluate(row)? == Null),
    215             Self::Is(expr, Float(f)) if f.is_nan() => match expr.evaluate(row)? {
    216                 Float(f) => Boolean(f.is_nan()),
    217                 Null => Null,
    218                 v => return errinput!("IS NAN can't be used with {}", v.datatype().unwrap()),
    219             },
    220             Self::Is(_, v) => panic!("invalid IS value {v}"), // enforced by parser
    221 
    222             // Mathematical operations. Inputs must be numbers, but integers and
    223             // floats are interchangeable (float when mixed). NULLs yield NULL.
    224             // Errors on integer overflow, while floats yield infinity or NaN.
    225             Self::Add(lhs, rhs) => lhs.evaluate(row)?.checked_add(&rhs.evaluate(row)?)?,
    226             Self::Divide(lhs, rhs) => lhs.evaluate(row)?.checked_div(&rhs.evaluate(row)?)?,
    227             Self::Exponentiate(lhs, rhs) => lhs.evaluate(row)?.checked_pow(&rhs.evaluate(row)?)?,
    228             Self::Factorial(expr) => match expr.evaluate(row)? {
    229                 Integer(i) if i < 0 => return errinput!("can't take factorial of negative number"),
    230                 Integer(i) => (1..=i).try_fold(Integer(1), |p, i| p.checked_mul(&Integer(i)))?,
    231                 Null => Null,
    232                 value => return errinput!("can't take factorial of {value}"),
    233             },
    234             Self::Identity(expr) => match expr.evaluate(row)? {
    235                 v @ (Integer(_) | Float(_) | Null) => v,
    236                 expr => return errinput!("can't take the identity of {expr}"),
    237             },
    238             Self::Multiply(lhs, rhs) => lhs.evaluate(row)?.checked_mul(&rhs.evaluate(row)?)?,
    239             Self::Negate(expr) => match expr.evaluate(row)? {
    240                 Integer(i) => Integer(-i),
    241                 Float(f) => Float(-f),
    242                 Null => Null,
    243                 value => return errinput!("can't negate {value}"),
    244             },
    245             Self::Remainder(lhs, rhs) => lhs.evaluate(row)?.checked_rem(&rhs.evaluate(row)?)?,
    246             Self::SquareRoot(expr) => match expr.evaluate(row)? {
    247                 Integer(i) if i < 0 => return errinput!("can't take negative square root"),
    248                 Integer(i) => Float((i as f64).sqrt()),
    249                 Float(f) => Float(f.sqrt()),
    250                 Null => Null,
    251                 value => return errinput!("can't take square root of {value}"),
    252             },
    253             Self::Subtract(lhs, rhs) => lhs.evaluate(row)?.checked_sub(&rhs.evaluate(row)?)?,
    254 
    255             // LIKE pattern matching, using _ and % as single- and
    256             // multi-character wildcards. Inputs must be strings. NULLs yield
    257             // NULL. There's no support for escaping an _ and %.
    258             Self::Like(lhs, rhs) => match (lhs.evaluate(row)?, rhs.evaluate(row)?) {
    259                 (String(lhs), String(rhs)) => {
    260                     // We could precompile the pattern if it's constant, instead
    261                     // of recompiling it for every row, but this is fine.
    262                     let pattern =
    263                         format!("^{}$", regex::escape(&rhs).replace('%', ".*").replace('_', "."));
    264                     Boolean(regex::Regex::new(&pattern)?.is_match(&lhs))
    265                 }
    266                 (String(_), Null) | (Null, String(_)) | (Null, Null) => Null,
    267                 (lhs, rhs) => return errinput!("can't LIKE {lhs} and {rhs}"),
    268             },
    269         })
    270     }

    272     /// Recursively walks the expression tree depth-first, calling the given
    273     /// closure until it returns false. Returns true otherwise.
    274     pub fn walk(&self, visitor: &mut impl FnMut(&Expression) -> bool) -> bool {
    275         if !visitor(self) {
    276             return false;
    277         }
    278         match self {
    279             Self::Add(lhs, rhs)
    280             | Self::And(lhs, rhs)
    281             | Self::Divide(lhs, rhs)
    282             | Self::Equal(lhs, rhs)
    283             | Self::Exponentiate(lhs, rhs)
    284             | Self::GreaterThan(lhs, rhs)
    285             | Self::LessThan(lhs, rhs)
    286             | Self::Like(lhs, rhs)
    287             | Self::Multiply(lhs, rhs)
    288             | Self::Or(lhs, rhs)
    289             | Self::Remainder(lhs, rhs)
    290             | Self::Subtract(lhs, rhs) => lhs.walk(visitor) && rhs.walk(visitor),
    291 
    292             Self::Factorial(expr)
    293             | Self::Identity(expr)
    294             | Self::Is(expr, _)
    295             | Self::Negate(expr)
    296             | Self::Not(expr)
    297             | Self::SquareRoot(expr) => expr.walk(visitor),
    298 
    299             Self::Constant(_) | Self::Column(_) => true,
    300         }
    301     }

    303     /// Recursively walks the expression tree depth-first, calling the given
    304     /// closure until it returns true. Returns false otherwise. This is the
    305     /// inverse of walk().
    306     pub fn contains(&self, visitor: &impl Fn(&Expression) -> bool) -> bool {
    307         !self.walk(&mut |e| !visitor(e))
    308     }

    310     /// Transforms the expression by recursively applying the given closures
    311     /// depth-first to each node before/after descending.
    312     pub fn transform(
    313         mut self,
    314         before: &impl Fn(Self) -> Result<Self>,
    315         after: &impl Fn(Self) -> Result<Self>,
    316     ) -> Result<Self> {
    317         // Helper for transforming boxed expressions.
    318         let xform = |mut expr: Box<Expression>| -> Result<Box<Expression>> {
    319             *expr = expr.transform(before, after)?;
    320             Ok(expr)
    321         };
    322 
    323         self = before(self)?;
    324         self = match self {
    325             Self::Add(lhs, rhs) => Self::Add(xform(lhs)?, xform(rhs)?),
    326             Self::And(lhs, rhs) => Self::And(xform(lhs)?, xform(rhs)?),
    327             Self::Divide(lhs, rhs) => Self::Divide(xform(lhs)?, xform(rhs)?),
    328             Self::Equal(lhs, rhs) => Self::Equal(xform(lhs)?, xform(rhs)?),
    329             Self::Exponentiate(lhs, rhs) => Self::Exponentiate(xform(lhs)?, xform(rhs)?),
    330             Self::GreaterThan(lhs, rhs) => Self::GreaterThan(xform(lhs)?, xform(rhs)?),
    331             Self::LessThan(lhs, rhs) => Self::LessThan(xform(lhs)?, xform(rhs)?),
    332             Self::Like(lhs, rhs) => Self::Like(xform(lhs)?, xform(rhs)?),
    333             Self::Multiply(lhs, rhs) => Self::Multiply(xform(lhs)?, xform(rhs)?),
    334             Self::Or(lhs, rhs) => Self::Or(xform(lhs)?, xform(rhs)?),
    335             Self::Remainder(lhs, rhs) => Self::Remainder(xform(lhs)?, xform(rhs)?),
    336             Self::SquareRoot(expr) => Self::SquareRoot(xform(expr)?),
    337             Self::Subtract(lhs, rhs) => Self::Subtract(xform(lhs)?, xform(rhs)?),
    338 
    339             Self::Factorial(expr) => Self::Factorial(xform(expr)?),
    340             Self::Identity(expr) => Self::Identity(xform(expr)?),
    341             Self::Is(expr, value) => Self::Is(xform(expr)?, value),
    342             Self::Negate(expr) => Self::Negate(xform(expr)?),
    343             Self::Not(expr) => Self::Not(xform(expr)?),
    344 
    345             expr @ (Self::Constant(_) | Self::Column(_)) => expr,
    346         };
    347         self = after(self)?;
    348         Ok(self)
    349     }

    351     /// Converts the expression into conjunctive normal form, i.e. an AND of
    352     /// ORs, which is useful when optimizing plans. This is done by converting
    353     /// to negation normal form and then applying De Morgan's distributive law.
    354     pub fn into_cnf(self) -> Self {
    355         use Expression::*;
    356         let xform = |expr| {
    357             // We can't use a single match, since it needs deref patterns.
    358             let Or(lhs, rhs) = expr else { return expr };
    359             match (*lhs, *rhs) {
    360                 // (x AND y) OR z → (x OR z) AND (y OR z)
    361                 (And(l, r), rhs) => And(Or(l, rhs.clone().into()).into(), Or(r, rhs.into()).into()),
    362                 // x OR (y AND z) → (x OR y) AND (x OR z)
    363                 (lhs, And(l, r)) => And(Or(lhs.clone().into(), l).into(), Or(lhs.into(), r).into()),
    364                 // Otherwise, do nothing.
    365                 (lhs, rhs) => Or(lhs.into(), rhs.into()),
    366             }
    367         };
    368         self.into_nnf().transform(&|e| Ok(xform(e)), &Ok).unwrap() // infallible
    369     }

    371     /// Converts the expression into negation normal form. This pushes NOT
    372     /// operators into the tree using De Morgan's laws, such that they're always
    373     /// below other logical operators. It is a useful intermediate form for
    374     /// applying other logical normalizations.
    375     pub fn into_nnf(self) -> Self {
    376         use Expression::*;
    377         let xform = |expr| {
    378             let Not(inner) = expr else { return expr };
    379             match *inner {
    380                 // NOT (x AND y) → (NOT x) OR (NOT y)
    381                 And(lhs, rhs) => Or(Not(lhs).into(), Not(rhs).into()),
    382                 // NOT (x OR y) → (NOT x) AND (NOT y)
    383                 Or(lhs, rhs) => And(Not(lhs).into(), Not(rhs).into()),
    384                 // NOT NOT x → x
    385                 Not(inner) => *inner,
    386                 // Otherwise, do nothing.
    387                 expr => Not(expr.into()),
    388             }
    389         };
    390         self.transform(&|e| Ok(xform(e)), &Ok).unwrap() // never fails
    391     }

    393     /// Converts the expression into conjunctive normal form as a vector of
    394     /// ANDed expressions (instead of nested ANDs).
    395     pub fn into_cnf_vec(self) -> Vec<Self> {
    396         let mut cnf = Vec::new();
    397         let mut stack = vec![self.into_cnf()];
    398         while let Some(expr) = stack.pop() {
    399             if let Self::And(lhs, rhs) = expr {
    400                 stack.extend([*rhs, *lhs]); // push lhs last to pop it first
    401             } else {
    402                 cnf.push(expr);
    403             }
    404         }
    405         cnf
    406     }

    408     /// Creates an expression by ANDing together a vector, or None if empty.
    409     pub fn and_vec(exprs: Vec<Expression>) -> Option<Self> {
    410         let mut iter = exprs.into_iter();
    411         let mut expr = iter.next()?;
    412         for rhs in iter {
    413             expr = Expression::And(expr.into(), rhs.into());
    414         }
    415         Some(expr)
    416     }

    418     /// Checks if an expression is a single column lookup (i.e. a disjunction of
    419     /// = or IS NULL/NAN for a single column), returning the column index.
    420     pub fn is_column_lookup(&self) -> Option<usize> {
    421         use Expression::*;
    422         match &self {
    423             // Column/constant equality can use index lookups. NULL and NaN are
    424             // handled in into_column_values().
    425             Equal(lhs, rhs) => match (lhs.as_ref(), rhs.as_ref()) {
    426                 (Column(c), Constant(_)) | (Constant(_), Column(c)) => Some(*c),
    427                 _ => None,
    428             },
    429             // IS NULL and IS NAN can use index lookups.
    430             Is(expr, _) => match expr.as_ref() {
    431                 Column(c) => Some(*c),
    432                 _ => None,
    433             },
    434             // All OR branches must be lookups on the same column:
    435             // id = 1 OR id = 2 OR id = 3.
    436             Or(lhs, rhs) => match (lhs.is_column_lookup(), rhs.is_column_lookup()) {
    437                 (Some(l), Some(r)) if l == r => Some(l),
    438                 _ => None,
    439             },
    440             _ => None,
    441         }
    442     }

    444     /// Extracts column lookup values for the given column. Panics if the
    445     /// expression isn't a lookup of the given column, i.e. is_column_lookup()
    446     /// must return true for the expression.
    447     pub fn into_column_values(self, index: usize) -> Vec<Value> {
    448         use Expression::*;
    449         match self {
    450             Equal(lhs, rhs) => match (*lhs, *rhs) {
    451                 (Column(column), Constant(value)) | (Constant(value), Column(column)) => {
    452                     assert_eq!(column, index, "unexpected column");
    453                     // NULL and NAN index lookups are for IS NULL and IS NAN.
    454                     // Equality shouldn't match anything, return empty vec.
    455                     if value.is_undefined() {
    456                         Vec::new()
    457                     } else {
    458                         vec![value]
    459                     }
    460                 }
    461                 (lhs, rhs) => panic!("unexpected expression {:?}", Equal(lhs.into(), rhs.into())),
    462             },
    463             // IS NULL and IS NAN can use index lookups.
    464             Is(expr, value) => match *expr {
    465                 Column(column) => {
    466                     assert_eq!(column, index, "unexpected column");
    467                     vec![value]
    468                 }
    469                 expr => panic!("unexpected expression {expr:?}"),
    470             },
    471             Or(lhs, rhs) => {
    472                 let mut values = lhs.into_column_values(index);
    473                 values.extend(rhs.into_column_values(index));
    474                 values
    475             }
    476             expr => panic!("unexpected expression {expr:?}"),
    477         }
    478     }

    480     /// Replaces column references with the given column.
    481     pub fn replace_column(self, from: usize, to: usize) -> Self {
    482         let xform = |expr| match expr {
    483             Expression::Column(i) if i == from => Expression::Column(to),
    484             expr => expr,
    485         };
    486         self.transform(&|e| Ok(xform(e)), &Ok).unwrap() // infallible
    487     }
    488 
    489     /// Shifts column references by the given amount.
    490     pub fn shift_column(self, diff: isize) -> Self {
    491         let xform = |expr| match expr {
    492             Expression::Column(i) => Expression::Column((i as isize + diff) as usize),
    493             expr => expr,
    494         };
    495         self.transform(&|e| Ok(xform(e)), &Ok).unwrap() // infallible
    496     }
    497 }
    498 
    499 impl From<Value> for Expression {
    500     fn from(value: Value) -> Self {
    501         Expression::Constant(value)
    502     }
    503 }
    504 
    505 impl From<Value> for Box<Expression> {
    506     fn from(value: Value) -> Self {
    507         Box::new(value.into())
    508     }
