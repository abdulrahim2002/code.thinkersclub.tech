     10 /// A table schema, which specifies its data structure and constraints.
     11 ///
     12 /// Tables can't change after they are created. There is no ALTER TABLE nor
     13 /// CREATE/DROP INDEX -- only CREATE TABLE and DROP TABLE.
     14 #[derive(Clone, Debug, PartialEq, Deserialize, Serialize)]
     15 pub struct Table {
     16     /// The table name. Can't be empty.
     17     pub name: String,
     18     /// The primary key column index. A table must have a primary key, and it
     19     /// can only be a single column.
     20     pub primary_key: usize,
     21     /// The table's columns. Must have at least one.
     22     pub columns: Vec<Column>,
     23 }

     25 impl encoding::Value for Table {}

     27 /// A table column.
     28 #[derive(Clone, Debug, PartialEq, Deserialize, Serialize)]
     29 pub struct Column {
     30     /// Column name. Can't be empty.
     31     pub name: String,
     32     /// Column datatype.
     33     pub datatype: DataType,
     34     /// Whether the column allows null values. Not legal for primary keys.
     35     pub nullable: bool,
     36     /// The column's default value. If None, the user must specify an explicit
     37     /// value. Must match the column datatype. Nullable columns require a
     38     /// default (often Null), and Null is only a valid default when nullable.
     39     pub default: Option<Value>,
     40     /// Whether the column should only allow unique values (ignoring NULLs).
     41     /// Must be true for a primary key column.
     42     pub unique: bool,
     43     /// Whether the column should have a secondary index. Must be false for
     44     /// primary keys, which are the implicit primary index. Must be true for
     45     /// unique or reference columns.
     46     pub index: bool,
     47     /// If set, this column is a foreign key reference to the given table's
     48     /// primary key. Must be of the same type as the target primary key.
     49     pub references: Option<String>,
     50 }

     52 impl std::fmt::Display for Table {
     53     fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
     54         writeln!(f, "CREATE TABLE {} (", format_ident(&self.name))?;
     55         for (i, column) in self.columns.iter().enumerate() {
     56             write!(f, "  {} {}", format_ident(&column.name), column.datatype)?;
     57             if i == self.primary_key {
     58                 write!(f, " PRIMARY KEY")?;
     59             } else if !column.nullable {
     60                 write!(f, " NOT NULL")?;
     61             }
     62             if let Some(default) = &column.default {
     63                 write!(f, " DEFAULT {default}")?;
     64             }
     65             if i != self.primary_key {
     66                 if column.unique {
     67                     write!(f, " UNIQUE")?;
     68                 }
     69                 if column.index {
     70                     write!(f, " INDEX")?;
     71                 }
     72             }
     73             if let Some(reference) = &column.references {
     74                 write!(f, " REFERENCES {reference}")?;
     75             }
     76             if i < self.columns.len() - 1 {
     77                 write!(f, ",")?;
     78             }
     79             writeln!(f)?;
     80         }
     81         write!(f, ")")
     82     }
     83 }

     85 impl Table {
     86     /// Validates the table schema, using the catalog to validate foreign key
     87     /// references.
     88     pub fn validate(&self, catalog: &impl Catalog) -> Result<()> {
     89         if self.name.is_empty() {
     90             return errinput!("table name can't be empty");
     91         }
     92         if self.columns.is_empty() {
     93             return errinput!("table has no columns");
     94         }
     95         if self.columns.get(self.primary_key).is_none() {
     96             return errinput!("invalid primary key index");
     97         }
     98 
     99         for (i, column) in self.columns.iter().enumerate() {
    100             if column.name.is_empty() {
    101                 return errinput!("column name can't be empty");
    102             }
    103             let (cname, ctype) = (&column.name, &column.datatype); // for formatting convenience
    104 
    105             // Validate primary key.
    106             let is_primary_key = i == self.primary_key;
    107             if is_primary_key {
    108                 if column.nullable {
    109                     return errinput!("primary key {cname} cannot be nullable");
    110                 }
    111                 if !column.unique {
    112                     return errinput!("primary key {cname} must be unique");
    113                 }
    114                 if column.index {
    115                     return errinput!("primary key {cname} can't have an index");
    116                 }
    117             }
    118 
    119             // Validate default value.
    120             match column.default.as_ref().map(|v| v.datatype()) {
    121                 None if column.nullable => {
    122                     return errinput!("nullable column {cname} must have a default value")
    123                 }
    124                 Some(None) if !column.nullable => {
    125                     return errinput!("invalid NULL default for non-nullable column {cname}")
    126                 }
    127                 Some(Some(vtype)) if vtype != column.datatype => {
    128                     return errinput!("invalid default type {vtype} for {ctype} column {cname}");
    129                 }
    130                 Some(_) | None => {}
    131             }
    132 
    133             // Validate unique index.
    134             if column.unique && !column.index && !is_primary_key {
    135                 return errinput!("unique column {cname} must have a secondary index");
    136             }
    137 
    138             // Validate references.
    139             if let Some(reference) = &column.references {
    140                 if !column.index && !is_primary_key {
    141                     return errinput!("reference column {cname} must have a secondary index");
    142                 }
    143                 let reftype = if reference == &self.name {
    144                     self.columns[self.primary_key].datatype
    145                 } else if let Some(target) = catalog.get_table(reference)? {
    146                     target.columns[target.primary_key].datatype
    147                 } else {
    148                     return errinput!("unknown table {reference} referenced by column {cname}");
    149                 };
    150                 if column.datatype != reftype {
    151                     return errinput!("can't reference {reftype} primary key of {reference} from {ctype} column {cname}");
    152                 }
    153             }
    154         }
    155         Ok(())
    156     }
    157 
    158     /// Validates a row, including uniqueness and reference checks using the
    159     /// given transaction.
    160     ///
    161     /// If update is true, the row replaces an existing entry with the same
    162     /// primary key. Otherwise, it is an insert. Primary key changes are
    163     /// implemented as a delete+insert.
    164     ///
    165     /// Validating uniqueness and references individually for each row is not
    166     /// performant, but it's fine for our purposes.
    167     pub fn validate_row(&self, row: &[Value], update: bool, txn: &impl Transaction) -> Result<()> {
    168         if row.len() != self.columns.len() {
    169             return errinput!("invalid row size for table {}", self.name);
    170         }
    171 
    172         // Validate primary key.
    173         let id = &row[self.primary_key];
    174         let idslice = &row[self.primary_key..=self.primary_key];
    175         if id.is_undefined() {
    176             return errinput!("invalid primary key {id}");
    177         }
    178         if !update && !txn.get(&self.name, idslice)?.is_empty() {
    179             return errinput!("primary key {id} already exists");
    180         }
    181 
    182         for (i, (column, value)) in self.columns.iter().zip(row).enumerate() {
    183             let (cname, ctype) = (&column.name, &column.datatype);
    184             let valueslice = &row[i..=i];
    185 
    186             // Validate datatype.
    187             if let Some(ref vtype) = value.datatype() {
    188                 if vtype != ctype {
    189                     return errinput!("invalid datatype {vtype} for {ctype} column {cname}");
    190                 }
    191             }
    192             if value == &Value::Null && !column.nullable {
    193                 return errinput!("NULL value not allowed for column {cname}");
    194             }
    195 
    196             // Validate outgoing references.
    197             if let Some(target) = &column.references {
    198                 match value {
    199                     // NB: NaN is not a valid primary key, and not valid as a
    200                     // missing foreign key marker.
    201                     Value::Null => {}
    202                     v if target == &self.name && v == id => {}
    203                     v if txn.get(target, valueslice)?.is_empty() => {
    204                         return errinput!("reference {v} not in table {target}");
    205                     }
    206                     _ => {}
    207                 }
    208             }
    209 
    210             // Validate uniqueness constraints. Unique columns are indexed.
    211             if column.unique && i != self.primary_key && !value.is_undefined() {
    212                 let mut index = txn.lookup_index(&self.name, &column.name, valueslice)?;
    213                 if update {
    214                     index.remove(id); // ignore existing version of this row
    215                 }
    216                 if !index.is_empty() {
    217                     return errinput!("value {value} already in unique column {cname}");
    218                 }
    219             }
    220         }
    221         Ok(())
    222     }
    223 }
