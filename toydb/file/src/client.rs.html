     12 /// A toyDB client. Connects to a server via TCP and submits SQL statements and
     13 /// other requests.
     14 pub struct Client {
     15     /// Inbound response stream.
     16     reader: std::io::BufReader<std::net::TcpStream>,
     17     /// Outbound request stream.
     18     writer: std::io::BufWriter<std::net::TcpStream>,
     19     /// The current transaction, if any.
     20     txn: Option<mvcc::TransactionState>,
     21 }
     22 
     23 impl Client {
     24     /// Connects to a toyDB server, creating a new client.
     25     pub fn connect(addr: impl std::net::ToSocketAddrs) -> Result<Self> {
     26         let socket = std::net::TcpStream::connect(addr)?;
     27         let reader = std::io::BufReader::new(socket.try_clone()?);
     28         let writer = std::io::BufWriter::new(socket);
     29         Ok(Self { reader, writer, txn: None })
     30     }
     31 
     32     /// Sends a request to the server, returning the response.
     33     fn request(&mut self, request: Request) -> Result<Response> {
     34         request.encode_into(&mut self.writer)?;
     35         self.writer.flush()?;
     36         Result::decode_from(&mut self.reader)?
     37     }
     38 
     39     /// Executes a SQL statement.
     40     pub fn execute(&mut self, statement: &str) -> Result<StatementResult> {
     41         let result = match self.request(Request::Execute(statement.to_string()))? {
     42             Response::Execute(result) => result,
     43             response => return errdata!("unexpected response {response:?}"),
     44         };
     45         // Update the transaction state.
     46         match &result {
     47             StatementResult::Begin(state) => self.txn = Some(state.clone()),
     48             StatementResult::Commit { .. } => self.txn = None,
     49             StatementResult::Rollback { .. } => self.txn = None,
     50             _ => {}
     51         }
     52         Ok(result)
     53     }
     54 
     55     /// Fetches a table schema.
     56     pub fn get_table(&mut self, table: &str) -> Result<Table> {
     57         match self.request(Request::GetTable(table.to_string()))? {
     58             Response::GetTable(table) => Ok(table),
     59             response => errdata!("unexpected response: {response:?}"),
     60         }
     61     }

     84     /// Runs the given closure, automatically retrying serialization and abort
     85     /// errors. If a transaction is open following an error, it is automatically
     86     /// rolled back. It is the caller's responsibility to use a transaction in
     87     /// the closure where appropriate (i.e. when it is not idempotent).
     88     pub fn with_retry<T>(&mut self, f: impl Fn(&mut Client) -> Result<T>) -> Result<T> {
     89         const MAX_RETRIES: u32 = 10;
     90         const MIN_WAIT: u64 = 10;
     91         const MAX_WAIT: u64 = 2_000;
     92         let mut retries: u32 = 0;
     93         loop {
     94             match f(self) {
     95                 Ok(result) => return Ok(result),
     96                 Err(Error::Serialization | Error::Abort) if retries < MAX_RETRIES => {
     97                     if self.txn().is_some() {
     98                         self.execute("ROLLBACK")?;
     99                     }
    100                     // Use exponential backoff starting at MIN_WAIT doubling up
    101                     // to MAX_WAIT, but randomize the wait time in this interval
    102                     // to reduce the chance of collisions.
    103                     let mut wait = std::cmp::min(MIN_WAIT * 2_u64.pow(retries), MAX_WAIT);
    104                     wait = rand::thread_rng().gen_range(MIN_WAIT..=wait);
    105                     std::thread::sleep(std::time::Duration::from_millis(wait));
    106                     retries += 1;
    107                 }
    108                 Err(error) => {
    109                     if self.txn().is_some() {
    110                         self.execute("ROLLBACK").ok(); // ignore rollback error
    111                     }
    112                     return Err(error);
    113                 }
    114             }
    115         }
    116     }
