      3 /// toyDB errors.
      4 #[derive(Clone, Debug, PartialEq, Serialize, Deserialize)]
      5 pub enum Error {
      6     /// The operation was aborted and must be retried. This typically happens
      7     /// with e.g. Raft leader changes. This is used instead of implementing
      8     /// complex retry logic and replay protection in Raft.
      9     Abort,
     10     /// Invalid data, typically decoding errors or unexpected internal values.
     11     InvalidData(String),
     12     /// Invalid user input, typically parser or query errors.
     13     InvalidInput(String),
     14     /// An IO error.
     15     IO(String),
     16     /// A write was attempted in a read-only transaction.
     17     ReadOnly,
     18     /// A write transaction conflicted with a different writer and lost. The
     19     /// transaction must be retried.
     20     Serialization,
     21 }

     25 impl std::fmt::Display for Error {
     26     fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
     27         match self {
     28             Error::Abort => write!(f, "operation aborted"),
     29             Error::InvalidData(msg) => write!(f, "invalid data: {msg}"),
     30             Error::InvalidInput(msg) => write!(f, "invalid input: {msg}"),
     31             Error::IO(msg) => write!(f, "io error: {msg}"),
     32             Error::ReadOnly => write!(f, "read-only transaction"),
     33             Error::Serialization => write!(f, "serialization failure, retry transaction"),
     34         }
     35     }
     36 }

     38 impl Error {
     39     /// Returns whether the error is considered deterministic. Raft state
     40     /// machine application needs to know whether a command failure is
     41     /// deterministic on the input command -- if it is, the command can be
     42     /// considered applied and the error returned to the client, but otherwise
     43     /// the state machine must panic to prevent replica divergence.
     44     pub fn is_deterministic(&self) -> bool {
     45         match self {
     46             // Aborts don't happen during application, only leader changes. But
     47             // we consider them non-deterministic in case an abort should happen
     48             // unexpectedly below Raft.
     49             Error::Abort => false,
     50             // Possible data corruption local to this node.
     51             Error::InvalidData(_) => false,
     52             // Input errors are (likely) deterministic. They might not be in
     53             // case data was corrupted in flight, but we ignore this case.
     54             Error::InvalidInput(_) => true,
     55             // IO errors are typically local to the node (e.g. faulty disk).
     56             Error::IO(_) => false,
     57             // Write commands in read-only transactions are deterministic.
     58             Error::ReadOnly => true,
     59             // Write conflicts are determinstic.
     60             Error::Serialization => true,
     61         }
     62     }
     63 }
