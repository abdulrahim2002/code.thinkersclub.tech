     16 /// The outbound Raft peer channel capacity. This buffers messages when a Raft
     17 /// peer is slow or unavailable. Beyond this, messages will be dropped.
     18 const RAFT_PEER_CHANNEL_CAPACITY: usize = 1000;

     23 /// A toyDB server. Routes messages to/from an inner Raft node.
     24 ///
     25 /// * Listens for inbound SQL connections from clients via TCP and passes
     26 ///   requests to the local Raft node.
     27 ///
     28 /// * Listens for inbound Raft connections from other toyDB nodes via TCP and
     29 ///   passes messages to the local Raft node.
     30 ///
     31 /// * Connects to other toyDB nodes via TCP and sends outbound Raft messages
     32 ///   from the local Raft node.
     33 pub struct Server {
     34     /// The inner Raft node.
     35     node: raft::Node,
     36     /// Outbound messages from the Raft node.
     37     node_rx: Receiver<raft::Envelope>,
     38     /// Raft peer IDs and addresses.
     39     peers: HashMap<raft::NodeID, String>,
     40 }

     42 impl Server {
     43     /// Creates a new toyDB server.
     44     pub fn new(
     45         id: raft::NodeID,
     46         peers: HashMap<raft::NodeID, String>,
     47         raft_log: raft::Log,
     48         raft_state: Box<dyn raft::State>,
     49     ) -> Result<Self> {
     50         let (node_tx, node_rx) = crossbeam::channel::unbounded();
     51         let node = raft::Node::new(
     52             id,
     53             peers.keys().copied().collect(),
     54             raft_log,
     55             raft_state,
     56             node_tx,
     57             raft::Options::default(),
     58         )?;
     59         Ok(Self { node, peers, node_rx })
     60     }

     62     /// Serves Raft and SQL requests indefinitely. Consumes the server.
     63     pub fn serve(self, raft_addr: impl ToSocketAddrs, sql_addr: impl ToSocketAddrs) -> Result<()> {
     64         let raft_listener = TcpListener::bind(raft_addr)?;
     65         let sql_listener = TcpListener::bind(sql_addr)?;
     66         info!(
     67             "Listening on {} (SQL) and {} (Raft)",
     68             sql_listener.local_addr()?,
     69             raft_listener.local_addr()?
     70         );
     71 
     72         std::thread::scope(move |s| {
     73             let id = self.node.id();
     74             let (raft_request_tx, raft_request_rx) = crossbeam::channel::unbounded();
     75             let (raft_step_tx, raft_step_rx) = crossbeam::channel::unbounded();
     76 
     77             // Serve inbound Raft connections.
     78             s.spawn(move || Self::raft_accept(raft_listener, raft_step_tx));
     79 
     80             // Establish outbound Raft connections to peers.
     81             let mut raft_peers_tx = HashMap::new();
     82             for (id, addr) in self.peers.into_iter() {
     83                 let (raft_peer_tx, raft_peer_rx) =
     84                     crossbeam::channel::bounded(RAFT_PEER_CHANNEL_CAPACITY);
     85                 raft_peers_tx.insert(id, raft_peer_tx);
     86                 s.spawn(move || Self::raft_send_peer(addr, raft_peer_rx));
     87             }
     88 
     89             // Route Raft messages between the local node, peers, and clients.
     90             s.spawn(move || {
     91                 Self::raft_route(
     92                     self.node,
     93                     self.node_rx,
     94                     raft_step_rx,
     95                     raft_peers_tx,
     96                     raft_request_rx,
     97                 )
     98             });
     99 
    100             // Serve inbound SQL connections.
    101             let sql_engine = sql::engine::Raft::new(raft_request_tx);
    102             s.spawn(move || Self::sql_accept(id, sql_listener, sql_engine));
    103         });
    104 
    105         Ok(())
    106     }

    108     /// Accepts new inbound Raft connections from peers and spawns threads
    109     /// routing inbound messages to the local Raft node.
    110     fn raft_accept(listener: TcpListener, raft_step_tx: Sender<raft::Envelope>) {
    111         std::thread::scope(|s| loop {
    112             let (socket, peer) = match listener.accept() {
    113                 Ok((socket, peer)) => (socket, peer),
    114                 Err(err) => {
    115                     error!("Raft peer accept failed: {err}");
    116                     continue;
    117                 }
    118             };
    119             let raft_step_tx = raft_step_tx.clone();
    120             s.spawn(move || {
    121                 debug!("Raft peer {peer} connected");
    122                 match Self::raft_receive_peer(socket, raft_step_tx) {
    123                     Ok(()) => debug!("Raft peer {peer} disconnected"),
    124                     Err(err) => error!("Raft peer {peer} error: {err}"),
    125                 }
    126             });
    127         });
    128     }

    130     /// Receives inbound messages from a peer via TCP, and queues them for
    131     /// stepping into the Raft node.
    132     fn raft_receive_peer(socket: TcpStream, raft_step_tx: Sender<raft::Envelope>) -> Result<()> {
    133         let mut socket = std::io::BufReader::new(socket);
    134         while let Some(message) = raft::Envelope::maybe_decode_from(&mut socket)? {
    135             raft_step_tx.send(message)?;
    136         }
    137         Ok(())
    138     }

    140     /// Sends outbound messages to a peer via TCP. Retries indefinitely if the
    141     /// connection fails.
    142     fn raft_send_peer(addr: String, raft_node_rx: Receiver<raft::Envelope>) {
    143         loop {
    144             let mut socket = match TcpStream::connect(&addr) {
    145                 Ok(socket) => std::io::BufWriter::new(socket),
    146                 Err(err) => {
    147                     error!("Failed connecting to Raft peer {addr}: {err}");
    148                     std::thread::sleep(RAFT_PEER_RETRY_INTERVAL);
    149                     continue;
    150                 }
    151             };
    152             while let Ok(message) = raft_node_rx.recv() {
    153                 if let Err(err) = message.encode_into(&mut socket).and_then(|_| Ok(socket.flush()?))
    154                 {
    155                     error!("Failed sending to Raft peer {addr}: {err}");
    156                     break;
    157                 }
    158             }
    159             debug!("Disconnected from Raft peer {addr}");
    160         }
    161     }

    163     /// Routes Raft messages:
    164     ///
    165     /// * node_rx: outbound messages from the local Raft node. Routed to peers
    166     ///   via TCP, or to local clients via a response channel.
    167     ///
    168     /// * request_rx: inbound requests from local SQL clients. Stepped into
    169     ///   the local Raft node as ClientRequest messages. Responses are returned
    170     ///   via the provided response channel.
    171     ///
    172     /// * peers_rx: inbound messages from remote Raft peers. Stepped into the
    173     ///   local Raft node.
    174     ///
    175     /// * peers_tx: outbound per-peer channels sent via TCP connections.
    176     ///   Messages from the local node's node_rx are sent here.
    177     ///
    178     /// Panics on any errors, since the Raft node can't recover from failed
    179     /// state transitions.
    180     fn raft_route(
    181         mut node: raft::Node,
    182         node_rx: Receiver<raft::Envelope>,
    183         peers_rx: Receiver<raft::Envelope>,
    184         mut peers_tx: HashMap<raft::NodeID, Sender<raft::Envelope>>,
    185         request_rx: Receiver<(raft::Request, Sender<Result<raft::Response>>)>,
    186     ) {
    187         // Track response channels by request ID. The Raft node will emit
    188         // ClientResponse messages that we forward to the response channel.
    189         let mut response_txs = HashMap::<raft::RequestID, Sender<Result<raft::Response>>>::new();
    190 
    191         let ticker = crossbeam::channel::tick(raft::TICK_INTERVAL);
    192         loop {
    193             crossbeam::select! {
    194                 // Periodically tick the node.
    195                 recv(ticker) -> _ => node = node.tick().expect("tick failed"),
    196 
    197                 // Step messages from peers into the node.
    198                 recv(peers_rx) -> result => {
    199                     let msg = result.expect("peers_rx disconnected");
    200                     node = node.step(msg).expect("step failed");
    201                 },
    202 
    203                 // Send outbound messages from the node to the appropriate peer.
    204                 // If we receive a client response addressed to the local node,
    205                 // forward it to the waiting client via the response channel.
    206                 recv(node_rx) -> result => {
    207                     let msg = result.expect("node_rx disconnected");
    208                     if msg.to == node.id() {
    209                         if let raft::Message::ClientResponse{ id, response } = msg.message {
    210                             if let Some(response_tx) = response_txs.remove(&id) {
    211                                 response_tx.send(response).expect("response_tx disconnected");
    212                             }
    213                             continue
    214                         }
    215                     }
    216                     let peer_tx = peers_tx.get_mut(&msg.to).expect("unknown peer");
    217                     match peer_tx.try_send(msg) {
    218                         Ok(()) => {},
    219                         Err(crossbeam::channel::TrySendError::Full(_)) => {
    220                             error!("Raft peer channel full, dropping message");
    221                         },
    222                         Err(crossbeam::channel::TrySendError::Disconnected(_)) => {
    223                             panic!("peer_tx disconnected");
    224                         },
    225                     };
    226                 }
    227 
    228                 // Track inbound client requests and step them into the node.
    229                 recv(request_rx) -> result => {
    230                     let (request, response_tx) = result.expect("request_rx disconnected");
    231                     let id = uuid::Uuid::new_v4();
    232                     let msg = raft::Envelope{
    233                         from: node.id(),
    234                         to: node.id(),
    235                         term: node.term(),
    236                         message: raft::Message::ClientRequest{id, request},
    237                     };
    238                     node = node.step(msg).expect("step failed");
    239                     response_txs.insert(id, response_tx);
    240                 }
    241             }
    242         }
    243     }

    245     /// Accepts new SQL client connections and spawns session threads for them.
    246     fn sql_accept(id: raft::NodeID, listener: TcpListener, sql_engine: sql::engine::Raft) {
    247         std::thread::scope(|s| loop {
    248             let (socket, peer) = match listener.accept() {
    249                 Ok((socket, peer)) => (socket, peer),
    250                 Err(err) => {
    251                     error!("Client accept failed: {err}");
    252                     continue;
    253                 }
    254             };
    255             let session = sql_engine.session();
    256             s.spawn(move || {
    257                 debug!("Client {peer} connected");
    258                 match Self::sql_session(id, socket, session) {
    259                     Ok(()) => debug!("Client {peer} disconnected"),
    260                     Err(err) => error!("Client {peer} error: {err}"),
    261                 }
    262             });
    263         })
    264     }

    266     /// Processes a client SQL session, executing SQL statements against the
    267     /// Raft node.
    268     fn sql_session(
    269         id: raft::NodeID,
    270         socket: TcpStream,
    271         mut session: sql::engine::Session<sql::engine::Raft>,
    272     ) -> Result<()> {
    273         let mut reader = std::io::BufReader::new(socket.try_clone()?);
    274         let mut writer = std::io::BufWriter::new(socket);
    275 
    276         while let Some(request) = Request::maybe_decode_from(&mut reader)? {
    277             // Execute request.
    278             debug!("Received request {request:?}");
    279             let response = match request {
    280                 Request::Execute(query) => session.execute(&query).map(Response::Execute),
    281                 Request::GetTable(table) => {
    282                     session.with_txn(true, |txn| txn.must_get_table(&table)).map(Response::GetTable)
    283                 }
    284                 Request::ListTables => session
    285                     .with_txn(true, |txn| {
    286                         Ok(txn.list_tables()?.into_iter().map(|t| t.name).collect())
    287                     })
    288                     .map(Response::ListTables),
    289                 Request::Status => session
    290                     .status()
    291                     .map(|s| Status { server: id, raft: s.raft, mvcc: s.mvcc })
    292                     .map(Response::Status),
    293             };
    294 
    295             // Process response.
    296             debug!("Returning response {response:?}");
    297             response.encode_into(&mut writer)?;
    298             writer.flush()?;
    299         }
    300         Ok(())
    301     }
    302 }
