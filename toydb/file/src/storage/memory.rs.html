      4 /// An in-memory key/value storage engine using the Rust standard library B-tree
      5 /// implementation. Data is not persisted.
      6 pub struct Memory {
      7     data: std::collections::BTreeMap<Vec<u8>, Vec<u8>>,
      8 }

     10 impl Memory {
     11     /// Creates a new Memory key-value storage engine.
     12     pub fn new() -> Self {
     13         Self { data: std::collections::BTreeMap::new() }
     14     }
     15 }

     17 impl Engine for Memory {
     18     type ScanIterator<'a> = ScanIterator<'a>;
     19 
     20     fn flush(&mut self) -> Result<()> {
     21         Ok(())
     22     }
     23 
     24     fn delete(&mut self, key: &[u8]) -> Result<()> {
     25         self.data.remove(key);
     26         Ok(())
     27     }
     28 
     29     fn get(&mut self, key: &[u8]) -> Result<Option<Vec<u8>>> {
     30         Ok(self.data.get(key).cloned())
     31     }
     32 
     33     fn scan(&mut self, range: impl std::ops::RangeBounds<Vec<u8>>) -> Self::ScanIterator<'_> {
     34         ScanIterator { inner: self.data.range(range) }
     35     }
     36 
     37     fn scan_dyn(
     38         &mut self,
     39         range: (std::ops::Bound<Vec<u8>>, std::ops::Bound<Vec<u8>>),
     40     ) -> Box<dyn super::ScanIterator + '_> {
     41         Box::new(self.scan(range))
     42     }
     43 
     44     fn set(&mut self, key: &[u8], value: Vec<u8>) -> Result<()> {
     45         self.data.insert(key.to_vec(), value);
     46         Ok(())
     47     }
     48 
     49     fn status(&mut self) -> Result<Status> {
     50         Ok(Status {
     51             name: "memory".to_string(),
     52             keys: self.data.len() as u64,
     53             size: self.data.iter().fold(0, |size, (k, v)| size + k.len() as u64 + v.len() as u64),
     54             total_disk_size: 0,
     55             live_disk_size: 0,
     56             garbage_disk_size: 0,
     57         })
     58     }
     59 }

     61 pub struct ScanIterator<'a> {
     62     inner: std::collections::btree_map::Range<'a, Vec<u8>, Vec<u8>>,

     65 impl<'a> ScanIterator<'a> {
     66     fn map(item: (&Vec<u8>, &Vec<u8>)) -> <Self as Iterator>::Item {
     67         let (key, value) = item;
     68         Ok((key.clone(), value.clone()))
     69     }
     70 }
     71 
     72 impl<'a> Iterator for ScanIterator<'a> {
     73     type Item = Result<(Vec<u8>, Vec<u8>)>;
     74 
     75     fn next(&mut self) -> Option<Self::Item> {
     76         self.inner.next().map(Self::map)
     77     }
     78 }

     80 impl<'a> DoubleEndedIterator for ScanIterator<'a> {
     81     fn next_back(&mut self) -> Option<Self::Item> {
     82         self.inner.next_back().map(Self::map)
     83     }
