      8 /// A very simple variant of BitCask, itself a very simple log-structured
      9 /// key-value engine used e.g. by the Riak database. It is not compatible with
     10 /// BitCask databases generated by other implementations. See:
     11 /// https://riak.com/assets/bitcask-intro.pdf
     12 ///
     13 /// BitCask writes key-value pairs to an append-only log file, and keeps a
     14 /// mapping of keys to file positions in memory. All live keys must fit in
     15 /// memory. Deletes write a tombstone value to the log file. To remove old
     16 /// garbage, logs can be compacted by writing new logs containing only live
     17 /// data, skipping replaced values and tombstones.
     18 ///
     19 /// This implementation makes several significant simplifications over
     20 /// standard BitCask:
     21 ///
     22 /// - Instead of writing multiple fixed-size log files, it uses a single
     23 ///   append-only log file of arbitrary size. This increases the compaction
     24 ///   volume, since the entire log file must be rewritten on every compaction,
     25 ///   and can exceed the filesystem's file size limit, but ToyDB databases are
     26 ///   expected to be small.
     27 ///
     28 /// - Compactions lock the database for reads and writes. This is ok since ToyDB
     29 ///   only compacts during node startup and files are expected to be small.
     30 ///
     31 /// - Hint files are not used, the log itself is scanned when opened to
     32 ///   build the keydir. Hint files only omit values, and ToyDB values are
     33 ///   expected to be small, so the hint files would be nearly as large as
     34 ///   the compacted log files themselves.
     35 ///
     36 /// - Log entries don't contain timestamps or checksums.
     37 ///
     38 /// The structure of a log entry is:
     39 ///
     40 /// - Key length as big-endian u32.
     41 /// - Value length as big-endian i32, or -1 for tombstones.
     42 /// - Key as raw bytes (max 2 GB).
     43 /// - Value as raw bytes (max 2 GB).
     44 pub struct BitCask {
     45     /// The active append-only log file.
     46     log: Log,
     47     /// Maps keys to a value position and length in the log file.
     48     keydir: KeyDir,

     54 impl BitCask {
     55     /// Opens or creates a BitCask database in the given file.
     56     pub fn new(path: PathBuf) -> Result<Self> {
     57         log::info!("Opening database {}", path.display());
     58         let mut log = Log::new(path.clone())?;

     62     }

     64     /// Opens a BitCask database, and automatically compacts it if the amount
     65     /// of garbage exceeds the given ratio and byte size when opened.
     66     pub fn new_compact(
     67         path: PathBuf,
     68         garbage_min_fraction: f64,
     69         garbage_min_bytes: u64,
     70     ) -> Result<Self> {
     71         let mut s = Self::new(path)?;
     72 
     73         let status = s.status()?;
     74         if Self::should_compact(
     75             status.garbage_disk_size,
     76             status.total_disk_size,
     77             garbage_min_fraction,
     78             garbage_min_bytes,
     79         ) {
     80             log::info!(
     81                 "Compacting {} to remove {:.0}% garbage ({} MB out of {} MB)",
     82                 s.log.path.display(),
     83                 status.garbage_percent(),
     84                 status.garbage_disk_size / 1024 / 1024,
     85                 status.total_disk_size / 1024 / 1024
     86             );
     87             s.compact()?;
     88             log::info!(
     89                 "Compacted {} to size {} MB",
     90                 s.log.path.display(),
     91                 (status.total_disk_size - status.garbage_disk_size) / 1024 / 1024
     92             );
     93         }
     94 
     95         Ok(s)
     96     }

     98     /// Returns true if the log file should be compacted.
     99     fn should_compact(
    100         garbage_size: u64,
    101         total_size: u64,
    102         min_fraction: f64,
    103         min_bytes: u64,
    104     ) -> bool {
    105         let garbage_fraction = garbage_size as f64 / total_size as f64;
    106         garbage_size > 0 && garbage_size >= min_bytes && garbage_fraction >= min_fraction
    107     }

    149         let value_len = value.len() as u32;

    154     fn status(&mut self) -> Result<Status> {
    155         let keys = self.keydir.len() as u64;
    156         let size = self
    157             .keydir
    158             .iter()
    159             .fold(0, |size, (key, (_, value_len))| size + key.len() as u64 + *value_len as u64);
    160         let total_disk_size = self.log.file.metadata()?.len();
    161         let live_disk_size = size + 8 * keys; // account for length prefixes
    162         let garbage_disk_size = total_disk_size - live_disk_size;
    163         Ok(Status {
    164             name: "bitcask".to_string(),
    165             keys,
    166             size,
    167             total_disk_size,
    168             live_disk_size,
    169             garbage_disk_size,
    170         })
    171     }

    201     /// Compacts the current log file by writing out a new log file containing
    202     /// only live keys and replacing the current file with it.
    203     pub fn compact(&mut self) -> Result<()> {
    204         let mut tmp_path = self.log.path.clone();
    205         tmp_path.set_extension("new");
    206         let (mut new_log, new_keydir) = self.write_log(tmp_path)?;
    207 
    208         std::fs::rename(&new_log.path, &self.log.path)?;
    209         new_log.path = self.log.path.clone();
    210 
    211         self.log = new_log;
    212         self.keydir = new_keydir;
    213         Ok(())
    214     }
    215 
    216     /// Writes out a new log file with the live entries of the current log file
    217     /// and returns it along with its keydir. Entries are written in key order.
    218     fn write_log(&mut self, path: PathBuf) -> Result<(Log, KeyDir)> {
    219         let mut new_keydir = KeyDir::new();
    220         let mut new_log = Log::new(path)?;
    221         new_log.file.set_len(0)?; // truncate file if it exists
    222         for (key, (value_pos, value_len)) in self.keydir.iter() {
    223             let value = self.log.read_value(*value_pos, *value_len)?;
    224             let (pos, len) = new_log.write_entry(key, Some(&value))?;
    225             new_keydir.insert(key.clone(), (pos + len as u64 - *value_len as u64, *value_len));
    226         }
    227         Ok((new_log, new_keydir))
    228     }
    229 }

    240 /// A BitCask append-only log file, containing a sequence of key/value
    241 /// entries encoded as follows;
    242 ///
    243 /// - Key length as big-endian u32.
    244 /// - Value length as big-endian i32, or -1 for tombstones.
    245 /// - Key as raw bytes (max 2 GB).
    246 /// - Value as raw bytes (max 2 GB).
    247 struct Log {
    248     /// Path to the log file.
    249     path: PathBuf,
    250     /// The opened file containing the log.
    251     file: std::fs::File,
    252 }
    253 
    254 impl Log {
    255     /// Opens a log file, or creates one if it does not exist. Takes out an
    256     /// exclusive lock on the file until it is closed, or errors if the lock is
    257     /// already held.
    258     fn new(path: PathBuf) -> Result<Self> {
    259         if let Some(dir) = path.parent() {
    260             std::fs::create_dir_all(dir)?
    261         }
    262         let file = std::fs::OpenOptions::new()
    263             .read(true)
    264             .write(true)
    265             .create(true)
    266             .truncate(false)
    267             .open(&path)?;
    268         file.try_lock_exclusive()?;
    269         Ok(Self { path, file })
    270     }
    271 
    272     /// Builds a keydir by scanning the log file. If an incomplete entry is
    273     /// encountered, it is assumed to be caused by an incomplete write operation
    274     /// and the remainder of the file is truncated.
    275     fn build_keydir(&mut self) -> Result<KeyDir> {
    276         let mut len_buf = [0u8; 4];
    277         let mut keydir = KeyDir::new();
    278         let file_len = self.file.metadata()?.len();
    279         let mut r = BufReader::new(&mut self.file);
    280         let mut pos = r.seek(SeekFrom::Start(0))?;
    281 
    282         while pos < file_len {
    283             // Read the next entry from the file, returning the key, value
    284             // position, and value length or None for tombstones.
    285             let result = || -> std::result::Result<(Vec<u8>, u64, Option<u32>), std::io::Error> {
    286                 r.read_exact(&mut len_buf)?;
    287                 let key_len = u32::from_be_bytes(len_buf);
    288                 r.read_exact(&mut len_buf)?;
    289                 let value_len_or_tombstone = match i32::from_be_bytes(len_buf) {
    290                     l if l >= 0 => Some(l as u32),
    291                     _ => None, // -1 for tombstones
    292                 };
    293                 let value_pos = pos + 4 + 4 + key_len as u64;
    294 
    295                 let mut key = vec![0; key_len as usize];
    296                 r.read_exact(&mut key)?;
    297 
    298                 if let Some(value_len) = value_len_or_tombstone {
    299                     if value_pos + value_len as u64 > file_len {
    300                         return Err(std::io::Error::new(
    301                             std::io::ErrorKind::UnexpectedEof,
    302                             "value extends beyond end of file",
    303                         ));
    304                     }
    305                     r.seek_relative(value_len as i64)?; // avoids discarding buffer
    306                 }
    307 
    308                 Ok((key, value_pos, value_len_or_tombstone))
    309             }();
    310 
    311             match result {
    312                 // Populate the keydir with the entry, or remove it on tombstones.
    313                 Ok((key, value_pos, Some(value_len))) => {
    314                     keydir.insert(key, (value_pos, value_len));
    315                     pos = value_pos + value_len as u64;
    316                 }
    317                 Ok((key, value_pos, None)) => {
    318                     keydir.remove(&key);
    319                     pos = value_pos;
    320                 }
    321                 // If an incomplete entry was found at the end of the file, assume an
    322                 // incomplete write and truncate the file.
    323                 Err(err) if err.kind() == std::io::ErrorKind::UnexpectedEof => {
    324                     log::error!("Found incomplete entry at offset {}, truncating file", pos);
    325                     self.file.set_len(pos)?;
    326                     break;
    327                 }
    328                 Err(err) => return Err(err.into()),
    329             }
    330         }
    331 
    332         Ok(keydir)
    333     }
    334 
    335     /// Reads a value from the log file.
    336     fn read_value(&mut self, value_pos: u64, value_len: u32) -> Result<Vec<u8>> {
    337         let mut value = vec![0; value_len as usize];
    338         self.file.seek(SeekFrom::Start(value_pos))?;
    339         self.file.read_exact(&mut value)?;
    340         Ok(value)
    341     }
    342 
    343     /// Appends a key/value entry to the log file, using a None value for
    344     /// tombstones. It returns the position and length of the entry.
    345     fn write_entry(&mut self, key: &[u8], value: Option<&[u8]>) -> Result<(u64, u32)> {
    346         let key_len = key.len() as u32;
    347         let value_len = value.map_or(0, |v| v.len() as u32);
    348         let value_len_or_tombstone = value.map_or(-1, |v| v.len() as i32);
    349         let len = 4 + 4 + key_len + value_len;
    350 
    351         let pos = self.file.seek(SeekFrom::End(0))?;
    352         let mut w = BufWriter::with_capacity(len as usize, &mut self.file);
    353         w.write_all(&key_len.to_be_bytes())?;
    354         w.write_all(&value_len_or_tombstone.to_be_bytes())?;
    355         w.write_all(key)?;
    356         if let Some(value) = value {
    357             w.write_all(value)?;
    358         }
    359         w.flush()?;
    360 
    361         Ok((pos, len))
    362     }
    363 }

    403     /// Tests that a log with an incomplete write at the end can be recovered by
    404     /// discarding the last entry.
    405     #[test]
    406     fn recovery() -> Result<()> {
    407         // Create an initial log file with a few entries. Keep track of where
    408         // each entry ends.
    409         let dir = tempfile::TempDir::with_prefix("toydb")?;
    410         let path = dir.path().join("complete");
    411         let mut log = Log::new(path.clone())?;
    412 
    413         let mut ends = vec![];
    414         let (pos, len) = log.write_entry("deleted".as_bytes(), Some(&[1, 2, 3]))?;
    415         ends.push(pos + len as u64);
    416         let (pos, len) = log.write_entry("deleted".as_bytes(), None)?;
    417         ends.push(pos + len as u64);
    418         let (pos, len) = log.write_entry(&[], Some(&[]))?;
    419         ends.push(pos + len as u64);
    420         let (pos, len) = log.write_entry("key".as_bytes(), Some(&[1, 2, 3, 4, 5]))?;
    421         ends.push(pos + len as u64);
    422         drop(log);
    423 
    424         // Copy the file, and truncate it at each byte, then try to open it
    425         // and assert that we always retain a prefix of entries.
    426         let truncpath = dir.path().join("truncated");
    427         let size = std::fs::metadata(&path)?.len();
    428         for pos in 0..=size {
    429             std::fs::copy(&path, &truncpath)?;
    430             let f = std::fs::OpenOptions::new().write(true).open(&truncpath)?;
    431             f.set_len(pos)?;
    432             drop(f);
    433 
    434             let mut expect = vec![];
    435             if pos >= ends[0] {
    436                 expect.push((b"deleted".to_vec(), vec![1, 2, 3]))
    437             }
    438             if pos >= ends[1] {
    439                 expect.pop(); // "deleted" key removed
    440             }
    441             if pos >= ends[2] {
    442                 expect.push((b"".to_vec(), vec![]))
    443             }
    444             if pos >= ends[3] {
    445                 expect.push((b"key".to_vec(), vec![1, 2, 3, 4, 5]))
    446             }
    447 
    448             let mut engine = BitCask::new(truncpath.clone())?;
    449             assert_eq!(expect, engine.scan(..).collect::<Result<Vec<_>>>()?);
    450         }
    451         Ok(())
    452     }

    496     /// A BitCask-specific goldenscript runner, which dispatches through to the
    497     /// standard Engine runner.
    498     struct BitCaskRunner {
    499         inner: Runner<BitCask>,
    500         tempdir: tempfile::TempDir,
    501     }
    502 
    503     impl goldenscript::Runner for BitCaskRunner {
    504         fn run(&mut self, command: &goldenscript::Command) -> StdResult<String, Box<dyn StdError>> {
    505             let mut output = String::new();
    506             match command.name.as_str() {
    507                 // compact
    508                 // Compacts the BitCask entry log.
    509                 "compact" => {
    510                     command.consume_args().reject_rest()?;
    511                     self.inner.engine.compact()?;
    512                 }
    513 
    514                 // dump
    515                 // Dumps the full BitCask entry log.
    516                 "dump" => {
    517                     command.consume_args().reject_rest()?;
    518                     self.dump(&mut output)?;
    519                 }
    520 
    521                 // reopen [compact_fraction=FLOAT]
    522                 // Closes and reopens the BitCask database. If compact_ratio is
    523                 // given, it specifies a garbage ratio beyond which the log
    524                 // should be auto-compacted on open.
    525                 "reopen" => {
    526                     let mut args = command.consume_args();
    527                     let compact_fraction = args.lookup_parse("compact_fraction")?;
    528                     args.reject_rest()?;
    529                     // We need to close the file before we can reopen it, which
    530                     // happens when the database is dropped. Replace the engine
    531                     // with a temporary empty engine then reopen the file.
    532                     let path = self.inner.engine.log.path.clone();
    533                     self.inner.engine = BitCask::new(self.tempdir.path().join("empty"))?;
    534                     if let Some(garbage_fraction) = compact_fraction {
    535                         self.inner.engine = BitCask::new_compact(path, garbage_fraction, 0)?;
    536                     } else {
    537                         self.inner.engine = BitCask::new(path)?;
    538                     }
    539                 }
    540 
    541                 // Pass other commands to the standard engine runner.
    542                 _ => return self.inner.run(command),
    543             }
    544             Ok(output)
    545         }
    546     }

    550             let tempdir = tempfile::TempDir::with_prefix("toydb").expect("tempdir failed");
    551             let engine = BitCask::new(tempdir.path().join("bitcask")).expect("bitcask failed");
    552             let inner = Runner::new(engine);
    553             Self { inner, tempdir }
    554         }
    555 
    556         /// Dumps the full BitCask entry log.
    557         fn dump(&mut self, output: &mut String) -> StdResult<(), Box<dyn StdError>> {
    558             let file = &mut self.inner.engine.log.file;
    559             let file_len = file.metadata()?.len();
    560             let mut r = BufReader::new(file);
    561             let mut pos = r.seek(SeekFrom::Start(0))?;
    562             let mut len_buf = [0; 4];
    563             let mut idx = 0;
    564 
    565             while pos < file_len {
    566                 if idx > 0 {
    567                     writeln!(output, "--------")?;
    568                 }
    569                 write!(output, "{:<7}", format!("{idx}@{pos}"))?;
    570 
    571                 r.read_exact(&mut len_buf)?;
    572                 let key_len = u32::from_be_bytes(len_buf);
    573                 write!(output, " keylen={key_len} [{}]", hex::encode(len_buf))?;
    574 
    575                 r.read_exact(&mut len_buf)?;
    576                 let value_len_or_tombstone = i32::from_be_bytes(len_buf); // NB: -1 for tombstones
    577                 let value_len = value_len_or_tombstone.max(0) as u32;
    578                 writeln!(output, " valuelen={value_len_or_tombstone} [{}]", hex::encode(len_buf))?;
    579 
    580                 let mut key = vec![0; key_len as usize];
    581                 r.read_exact(&mut key)?;
    582                 let mut value = vec![0; value_len as usize];
    583                 r.read_exact(&mut value)?;
    584                 let size = 4 + 4 + key_len as u64 + value_len as u64;
    585                 writeln!(
    586                     output,
    587                     "{:<7} key={} [{}] {}",
    588                     format!("{size}b"),
    589                     format::Raw::key(&key),
    590                     hex::encode(key),
    591                     match value_len_or_tombstone {
    592                         -1 => "tombstone".to_string(),
    593                         _ => format!(
    594                             "value={} [{}]",
    595                             format::Raw::bytes(&value),
    596                             hex::encode(&value)
    597                         ),
    598                     },
    599                 )?;
    600 
    601                 pos += size;
    602                 idx += 1;
    603             }
    604             Ok(())
    605         }
    606     }
