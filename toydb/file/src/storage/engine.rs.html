      6 /// A key/value storage engine storing arbitrary byte strings in lexicographical
      7 /// key order. Storing keys in order allows for efficient range scans, which is
      8 /// needed to e.g. scan a single table during SQL execution (where all rows have
      9 /// keys with a common key prefix for the table). Keys should use the KeyCode
     10 /// order-preserving encoding, see src/encoding/keycode. Writes are only
     11 /// guaranteed durable after calling flush().
     12 ///
     13 /// Only supports single-threaded use since all methods (including reads) take a
     14 /// mutable reference -- serialized access can't be avoided anyway, since both
     15 /// Raft execution and file access is serial.
     16 pub trait Engine: Send {
     17     /// The iterator returned by scan().
     18     type ScanIterator<'a>: ScanIterator + 'a
     19     where
     20         Self: Sized + 'a; // omit in trait objects, for object safety
     21 
     22     /// Deletes a key, or does nothing if it does not exist.
     23     fn delete(&mut self, key: &[u8]) -> Result<()>;
     24 
     25     /// Flushes any buffered data to the underlying storage medium.
     26     fn flush(&mut self) -> Result<()>;
     27 
     28     /// Gets a value for a key, if it exists.
     29     fn get(&mut self, key: &[u8]) -> Result<Option<Vec<u8>>>;
     30 
     31     /// Iterates over an ordered range of key/value pairs.
     32     fn scan(&mut self, range: impl std::ops::RangeBounds<Vec<u8>>) -> Self::ScanIterator<'_>
     33     where
     34         Self: Sized; // omit in trait objects, for object safety
     35 
     36     /// Like scan, but can be used from trait objects. The iterator will use
     37     /// dynamic dispatch, which has a minor performance penalty.
     38     fn scan_dyn(
     39         &mut self,
     40         range: (std::ops::Bound<Vec<u8>>, std::ops::Bound<Vec<u8>>),
     41     ) -> Box<dyn ScanIterator + '_>;
     42 
     43     /// Iterates over all key/value pairs starting with prefix.
     44     fn scan_prefix(&mut self, prefix: &[u8]) -> Self::ScanIterator<'_>
     45     where
     46         Self: Sized, // omit in trait objects, for object safety
     47     {
     48         self.scan(keycode::prefix_range(prefix))
     49     }
     50 
     51     /// Sets a value for a key, replacing the existing value if any.
     52     fn set(&mut self, key: &[u8], value: Vec<u8>) -> Result<()>;
     53 
     54     /// Returns engine status.
     55     fn status(&mut self) -> Result<Status>;
     56 }

     80 impl Status {
     81     pub fn garbage_percent(&self) -> f64 {
     82         if self.total_disk_size == 0 {
     83             return 0.0;
     84         }
     85         self.garbage_disk_size as f64 / self.total_disk_size as f64 * 100.0

    114     impl<E: Engine> goldenscript::Runner for Runner<E> {
    115         fn run(&mut self, command: &goldenscript::Command) -> StdResult<String, Box<dyn StdError>> {
    116             let mut output = String::new();
    117             match command.name.as_str() {
    118                 // delete KEY
    119                 "delete" => {
    120                     let mut args = command.consume_args();
    121                     let key = decode_binary(&args.next_pos().ok_or("key not given")?.value);
    122                     args.reject_rest()?;
    123                     self.engine.delete(&key)?;
    124                 }
    125 
    126                 // get KEY
    127                 "get" => {
    128                     let mut args = command.consume_args();
    129                     let key = decode_binary(&args.next_pos().ok_or("key not given")?.value);
    130                     args.reject_rest()?;
    131                     let value = self.engine.get(&key)?;
    132                     writeln!(output, "{}", format::Raw::key_maybe_value(&key, value.as_deref()))?;
    133                 }
    134 
    135                 // scan [reverse=BOOL] RANGE
    136                 "scan" => {
    137                     let mut args = command.consume_args();
    138                     let reverse = args.lookup_parse("reverse")?.unwrap_or(false);
    139                     let range =
    140                         parse_key_range(args.next_pos().map(|a| a.value.as_str()).unwrap_or(".."))?;
    141                     args.reject_rest()?;
    142                     let items: Vec<_> = if reverse {
    143                         self.engine.scan(range).rev().try_collect()?
    144                     } else {
    145                         self.engine.scan(range).try_collect()?
    146                     };
    147                     for (key, value) in items {
    148                         let fmtkv = format::Raw::key_value(&key, &value);
    149                         writeln!(output, "{fmtkv}")?;
    150                     }
    151                 }
    152 
    153                 // scan_prefix PREFIX
    154                 "scan_prefix" => {
    155                     let mut args = command.consume_args();
    156                     let prefix = decode_binary(&args.next_pos().ok_or("prefix not given")?.value);
    157                     args.reject_rest()?;
    158                     let mut scan = self.engine.scan_prefix(&prefix);
    159                     while let Some((key, value)) = scan.next().transpose()? {
    160                         let fmtkv = format::Raw::key_value(&key, &value);
    161                         writeln!(output, "{fmtkv}")?;
    162                     }
    163                 }
    164 
    165                 // set KEY=VALUE
    166                 "set" => {
    167                     let mut args = command.consume_args();
    168                     let kv = args.next_key().ok_or("key=value not given")?.clone();
    169                     let key = decode_binary(&kv.key.unwrap());
    170                     let value = decode_binary(&kv.value);
    171                     args.reject_rest()?;
    172                     self.engine.set(&key, value)?;
    173                 }
    174 
    175                 // status
    176                 "status" => {
    177                     command.consume_args().reject_rest()?;
    178                     writeln!(output, "{:#?}", self.engine.status()?)?;
    179                 }
    180 
    181                 name => return Err(format!("invalid command {name}").into()),
    182             }
    183             Ok(output)
    184         }
    185     }

    187     /// Decodes a raw byte vector from a Unicode string. Code points in the
    188     /// range U+0080 to U+00FF are converted back to bytes 0x80 to 0xff.
    189     /// This allows using e.g. \xff in the input string literal, and getting
    190     /// back a 0xff byte in the byte vector. Otherwise, char(0xff) yields
    191     /// the UTF-8 bytes 0xc3bf, which is the U+00FF code point as UTF-8.
    192     /// These characters are effectively represented as ISO-8859-1 rather
    193     /// than UTF-8, but it allows precise use of the entire u8 value range.
    194     pub fn decode_binary(s: &str) -> Vec<u8> {
    195         let mut buf = [0; 4];
    196         let mut bytes = Vec::new();
    197         for c in s.chars() {
    198             // u32 is the Unicode code point, not the UTF-8 encoding.
    199             match c as u32 {
    200                 b @ 0x80..=0xff => bytes.push(b as u8),
    201                 _ => bytes.extend(c.encode_utf8(&mut buf).as_bytes()),
    202             }
    203         }
    204         bytes
    205     }

    207     /// Parses an binary key range, using Rust range syntax.
    208     pub fn parse_key_range(
    209         s: &str,
    210     ) -> StdResult<impl std::ops::RangeBounds<Vec<u8>>, Box<dyn StdError>> {
    211         use std::ops::Bound;
    212         let mut bound = (Bound::<Vec<u8>>::Unbounded, Bound::<Vec<u8>>::Unbounded);
    213         let re = Regex::new(r"^(\S+)?\.\.(=)?(\S+)?").expect("invalid regex");
    214         let groups = re.captures(s).ok_or_else(|| format!("invalid range {s}"))?;
    215         if let Some(start) = groups.get(1) {
    216             bound.0 = Bound::Included(decode_binary(start.as_str()));
    217         }
    218         if let Some(end) = groups.get(3) {
    219             let end = decode_binary(end.as_str());
    220             if groups.get(2).is_some() {
    221                 bound.1 = Bound::Included(end)
    222             } else {
    223                 bound.1 = Bound::Excluded(end)
    224             }
    225         }
    226         Ok(bound)
    227     }

    291     /// An engine that wraps two others and mirrors operations across them,
    292     /// panicking if they produce different results. Engine implementations
    293     /// should not have any observable differences in behavior.
    294     pub struct Mirror<A: Engine, B: Engine> {
    295         pub a: A,
    296         pub b: B,
    297     }

    299     impl<A: Engine, B: Engine> Mirror<A, B> {
    300         pub fn new(a: A, b: B) -> Self {
    301             Self { a, b }
    302         }
    303     }

    305     impl<A: Engine, B: Engine> Engine for Mirror<A, B> {
    306         type ScanIterator<'a> = MirrorIterator<'a, A, B>
    307         where
    308             Self: Sized,
    309             A: 'a,
    310             B: 'a;
    311 
    312         fn delete(&mut self, key: &[u8]) -> Result<()> {
    313             self.a.delete(key)?;
    314             self.b.delete(key)
    315         }
    316 
    317         fn flush(&mut self) -> Result<()> {
    318             self.a.flush()?;
    319             self.b.flush()
    320         }
    321 
    322         fn get(&mut self, key: &[u8]) -> Result<Option<Vec<u8>>> {
    323             let a = self.a.get(key)?;
    324             let b = self.b.get(key)?;
    325             assert_eq!(a, b);
    326             Ok(a)
    327         }
    328 
    329         fn scan(&mut self, range: impl std::ops::RangeBounds<Vec<u8>>) -> Self::ScanIterator<'_>
    330         where
    331             Self: Sized,
    332         {
    333             let a = self.a.scan((range.start_bound().cloned(), range.end_bound().cloned()));
    334             let b = self.b.scan(range);
    335             MirrorIterator { a, b }
    336         }
    337 
    338         fn scan_dyn(
    339             &mut self,
    340             range: (std::ops::Bound<Vec<u8>>, std::ops::Bound<Vec<u8>>),
    341         ) -> Box<dyn ScanIterator + '_> {
    342             let a = self.a.scan(range.clone());
    343             let b = self.b.scan(range);
    344             Box::new(MirrorIterator::<A, B> { a, b })
    345         }
    346 
    347         fn set(&mut self, key: &[u8], value: Vec<u8>) -> Result<()> {
    348             self.a.set(key, value.clone())?;
    349             self.b.set(key, value)
    350         }
    351 
    352         fn status(&mut self) -> Result<Status> {
    353             let a = self.a.status()?;
    354             let b = self.b.status()?;
    355             // Only some items are comparable.
    356             assert_eq!(a.keys, b.keys);
    357             assert_eq!(a.size, b.size);
    358             Ok(a)
    359         }
    360     }
