      1 //! toydump is a debug tool that prints a toyDB BitCask database in
      2 //! human-readable form. It can print both the SQL database and the Raft log
      3 //! (via --raft). It only outputs live BitCask data, not garbage entries.
      4 
      5 #![warn(clippy::all)]
      6 
      7 use toydb::encoding::format::{self, Formatter as _};
      8 use toydb::error::Result;
      9 use toydb::storage::{BitCask, Engine as _};
     10 
     11 use clap::Parser as _;
     12 
     13 fn main() {
     14     if let Err(error) = Command::parse().run() {
     15         eprintln!("Error: {error}")
     16     }
     17 }
     18 
     19 /// The toydump command.
     20 #[derive(clap::Parser)]
     21 #[command(about = "Prints toyDB file contents.", version, propagate_version = true)]
     22 struct Command {
     23     /// The BitCask file to dump (SQL database unless --raft).
     24     file: String,
     25     /// The file is a Raft log, not SQL database.
     26     #[arg(long)]
     27     raft: bool,
     28     /// Also show raw key and value.
     29     #[arg(long)]
     30     raw: bool,
     31 }
     32 
     33 impl Command {
     34     /// Runs the command.
     35     fn run(self) -> Result<()> {
     36         let mut engine = BitCask::new(self.file.into())?;
     37         let mut scan = engine.scan(..);
     38         while let Some((key, value)) = scan.next().transpose()? {
     39             let mut string = match self.raft {
     40                 true => format::Raft::<format::SQLCommand>::key_value(&key, &value),
     41                 false => format::MVCC::<format::SQL>::key_value(&key, &value),
     42             };
     43             if self.raw {
     44                 string = format!("{string} [{}]", format::Raw::key_value(&key, &value))
     45             }
     46             println!("{string}");
     47         }
     48         Ok(())
     49     }
     50 }
