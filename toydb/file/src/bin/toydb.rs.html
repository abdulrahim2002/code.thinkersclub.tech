      2 //! config/toydb.yaml) or corresponding TOYDB_ environment variables. Listens
      3 //! for SQL clients (default port 9605) and Raft connections from other toyDB

      5 //! data/raft and data/sql by default.

     28 /// The toyDB server configuration. Can be provided via config file (default
     29 /// config/toydb.yaml) or TOYDB_ environment variables.
     30 #[derive(Debug, Deserialize)]
     31 struct Config {
     32     /// The node ID. Must be unique in the cluster.
     33     id: raft::NodeID,
     34     /// The other nodes in the cluster, and their Raft TCP addresses.
     35     peers: HashMap<raft::NodeID, String>,
     36     /// The Raft listen address.
     37     listen_raft: String,
     38     /// The SQL listen address.
     39     listen_sql: String,
     40     /// The log level.
     41     log_level: String,
     42     /// The path to this node's data directory. The Raft log is stored in
     43     /// the file "raft", and the SQL state machine in "sql".
     44     data_dir: String,
     45     /// The Raft storage engine: bitcask or memory.
     46     storage_raft: String,
     47     /// The SQL storage engine: bitcask or memory.
     48     storage_sql: String,
     49     /// If false, don't fsync Raft log writes to disk. Disabling this
     50     /// will yield much better write performance, but may lose data on
     51     /// host crashes which compromises Raft safety guarantees.
     52     fsync: bool,
     53     /// The garbage fraction threshold at which to trigger compaction.
     54     compact_threshold: f64,
     55     /// The minimum bytes of garbage before triggering compaction.
     56     compact_min_bytes: u64,
     57 }
     58 
     59 impl Config {
     60     /// Loads the configuration from the given file.
     61     fn load(file: &str) -> Result<Self> {
     62         Ok(config::Config::builder()
     63             .set_default("id", "1")?
     64             .set_default("listen_sql", "localhost:9605")?
     65             .set_default("listen_raft", "localhost:9705")?
     66             .set_default("log_level", "info")?
     67             .set_default("data_dir", "data")?
     68             .set_default("storage_raft", "bitcask")?
     69             .set_default("storage_sql", "bitcask")?
     70             .set_default("fsync", true)?
     71             .set_default("compact_threshold", 0.2)?
     72             .set_default("compact_min_bytes", 1_000_000)?
     73             .add_source(config::File::with_name(file))
     74             .add_source(config::Environment::with_prefix("TOYDB"))
     75             .build()?
     76             .try_deserialize()?)
     77     }
     78 }

     80 /// The toyDB server command.
     81 #[derive(clap::Parser)]
     82 #[command(about = "Starts a toyDB server.", version, propagate_version = true)]
     83 struct Command {
     84     /// The configuration file path.
     85     #[arg(short = 'c', long, default_value = "config/toydb.yaml")]
     86     config: String,
     87 }
     88 
     89 impl Command {
     90     /// Runs the toyDB server.
     91     fn run(self) -> Result<()> {
     92         // Load the configuration.
     93         let cfg = Config::load(&self.config)?;
     94 
     95         // Initialize logging.
     96         let loglevel = cfg.log_level.parse()?;
     97         let mut logconfig = simplelog::ConfigBuilder::new();
     98         if loglevel != simplelog::LevelFilter::Debug {
     99             logconfig.add_filter_allow_str("toydb");
    100         }
    101         simplelog::SimpleLogger::init(loglevel, logconfig.build())?;
    102 
    103         // Initialize the Raft log storage engine.
    104         let datadir = std::path::Path::new(&cfg.data_dir);
    105         let mut raft_log = match cfg.storage_raft.as_str() {
    106             "bitcask" | "" => {
    107                 let engine = storage::BitCask::new_compact(
    108                     datadir.join("raft"),
    109                     cfg.compact_threshold,
    110                     cfg.compact_min_bytes,
    111                 )?;
    112                 raft::Log::new(Box::new(engine))?
    113             }
    114             "memory" => raft::Log::new(Box::new(storage::Memory::new()))?,
    115             name => return errinput!("invalid Raft storage engine {name}"),
    116         };
    117         raft_log.enable_fsync(cfg.fsync);
    118 
    119         // Initialize the SQL storage engine.
    120         let raft_state: Box<dyn raft::State> = match cfg.storage_sql.as_str() {
    121             "bitcask" | "" => {
    122                 let engine = storage::BitCask::new_compact(
    123                     datadir.join("sql"),
    124                     cfg.compact_threshold,
    125                     cfg.compact_min_bytes,
    126                 )?;
    127                 Box::new(sql::engine::Raft::new_state(engine)?)
    128             }
    129             "memory" => Box::new(sql::engine::Raft::new_state(storage::Memory::new())?),
    130             name => return errinput!("invalid SQL storage engine {name}"),
    131         };
    132 
    133         // Start the server.
    134         Server::new(cfg.id, cfg.peers, raft_log, raft_state)?
    135             .serve(&cfg.listen_raft, &cfg.listen_sql)
    136     }
    137 }
