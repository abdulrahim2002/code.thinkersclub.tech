      1 //! Runs toyDB workload benchmarks. By default, it assumes a running 5-node
      2 //! cluster as launched via cluster/run.sh, but this can be modified via -H.
      3 //! For example, a read-only workload can be run as:
      4 //!
      5 //! cargo run --release --bin workload -- read
      6 //!
      7 //! See --help for a list of available workloads and arguments.

     26 fn main() {
     27     let Command { runner, subcommand } = Command::parse();
     28     let result = match subcommand {
     29         Subcommand::Read(read) => runner.run(read),
     30         Subcommand::Write(write) => runner.run(write),
     31         Subcommand::Bank(bank) => runner.run(bank),
     32     };
     33     if let Err(error) = result {
     34         eprintln!("Error: {error}")
     35     }
     36 }

     56 /// Runs a workload benchmark.
     57 #[derive(clap::Args)]
     58 struct Runner {
     59     /// Hosts to connect to (optionally with port number).
     60     #[arg(
     61         short = 'H',
     62         long,
     63         value_delimiter = ',',
     64         default_value = "localhost:9605,localhost:9604,localhost:9603,localhost:9602,localhost:9601"
     65     )]
     66     hosts: Vec<String>,
     67 
     68     /// Number of concurrent workers to spawn.
     69     #[arg(short, long, default_value = "16")]
     70     concurrency: usize,
     71 
     72     /// Number of transactions to execute.
     73     #[arg(short = 'n', long, default_value = "100000")]
     74     count: usize,
     75 
     76     /// Seed to use for random number generation.
     77     #[arg(short, long, default_value = "16791084677885396490")]
     78     seed: u64,
     79 }

     81 impl Runner {
     82     /// Runs the specified workload.
     83     fn run<W: Workload>(self, workload: W) -> Result<()> {
     84         let mut rng = StdRng::seed_from_u64(self.seed);
     85         let mut client = Client::connect(&self.hosts[0])?;
     86 
     87         // Set up a histogram recording txn latencies as nanoseconds. The
     88         // buckets range from 0.001s to 10s.
     89         let mut hist = Histogram::<u32>::new_with_bounds(1_000, 10_000_000_000, 3)?.into_sync();
     90 
     91         // Prepare the dataset.
     92         print!("Preparing initial dataset... ");
     93         std::io::stdout().flush()?;
     94         let start = std::time::Instant::now();
     95         workload.prepare(&mut client, &mut rng)?;
     96         println!("done ({:.3}s)", start.elapsed().as_secs_f64());
     97 
     98         // Spawn workers, round robin across hosts.
     99         std::thread::scope(|s| -> Result<()> {
    100             print!("Spawning {} workers... ", self.concurrency);
    101             std::io::stdout().flush()?;
    102             let start = std::time::Instant::now();
    103 
    104             let (work_tx, work_rx) = crossbeam::channel::bounded(self.concurrency);
    105             let (done_tx, done_rx) = crossbeam::channel::bounded::<()>(0);
    106 
    107             for addr in self.hosts.iter().cycle().take(self.concurrency) {
    108                 let mut client = Client::connect(addr)?;
    109                 let mut recorder = hist.recorder();
    110                 let work_rx = work_rx.clone();
    111                 let done_tx = done_tx.clone();
    112                 s.spawn(move || -> Result<()> {
    113                     while let Ok(item) = work_rx.recv() {
    114                         let start = std::time::Instant::now();
    115                         client.with_retry(|client| W::execute(client, &item))?;
    116                         recorder.record(start.elapsed().as_nanos() as u64)?;
    117                     }
    118                     drop(done_tx); // disconnects done_rx once all workers exit
    119                     Ok(())
    120                 });
    121             }
    122             drop(done_tx); // drop local copy
    123 
    124             println!("done ({:.3}s)", start.elapsed().as_secs_f64());
    125 
    126             // Spawn work generator.
    127             {
    128                 println!("Running workload {}...", workload);
    129                 let generator = workload.generate(rng).take(self.count);
    130                 s.spawn(move || -> Result<()> {
    131                     for item in generator {
    132                         work_tx.send(item)?;
    133                     }
    134                     Ok(())
    135                 });
    136             }
    137 
    138             // Periodically print stats until all workers are done.
    139             let start = std::time::Instant::now();
    140             let ticker = crossbeam::channel::tick(Duration::from_secs(1));
    141 
    142             println!();
    143             println!("Time   Progress     Txns      Rate       p50       p90       p99      pMax");
    144 
    145             while let Err(crossbeam::channel::TryRecvError::Empty) = done_rx.try_recv() {
    146                 crossbeam::select! {
    147                     recv(ticker) -> _ => {},
    148                     recv(done_rx) -> _ => {},
    149                 }
    150 
    151                 let duration = start.elapsed().as_secs_f64();
    152                 hist.refresh_timeout(Duration::from_secs(1));
    153 
    154                 println!(
    155                     "{:<8} {:>5.1}%  {:>7}  {:>6.0}/s  {:>6.1}ms  {:>6.1}ms  {:>6.1}ms  {:>6.1}ms",
    156                     format!("{:.1}s", duration),
    157                     hist.len() as f64 / self.count as f64 * 100.0,
    158                     hist.len(),
    159                     hist.len() as f64 / duration,
    160                     Duration::from_nanos(hist.value_at_quantile(0.5)).as_secs_f64() * 1000.0,
    161                     Duration::from_nanos(hist.value_at_quantile(0.9)).as_secs_f64() * 1000.0,
    162                     Duration::from_nanos(hist.value_at_quantile(0.99)).as_secs_f64() * 1000.0,
    163                     Duration::from_nanos(hist.max()).as_secs_f64() * 1000.0,
    164                 );
    165             }
    166             Ok(())
    167         })?;
    168 
    169         // Verify the final dataset.
    170         println!();
    171         print!("Verifying dataset... ");
    172         std::io::stdout().flush()?;
    173         let start = std::time::Instant::now();
    174         workload.verify(&mut client, self.count)?;
    175         println!("done ({:.3}s)", start.elapsed().as_secs_f64());
    176 
    177         Ok(())
    178     }
    179 }

    181 /// A workload.
    182 trait Workload: std::fmt::Display {
    183     /// A work item.
    184     type Item: Send;
    185 
    186     /// Prepares the workload by creating initial tables and data.
    187     fn prepare(&self, client: &mut Client, rng: &mut StdRng) -> Result<()>;
    188 
    189     /// Generates work items as an iterator.
    190     fn generate(&self, rng: StdRng) -> impl Iterator<Item = Self::Item> + Send + 'static;
    191 
    192     /// Executes a single work item. This will automatically be retried on
    193     /// certain errors, and must use a transaction where appropriate.
    194     fn execute(client: &mut Client, item: &Self::Item) -> Result<()>;
    195 
    196     /// Verifies the dataset after the workload has completed.
    197     fn verify(&self, _client: &mut Client, _txns: usize) -> Result<()> {
    198         Ok(())
    199     }
    200 }

    202 /// A read-only workload. Creates an id,value table and populates it with the
    203 /// given row count and value size. Then runs batches of random primary key
    204 /// lookups (SELECT * FROM read WHERE id = 1 OR id = 2 ...).
    205 #[derive(clap::Args, Clone)]
    206 #[command(about = "A read-only workload using primary key lookups")]
    207 struct Read {
    208     /// Total number of rows in data set.
    209     #[arg(short, long, default_value = "1000")]
    210     rows: u64,
    211 
    212     /// Row value size (excluding primary key).
    213     #[arg(short, long, default_value = "64")]
    214     size: usize,
    215 
    216     /// Number of rows to fetch in a single select.
    217     #[arg(short, long, default_value = "1")]
    218     batch: usize,
    219 }
    220 
    221 impl std::fmt::Display for Read {
    222     fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
    223         write!(f, "read (rows={} size={} batch={})", self.rows, self.size, self.batch)
    224     }
    225 }
    226 
    227 impl Workload for Read {
    228     type Item = HashSet<u64>;
    229 
    230     fn prepare(&self, client: &mut Client, rng: &mut StdRng) -> Result<()> {
    231         client.execute("BEGIN")?;
    232         client.execute(r#"DROP TABLE IF EXISTS "read""#)?;
    233         client.execute(r#"CREATE TABLE "read" (id INT PRIMARY KEY, value STRING NOT NULL)"#)?;
    234 
    235         let chars = &mut rand::distributions::Alphanumeric.sample_iter(rng).map(|b| b as char);
    236         let rows = (1..=self.rows).map(|id| (id, chars.take(self.size).collect::<String>()));
    237         let chunks = rows.chunks(100);
    238         let queries = chunks.into_iter().map(|chunk| {
    239             format!(
    240                 r#"INSERT INTO "read" (id, value) VALUES ({})"#,
    241                 chunk.map(|(id, value)| format!("{}, '{}'", id, value)).join("), (")
    242             )
    243         });
    244         for query in queries {
    245             client.execute(&query)?;
    246         }
    247         client.execute("COMMIT")?;
    248         Ok(())
    249     }
    250 
    251     fn generate(&self, rng: StdRng) -> impl Iterator<Item = Self::Item> + 'static {
    252         ReadGenerator {
    253             batch: self.batch,
    254             dist: rand::distributions::Uniform::new(1, self.rows + 1),
    255             rng,
    256         }
    257     }
    258 
    259     fn execute(client: &mut Client, item: &Self::Item) -> Result<()> {
    260         let batch_size = item.len();
    261         let query = format!(
    262             r#"SELECT * FROM "read" WHERE {}"#,
    263             item.iter().map(|id| format!("id = {}", id)).join(" OR ")
    264         );
    265         let rows: Rows = client.execute(&query)?.try_into()?;
    266         assert_eq!(rows.count(), batch_size, "Unexpected row count");
    267         Ok(())
    268     }
    269 
    270     fn verify(&self, client: &mut Client, _: usize) -> Result<()> {
    271         let count: i64 = client.execute(r#"SELECT COUNT(*) FROM "read""#)?.try_into()?;
    272         assert_eq!(count, self.rows as i64, "Unexpected row count");
    273         Ok(())
    274     }
    275 }

    277 /// A Read workload generator, yielding batches of random, unique primary keys.
    278 struct ReadGenerator {
    279     batch: usize,
    280     rng: StdRng,
    281     dist: rand::distributions::Uniform<u64>,
    282 }
    283 
    284 impl Iterator for ReadGenerator {
    285     type Item = <Read as Workload>::Item;
    286 
    287     fn next(&mut self) -> Option<Self::Item> {
    288         let mut ids = HashSet::new();
    289         for id in self.dist.sample_iter(&mut self.rng) {
    290             ids.insert(id);
    291             if ids.len() >= self.batch {
    292                 break;
    293             }
    294         }
    295         Some(ids)
    296     }
    297 }

    299 /// A write-only workload. Creates an id,value table, and writes rows with
    300 /// sequential primary keys and the given value size, in the given batch size
    301 /// (INSERT INTO write (id, value) VALUES ...). The number of rows written
    302 /// is given by Runner.count * Write.batch.
    303 #[derive(clap::Args, Clone)]
    304 #[command(about = "A write-only workload writing sequential rows")]
    305 struct Write {
    306     /// Row value size (excluding primary key).
    307     #[arg(short, long, default_value = "64")]
    308     size: usize,
    309 
    310     /// Number of rows to write in a single insert query.
    311     #[arg(short, long, default_value = "1")]
    312     batch: usize,
    313 }
    314 
    315 impl std::fmt::Display for Write {
    316     fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
    317         write!(f, "write (size={} batch={})", self.size, self.batch)
    318     }
    319 }

    321 impl Workload for Write {
    322     type Item = Vec<(u64, String)>;
    323 
    324     fn prepare(&self, client: &mut Client, _: &mut StdRng) -> Result<()> {
    325         client.execute("BEGIN")?;
    326         client.execute(r#"DROP TABLE IF EXISTS "write""#)?;
    327         client.execute(r#"CREATE TABLE "write" (id INT PRIMARY KEY, value STRING NOT NULL)"#)?;
    328         client.execute("COMMIT")?;
    329         Ok(())
    330     }
    331 
    332     fn generate(&self, rng: StdRng) -> impl Iterator<Item = Self::Item> + 'static {
    333         WriteGenerator { next_id: 1, size: self.size, batch: self.batch, rng }
    334     }
    335 
    336     fn execute(client: &mut Client, item: &Self::Item) -> Result<()> {
    337         let batch_size = item.len();
    338         let query = format!(
    339             r#"INSERT INTO "write" (id, value) VALUES {}"#,
    340             item.iter().map(|(id, value)| format!("({}, '{}')", id, value)).join(", ")
    341         );
    342         if let StatementResult::Insert { count } = client.execute(&query)? {
    343             assert_eq!(count as usize, batch_size, "Unexpected row count");
    344         } else {
    345             panic!("Unexpected result")
    346         }
    347         Ok(())
    348     }
    349 
    350     fn verify(&self, client: &mut Client, txns: usize) -> Result<()> {
    351         let count: i64 = client.execute(r#"SELECT COUNT(*) FROM "write""#)?.try_into()?;
    352         assert_eq!(count as usize, txns * self.batch, "Unexpected row count");
    353         Ok(())
    354     }
    355 }

    357 /// A Write workload generator, yielding batches of sequential primary keys and
    358 /// random rows.
    359 struct WriteGenerator {
    360     next_id: u64,
    361     size: usize,
    362     batch: usize,
    363     rng: StdRng,
    364 }
    365 
    366 impl Iterator for WriteGenerator {
    367     type Item = <Write as Workload>::Item;
    368 
    369     fn next(&mut self) -> Option<Self::Item> {
    370         let chars =
    371             &mut rand::distributions::Alphanumeric.sample_iter(&mut self.rng).map(|b| b as char);
    372         let mut rows = Vec::with_capacity(self.batch);
    373         while rows.len() < self.batch {
    374             rows.push((self.next_id, chars.take(self.size).collect()));
    375             self.next_id += 1;
    376         }
    377         Some(rows)
    378     }
    379 }

    381 /// A bank workload. Creates a set of customers and accounts, and makes random
    382 /// transfers between them. Specifically, it picks two random customers A and B,
    383 /// and then finds A's highest-balance account and B's lowest-balance account,
    384 /// and transfers a random amount without overdrawing the account. This
    385 /// somewhat convoluted scheme is used to make the workload slightly less
    386 /// trivial, including joins, ordering, and secondary indexes.
    387 #[derive(clap::Args, Clone)]
    388 #[command(about = "A bank workload, making transfers between customer accounts")]
    389 struct Bank {
    390     /// Number of customers.
    391     #[arg(short, long, default_value = "100")]
    392     customers: u64,
    393 
    394     /// Number of accounts per customer.
    395     #[arg(short, long, default_value = "10")]
    396     accounts: u64,
    397 
    398     /// Initial account balance.
    399     #[arg(short, long, default_value = "100")]
    400     balance: u64,
    401 
    402     /// Max amount to transfer.
    403     #[arg(short, long, default_value = "50")]
    404     max_transfer: u64,
    405 }
    406 
    407 impl std::fmt::Display for Bank {
    408     fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
    409         write!(f, "bank (customers={} accounts={})", self.customers, self.accounts)
    410     }
    411 }
    412 
    413 impl Workload for Bank {
    414     type Item = (u64, u64, u64); // from,to,amount
    415 
    416     fn prepare(&self, client: &mut Client, rng: &mut StdRng) -> Result<()> {
    417         let petnames = petname::Petnames::default();
    418         client.execute("BEGIN")?;
    419         client.execute("DROP TABLE IF EXISTS account")?;
    420         client.execute("DROP TABLE IF EXISTS customer")?;
    421         client.execute(
    422             "CREATE TABLE customer (
    423                     id INTEGER PRIMARY KEY,
    424                     name STRING NOT NULL
    425                 )",
    426         )?;
    427         client.execute(
    428             "CREATE TABLE account (
    429                     id INTEGER PRIMARY KEY,
    430                     customer_id INTEGER NOT NULL INDEX REFERENCES customer,
    431                     balance INTEGER NOT NULL
    432                 )",
    433         )?;
    434         client.execute(&format!(
    435             "INSERT INTO customer VALUES {}",
    436             (1..=self.customers)
    437                 .zip(petnames.iter(rng, 3, " "))
    438                 .map(|(id, name)| format!("({}, '{}')", id, name))
    439                 .join(", ")
    440         ))?;
    441         client.execute(&format!(
    442             "INSERT INTO account VALUES {}",
    443             (1..=self.customers)
    444                 .flat_map(|c| (1..=self.accounts).map(move |a| (c, (c - 1) * self.accounts + a)))
    445                 .map(|(c, a)| (format!("({}, {}, {})", a, c, self.balance)))
    446                 .join(", ")
    447         ))?;
    448         client.execute("COMMIT")?;
    449         Ok(())
    450     }
    451 
    452     fn generate(&self, rng: StdRng) -> impl Iterator<Item = Self::Item> + 'static {
    453         let customers = self.customers;
    454         let max_transfer = self.max_transfer;
    455         // Generate random u64s, then pick random from,to,amount as the
    456         // remainder of the max customer and amount.
    457         rand::distributions::Uniform::new_inclusive(0, u64::MAX)
    458             .sample_iter(rng)
    459             .tuples()
    460             .map(move |(a, b, c)| (a % customers + 1, b % customers + 1, c % max_transfer + 1))
    461             .filter(|(from, to, _)| from != to)
    462     }
    463 
    464     fn execute(client: &mut Client, item: &Self::Item) -> Result<()> {
    465         let (from, to, mut amount) = item;
    466 
    467         client.execute("BEGIN")?;
    468 
    469         let row: Row = client
    470             .execute(&format!(
    471                 "SELECT a.id, a.balance
    472                         FROM account a JOIN customer c ON a.customer_id = c.id
    473                         WHERE c.id = {}
    474                         ORDER BY a.balance DESC
    475                         LIMIT 1",
    476                 from
    477             ))?
    478             .try_into()?;
    479         let mut row = row.into_iter();
    480         let from_account: i64 = row.next().unwrap().try_into()?;
    481         let from_balance: i64 = row.next().unwrap().try_into()?;
    482         amount = std::cmp::min(amount, from_balance as u64);
    483 
    484         let to_account: i64 = client
    485             .execute(&format!(
    486                 "SELECT a.id, a.balance
    487                         FROM account a JOIN customer c ON a.customer_id = c.id
    488                         WHERE c.id = {}
    489                         ORDER BY a.balance ASC
    490                         LIMIT 1",
    491                 to
    492             ))?
    493             .try_into()?;
    494 
    495         client.execute(&format!(
    496             "UPDATE account SET balance = balance - {} WHERE id = {}",
    497             amount, from_account,
    498         ))?;
    499         client.execute(&format!(
    500             "UPDATE account SET balance = balance + {} WHERE id = {}",
    501             amount, to_account,
    502         ))?;
    503 
    504         client.execute("COMMIT")?;
    505 
    506         Ok(())
    507     }
    508 
    509     fn verify(&self, client: &mut Client, _: usize) -> Result<()> {
    510         let balance: i64 = client.execute("SELECT SUM(balance) FROM account")?.try_into()?;
    511         assert_eq!(balance as u64, self.customers * self.accounts * self.balance);
    512         let negative: i64 =
    513             client.execute("SELECT COUNT(*) FROM account WHERE balance < 0")?.try_into()?;
    514         assert_eq!(negative, 0);
    515         Ok(())
    516     }
