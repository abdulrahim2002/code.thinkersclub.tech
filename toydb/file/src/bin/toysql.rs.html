     27 /// The toySQL command.
     28 #[derive(clap::Parser)]
     29 #[command(about = "A toyDB client.", version, propagate_version = true)]
     30 struct Command {
     31     /// A SQL statement to execute, then exit.
     32     #[arg()]
     33     statement: Option<String>,
     34     /// Host to connect to.
     35     #[arg(short = 'H', long, default_value = "localhost")]
     36     host: String,
     37     /// Port number to connect to.
     38     #[arg(short = 'p', long, default_value = "9605")]
     39     port: u16,
     40 }
     41 
     42 impl Command {
     43     /// Runs the command.
     44     fn run(self) -> Result<()> {
     45         let mut shell = Shell::new(&self.host, self.port)?;
     46         match self.statement {
     47             Some(statement) => shell.execute(&statement),
     48             None => shell.run(),
     49         }
     50     }
     51 }

     53 /// An interactive toySQL shell.
     54 struct Shell {
     55     /// The toyDB client.
     56     client: Client,
     57     /// The Rustyline command editor.
     58     editor: Editor<InputValidator, DefaultHistory>,
     59     /// The path to the history file, if any.
     60     history_path: Option<std::path::PathBuf>,
     61     /// If true, SELECT column headers will be displayed.
     62     show_headers: bool,
     63 }
     64 
     65 impl Shell {
     66     /// Creates a new shell connected to the given server.
     67     fn new(host: &str, port: u16) -> Result<Self> {
     68         let client = Client::connect((host, port))?;
     69         // Set up Rustyline. Make sure multiline pastes are handled normally.
     70         let mut editor = Editor::new()?;
     71         editor.set_helper(Some(InputValidator));
     72         editor.bind_sequence(
     73             rustyline::KeyEvent(rustyline::KeyCode::BracketedPasteStart, Modifiers::NONE),
     74             rustyline::Cmd::Noop,
     75         );
     76         let history_path = std::env::var_os("HOME")
     77             .map(|home| std::path::PathBuf::from(home).join(".toysql.history"));
     78         Ok(Self { client, editor, history_path, show_headers: false })
     79     }
     80 
     81     /// Executes a SQL statement or ! command.
     82     fn execute(&mut self, input: &str) -> Result<()> {
     83         if input.starts_with('!') {
     84             self.execute_command(input)
     85         } else if !input.is_empty() {
     86             self.execute_sql(input)
     87         } else {
     88             Ok(())
     89         }
     90     }
     91 
     92     /// Executes a toySQL ! command (e.g. !help)
     93     fn execute_command(&mut self, input: &str) -> Result<()> {
     94         let mut input = input.split_ascii_whitespace();
     95         let Some(command) = input.next() else {
     96             return errinput!("expected command");
     97         };
     98         let args = input.collect_vec();
     99 
    100         match (command, args.as_slice()) {
    101             // Toggles column headers.
    102             ("!headers", []) => {
    103                 self.show_headers = !self.show_headers;
    104                 match self.show_headers {
    105                     true => println!("Headers enabled"),
    106                     false => println!("Headers disabled"),
    107                 }
    108             }
    109             ("!headers", _) => return errinput!("!headers takes no arguments"),
    110 
    111             // Displays help.
    112             ("!help", []) => println!(
    113                 r#"
    114 Enter a SQL statement terminated by a semicolon (;) to execute it, or Ctrl-D to
    115 exit. The following commands are also available:
    116 
    117     !headers           Toggles column headers
    118     !help              This help message
    119     !status            Display server status
    120     !table NAME        Display a table schema
    121     !tables            List tables
    122 "#
    123             ),
    124             ("!help", _) => return errinput!("!help takes no arguments"),
    125 
    126             // Displays server status.
    127             ("!status", []) => {
    128                 let status = self.client.status()?;
    129                 println!(
    130                     r#"
    131 Server:       n{server} with Raft leader n{leader} in term {term} for {nodes} nodes
    132 Raft log:     {committed} committed, {applied} applied, {raft_size} MB, {raft_garbage}% garbage ({raft_storage} engine)
    133 Replication:  {raft_match}
    134 SQL storage:  {sql_keys} keys, {sql_size} MB logical, {nodes}x {sql_disk_size} MB disk, {sql_garbage}% garbage ({sql_storage} engine)
    135 Transactions: {active_txns} active, {versions} total
    136 "#,
    137                     server = status.server,
    138                     leader = status.raft.leader,
    139                     term = status.raft.term,
    140                     nodes = status.raft.match_index.len(),
    141                     committed = status.raft.commit_index,
    142                     applied = status.raft.applied_index,
    143                     raft_size = format_args!("{:.3}", status.raft.storage.size as f64 / 1000000.0),
    144                     raft_garbage = format_args!("{:.0}", status.raft.storage.garbage_percent()),
    145                     raft_storage = status.raft.storage.name,
    146                     raft_match =
    147                         status.raft.match_index.iter().map(|(n, m)| format!("n{n}:{m}")).join(" "),
    148                     sql_keys = status.mvcc.storage.keys,
    149                     sql_size = format_args!("{:.3}", status.mvcc.storage.size as f64 / 1000000.0),
    150                     sql_disk_size = format_args!(
    151                         "{:.3}",
    152                         status.mvcc.storage.total_disk_size as f64 / 1000000.0
    153                     ),
    154                     sql_garbage = format_args!("{:.0}", status.mvcc.storage.garbage_percent()),
    155                     sql_storage = status.mvcc.storage.name,
    156                     active_txns = status.mvcc.active_txns,
    157                     versions = status.mvcc.versions,
    158                 )
    159             }
    160             ("!status", _) => return errinput!("!status takes no arguments"),
    161 
    162             ("!table", [name]) => println!("{}", self.client.get_table(name)?),
    163             ("!table", _) => return errinput!("!table takes 1 argument"),
    164 
    165             ("!tables", []) => self.client.list_tables()?.iter().for_each(|t| println!("{t}")),
    166             ("!tables", _) => return errinput!("!tables takes no arguments"),
    167 
    168             (command, _) => return errinput!("unknown command {command}"),
    169         }
    170         Ok(())
    171     }

    173     /// Executes a SQL statement and displays the results.
    174     fn execute_sql(&mut self, statement: &str) -> Result<()> {
    175         use StatementResult::*;
    176         match self.client.execute(statement)? {
    177             Begin(state) => match state.read_only {
    178                 true => println!("Began read-only transaction at version {}", state.version),
    179                 false => println!("Began transaction {}", state.version),
    180             },
    181             Commit { version } => println!("Committed transaction {version}"),
    182             Rollback { version } => println!("Rolled back transaction {version}"),
    183             Insert { count } => println!("Inserted {count} rows"),
    184             Delete { count } => println!("Deleted {count} rows"),
    185             Update { count } => println!("Updated {count} rows"),
    186             CreateTable { name } => println!("Created table {name}"),
    187             DropTable { name, existed } => match existed {
    188                 true => println!("Dropped table {name}"),
    189                 false => println!("Table {name} does not exist"),
    190             },
    191             Explain(plan) => println!("{plan}"),
    192             Select { columns, rows } => {
    193                 if self.show_headers {
    194                     println!("{}", columns.iter().map(|c| c.as_header()).join(", "));
    195                 }
    196                 for row in rows {
    197                     println!("{}", row.iter().join(", "));
    198                 }
    199             }
    200         }
    201         Ok(())
    202     }

    204     /// Prompts the user for input. Returns None if the shell should close.
    205     fn prompt(&mut self) -> rustyline::Result<String> {
    206         let prompt = match self.client.txn() {
    207             Some(txn) if txn.read_only => format!("toydb@{}> ", txn.version),
    208             Some(txn) => format!("toydb:{}> ", txn.version),
    209             None => "toydb> ".to_string(),
    210         };
    211         self.editor.readline(&prompt)
    212     }
    213 
    214     /// Runs the interactive shell.
    215     fn run(&mut self) -> Result<()> {
    216         // Load the history file, if any.
    217         if let Some(history_path) = &self.history_path {
    218             match self.editor.load_history(history_path) {
    219                 Ok(()) => {}
    220                 Err(ReadlineError::Io(error)) if error.kind() == std::io::ErrorKind::NotFound => {}
    221                 Err(error) => return Err(error.into()),
    222             }
    223         }
    224 
    225         // Print welcome message.
    226         let server = self.client.status()?.server;
    227         println!("Connected to toyDB node n{server}. Enter !help for instructions.");
    228 
    229         // Prompt for commands and execute them.
    230         loop {
    231             let input = match self.prompt() {
    232                 Ok(input) => input.trim().to_string(),
    233                 Err(ReadlineError::Interrupted) => continue,
    234                 Err(ReadlineError::Eof) => break,
    235                 Err(error) => return Err(error.into()),
    236             };
    237             self.editor.add_history_entry(&input)?;
    238             if let Err(error) = self.execute(&input) {
    239                 eprintln!("Error: {error}");
    240             };
    241         }
    242 
    243         // Save the history file.
    244         if let Some(history_path) = &self.history_path {
    245             self.editor.save_history(history_path)?;
    246         }
    247         Ok(())
    248     }
    249 }

    251 /// A Rustyline helper for multiline editing. After a new line is entered, it
    252 /// determines whether the input makes up a complete SQL statement that should
    253 /// be submitted to the server (i.e. it's terminated by ;), or wait for further
    254 /// input.
    255 #[derive(Completer, Helper, Highlighter, Hinter)]
    256 struct InputValidator;
    257 
    258 impl Validator for InputValidator {
    259     fn validate(&self, ctx: &mut ValidationContext) -> rustyline::Result<ValidationResult> {
    260         let input = ctx.input();
    261         // Empty lines and ! commands are ready.
    262         if input.is_empty() || input.starts_with('!') || input == ";" {
    263             return Ok(ValidationResult::Valid(None));
    264         }
    265         // For SQL statements, just look for any semicolon or lexer error, and
    266         // rely on the server for further validation and error handling.
    267         if Lexer::new(input).any(|r| matches!(r, Ok(Token::Semicolon) | Err(_))) {
    268             return Ok(ValidationResult::Valid(None));
    269         }
    270         // Otherwise, wait for more input.
    271         Ok(ValidationResult::Incomplete)
    272     }
    273 
    274     fn validate_while_typing(&self) -> bool {
    275         false // only check after completed lines
    276     }
