      1 //! Bincode is used to encode values, both in key/value stores and the toyDB
      2 //! network protocol. It is a Rust-specific encoding that depends on the
      3 //! internal data structures being stable, but it's sufficient for toyDB. See:
      4 //! https://github.com/bincode-org/bincode
      5 //!
      6 //! This module wraps the standard bincode crate to change the default options,
      7 //! in particular to use variable-length rather than fixed-length integers.
      8 //! Confusingly, upstream bincode::(de)serialize uses different options (fixed)
      9 //! than DefaultOptions (variable) -- this module always uses DefaultOptions.
     10 
     11 use crate::error::{Error, Result};
     12 
     13 use bincode::Options as _;
     14 use serde::{de::DeserializeOwned, Deserialize, Serialize};
     15 
     16 /// Returns the default Bincode options, initialized on first use.
     17 fn bincode() -> &'static bincode::DefaultOptions {
     18     static BINCODE: std::sync::OnceLock<bincode::DefaultOptions> = std::sync::OnceLock::new();
     19     BINCODE.get_or_init(bincode::DefaultOptions::new)
     20 }
     21 
     22 /// Deserializes a value using Bincode.
     23 pub fn deserialize<'de, T: Deserialize<'de>>(bytes: &'de [u8]) -> Result<T> {
     24     Ok(bincode().deserialize(bytes)?)
     25 }
     26 
     27 /// Deserializes a value from a reader using Bincode.
     28 pub fn deserialize_from<R: std::io::Read, T: DeserializeOwned>(reader: R) -> Result<T> {
     29     Ok(bincode().deserialize_from(reader)?)
     30 }
     31 
     32 /// Deserializes a value from a reader using Bincode, or returns None if the
     33 /// reader is closed.
     34 pub fn maybe_deserialize_from<R: std::io::Read, T: DeserializeOwned>(
     35     reader: R,
     36 ) -> Result<Option<T>> {
     37     match bincode().deserialize_from(reader) {
     38         Ok(v) => Ok(Some(v)),
     39         Err(e) => match *e {
     40             bincode::ErrorKind::Io(e) => match e.kind() {
     41                 std::io::ErrorKind::UnexpectedEof => Ok(None),
     42                 std::io::ErrorKind::ConnectionReset => Ok(None),
     43                 _ => Err(Error::from(e)),
     44             },
     45             _ => Err(Error::from(e)),
     46         },
     47     }
     48 }
     49 
     50 /// Serializes a value using Bincode.
     51 pub fn serialize<T: Serialize>(value: &T) -> Vec<u8> {
     52     // Panic on serialization failures, as this is typically an issue with the
     53     // provided data structure.
     54     bincode().serialize(value).expect("bincode serialization failed")
     55 }
     56 
     57 /// Serializes a value to a writer using Bincode.
     58 pub fn serialize_into<W: std::io::Write, T: Serialize>(writer: W, value: &T) -> Result<()> {
     59     Ok(bincode().serialize_into(writer, value)?)
     60 }
