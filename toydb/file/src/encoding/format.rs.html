     12 /// Formats raw key/value pairs.
     13 pub trait Formatter {
     14     /// Formats a key.
     15     fn key(key: &[u8]) -> String;
     16 
     17     /// Formats a value.
     18     fn value(key: &[u8], value: &[u8]) -> String;
     19 
     20     /// Formats a key/value pair.
     21     fn key_value(key: &[u8], value: &[u8]) -> String {
     22         Self::key_maybe_value(key, Some(value))
     23     }
     24 
     25     /// Formats a key/value pair, where the value may not exist.
     26     fn key_maybe_value(key: &[u8], value: Option<&[u8]>) -> String {
     27         let fkey = Self::key(key);
     28         let fvalue = value.map(|v| Self::value(key, v)).unwrap_or("None".to_string());
     29         format!("{fkey} → {fvalue}")
     30     }
     31 }

     33 /// Formats raw byte slices.
     34 pub struct Raw;
     35 
     36 impl Raw {
     37     pub fn bytes(bytes: &[u8]) -> String {
     38         let escaped = bytes.iter().copied().flat_map(std::ascii::escape_default).collect_vec();
     39         let string = String::from_utf8_lossy(&escaped);
     40         format!("\"{string}\"")
     41     }
     42 }
     43 
     44 impl Formatter for Raw {
     45     fn key(key: &[u8]) -> String {
     46         Self::bytes(key)
     47     }
     48 
     49     fn value(_: &[u8], value: &[u8]) -> String {
     50         Self::bytes(value)
     51     }
     52 }

     54 /// Formats Raft log entries. Dispatches to I for command formatting.
     55 pub struct Raft<I: Formatter>(std::marker::PhantomData<I>);
     56 
     57 impl<I: Formatter> Raft<I> {
     58     pub fn entry(entry: &raft::Entry) -> String {
     59         let fcommand =
     60             entry.command.as_deref().map(|c| I::value(&[], c)).unwrap_or("None".to_string());
     61         format!("{}@{} {fcommand}", entry.index, entry.term)
     62     }
     63 }
     64 
     65 impl<I: Formatter> Formatter for Raft<I> {
     66     fn key(key: &[u8]) -> String {
     67         let Ok(key) = raft::Key::decode(key) else {
     68             return Raw::key(key);
     69         };
     70         format!("raft:{key:?}")
     71     }
     72 
     73     fn value(key: &[u8], value: &[u8]) -> String {
     74         let Ok(key) = raft::Key::decode(key) else {
     75             return Raw::value(key, value);
     76         };
     77         match key {
     78             raft::Key::CommitIndex => {
     79                 match bincode::deserialize::<(raft::Index, raft::Term)>(value) {
     80                     Ok((index, term)) => format!("{index}@{term}"),
     81                     Err(_) => Raw::bytes(value),
     82                 }
     83             }
     84             raft::Key::TermVote => {
     85                 match bincode::deserialize::<(raft::Term, Option<raft::NodeID>)>(value) {
     86                     Ok((term, vote)) => format!(
     87                         "term={term} vote={}",
     88                         vote.map(|v| v.to_string()).unwrap_or("None".to_string())
     89                     ),
     90                     Err(_) => Raw::bytes(value),
     91                 }
     92             }
     93             raft::Key::Entry(_) => match bincode::deserialize::<raft::Entry>(value) {
     94                 Ok(entry) => Self::entry(&entry),
     95                 Err(_) => Raw::bytes(value),
     96             },
     97         }
     98     }
     99 }

    101 /// Formats MVCC keys/values. Dispatches to I for inner key/value formatting.
    102 pub struct MVCC<I: Formatter>(std::marker::PhantomData<I>);
    103 
    104 impl<I: Formatter> Formatter for MVCC<I> {
    105     fn key(key: &[u8]) -> String {
    106         let Ok(key) = mvcc::Key::decode(key) else { return Raw::key(key) };
    107         match key {
    108             mvcc::Key::TxnWrite(version, innerkey) => {
    109                 format!("mvcc:TxnWrite({version}, {})", I::key(&innerkey))
    110             }
    111             mvcc::Key::Version(innerkey, version) => {
    112                 format!("mvcc:Version({}, {version})", I::key(&innerkey))
    113             }
    114             mvcc::Key::Unversioned(innerkey) => {
    115                 format!("mvcc:Unversioned({})", I::key(&innerkey))
    116             }
    117             mvcc::Key::NextVersion | mvcc::Key::TxnActive(_) | mvcc::Key::TxnActiveSnapshot(_) => {
    118                 format!("mvcc:{key:?}")
    119             }
    120         }
    121     }
    122 
    123     fn value(key: &[u8], value: &[u8]) -> String {
    124         let Ok(key) = mvcc::Key::decode(key) else { return Raw::bytes(value) };
    125         match key {
    126             mvcc::Key::NextVersion => {
    127                 let Ok(version) = bincode::deserialize::<mvcc::Version>(value) else {
    128                     return Raw::bytes(value);
    129                 };
    130                 version.to_string()
    131             }
    132             mvcc::Key::TxnActiveSnapshot(_) => {
    133                 let Ok(active) = bincode::deserialize::<BTreeSet<u64>>(value) else {
    134                     return Raw::bytes(value);
    135                 };
    136                 format!("{{{}}}", active.iter().map(|v| v.to_string()).join(","))
    137             }
    138             mvcc::Key::TxnActive(_) | mvcc::Key::TxnWrite(_, _) => Raw::bytes(value),
    139             mvcc::Key::Version(userkey, _) => match bincode::deserialize(value) {
    140                 Ok(Some(value)) => I::value(&userkey, value),
    141                 Ok(None) => "None".to_string(),
    142                 Err(_) => Raw::bytes(value),
    143             },
    144             mvcc::Key::Unversioned(userkey) => I::value(&userkey, value),
    145         }
    146     }
    147 }

    149 /// Formats SQL keys/values.
    150 pub struct SQL;
    151 
    152 impl SQL {
    153     fn values(values: impl IntoIterator<Item = sql::types::Value>) -> String {
    154         values.into_iter().join(",")
    155     }
    156 
    157     fn schema(table: sql::types::Table) -> String {
    158         let re = regex::Regex::new(r#"\n\s*"#).expect("regex failed");
    159         re.replace_all(&table.to_string(), " ").into_owned()
    160     }
    161 }
    162 
    163 impl Formatter for SQL {
    164     fn key(key: &[u8]) -> String {
    165         // Special-case the applied_index key.
    166         if key == sql::engine::Raft::APPLIED_INDEX_KEY {
    167             return String::from_utf8_lossy(key).into_owned();
    168         }
    169 
    170         let Ok(key) = sql::engine::Key::decode(key) else { return Raw::key(key) };
    171         match key {
    172             sql::engine::Key::Table(name) => format!("sql:Table({name})"),
    173             sql::engine::Key::Index(table, column, value) => {
    174                 format!("sql:Index({table}.{column}, {value})")
    175             }
    176             sql::engine::Key::Row(table, id) => {
    177                 format!("sql:Row({table}, {id})")
    178             }
    179         }
    180     }
    181 
    182     fn value(key: &[u8], value: &[u8]) -> String {
    183         // Special-case the applied_index key.
    184         if key == sql::engine::Raft::APPLIED_INDEX_KEY {
    185             if let Ok(applied_index) = bincode::deserialize::<raft::Index>(value) {
    186                 return applied_index.to_string();
    187             }
    188         }
    189 
    190         let Ok(key) = sql::engine::Key::decode(key) else { return Raw::key(value) };
    191         match key {
    192             sql::engine::Key::Table(_) => {
    193                 let Ok(table) = bincode::deserialize::<sql::types::Table>(value) else {
    194                     return Raw::bytes(value);
    195                 };
    196                 Self::schema(table)
    197             }
    198             sql::engine::Key::Row(_, _) => {
    199                 let Ok(row) = bincode::deserialize::<sql::types::Row>(value) else {
    200                     return Raw::bytes(value);
    201                 };
    202                 Self::values(row)
    203             }
    204             sql::engine::Key::Index(_, _, _) => {
    205                 let Ok(index) = bincode::deserialize::<BTreeSet<sql::types::Value>>(value) else {
    206                     return Raw::bytes(value);
    207                 };
    208                 Self::values(index)
    209             }
    210         }
    211     }
    212 }

    214 /// Formats SQL Raft write commands, from the Raft log.
    215 pub struct SQLCommand;
    216 
    217 impl Formatter for SQLCommand {
    218     /// There is no key, since they're wrapped in a Raft log entry.
    219     fn key(_: &[u8]) -> String {
    220         panic!("SQL commands don't have a key");
    221     }
    222 
    223     fn value(_: &[u8], value: &[u8]) -> String {
    224         let Ok(write) = sql::engine::Write::decode(value) else {
    225             return Raw::bytes(value);
    226         };
    227 
    228         let txn = match &write {
    229             sql::engine::Write::Begin => None,
    230             sql::engine::Write::Commit(txn)
    231             | sql::engine::Write::Rollback(txn)
    232             | sql::engine::Write::Delete { txn, .. }
    233             | sql::engine::Write::Insert { txn, .. }
    234             | sql::engine::Write::Update { txn, .. }
    235             | sql::engine::Write::CreateTable { txn, .. }
    236             | sql::engine::Write::DropTable { txn, .. } => Some(txn),
    237         };
    238         let ftxn =
    239             txn.filter(|t| !t.read_only).map(|t| format!("t{} ", t.version)).unwrap_or_default();
    240 
    241         let fcommand = match write {
    242             sql::engine::Write::Begin => "BEGIN".to_string(),
    243             sql::engine::Write::Commit(_) => "COMMIT".to_string(),
    244             sql::engine::Write::Rollback(_) => "ROLLBACK".to_string(),
    245             sql::engine::Write::Delete { table, ids, .. } => {
    246                 format!("DELETE {table} {}", ids.iter().map(|id| id.to_string()).join(","))
    247             }
    248             sql::engine::Write::Insert { table, rows, .. } => {
    249                 format!(
    250                     "INSERT {table} {}",
    251                     rows.into_iter().map(|row| format!("({})", SQL::values(row))).join(" ")
    252                 )
    253             }
    254             sql::engine::Write::Update { table, rows, .. } => format!(
    255                 "UPDATE {table} {}",
    256                 rows.into_iter().map(|(id, row)| format!("{id}→({})", SQL::values(row))).join(" ")
    257             ),
    258             sql::engine::Write::CreateTable { schema, .. } => SQL::schema(schema),
    259             sql::engine::Write::DropTable { table, .. } => format!("DROP TABLE {table}"),
    260         };
    261         format!("{ftxn}{fcommand}")
    262     }
