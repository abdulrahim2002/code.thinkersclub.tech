     14 /// Adds automatic Keycode encode/decode methods to key enums. These are
     15 /// primarily meant for keys stored in key/value storage engines.
     16 pub trait Key<'de>: Serialize + Deserialize<'de> {
     17     /// Decodes a key from a byte slice using Keycode.
     18     fn decode(bytes: &'de [u8]) -> Result<Self> {
     19         keycode::deserialize(bytes)
     20     }
     21 
     22     /// Encodes a key to a byte vector using Keycode.
     23     ///
     24     /// In the common case, the encoded key is borrowed for a storage engine
     25     /// call and then thrown away. We could avoid a bunch of allocations by
     26     /// taking a reusable byte vector to encode into and return a reference to
     27     /// it, but we keep it simple.
     28     fn encode(&self) -> Vec<u8> {
     29         keycode::serialize(self)
     30     }
     31 }
     32 
     33 /// Adds automatic Bincode encode/decode methods to value types. These are used
     34 /// not only for values in key/value storage engines, but also for e.g. network
     35 /// protocol messages and other values.
     36 pub trait Value: Serialize + DeserializeOwned {
     37     /// Decodes a value from a byte slice using Bincode.
     38     fn decode(bytes: &[u8]) -> Result<Self> {
     39         bincode::deserialize(bytes)
     40     }
     41 
     42     /// Decodes a value from a reader using Bincode.
     43     fn decode_from<R: std::io::Read>(reader: R) -> Result<Self> {
     44         bincode::deserialize_from(reader)
     45     }
     46 
     47     /// Decodes a value from a reader using Bincode, or returns None if the
     48     /// reader is closed.
     49     fn maybe_decode_from<R: std::io::Read>(reader: R) -> Result<Option<Self>> {
     50         bincode::maybe_deserialize_from(reader)
     51     }
     52 
     53     /// Encodes a value to a byte vector using Bincode.
     54     fn encode(&self) -> Vec<u8> {
     55         bincode::serialize(self)
     56     }
     57 
     58     /// Encodes a value into a writer using Bincode.
     59     fn encode_into<W: std::io::Write>(&self, writer: W) -> Result<()> {
     60         bincode::serialize_into(writer, self)
     61     }
     62 }
     63 
     64 /// Blanket implementations for various types wrapping a value type.
     65 impl<V: Value> Value for Option<V> {}
     66 impl<V: Value> Value for Result<V> {}
     67 impl<V: Value> Value for Vec<V> {}
     68 impl<V1: Value, V2: Value> Value for (V1, V2) {}
     69 impl<V: Value + std::cmp::Eq + std::hash::Hash> Value for HashSet<V> {}
