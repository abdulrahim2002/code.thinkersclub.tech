      1 //! KeyCode is a lexicographical order-preserving binary encoding for use with
      2 //! keys. It is designed for simplicity, not efficiency (i.e. it does not use
      3 //! varints or other compression methods).
      4 //!
      5 //! Ordering is important because it allows limited scans across specific parts
      6 //! of the keyspace, e.g. scanning an individual table or using an index range
      7 //! predicate like WHERE id < 100. It also avoids sorting in some cases where
      8 //! the keys are already in the desired order, e.g. in the Raft log.
      9 //!
     10 //! The encoding is not self-describing: the caller must provide a concrete type
     11 //! to decode into, and the binary key must conform to its structure.
     12 //!
     13 //! KeyCode supports a subset of primitive data types, encoded as follows:
     14 //!
     15 //! bool:    0x00 for false, 0x01 for true.
     16 //! u64:     Big-endian binary representation.
     17 //! i64:     Big-endian binary representation, with sign bit flipped.
     18 //! f64:     Big-endian binary representation, with sign bit flipped, and rest if negative.
     19 //! Vec<u8>: 0x00 is escaped as 0x00ff, terminated with 0x0000.
     20 //! String:  Like Vec<u8>.
     21 //!
     22 //! Additionally, several container types are supported:
     23 //!
     24 //! Tuple:  Concatenation of elements, with no surrounding structure.
     25 //! Array:  Like tuple.
     26 //! Vec:    Like tuple.
     27 //! Enum:   The variant's enum index as a single u8 byte.
     28 //!
     29 //! SQL Value enums are encoded according to the above scheme, i.e. a single
     30 //! byte identifying the enum variant by index, then the primitive value.
     31 //!
     32 //! The canonical key reprentation is an enum -- for example:
     33 //!
     34 //! ```
     35 //! #[derive(Debug, Deserialize, Serialize)]
     36 //! enum Key {
     37 //!     Foo,
     38 //!     Bar(String),
     39 //!     Baz(bool, u64, #[serde(with = "serde_bytes")] Vec<u8>),
     40 //! }
     41 //! ```
     42 //!
     43 //! Unfortunately, byte vectors and slices such as Vec<u8> must be wrapped with
     44 //! serde_bytes::ByteBuf or use the #[serde(with="serde_bytes")] attribute. See
     45 //! https://github.com/serde-rs/bytes
     46

     52 /// Serializes a key to a binary KeyCode representation.
     53 ///
     54 /// In the common case, the encoded key is borrowed for a storage engine call
     55 /// and then thrown away. We could avoid a bunch of allocations by taking a
     56 /// reusable byte vector to encode into and return a reference to it, but we
     57 /// keep it simple.
     58 pub fn serialize<T: ser::Serialize>(key: &T) -> Vec<u8> {
     59     let mut serializer = Serializer { output: Vec::new() };
     60     // Panic on serialization failures, as this is typically an issue with the
     61     // provided data structure.
     62     key.serialize(&mut serializer).expect("keycode serialization failed");
     63     serializer.output
     64 }

     66 /// Deserializes a key from a binary KeyCode representation.
     67 pub fn deserialize<'a, T: de::Deserialize<'a>>(input: &'a [u8]) -> Result<T> {
     68     let mut deserializer = Deserializer::from_bytes(input);
     69     let t = T::deserialize(&mut deserializer)?;
     70     if !deserializer.input.is_empty() {
     71         return errdata!(
     72             "unexpected trailing bytes {:x?} at end of key {input:x?}",
     73             deserializer.input,
     74         );
     75     }
     76     Ok(t)
     77 }

     79 /// Generates a key range for a key prefix, used e.g. for prefix scans.
     80 ///
     81 /// The exclusive end bound is generated by adding 1 to the value of the last
     82 /// byte. If the last byte(s) is 0xff (so adding 1 would overflow), we instead
     83 /// find the latest non-0xff byte, increment that, and truncate the rest. If all
     84 /// bytes are 0xff, we scan to the end of the range, since there can't be other
     85 /// prefixes after it.
     86 pub fn prefix_range(prefix: &[u8]) -> (std::ops::Bound<Vec<u8>>, std::ops::Bound<Vec<u8>>) {
     87     let start = std::ops::Bound::Included(prefix.to_vec());
     88     let end = match prefix.iter().rposition(|b| *b != 0xff) {
     89         Some(i) => std::ops::Bound::Excluded(
     90             prefix.iter().take(i).copied().chain(std::iter::once(prefix[i] + 1)).collect(),
     91         ),
     92         None => std::ops::Bound::Unbounded,
     93     };
     94     (start, end)
     95 }

    132     /// i64 uses the big-endian two's completement encoding, but flips the
    133     /// left-most sign bit such that negative numbers are ordered before
    134     /// positive numbers.
    135     ///
    136     /// The relative ordering of the remaining bits is already correct: -1, the
    137     /// largest negative integer, is encoded as 01111111...11111111, ordered
    138     /// after all other negative integers but before positive integers.
    139     fn serialize_i64(self, v: i64) -> Result<()> {
    140         let mut bytes = v.to_be_bytes();
    141         bytes[0] ^= 1 << 7; // flip sign bit
    142         self.output.extend(bytes);
    143         Ok(())
    144     }

    169     /// positive numbers after negative numbers, and also flips all other bits
    170     /// for negative numbers to order them from smallest to greatest. NaN is
    171     /// ordered at the end.
    172     fn serialize_f64(self, v: f64) -> Result<()> {
    173         let mut bytes = v.to_be_bytes();
    174         if bytes[0] & 1 << 7 == 0 {
    175             bytes[0] ^= 1 << 7; // positive, flip sign bit
    176         } else {
    177             bytes.iter_mut().for_each(|b| *b = !*b); // negative, flip all bits
    178         }
    179         self.output.extend(bytes);
    180         Ok(())
    181     }

    347 impl<'de> Deserializer<'de> {
    348     /// Creates a deserializer for a byte slice.
    349     pub fn from_bytes(input: &'de [u8]) -> Self {
    350         Deserializer { input }
    351     }
    352 
    353     /// Chops off and returns the next len bytes of the byte slice, or errors if
    354     /// there aren't enough bytes left.
    355     fn take_bytes(&mut self, len: usize) -> Result<&[u8]> {
    356         if self.input.len() < len {
    357             return errdata!("insufficient bytes, expected {len} bytes for {:x?}", self.input);
    358         }
    359         let bytes = &self.input[..len];
    360         self.input = &self.input[len..];
    361         Ok(bytes)
    362     }
    363 
    364     /// Decodes and chops off the next encoded byte slice.
    365     fn decode_next_bytes(&mut self) -> Result<Vec<u8>> {
    366         // We can't easily share state between Iterator.scan() and
    367         // Iterator.filter() when processing escape sequences, so use a
    368         // straightforward loop.
    369         let mut decoded = Vec::new();
    370         let mut iter = self.input.iter().enumerate();
    371         let taken = loop {
    372             match iter.next() {
    373                 Some((_, 0x00)) => match iter.next() {
    374                     Some((i, 0x00)) => break i + 1,        // terminator
    375                     Some((_, 0xff)) => decoded.push(0x00), // escaped 0x00
    376                     _ => return errdata!("invalid escape sequence"),
    377                 },
    378                 Some((_, b)) => decoded.push(*b),
    379                 None => return errdata!("unexpected end of input"),
    380             }
    381         };
    382         self.input = &self.input[taken..];
    383         Ok(decoded)
    384     }
    385 }

    612     #[derive(Debug, Deserialize, Serialize, PartialEq)]
    613     enum Key<'a> {
    614         Unit,
    615         NewType(String),
    616         Tuple(bool, #[serde(with = "serde_bytes")] Vec<u8>, u64),
    617         Cow(
    618             #[serde(with = "serde_bytes")]
    619             #[serde(borrow)]
    620             Cow<'a, [u8]>,
    621             bool,
    622             #[serde(borrow)] Cow<'a, str>,
    623         ),
    624     }

    626     /// Assert that serializing a value yields the expected byte sequence (as a
    627     /// hex-encoded string), and that deserializing it yields the original value.
    628     macro_rules! test_serialize_deserialize {
    629         ( $( $name:ident: $input:expr => $expect:literal, )* ) => {
    630         $(
    631             #[test]
    632             fn $name() -> Result<()> {
    633                 let mut input = $input;
    634                 let expect = $expect;
    635                 let output = serialize(&input);
    636                 assert_eq!(hex::encode(&output), expect, "encode failed");
    637 
    638                 let expect = input;
    639                 input = deserialize(&output)?; // reuse input variable for proper type
    640                 assert_eq!(input, expect, "decode failed");
    641                 Ok(())
    642             }
    643         )*
    644         };
    645     }

    647     /// Assert that deserializing invalid inputs results in errors. Takes byte
    648     /// slices (as hex-encoded strings) and the type to deserialize into.
    649     macro_rules! test_deserialize_error {
    650         ( $( $name:ident: $input:literal as $type:ty, )* ) => {
    651         paste! {
    652         $(
    653             #[test]
    654             #[should_panic]
    655             fn [< $name _deserialize_error >]() {
    656                 let bytes = hex::decode($input).unwrap();
    657                 deserialize::<$type>(&bytes).unwrap();
    658             }
    659         )*
    660         }
    661         };
    662     }
