+#[derive(Clone, Debug, PartialEq)]
+pub enum Keyword {
+    As,
+    Select,

+impl Keyword {
+    fn from_str(ident: &str) -> Option<Self> {
+        Some(match ident.to_uppercase().as_ref() {
+            "AS" => Self::As,
+            "SELECT" => Self::Select,
+            _ => return None,
+        })
+    }
+
+    fn to_str(&self) -> &str {
+        match self {
+            Self::As => "AS",
+            Self::Select => "SELECT",
+        }
+    }
+}
