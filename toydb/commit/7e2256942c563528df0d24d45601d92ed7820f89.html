+fn execute<E: Engine>(engine: &E, query: &str) -> Result<ResultSet, Error> {
+    let mut txn = engine.begin()?;
+    let ast = Parser::new(query).parse()?;
+    let plan = Plan::build(ast)?.optimize()?;
+    let result = plan.execute(Context { txn: &mut txn });
+    if result.is_ok() {
+        txn.commit()?;
+    } else {
+        txn.rollback()?;
+    }
+    result
+}
