     24 /* sorts the linked list by changing next pointers (not sub1) */
     25 void MergeSort(struct Node** headRef)
     26 {
     27 	struct Node* head = *headRef;
     28 	struct Node* a;
     29 	struct Node* b;
     30 
     31 	/* Base case -- length 0 or 1 */
     32 	if ((head == NULL) || (head->next == NULL)) {
     33 		return;
     34 	}
     35 
     36 	/* Split head into 'a' and 'b' sublists */
     37 	FrontBackSplit(head, &a, &b);
     38 
     39 	/* Recursively sort the sublists */
     40 	MergeSort(&a);
     41 	MergeSort(&b);
     42 
     43 	/* answer = merge the two sorted lists together */
     44 	*headRef = SortedMerge(a, b);
     45 }

     48 struct Node* SortedMerge(struct Node* a, struct Node* b)
     49 {
     50 	struct Node* result = NULL;
     51 
     52 	/* Base cases */
     53 	if (a == NULL)
     54 		return (b);
     55 	else if (b == NULL)
     56 		return (a);
     57 
     58 	/* Pick either a or b, and recur */
     59 	if (a->sub1 >= b->sub1) {
     60 		result = a;
     61 		result->next = SortedMerge(a->next, b);
     62 	}
     63 	else {
     64 		result = b;
     65 		result->next = SortedMerge(a, b->next);
     66 	}
     67 	return (result);
     68 }

     71 /* Split the nodes of the given list into front and back halves,
     72 	and return the two lists using the reference parameters.
     73 	If the length is odd, the extra node should go in the front list.
     74 	Uses the fast/slow pointer strategy. */
     75 void FrontBackSplit(struct Node* source,
     76 					struct Node** frontRef, struct Node** backRef)
     77 {
     78 	struct Node* fast;
     79 	struct Node* slow;
     80 	slow = source;
     81 	fast = source->next;
     82 
     83 	/* Advance 'fast' two nodes, and advance 'slow' one node */
     84 	while (fast != NULL) {
     85 		fast = fast->next;
     86 		if (fast != NULL) {
     87 			slow = slow->next;
     88 			fast = fast->next;
     89 		}
     90 	}
     91 
     92 	/* 'slow' is before the midpoint in the list, so split it in two
     93 	at that point. */
     94 	*frontRef = source;
     95 	*backRef = slow->next;
     96 	slow->next = NULL;
     97 }
