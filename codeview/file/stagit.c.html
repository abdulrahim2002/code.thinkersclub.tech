    116 commitinfo_getstats(struct commitinfo *ci)
    117 {
    118 	struct deltainfo *di;
    119 	git_diff_options opts;
    120 	git_diff_find_options fopts;
    121 	const git_diff_delta *delta;
    122 	const git_diff_hunk *hunk;
    123 	const git_diff_line *line;
    124 	git_patch *patch = NULL;
    125 	size_t ndeltas, nhunks, nhunklines;
    126 	size_t i, j, k;
    127 
    128 	if (git_tree_lookup(&(ci->commit_tree), repo, git_commit_tree_id(ci->commit)))
    129 		goto err;
    130 	if (!git_commit_parent(&(ci->parent), ci->commit, 0)) {
    131 		if (git_tree_lookup(&(ci->parent_tree), repo, git_commit_tree_id(ci->parent))) {
    132 			ci->parent = NULL;
    133 			ci->parent_tree = NULL;
    134 		}
    135 	}
    136 
    137 	git_diff_init_options(&opts, GIT_DIFF_OPTIONS_VERSION);
    138 	opts.flags |= GIT_DIFF_DISABLE_PATHSPEC_MATCH |
    139 	              GIT_DIFF_IGNORE_SUBMODULES |
    140 		      GIT_DIFF_INCLUDE_TYPECHANGE;
    141 	if (git_diff_tree_to_tree(&(ci->diff), repo, ci->parent_tree, ci->commit_tree, &opts))
    142 		goto err;
    143 
    144 	if (git_diff_find_init_options(&fopts, GIT_DIFF_FIND_OPTIONS_VERSION))
    145 		goto err;
    146 	/* find renames and copies, exact matches (no heuristic) for renames. */
    147 	fopts.flags |= GIT_DIFF_FIND_RENAMES | GIT_DIFF_FIND_COPIES |
    148 	               GIT_DIFF_FIND_EXACT_MATCH_ONLY;
    149 	if (git_diff_find_similar(ci->diff, &fopts))
    150 		goto err;
    151 
    152 	ndeltas = git_diff_num_deltas(ci->diff);
    153 	if (ndeltas && !(ci->deltas = calloc(ndeltas, sizeof(struct deltainfo *))))
    154 		err(1, "calloc");
    155 
    156 	for (i = 0; i < ndeltas; i++) {
    157 		if (git_patch_from_diff(&patch, ci->diff, i))
    158 			goto err;
    159 
    160 		if (!(di = calloc(1, sizeof(struct deltainfo))))
    161 			err(1, "calloc");
    162 		di->patch = patch;
    163 		ci->deltas[i] = di;
    164 
    165 		delta = git_patch_get_delta(patch);
    166 
    167 		/* skip stats for binary data */
    168 		if (delta->flags & GIT_DIFF_FLAG_BINARY)
    169 			continue;
    170 
    171 		nhunks = git_patch_num_hunks(patch);
    172 		for (j = 0; j < nhunks; j++) {
    173 			if (git_patch_get_hunk(&hunk, &nhunklines, patch, j))
    174 				break;
    175 			for (k = 0; ; k++) {
    176 				if (git_patch_get_line_in_hunk(&line, patch, j, k))
    177 					break;
    178 				if (line->old_lineno == -1) {
    179 					di->addcount++;
    180 					ci->addcount++;
    181 				} else if (line->new_lineno == -1) {
    182 					di->delcount++;
    183 					ci->delcount++;
    184 				}
    185 			}
    186 		}
    187 	}
    188 	ci->ndeltas = i;
    189 	ci->filecount = i;
    190 
    191 	return 0;
    192 
    193 err:
    194 	git_diff_free(ci->diff);
    195 	ci->diff = NULL;
    196 	git_tree_free(ci->commit_tree);
    197 	ci->commit_tree = NULL;
    198 	git_tree_free(ci->parent_tree);
    199 	ci->parent_tree = NULL;
    200 	git_commit_free(ci->parent);
    201 	ci->parent = NULL;
    202 
    203 	if (ci->deltas)
    204 		for (i = 0; i < ci->ndeltas; i++)
    205 			deltainfo_free(ci->deltas[i]);
    206 	free(ci->deltas);
    207 	ci->deltas = NULL;
    208 	ci->ndeltas = 0;
    209 	ci->addcount = 0;
    210 	ci->delcount = 0;
    211 	ci->filecount = 0;
    212 
    213 	return -1;
    214 }

    265 int
    266 refs_cmp(const void *v1, const void *v2)
    267 {
    268 	const struct referenceinfo *r1 = v1, *r2 = v2;
    269 	time_t t1, t2;
    270 	int r;
    271 
    272 	if ((r = git_reference_is_tag(r1->ref) - git_reference_is_tag(r2->ref)))
    273 		return r;
    274 
    275 	t1 = r1->ci->author ? r1->ci->author->when.time : 0;
    276 	t2 = r2->ci->author ? r2->ci->author->when.time : 0;
    277 	if ((r = t1 > t2 ? -1 : (t1 == t2 ? 0 : 1)))
    278 		return r;
    279 
    280 	return strcmp(git_reference_shorthand(r1->ref),
    281 	              git_reference_shorthand(r2->ref));
    282 }

    284 int
    285 getrefs(struct referenceinfo **pris, size_t *prefcount)
    286 {
    287 	struct referenceinfo *ris = NULL;
    288 	struct commitinfo *ci = NULL;
    289 	git_reference_iterator *it = NULL;
    290 	const git_oid *id = NULL;
    291 	git_object *obj = NULL;
    292 	git_reference *dref = NULL, *r, *ref = NULL;
    293 	size_t i, refcount;
    294 
    295 	*pris = NULL;
    296 	*prefcount = 0;
    297 
    298 	if (git_reference_iterator_new(&it, repo))
    299 		return -1;
    300 
    301 	for (refcount = 0; !git_reference_next(&ref, it); ) {
    302 		if (!git_reference_is_branch(ref) && !git_reference_is_tag(ref)) {
    303 			git_reference_free(ref);
    304 			ref = NULL;
    305 			continue;
    306 		}
    307 
    308 		switch (git_reference_type(ref)) {
    309 		case GIT_REF_SYMBOLIC:
    310 			if (git_reference_resolve(&dref, ref))
    311 				goto err;
    312 			r = dref;
    313 			break;
    314 		case GIT_REF_OID:
    315 			r = ref;
    316 			break;
    317 		default:
    318 			continue;
    319 		}
    320 		if (!git_reference_target(r) ||
    321 		    git_reference_peel(&obj, r, GIT_OBJ_ANY))
    322 			goto err;
    323 		if (!(id = git_object_id(obj)))
    324 			goto err;
    325 		if (!(ci = commitinfo_getbyoid(id)))
    326 			break;
    327 
    328 		if (!(ris = reallocarray(ris, refcount + 1, sizeof(*ris))))
    329 			err(1, "realloc");
    330 		ris[refcount].ci = ci;
    331 		ris[refcount].ref = r;
    332 		refcount++;
    333 
    334 		git_object_free(obj);
    335 		obj = NULL;
    336 		git_reference_free(dref);
    337 		dref = NULL;
    338 	}
    339 	git_reference_iterator_free(it);
    340 
    341 	/* sort by type, date then shorthand name */
    342 	qsort(ris, refcount, sizeof(*ris), refs_cmp);
    343 
    344 	*pris = ris;
    345 	*prefcount = refcount;
    346 
    347 	return 0;
    348 
    349 err:
    350 	git_object_free(obj);
    351 	git_reference_free(dref);
    352 	commitinfo_free(ci);
    353 	for (i = 0; i < refcount; i++) {
    354 		commitinfo_free(ris[i].ci);
    355 		git_reference_free(ris[i].ref);
    356 	}
    357 	free(ris);
    358 
    359 	return -1;
    360 }

    373 /* Percent-encode, see RFC3986 section 2.1. */
    374 void
    375 percentencode(FILE *fp, const char *s, size_t len)
    376 {
    377 	static char tab[] = "0123456789ABCDEF";
    378 	unsigned char uc;
    379 	size_t i;
    380 
    381 	for (i = 0; *s && i < len; s++, i++) {
    382 		uc = *s;
    383 		/* NOTE: do not encode '/' for paths or ",-." */
    384 		if (uc < ',' || uc >= 127 || (uc >= ':' && uc <= '@') ||
    385 		    uc == '[' || uc == ']') {
    386 			putc('%', fp);
    387 			putc(tab[(uc >> 4) & 0x0f], fp);
    388 			putc(tab[uc & 0x0f], fp);
    389 		} else {
    390 			putc(uc, fp);
    391 		}
    392 	}
    393 }

    662 		}
    663 	}
    664

    698 void
    699 printshowfile(FILE *fp, struct commitinfo *ci)
    700 {
    701 	const git_diff_delta *delta;
    702 	const git_diff_hunk *hunk;
    703 	const git_diff_line *line;
    704 	git_patch *patch;
    705 	size_t nhunks, nhunklines, changed, add, del, total, i, j, k;
    706 	char linestr[80];
    707 	int c;
    708 
    709 	printcommit(fp, ci);
    710 
    711 	if (!ci->deltas)
    712 		return;
    713 
    714 	if (ci->filecount > 1000   ||
    715 	    ci->ndeltas   > 1000   ||
    716 	    ci->addcount  > 100000 ||
    717 	    ci->delcount  > 100000) {
    718 		fputs("Diff is too large, output suppressed.\n", fp);
    719 		return;
    720 	}
    721 
    722 	/* diff stat */
    723 	fputs("<b>Diffstat:</b>\n<table>", fp);
    724 	for (i = 0; i < ci->ndeltas; i++) {
    725 		delta = git_patch_get_delta(ci->deltas[i]->patch);
    726 
    727 		switch (delta->status) {
    728 		case GIT_DELTA_ADDED:      c = 'A'; break;
    729 		case GIT_DELTA_COPIED:     c = 'C'; break;
    730 		case GIT_DELTA_DELETED:    c = 'D'; break;
    731 		case GIT_DELTA_MODIFIED:   c = 'M'; break;
    732 		case GIT_DELTA_RENAMED:    c = 'R'; break;
    733 		case GIT_DELTA_TYPECHANGE: c = 'T'; break;
    734 		default:                   c = ' '; break;
    735 		}
    736 		if (c == ' ')
    737 			fprintf(fp, "<tr><td>%c", c);
    738 		else
    739 			fprintf(fp, "<tr><td class=\"%c\">%c", c, c);
    740 
    741 		fprintf(fp, "</td><td><a href=\"#h%zu\">", i);
    742 		xmlencode(fp, delta->old_file.path, strlen(delta->old_file.path));
    743 		if (strcmp(delta->old_file.path, delta->new_file.path)) {
    744 			fputs(" -&gt; ", fp);
    745 			xmlencode(fp, delta->new_file.path, strlen(delta->new_file.path));
    746 		}
    747 
    748 		add = ci->deltas[i]->addcount;
    749 		del = ci->deltas[i]->delcount;
    750 		changed = add + del;
    751 		total = sizeof(linestr) - 2;
    752 		if (changed > total) {
    753 			if (add)
    754 				add = ((float)total / changed * add) + 1;
    755 			if (del)
    756 				del = ((float)total / changed * del) + 1;
    757 		}
    758 		memset(&linestr, '+', add);
    759 		memset(&linestr[add], '-', del);
    760 
    761 		fprintf(fp, "</a></td><td> | </td><td class=\"num\">%zu</td><td><span class=\"i\">",
    762 		        ci->deltas[i]->addcount + ci->deltas[i]->delcount);
    763 		fwrite(&linestr, 1, add, fp);
    764 		fputs("</span><span class=\"d\">", fp);
    765 		fwrite(&linestr[add], 1, del, fp);
    766 		fputs("</span></td></tr>\n", fp);
    767 	}
    768 	fprintf(fp, "</table></pre><pre>%zu file%s changed, %zu insertion%s(+), %zu deletion%s(-)\n",
    769 		ci->filecount, ci->filecount == 1 ? "" : "s",
    770 	        ci->addcount,  ci->addcount  == 1 ? "" : "s",
    771 	        ci->delcount,  ci->delcount  == 1 ? "" : "s");
    772 
    773 	fputs("<hr/>", fp);
    774 
    775 	for (i = 0; i < ci->ndeltas; i++) {
    776 		patch = ci->deltas[i]->patch;
    777 		delta = git_patch_get_delta(patch);
    778 		fprintf(fp, "<b>diff --git a/<a id=\"h%zu\" href=\"%sfile/", i, relpath);
    779 		percentencode(fp, delta->old_file.path, strlen(delta->old_file.path));
    780 		fputs(".html\">", fp);
    781 		xmlencode(fp, delta->old_file.path, strlen(delta->old_file.path));
    782 		fprintf(fp, "</a> b/<a href=\"%sfile/", relpath);
    783 		percentencode(fp, delta->new_file.path, strlen(delta->new_file.path));
    784 		fprintf(fp, ".html\">");
    785 		xmlencode(fp, delta->new_file.path, strlen(delta->new_file.path));
    786 		fprintf(fp, "</a></b>\n");
    787 
    788 		/* check binary data */
    789 		if (delta->flags & GIT_DIFF_FLAG_BINARY) {
    790 			fputs("Binary files differ.\n", fp);
    791 			continue;
    792 		}
    793 
    794 		nhunks = git_patch_num_hunks(patch);
    795 		for (j = 0; j < nhunks; j++) {
    796 			if (git_patch_get_hunk(&hunk, &nhunklines, patch, j))
    797 				break;
    798 
    799 			fprintf(fp, "<a href=\"#h%zu-%zu\" id=\"h%zu-%zu\" class=\"h\">", i, j, i, j);
    800 			xmlencode(fp, hunk->header, hunk->header_len);
    801 			fputs("</a>", fp);
    802 
    803 			for (k = 0; ; k++) {
    804 				if (git_patch_get_line_in_hunk(&line, patch, j, k))
    805 					break;
    806 				if (line->old_lineno == -1)
    807 					fprintf(fp, "<a href=\"#h%zu-%zu-%zu\" id=\"h%zu-%zu-%zu\" class=\"i\">+",
    808 						i, j, k, i, j, k);
    809 				else if (line->new_lineno == -1)
    810 					fprintf(fp, "<a href=\"#h%zu-%zu-%zu\" id=\"h%zu-%zu-%zu\" class=\"d\">-",
    811 						i, j, k, i, j, k);
    812 				else
    813 					putc(' ', fp);
    814 				xmlencodeline(fp, line->content, line->content_len);
    815 				putc('\n', fp);
    816 				if (line->old_lineno == -1 || line->new_lineno == -1)
    817 					fputs("</a>", fp);
    818 			}
    819 		}
    820 	}
    821 }

    870 			errx(1, "path truncated: 'commit/%s.html'", oidstr);
    871 		r = access(path, F_OK);
    872 
    873 		/* optimization: if there are no log lines to write and
    874 		   the commit file already exists: skip the diffstat */
    875 		if (!nlogcommits) {
    876 			remcommits++;
    877 			if (!r)
    878 				continue;
    879 		}
    880 
    881 		if (!(ci = commitinfo_getbyoid(&id)))
    882 			break;

    884 		if (commitinfo_getstats(ci) == -1)
    885 			goto err;
    886

    888 			writelogline(fp, ci);
    889 			if (nlogcommits > 0)
    890 				nlogcommits--;
    891 		}
    892 
    893 		if (cachefile)
    894 			writelogline(wcachefp, ci);
    895 
    896 		/* check if file exists if so skip it */
    897 		if (r) {
    898 			relpath = "../";
    899 			fpfile = efopen(path, "w");
    900 			writeheader(fpfile, ci->summary);
    901 			fputs("<pre>", fpfile);
    902 			printshowfile(fpfile, ci);
    903 			fputs("</pre>\n", fpfile);
    904 			writefooter(fpfile);
    905 			checkfileerror(fpfile, path, 'w');
    906 			fclose(fpfile);
    907 		}
    908 err:
    909 		commitinfo_free(ci);
    910 	}
    911 	git_revwalk_free(w);
    912 
    913 	if (nlogcommits == 0 && remcommits != 0) {
    914 		fprintf(fp, "<tr><td></td><td colspan=\"5\">"
    915 		        "%zu more commits remaining, fetch the repository"
    916 		        "</td></tr>\n", remcommits);
    917 	}
    918 
    919 	relpath = "";
    920 
    921 	return 0;

   1004 				break;

   1016 			commitinfo_free(ris[i].ci);
   1017 			git_reference_free(ris[i].ref);
   1018 		}
   1019 		free(ris);
   1020 	}
   1021 
   1022 	fputs("</feed>\n", fp);
   1023 
   1024 	return 0;
   1025 }

   1027 size_t
   1028 writeblob(git_object *obj, const char *fpath, const char *filename, size_t filesize)
   1029 {
   1030 	char tmp[PATH_MAX] = "", *d;
   1031 	const char *p;
   1032 	size_t lc = 0;
   1033 	FILE *fp;
   1034 
   1035 	if (strlcpy(tmp, fpath, sizeof(tmp)) >= sizeof(tmp))
   1036 		errx(1, "path truncated: '%s'", fpath);
   1037 	if (!(d = dirname(tmp)))
   1038 		err(1, "dirname");
   1039 	if (mkdirp(d))
   1040 		return -1;
   1041 
   1042 	for (p = fpath, tmp[0] = '\0'; *p; p++) {
   1043 		if (*p == '/' && strlcat(tmp, "../", sizeof(tmp)) >= sizeof(tmp))
   1044 			errx(1, "path truncated: '../%s'", tmp);
   1045 	}
   1046 	relpath = tmp;
   1047 
   1048 	fp = efopen(fpath, "w");
   1049 	writeheader(fp, filename);
   1050 	fputs("<p> ", fp);
   1051 	xmlencode(fp, filename, strlen(filename));
   1052 	fprintf(fp, " (%zuB)", filesize);
   1053 	fputs("</p><hr/>", fp);
   1054 
   1055 	if (git_blob_is_binary((git_blob *)obj))
   1056 		fputs("<p>Binary file.</p>\n", fp);
   1057 	else
   1058 		// passing parameter to be used by the code
   1059 		lc = writeblobhtml(fp, (git_blob *)obj, filename);
   1060 
   1061 	writefooter(fp);
   1062 	checkfileerror(fp, fpath, 'w');
   1063 	fclose(fp);
   1064 
   1065 	relpath = "";
   1066 
   1067 	return lc;
   1068 }

   1070 const char *
   1071 filemode(git_filemode_t m)
   1072 {
   1073 	static char mode[11];
   1074 
   1075 	memset(mode, '-', sizeof(mode) - 1);
   1076 	mode[10] = '\0';
   1077 
   1078 	if (S_ISREG(m))
   1079 		mode[0] = '-';
   1080 	else if (S_ISBLK(m))
   1081 		mode[0] = 'b';
   1082 	else if (S_ISCHR(m))
   1083 		mode[0] = 'c';
   1084 	else if (S_ISDIR(m))
   1085 		mode[0] = 'd';
   1086 	else if (S_ISFIFO(m))
   1087 		mode[0] = 'p';
   1088 	else if (S_ISLNK(m))
   1089 		mode[0] = 'l';
   1090 	else if (S_ISSOCK(m))
   1091 		mode[0] = 's';
   1092 	else
   1093 		mode[0] = '?';
   1094 
   1095 	if (m & S_IRUSR) mode[1] = 'r';
   1096 	if (m & S_IWUSR) mode[2] = 'w';
   1097 	if (m & S_IXUSR) mode[3] = 'x';
   1098 	if (m & S_IRGRP) mode[4] = 'r';
   1099 	if (m & S_IWGRP) mode[5] = 'w';
   1100 	if (m & S_IXGRP) mode[6] = 'x';
   1101 	if (m & S_IROTH) mode[7] = 'r';
   1102 	if (m & S_IWOTH) mode[8] = 'w';
   1103 	if (m & S_IXOTH) mode[9] = 'x';
   1104 
   1105 	if (m & S_ISUID) mode[3] = (mode[3] == 'x') ? 's' : 'S';
   1106 	if (m & S_ISGID) mode[6] = (mode[6] == 'x') ? 's' : 'S';
   1107 	if (m & S_ISVTX) mode[9] = (mode[9] == 'x') ? 't' : 'T';
   1108 
   1109 	return mode;

   1112 int
   1113 writefilestree(FILE *fp, git_tree *tree, const char *path)
   1114 {
   1115 	const git_tree_entry *entry = NULL;
   1116 	git_object *obj = NULL;
   1117 	const char *entryname;
   1118 	char filepath[PATH_MAX], entrypath[PATH_MAX], oid[8];
   1119 	size_t count, i, lc, filesize;
   1120 	int r, ret;
   1121 
   1122 	count = git_tree_entrycount(tree);
   1123 	for (i = 0; i < count; i++) {
   1124 		if (!(entry = git_tree_entry_byindex(tree, i)) ||
   1125 		    !(entryname = git_tree_entry_name(entry)))
   1126 			return -1;
   1127 		joinpath(entrypath, sizeof(entrypath), path, entryname);
   1128 
   1129 		r = snprintf(filepath, sizeof(filepath), "file/%s.html",
   1130 		         entrypath);
   1131 		if (r < 0 || (size_t)r >= sizeof(filepath))
   1132 			errx(1, "path truncated: 'file/%s.html'", entrypath);
   1133 
   1134 		if (!git_tree_entry_to_object(&obj, repo, entry)) {
   1135 			switch (git_object_type(obj)) {
   1136 			case GIT_OBJ_BLOB:
   1137 				break;
   1138 			case GIT_OBJ_TREE:
   1139 				/* NOTE: recurses */
   1140 				ret = writefilestree(fp, (git_tree *)obj,
   1141 				                     entrypath);
   1142 				git_object_free(obj);
   1143 				if (ret)
   1144 					return ret;
   1145 				continue;
   1146 			default:
   1147 				git_object_free(obj);
   1148 				continue;
   1149 			}
   1150 
   1151 			filesize = git_blob_rawsize((git_blob *)obj);
   1152 			lc = writeblob(obj, filepath, entryname, filesize);
   1153 
   1154 			fputs("<tr><td>", fp);
   1155 			fputs(filemode(git_tree_entry_filemode(entry)), fp);
   1156 			fprintf(fp, "</td><td><a href=\"%s", relpath);
   1157 			percentencode(fp, filepath, strlen(filepath));
   1158 			fputs("\">", fp);
   1159 			xmlencode(fp, entrypath, strlen(entrypath));
   1160 			fputs("</a></td><td class=\"num\" align=\"right\">", fp);
   1161 			if (lc > 0)
   1162 				fprintf(fp, "%zuL", lc);
   1163 			else
   1164 				fprintf(fp, "%zuB", filesize);
   1165 			fputs("</td></tr>\n", fp);
   1166 			git_object_free(obj);
   1167 		} else if (git_tree_entry_type(entry) == GIT_OBJ_COMMIT) {
   1168 			/* commit object in tree is a submodule */
   1169 			fprintf(fp, "<tr><td>m---------</td><td><a href=\"%sfile/.gitmodules.html\">",
   1170 				relpath);
   1171 			xmlencode(fp, entrypath, strlen(entrypath));
   1172 			fputs("</a> @ ", fp);
   1173 			git_oid_tostr(oid, sizeof(oid), git_tree_entry_id(entry));
   1174 			xmlencode(fp, oid, strlen(oid));
   1175 			fputs("</td><td class=\"num\" align=\"right\"></td></tr>\n", fp);
   1176 		}
   1177 	}
   1178 
   1179 	return 0;
   1180 }

   1206 int
   1207 writerefs(FILE *fp)
   1208 {
   1209 	struct referenceinfo *ris = NULL;
   1210 	struct commitinfo *ci;
   1211 	size_t count, i, j, refcount;
   1212 	const char *titles[] = { "Branches", "Tags" };
   1213 	const char *ids[] = { "branches", "tags" };
   1214 	const char *s;
   1215 
   1216 	if (getrefs(&ris, &refcount) == -1)
   1217 		return -1;
   1218 
   1219 	for (i = 0, j = 0, count = 0; i < refcount; i++) {
   1220 		if (j == 0 && git_reference_is_tag(ris[i].ref)) {
   1221 			if (count)
   1222 				fputs("</tbody></table><br/>\n", fp);
   1223 			count = 0;
   1224 			j = 1;
   1225 		}
   1226 
   1227 		/* print header if it has an entry (first). */
   1228 		if (++count == 1) {
   1229 			fprintf(fp, "<h2>%s</h2><table id=\"%s\">"
   1230 		                "<thead>\n<tr><td><b>Name</b></td>"
   1231 			        "<td><b>Last commit date</b></td>"
   1232 			        "<td><b>Author</b></td>\n</tr>\n"
   1233 			        "</thead><tbody>\n",
   1234 			         titles[j], ids[j]);
   1235 		}
   1236 
   1237 		ci = ris[i].ci;
   1238 		s = git_reference_shorthand(ris[i].ref);
   1239 
   1240 		fputs("<tr><td>", fp);
   1241 		xmlencode(fp, s, strlen(s));
   1242 		fputs("</td><td>", fp);
   1243 		if (ci->author)
   1244 			printtimeshort(fp, &(ci->author->when));
   1245 		fputs("</td><td>", fp);
   1246 		if (ci->author)
   1247 			xmlencode(fp, ci->author->name, strlen(ci->author->name));
   1248 		fputs("</td></tr>\n", fp);
   1249 	}
   1250 	/* table footer */
   1251 	if (count)
   1252 		fputs("</tbody></table><br/>\n", fp);
   1253 
   1254 	for (i = 0; i < refcount; i++) {
   1255 		commitinfo_free(ris[i].ci);
   1256 		git_reference_free(ris[i].ref);
   1257 	}
   1258 	free(ris);
   1259 
   1260 	return 0;
   1261 }

   1271 int
   1272 main(int argc, char *argv[])
   1273 {
   1274 	git_object *obj = NULL;
   1275 	const git_oid *head = NULL;
   1276 	mode_t mask;
   1277 	FILE *fp, *fpread;
   1278 	char path[PATH_MAX], repodirabs[PATH_MAX + 1], *p;
   1279 	char tmppath[64] = "cache.XXXXXXXXXXXX", buf[BUFSIZ];
   1280 	size_t n;
   1281 	int i, fd;
   1282 
   1283 	for (i = 1; i < argc; i++) {
   1284 		if (argv[i][0] != '-') {
   1285 			if (repodir)
   1286 				usage(argv[0]);
   1287 			repodir = argv[i];
   1288 		} else if (argv[i][1] == 'c') {
   1289 			if (nlogcommits > 0 || i + 1 >= argc)
   1290 				usage(argv[0]);
   1291 			cachefile = argv[++i];
   1292 		} else if (argv[i][1] == 'l') {
   1293 			if (cachefile || i + 1 >= argc)
   1294 				usage(argv[0]);
   1295 			errno = 0;
   1296 			nlogcommits = strtoll(argv[++i], &p, 10);
   1297 			if (argv[i][0] == '\0' || *p != '\0' ||
   1298 			    nlogcommits <= 0 || errno)
   1299 				usage(argv[0]);
   1300 		} else if (argv[i][1] == 'u') {
   1301 			if (i + 1 >= argc)
   1302 				usage(argv[0]);
   1303 			baseurl = argv[++i];
   1304 		}
   1305 	}
   1306 	if (!repodir)
   1307 		usage(argv[0]);
   1308 
   1309 	if (!realpath(repodir, repodirabs))
   1310 		err(1, "realpath");
   1311 
   1312 	/* do not search outside the git repository:
   1313 	   GIT_CONFIG_LEVEL_APP is the highest level currently */
   1314 	git_libgit2_init();
   1315 	for (i = 1; i <= GIT_CONFIG_LEVEL_APP; i++)
   1316 		git_libgit2_opts(GIT_OPT_SET_SEARCH_PATH, i, "");
   1317 	/* do not require the git repository to be owned by the current user */
   1318 	git_libgit2_opts(GIT_OPT_SET_OWNER_VALIDATION, 0);
   1319 
   1320 #ifdef __OpenBSD__
   1321 	if (unveil(repodir, "r") == -1)
   1322 		err(1, "unveil: %s", repodir);
   1323 	if (unveil(".", "rwc") == -1)
   1324 		err(1, "unveil: .");
   1325 	if (cachefile && unveil(cachefile, "rwc") == -1)
   1326 		err(1, "unveil: %s", cachefile);
   1327 
   1328 	if (cachefile) {
   1329 		if (pledge("stdio rpath wpath cpath fattr", NULL) == -1)
   1330 			err(1, "pledge");
   1331 	} else {
   1332 		if (pledge("stdio rpath wpath cpath", NULL) == -1)
   1333 			err(1, "pledge");
   1334 	}
   1335 #endif
   1336 
   1337 	if (git_repository_open_ext(&repo, repodir,
   1338 		GIT_REPOSITORY_OPEN_NO_SEARCH, NULL) < 0) {
   1339 		fprintf(stderr, "%s: cannot open repository\n", argv[0]);
   1340 		return 1;
   1341 	}
   1342 
   1343 	/* find HEAD */
   1344 	if (!git_revparse_single(&obj, repo, "HEAD"))
   1345 		head = git_object_id(obj);
   1346 	git_object_free(obj);
   1347 
   1348 	/* use directory name as name */
   1349 	if ((name = strrchr(repodirabs, '/')))
   1350 		name++;
   1351 	else
   1352 		name = "";
   1353 
   1354 	/* strip .git suffix */
   1355 	if (!(strippedname = strdup(name)))
   1356 		err(1, "strdup");
   1357 	if ((p = strrchr(strippedname, '.')))
   1358 		if (!strcmp(p, ".git"))
   1359 			*p = '\0';
   1360 
   1361 	/* read description or .git/description */
   1362 	joinpath(path, sizeof(path), repodir, "description");
   1363 	if (!(fpread = fopen(path, "r"))) {
   1364 		joinpath(path, sizeof(path), repodir, ".git/description");
   1365 		fpread = fopen(path, "r");
   1366 	}
   1367 	if (fpread) {
   1368 		if (!fgets(description, sizeof(description), fpread))
   1369 			description[0] = '\0';
   1370 		checkfileerror(fpread, path, 'r');
   1371 		fclose(fpread);
   1372 	}
   1373 
   1374 	/* read url or .git/url */
   1375 	joinpath(path, sizeof(path), repodir, "url");
   1376 	if (!(fpread = fopen(path, "r"))) {
   1377 		joinpath(path, sizeof(path), repodir, ".git/url");
   1378 		fpread = fopen(path, "r");
   1379 	}
   1380 	if (fpread) {
   1381 		if (!fgets(cloneurl, sizeof(cloneurl), fpread))
   1382 			cloneurl[0] = '\0';
   1383 		checkfileerror(fpread, path, 'r');
   1384 		fclose(fpread);
   1385 		cloneurl[strcspn(cloneurl, "\n")] = '\0';
   1386 	}
   1387 
   1388 	/* check LICENSE */
   1389 	for (i = 0; i < LEN(licensefiles) && !license; i++) {
   1390 		if (!git_revparse_single(&obj, repo, licensefiles[i]) &&
   1391 		    git_object_type(obj) == GIT_OBJ_BLOB)
   1392 			license = licensefiles[i] + strlen("HEAD:");
   1393 		git_object_free(obj);
   1394 	}
   1395 
   1396 	/* check README */
   1397 	for (i = 0; i < LEN(readmefiles) && !readme; i++) {
   1398 		if (!git_revparse_single(&obj, repo, readmefiles[i]) &&
   1399 		    git_object_type(obj) == GIT_OBJ_BLOB)
   1400 			readme = readmefiles[i] + strlen("HEAD:");
   1401 		git_object_free(obj);
   1402 	}
   1403 
   1404 	if (!git_revparse_single(&obj, repo, "HEAD:.gitmodules") &&
   1405 	    git_object_type(obj) == GIT_OBJ_BLOB)
   1406 		submodules = ".gitmodules";
   1407 	git_object_free(obj);
   1408 
   1409 	/* log for HEAD */
   1410 	fp = efopen("log.html", "w");
   1411 	relpath = "";
   1412 	mkdir("commit", S_IRWXU | S_IRWXG | S_IRWXO);
   1413 	writeheader(fp, "Log");
   1414 	fputs("<table id=\"log\"><thead>\n<tr><td><b>Date</b></td>"
   1415 	      "<td><b>Commit message</b></td>"
   1416 	      "<td><b>Author</b></td><td class=\"num\" align=\"right\"><b>Files</b></td>"
   1417 	      "<td class=\"num\" align=\"right\"><b>+</b></td>"
   1418 	      "<td class=\"num\" align=\"right\"><b>-</b></td></tr>\n</thead><tbody>\n", fp);
   1419 
   1420 	if (cachefile && head) {
   1421 		/* read from cache file (does not need to exist) */
   1422 		if ((rcachefp = fopen(cachefile, "r"))) {
   1423 			if (!fgets(lastoidstr, sizeof(lastoidstr), rcachefp))
   1424 				errx(1, "%s: no object id", cachefile);
   1425 			if (git_oid_fromstr(&lastoid, lastoidstr))
   1426 				errx(1, "%s: invalid object id", cachefile);
   1427 		}
   1428 
   1429 		/* write log to (temporary) cache */
   1430 		if ((fd = mkstemp(tmppath)) == -1)
   1431 			err(1, "mkstemp");
   1432 		if (!(wcachefp = fdopen(fd, "w")))
   1433 			err(1, "fdopen: '%s'", tmppath);
   1434 		/* write last commit id (HEAD) */
   1435 		git_oid_tostr(buf, sizeof(buf), head);
   1436 		fprintf(wcachefp, "%s\n", buf);
   1437 
   1438 		writelog(fp, head);
   1439 
   1440 		if (rcachefp) {
   1441 			/* append previous log to log.html and the new cache */
   1442 			while (!feof(rcachefp)) {
   1443 				n = fread(buf, 1, sizeof(buf), rcachefp);
   1444 				if (ferror(rcachefp))
   1445 					break;
   1446 				if (fwrite(buf, 1, n, fp) != n ||
   1447 				    fwrite(buf, 1, n, wcachefp) != n)
   1448 					    break;
   1449 			}
   1450 			checkfileerror(rcachefp, cachefile, 'r');
   1451 			fclose(rcachefp);
   1452 		}
   1453 		checkfileerror(wcachefp, tmppath, 'w');
   1454 		fclose(wcachefp);
   1455 	} else {
   1456 		if (head)
   1457 			writelog(fp, head);
   1458 	}
   1459 
   1460 	fputs("</tbody></table>", fp);
   1461 	writefooter(fp);
   1462 	checkfileerror(fp, "log.html", 'w');
   1463 	fclose(fp);
   1464 
   1465 	/* files for HEAD */
   1466 	fp = efopen("files.html", "w");
   1467 	writeheader(fp, "Files");
   1468 	if (head)
   1469 		writefiles(fp, head);
   1470 	writefooter(fp);
   1471 	checkfileerror(fp, "files.html", 'w');
   1472 	fclose(fp);
   1473 
   1474 	/* summary page with branches and tags */
   1475 	fp = efopen("refs.html", "w");
   1476 	writeheader(fp, "Refs");
   1477 	writerefs(fp);
   1478 	writefooter(fp);
   1479 	checkfileerror(fp, "refs.html", 'w');
   1480 	fclose(fp);
   1481 
   1482 	/* Atom feed */
   1483 	fp = efopen("atom.xml", "w");
   1484 	writeatom(fp, 1);
   1485 	checkfileerror(fp, "atom.xml", 'w');
   1486 	fclose(fp);
   1487 
   1488 	/* Atom feed for tags / releases */
   1489 	fp = efopen("tags.xml", "w");
   1490 	writeatom(fp, 0);
   1491 	checkfileerror(fp, "tags.xml", 'w');
   1492 	fclose(fp);
   1493 
   1494 	/* rename new cache file on success */
   1495 	if (cachefile && head) {
   1496 		if (rename(tmppath, cachefile))
   1497 			err(1, "rename: '%s' to '%s'", tmppath, cachefile);
   1498 		umask((mask = umask(0)));
   1499 		if (chmod(cachefile,
   1500 		    (S_IRUSR|S_IWUSR|S_IRGRP|S_IWGRP|S_IROTH|S_IWOTH) & ~mask))
   1501 			err(1, "chmod: '%s'", cachefile);
   1502 	}
   1503 
   1504 	/* cleanup */
   1505 	git_repository_free(repo);
   1506 	git_libgit2_shutdown();
   1507 
   1508 	return 0;
